​ 14.1 单总线概述

单总线（One-Wire）是一种串行通信协议和硬件总线，用于在电子设备之间传输数据和控制信号。它是由独立的芯片制造商Dallas Semiconductor开发的，并且在多种应用中得到了广泛应用。与SPI、I2C等串行数据通信方式不同，单总线的特点是只需要一根信号线，既可以传输时钟又可以传输数据，而且数据是双向的。所以单总线具有节省IO口，结构简单，便于扩展和维护等特点。

单总线用于各种应用，包括温度传感器，湿度传感器，EEPROM存储器，时钟等。它在许多领域中得到了广泛应用，例如工业自动化，家庭自动化，物联网和电子设备监测等。在本章节中将使用温度传感器DS18b20来学习单总线。

单总线由以下硬件部分来组成：

信号线：单总线使用一根信号线进行数据传输和通信。这根线被称为数据线，也是提供电源的线路。单总线上的所有设备都连接到这根信号线上。

上拉电阻：单总线需要一个上拉电阻连接到信号线和电源之间，以确保在没有设备发送数据时，信号线上的电平保持为高电平（逻辑1）。上拉电阻的值通常在4.7千欧姆到10千欧姆之间。

设备：单总线支持多个设备连接到信号线上。每个设备都具有唯一的64位地址，通过这个地址来识别和选择通信的目标设备。设备可以是各种类型的传感器，存储器，时钟等。

处理器：处理器是单总线上的控制器，处理器负责发送命令，读取响应和控制单总线上的从设备。

​ 14.2 DS18B20介绍

DS18B20是一种基于单总线协议的数字温度传感器，提供9到 12 位分辨率，测量精度可达 ±0.4°C（在 -10°C70°C 范围内），分辨率为 0.0625°C，测温范围为-55°C 至 +125°C。它支持通过数据线供电（电压范围 2.5V5.5V）或外部电源供电，具有低功耗、高精度和强抗静电能力（HBM 8000V，MM 800V）。每颗芯片内置全球唯一的 64 位序列号，允许多个设备挂载在同一根单总线上实现多点分布式测温，适用于大面积温度监测场景。

该芯片无需外围元件即可实现温度采集，并可通过寄存器设置温度上下限报警功能，利用“报警搜索”命令快速识别超出设定范围的设备。其常见的封装形式包括TO-92、TO-92S、MSOP8和SOP8 等，引脚分别为VDD（电源）、DQ（数据线）和GND（地）。

DS18B20 广泛应用于室内环境监测、温室农业、水族箱控制、工业自动化、太阳能热水器、冷链物流以及气象站等场景，特别适合需要多点测温和远距离布设的应用场合。

/* 自定义 IOCTL 命令 */
#define SET_RESOLUTION       _IOW('A', 0, int)
#define READ_RESOLUTION      _IOR('A', 1, int)

/* 支持的分辨率选项 */
#define SET_RESOLUTION_9     9
#define SET_RESOLUTION_10    10
#define SET_RESOLUTION_11    11
#define SET_RESOLUTION_12    12

/* 驱动私有数据结构 */
struct ds18b20_data {
    dev_t dev_num;
    struct cdev ds18b20_cdev;
    struct class *ds18b20_class;
    struct device *ds18b20_device;
    struct gpio_desc *ds18b20_gpio;
};

static struct ds18b20_data *ds18b20;

/**
 * ds18b20_reset - 向 DS18B20 发送复位信号
 */
void ds18b20_reset(void)
{
    gpiod_direction_output(ds18b20->ds18b20_gpio, 0);
    udelay(700);

    gpiod_set_value(ds18b20->ds18b20_gpio, 1);
    gpiod_direction_input(ds18b20->ds18b20_gpio);

    while (gpiod_get_value(ds18b20->ds18b20_gpio)); // 等待应答低电平
    while (!gpiod_get_value(ds18b20->ds18b20_gpio)); // 等待恢复高电平

    udelay(480);
}

/**
 * ds18b20_writebit - 向 DS18B20 写入一个 bit
 * @bit: 要写入的 bit (0 或 1)
 */
void ds18b20_writebit(unsigned char bit)
{
    gpiod_direction_output(ds18b20->ds18b20_gpio, 0);
    udelay(2);

    if (bit) {
        udelay(10);
    } else {
        udelay(65);
    }

    gpiod_set_value(ds18b20->ds18b20_gpio, 1);
    udelay(2);
}

/**
 * ds18b20_writebyte - 向 DS18B20 写入一个字节
 * @data: 要写入的数据
 */
void ds18b20_writebyte(int data)
{
    int i;
    for (i = 0; i < 8; i++) {
        ds18b20_writebit(data & 0x01);
        data >>= 1;
    }
}

/**
 * ds18b20_readbit - 从 DS18B20 读取一个 bit
 * Return: 读取到的 bit 值
 */
unsigned char ds18b20_readbit(void)
{
    unsigned char bit;

    gpiod_direction_output(ds18b20->ds18b20_gpio, 0);
    udelay(2);

    gpiod_direction_input(ds18b20->ds18b20_gpio);
    udelay(10);

    bit = gpiod_get_value(ds18b20->ds18b20_gpio);
    udelay(60);

    return bit;
}

/**
 * ds18b20_readbyte - 从 DS18B20 读取一个字节
 * Return: 读取到的字节数据
 */
int ds18b20_readbyte(void)
{
    int data = 0;
    int i;

    for (i = 0; i < 8; i++) {
        data |= ds18b20_readbit() << i;
    }

    return data;
}

/**
 * ds18b20_readtemp - 从 DS18B20 读取温度值
 * Return: 温度原始值（16位补码格式）
 */
int ds18b20_readtemp(void)
{
    int temp_l, temp_h, temp;

    ds18b20_reset();
    ds18b20_writebyte(0xCC); /* Skip ROM */
    ds18b20_writebyte(0x44); /* Start conversion */

    mdelay(750); /* 等待转换完成 */

    ds18b20_reset();
    ds18b20_writebyte(0xCC); /* Skip ROM */
    ds18b20_writebyte(0xBE); /* Read scratchpad */

    temp_l = ds18b20_readbyte();
    temp_h = ds18b20_readbyte();

    temp = (temp_h << 8) | temp_l;

    return temp;
}

/**
 * set_resolution - 设置 DS18B20 的温度分辨率
 * @args: 分辨率参数（9~12）
 */
void set_resolution(int args)
{
    ds18b20_reset();
    ds18b20_writebyte(0xCC); /* Skip ROM */
    ds18b20_writebyte(0x4E); /* Write scratchpad */

    ds18b20_writebyte(60);   /* TH register */
    ds18b20_writebyte(10);   /* TL register */

    switch (args) {
    case SET_RESOLUTION_9:
        ds18b20_writebyte(0x1F);
        break;
    case SET_RESOLUTION_10:
        ds18b20_writebyte(0x3F);
        break;
    case SET_RESOLUTION_11:
        ds18b20_writebyte(0x5F);
        break;
    case SET_RESOLUTION_12:
        ds18b20_writebyte(0x7F);
        break;
    default:
        break;
    }
}

/**
 * check_args - 检查设置的分辨率是否生效
 * @args: 分辨率参数
 * Return: 成功返回 0，失败返回 -1
 */
int check_args(int args)
{
    int ret = -1;

    ds18b20_reset();
    ds18b20_writebyte(0xCC); /* Skip ROM */
    ds18b20_writebyte(0xBE); /* Read scratchpad */

    ds18b20_readbyte(); /* TH */
    ds18b20_readbyte(); /* TL */

    switch (args) {
    case SET_RESOLUTION_9:
        if (ds18b20_readbyte() == 0x1F)
            ret = 0;
        break;
    case SET_RESOLUTION_10:
        if (ds18b20_readbyte() == 0x3F)
            ret = 0;
        break;
    case SET_RESOLUTION_11:
        if (ds18b20_readbyte() == 0x5F)
            ret = 0;
        break;
    case SET_RESOLUTION_12:
        if (ds18b20_readbyte() == 0x7F)
            ret = 0;
        break;
    default:
        break;
    }

    return ret;
}

/**
 * read_resolution - 读取当前设置的分辨率
 * Return: 当前配置寄存器值
 */
int read_resolution(void)
{
    int ret;

    ds18b20_reset();
    ds18b20_writebyte(0xCC); /* Skip ROM */
    ds18b20_writebyte(0xBE); /* Read scratchpad */

    ds18b20_readbyte(); /* TH */
    ds18b20_readbyte(); /* TL */
    ds18b20_readbyte(); /* reserved */
    ds18b20_readbyte(); /* reserved */

    ret = ds18b20_readbyte(); /* Config reg */

    return ret;
}

static int ds18b20_open(struct inode *inode, struct file *file)

static long ds18b20_ioctl(struct file *file, unsigned int cmd, unsigned long args)

    int resolution;

    switch (cmd) {
    case SET_RESOLUTION:
        if (args >= SET_RESOLUTION_9 && args <= SET_RESOLUTION_12) {

        ret = PTR_ERR(ds18b20->ds18b20_class);
        goto err_unregister;
    }

    ds18b20->ds18b20_device = device_create(ds18b20->ds18b20_class, NULL, ds18b20->dev_num, NULL, "ds18b20");

        ret = PTR_ERR(ds18b20->ds18b20_device);
        goto err_class_destroy;
    }

    ds18b20->ds18b20_gpio = gpiod_get_optional(&pdev->dev, "ds18b20", GPIOD_OUT_HIGH);
    if (!ds18b20->ds18b20_gpio) {
        ret = -EBUSY;
        goto err_device_destroy;
    }

    gpiod_direction_output(ds18b20->ds18b20_gpio, 1);
    return 0;

err_device_destroy:
    device_destroy(ds18b20->ds18b20_class, ds18b20->dev_num);
err_class_destroy:
    class_destroy(ds18b20->ds18b20_class);
err_unregister:
    unregister_chrdev_region(ds18b20->dev_num, 1);
err_kfree:
    kfree(ds18b20);
    return ret;
}

/**
 * 根据传感器读取的原始数据计算温度值
 * @param value 传感器读取的原始数据
 */
void ds18b20_get_temp(int value) {
    char sig;   // 温度正负号
    float temp; // 温度值

    // 判断温度正负号
    if ((value >> 11) & 0x01) {
        sig = '-';
        value = ~value + 1;
        value &= ~(0xf8 << 8);
    } else {
        sig = '+';
    }

    // 计算温度值

第15章 I2C总线

​ 15.1 I2C总线概述

在现代电子设备中，常常需要多个集成电路芯片协同工作，实现各种复杂功能。这些芯片之间需要进行数据通信，传统的大量导线连接方式会导致引脚数量多、PCB布线复杂、成本高、功耗大等问题，给设计和制造带来很大困难。为了解决这些问题，1982年荷兰飞利浦公司提出了I²C（Inter-Integrated Circuit）协议，这是一种简单高效的芯片间通信协议，极大地简化了硬件电路设计，提高了资源利用率。

I²C总线仅需两根信号线：SDA（数据线）和SCL（时钟线），通常还需要两个上拉电阻将这两根线在空闲状态下保持高电平。主设备控制通信过程，包括发起通信、发送地址和方向位、提供时钟信号等，而从设备则根据接收到的地址做出响应。I²C采用同步串行通信方式，以字节为单位传输数据，高位先传，每个字节传输后都伴随一个应答信号，用于确认数据是否被正确接收。

I²C支持多种通信速率，包括标准模式（100kbps）、快速模式（400kbps）和高速模式（3.4Mbps），并且使用7位地址空间，最多可连接127个从设备，每个从设备都有唯一的地址。这种寻址机制使得主机可以准确地与目标从机进行通信。此外，I²C总线还支持多主机结构，通过仲裁机制避免多个主机同时操作总线造成的冲突，确保系统稳定运行。

总的来说，I²C协议因其硬件简单、易于扩展、成本低廉等优点，广泛应用于嵌入式系统中，如传感器、EEPROM、LCD显示屏等外设之间的通信。它已经成为现代电子系统内部通信的重要标准之一。

​ 15.2 I2C时序分析

I²C通信以起始信号（START）开始，以终止信号（STOP）结束。起始信号由主设备发出，表现为SCL为高电平时SDA出现一个下降沿；终止信号同样是主设备发出，表现为SCL为高电平时SDA出现一个上升沿。在起始信号之后，总线处于忙碌状态，直到收到终止信号后才释放。

每次数据传输以字节为单位进行，每个字节为8位，高位（MSB）先传。每个字节传输完成后，接收方需发送一个应答信号（ACK）或非应答信号（NACK）作为回应。如果接收方在第9个时钟周期将SDA拉低，表示成功接收（ACK）；若SDA保持高电平，则表示未成功接收（NACK）。NACK可能由多种情况引起，如无设备响应地址、设备正忙、无法识别命令、无法继续接收数据或表示传输结束等。

数据传输过程中，首先发送7位从设备地址，随后是1位读写方向位：0表示写操作，1表示读操作。一次完整的传输通常由主设备发送停止信号结束，但也可以通过发送重复起始信号（Repeated START）来切换目标地址或读写方向，从而实现连续的多段通信而无需释放总线。这种方式支持灵活的读写组合，适用于复杂的数据交互场景。

​ 15.3 I2C子系统框架

Linux系统中的I²C子系统可以划分为三个主要层次：用户空间、内核空间和硬件层。其中，内核空间是整个I²C通信机制的核心部分，主要包括 I²C设备驱动层**、I²C核心层 和 I²C适配器驱动层。

I²C设备驱动层负责实现与具体外设的交互，为上层应用提供统一的访问接口。该层通过 i2c_client 结构体表示一个连接到I²C总线上的从设备，包含设备地址、所属适配器等信息；同时，它还提供 i2c_driver 驱动结构，用于实现设备的初始化、数据读写和配置操作，并通过标准接口与用户空间进行交互。此外，I²C总线子系统负责管理整个I²C总线，协调 i2c_client 与 i2c_driver 的匹配与通信。

I²C适配器驱动层则面向具体的硬件控制器，负责实现I²C协议的底层时序控制和数据传输功能。它向上层提供统一的数据发送和接收接口，管理I²C总线上的从设备，并处理可能发生的总线错误或异常情况。

I²C核心层位于中间，起到承上启下的作用，是连接设备驱动层和适配器驱动层的桥梁。其主要功能由一组核心函数实现，如 i2c_master_send、i2c_master_recv 和更通用的 i2c_transfer 函数。其中，前两者是对后者封装的简化接口，而 i2c_transfer 则接受适配器指针、消息数组和消息数量作为参数，能够灵活完成复杂的I²C通信任务，常用于实际开发中深入理解I²C数据收发过程。

int i2c_read_reg(int fd, unsigned char addr, unsigned char reg) {
    unsigned char data;
    struct i2c_msg msgs[2] = {
        {
            .addr = addr,
            .flags = 0,
            .len = 1,
            .buf = &reg
        },
        {
            .addr = addr,
            .flags = I2C_M_RD,
            .len = 1,
            .buf = &data
        }
    };
    struct i2c_rdwr_ioctl_data msgset = {
        .msgs = msgs,
        .nmsgs = 2
    };

    if (ioctl(fd, I2C_RDWR, &msgset) < 0) {
        perror("I2C read failed");
        return -1;
    }

    return data;
}

// 向指定寄存器写入多个字节数据
int i2c_write_reg(int fd, unsigned char addr, unsigned char reg, const unsigned char *data, int len) {
    unsigned char buf[256];
    if (len >= sizeof(buf)) return -1;

    buf[0] = reg;
    memcpy(&buf[1], data, len);

    struct i2c_msg msg = {
        .addr = addr,
        .flags = 0,
        .len = len + 1,
        .buf = buf
    };
    struct i2c_rdwr_ioctl_data msgset = {
        .msgs = &msg,
        .nmsgs = 1
    };

    if (ioctl(fd, I2C_RDWR, &msgset) < 0) {
        perror("I2C write failed");
        return -1;
    }

    return 0;
}
