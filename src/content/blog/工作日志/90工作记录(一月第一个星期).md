~~~shell
# 安装编译所需的依赖包

    # 检查网络连接

            udev unzip uuid-dev wget zip zlib1g-dev distcc lib32ncurses-dev

        )

        # Ubuntu 18.04 特定的依赖包
        UBUNTU_18_PACKAGES=(
            liblz-dev liblzo2-2 liblzo2-dev mtd-utils squashfs-tools schedtool
            g++-multilib lib32z1-dev lib32ncurses5-dev lib32readline-dev gcc-multilib
            patchelf chrpath texinfo diffstat python3-pip subversion sed binutils

            python2 python3-distutils libpython2.7-dev
        )

            echo -e "\e[33m正在安装 Ubuntu 18.04 编译所需依赖包...\e[0m"

            sudo apt-get -y upgrade

            echo -e "\e[33m正在安装 Ubuntu 20.04 / 22.04 编译所需依赖包...\e[0m"
            sudo apt-get update
            sudo apt-get -y upgrade

    fi
}
~~~

~~~shell
#!/bin/bash
# 获取设备的 BOARD 值
BOARD=$(cat /proc/device-tree/compatible | tail -c 7 | head -c 6)
AIC_ID="a69c"
RTL_ID="0bda"
# 根据 BOARD 值执行相应的命令
if [ "$BOARD" == "rk3588" ]; then
    # 针对 rk3588 的命令
    insmod /usr/local/modules/rk3588/aic_load_fw.ko
    insmod /usr/local/modules/rk3588/aic8800_fdrv.ko
    rfkill unblock bluetooth
    hciconfig hci0 up

elif [ "$BOARD" == "rk3568" ]; then
    # 针对 rk3568 的命令
    insmod /usr/local/modules/rk3568/aic_load_fw.ko
    insmod /usr/local/modules/rk3568/aic8800_fdrv.ko
    rfkill unblock bluetooth
    hciconfig hci0 up

elif [ "$BOARD" == "rk3562" ]; then
    # 针对 rk3562 的命令
    insmod /usr/local/modules/rk3562/aic_load_fw.ko
    insmod /usr/local/modules/rk3562/aic8800_fdrv.ko
    rfkill unblock bluetooth
    hciconfig hci0 up

elif [ "$BOARD" == "rk3576" ]; then
    # 针对 rk3576 的命令
    # 检测 AIC8800D80（仅匹配前四位ID）
    if lsusb | grep -q "$AIC_ID"; then
        echo "[检测到] AICSemi AIC8800D80"
        insmod /usr/local/modules/rk3576/aic_load_fw.ko
        insmod /usr/local/modules/rk3576/aic8800_fdrv.ko
        rfkill unblock bluetooth
        hciconfig hci0 up
    # 检测 Realtek 8733（仅匹配前四位ID）
    elif lsusb | grep -q "$RTL_ID"; then
        echo "[检测到] Realtek 8723 (802.11n)"
        insmod /usr/local/modules/rk3576/RTL8733BU.ko
        insmod /usr/local/modules/rk3576/rtk_btusb.ko
        rfkill unblock bluetooth
        hciconfig hci0 up
    fi
else
    echo "未知的 BOARD 值: $BOARD"
fi
~~~
