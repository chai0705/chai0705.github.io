#!/bin/bash

# 获取make命令
function get_make_command()
{
  echo command make
}

# 从Makefile中提取LOCAL_MODULE模块名称
function get_make_modules()
{
    # 初始化模块名称变量为空
    local modules=""
    # 检查Makefile中是否包含LOCAL_MODULE关键字
    if [ ! -z "$(grep LOCAL_MODULE Makefile)" ]; then
        # 逐行读取Makefile文件
        while read -r mk; do
            # 匹配包含LOCAL_MODULE的行，提取等号后的值（模块名）
            if [[ "${mk}" =~ "LOCAL_MODULE" ]]; then modules+="${mk##*:=}"; fi
        done < Makefile
    fi
    # 返回提取到的模块名称
    echo $modules
}

# 自定义make构建函数，封装原生make命令并添加构建计时功能
function make()
{
    # 获取项目根目录
    local top_dir=$(gettop)
    # 保存当前工作目录
    local here=$PWD
    # 获取需要构建的模块列表
    local modules=$(get_make_modules)
    
    # 如果当前目录有build/env.sh（根目录），直接构建指定模块
    if [ -f build/env.sh ]; then
        _wrap_build $modules "$@"
    # 如果有指定模块，切换到根目录构建
    elif [ -n "$modules" ]; then
        cd $top_dir; _wrap_build $modules "$@"
        cd $here  # 切回原目录
    # 否则执行原生make命令
    else
        $(get_make_command) "$@"
    fi
}

# 构建包装函数：执行构建并输出带耗时和状态的结果
function _wrap_build()
{
    # 记录构建开始时间（秒级时间戳）
    local start_time=$(date +"%s")
    # 执行make命令，静默输出
    $(get_make_command) "$@" && echo > /dev/null
    # 保存命令执行返回值（0为成功，非0为失败）
    local ret=$?
    # 记录构建结束时间
    local end_time=$(date +"%s")
    # 计算总耗时（秒）
    local tdiff=$(($end_time-$start_time))
    # 转换为小时、分钟、秒
    local hours=$(($tdiff / 3600 ))
    local mins=$((($tdiff % 3600) / 60))
    local secs=$(($tdiff % 60))

    echo
    # 根据返回值输出不同颜色的结果信息
    if [ $ret -eq 0 ] ; then
        # 绿色输出构建成功信息
        echo -n -e "\033[1;32m #### make \""$@"\" completed successfully!"
    else
        # 红色输出构建失败信息
        echo -n -e "\033[1;31m #### make target \""$@"\" failed !!!"
    fi
    # 格式化输出耗时
    if [ $hours -gt 0 ] ; then
        printf "(%02g:%02g:%02g (hh:mm:ss))" $hours $mins $secs
    elif [ $mins -gt 0 ] ; then
        printf "(%02g:%02g (mm:ss))" $mins $secs
    else
        printf "(%s seconds)" $secs
    fi
    # 重置终端颜色
    echo -e " ####\033[1;0m"
    echo
    # 返回原命令的执行结果
    return $ret
}

# 查找最近的Makefile文件
function findmakefile()
{
    # 获取项目根目录
    local top_dir=$(gettop)
    # 根目录标识文件
    local top_file=build/env.sh
    # 保存当前工作目录
    local here=$PWD
    
    # 如果传入了目录参数且目录存在，切换到该目录
    if [[ ! -z "${1}" && -d "${top_dir}/${1}" ]]; then
        \cd ${top_dir}/${1}
    fi;
    
    local top_dir=""
    # 向上遍历目录直到找到Makefile或到达根目录
    while [ \( ! \( -f "${top_file}" \) \) -a \( "$PWD" != "/" \) ]; do
        top_dir=$(pwd)
        # 找到Makefile则输出路径并返回
        if [ -f "$top_dir/Makefile" ]; then
            echo $top_dir/Makefile
            \cd $here
            return
        fi
        \cd ..
    done
    # 切回原目录，未找到返回1
    \cd $here
    return 1
}

# 构建当前目录模块（mm命令）
function mm()
{
    # 获取项目根目录
    local top_dir=$(gettop)
    
    # 如果在项目根目录，直接执行make
    if [ -f "build/env.sh" ]; then
        make $@
    else
        # 查找最近的Makefile
        local makefile=$(findmakefile)
        
        # 检查根目录是否有效
        if [ ! "$top_dir" ]; then
            echo "Couldn't locate the top of the tree.  Try setting TOP."
            return 1
        # 检查Makefile是否找到
        elif [ ! "$makefile" ]; then
            echo "Couldn't locate a makefile from the current directory."
            return 1
        else
            # 保存当前目录，切换到Makefile所在目录执行构建
            local here=$(pwd)
            local there=$(dirname $makefile)
            cd ${there}; make $@
            cd ${here}
        fi
    fi
}

# 构建指定目录模块（mmm命令）
function mmm()
{
    # 获取项目根目录
    local top_dir=$(gettop)
    
    if [ "$top_dir" ]; then
        # 初始化变量
        local makefile=""
        local modules=""
        local modules_in_paths=""
        local args=""
        local dir=""
        local to_chop=""
        local dir_modules=""
        local get_install_path=""
        local get_install_paths=""
        
        # 分离参数：提取以-开头的参数
        local dash_args=$(echo "$@" | awk -v RS=" " -v ORS=" " '/^-.*$/')
        # 分离参数：提取目录参数
        local dirs=$(echo "$@" | awk -v RS=" " -v ORS=" " '/^[^-].*$/')
        
        # 遍历所有目录参数
        for dir in ${dirs} ; do
            # 提取目录中指定的模块名（格式：目录:模块1,模块2）
            dir_modules=$(echo ${dir} | sed -n -e 's/.*:\(.*$\)/\1/p' | sed 's/,/ /')
            # 提取纯目录路径（去掉模块部分）
            dir=$(echo ${dir} | sed -e 's/:.*//' -e 's:/$::')
            # 清理路径：移除开头的./和结尾的/
            dir=${dir#./}
            dir=${dir%/}
            
            local make_file=""
            # 如果未指定模块且目录存在，查找该目录的Makefile
            if [[ "${dir_modules}" = "" && -d "${top_dir}/${dir}" ]]; then
                make_file=$(findmakefile ${dir})
            else
                # 如果指定了模块，直接检查目录下的Makefile
                if [ -f "${dir}/Makefile" ]; then
                    local here=$PWD
                    cd $dir
                    make_file=$(pwd)
                    make_file=$make_file/Makefile
                    cd $here
                fi
            fi
            
            # 如果找到Makefile
            if [ "${make_file}" ]; then
                # 转换为相对根目录的路径
                local make_file=$(echo ${make_file}|sed 's:'${top_dir}'/::')
                # 如果未指定模块，添加modules-in-路径参数
                if [ "${dir_modules}" = "" ]; then
                    modules_in_paths="$modules_in_paths modules-in-$(dirname ${make_file})"
                    get_install_paths="$get_install_paths get-install-path-in-$(dirname ${make_file})"
                else
                    # 否则添加指定模块
                    modules="${modules} ${dir_modules}"
                fi
                makefile="${makefile} ${make_file}"
            else
                # 处理特殊参数
                case "${dir}" in
                  showcommands | snod | dist | *=* | -j*) args="${args} ${dir}";;
                  j*) args="${args} -${dir}";;
                  get-install-path) get_install_path=${dir};;
                  *) 
                    # 目录不存在或无Makefile，报错返回
                    if [ -d "${dir}" ]; then
                         echo "No Makefile in ${dir}.";
                    else
                         echo "Couldn't locate the directory ${dir}";
                    fi
                    return 1;;
                esac
            fi
        done
        
        # 如果指定了get-install-path参数，调整参数
        if [ -n "$get_install_path" ]; then
          args=${get_install_paths//\//-}
          modules=""
          modules_in_paths=""
        fi
        
        # 保存当前目录
        local here=$(pwd)
        
        # 如果找到Makefile，逐个目录执行构建
        if [ -n "${makefile}" ]; then
            for mk_file in ${makefile} ; do
                local there=${top_dir}/$(dirname $mk_file)
                cd ${there}; make ${args}
            done
        else
            # 否则在根目录执行构建
            cd ${top_dir}; make ${args}
        fi
        
        # 切回原目录
        cd ${here}
    else
        # 根目录无效，报错
        echo "Couldn't locate the top of the tree.  Try setting TOP."
        return 1
    fi
}

# 设置编译环境路径
function setpaths()
{
    # 定义各架构工具链路径
    arm_soc100_path=${PWD}/vendor/tools/host/huanglong/toolchains/arm-v100-linux/bin
    aarch64_soc100_path=${PWD}/vendor/tools/host/huanglong/toolchains/aarch64-v100-linux/bin
    host_path=${PWD}/vendor/tools/host/huanglong/image_tools/bin

    # 将工具链路径添加到PATH开头（避免重复添加）
    PATH=${arm_soc100_path}:${PATH//${arm_soc100_path}:/}
    PATH=${aarch64_soc100_path}:${PATH//${aarch64_soc100_path}:/}
    PATH=${host_path}:${PATH//${host_path}:/}

    # 为单模块驱动编译设置环境变量
    unset LINUX_PRODUCT_OUT
    export LINUX_PRODUCT_OUT=$(get_abs_build_var PRODUCT_OUT)
    export OUT=$LINUX_PRODUCT_OUT

    # 设置内核输出目录（根据芯片版本区分）
    KERNEL_OUT=$LINUX_PRODUCT_OUT/obj/KERNEL_OBJ_$(echo $(get_build_var CHIP_REVISION) | tr a-z A-Z)
    export KERNEL_OUT=$KERNEL_OUT

    # 设置主机端输出目录
    unset LINUX_HOST_OUT
    export LINUX_HOST_OUT=$(get_abs_build_var HOST_OUT)
}

# 检查当前shell是否为bash
function check_bash()
{
    # 获取shell版本信息
    is_bash=`${SHELL} --version | grep "GNU bash"`
    is_bash=${is_bash:0:8}
    
    # 检查是否为bash
    if [ "${is_bash}" != "GNU bash" ] ; then
        echo "The shell may NOT be a bash, we need a bash to build SDK, please use bash!";
        echo "To change your shell, you should be root first, and then set by steps followed:";
        echo "    rm -f /bin/sh";
        echo "    ln -s /bin/bash  /bin/sh";
        echo ""
    fi
}

# 获取项目根目录
function gettop
{
    # 根目录标识文件
    local TOPFILE=build/env.sh
    
    # 如果已设置TOP变量且有效，返回其绝对路径
    if [ -n "$TOP" -a -f "$TOP/$TOPFILE" ] ; then
        # 消除符号链接影响
        (cd $TOP; PWD= /bin/pwd)
    else
        # 如果当前目录有标识文件，返回当前目录
        if [ -f $TOPFILE ] ; then
            PWD= /bin/pwd
        else
            # 保存当前目录
            local HERE=$PWD
            T=
            # 向上遍历直到找到标识文件或到达根目录
            while [ \( ! \( -f $TOPFILE \) \) -a \( $PWD != "/" \) ]; do
                \cd ..
                T=`PWD= /bin/pwd -P`
            done
            \cd $HERE
            # 如果找到根目录，返回路径
            if [ -f "$T/$TOPFILE" ]; then
                echo $T
            fi
        fi
    fi
}

# 从LinuxProducts.mk获取可用的lunch选项
function get_lunch_choices()
{
    local product_mk=$1
    local choice_line
    
    # 逐行读取文件
    while read -r choice_line; do
        # 过滤不匹配的LOCAL_LUNCH_PATH
        if [[ "${choice_line}" =~ "LOCAL_LUNCH_PATH :=" && ! "${product_mk}" =~ "${choice_line##*\ }" ]]; then
            return
        fi
        # 提取COMMON_LUNCH_CHOICES并添加到选项列表
        if [[ "${choice_line}" =~ "COMMON_LUNCH_CHOICES" ]]; then
            add_lunch_combo "${choice_line##*\ }"
        fi
    done < "$1"
}

# 添加lunch选项到菜单（去重）
unset LUNCH_MENU_CHOICES
function add_lunch_combo()
{
    local new_combo=$1
    local c
    
    # 检查是否已存在，避免重复
    for c in ${LUNCH_MENU_CHOICES[@]} ; do
        if [ "$new_combo" = "$c" ] ; then
            return
        fi
    done
    # 添加新选项
    LUNCH_MENU_CHOICES=(${LUNCH_MENU_CHOICES[@]} $new_combo)
}

# 打印lunch菜单
function print_lunch_menu()
{
    # 获取系统类型
    local uname=$(uname)
    echo
    echo "You're building on" $uname
    echo
    echo "Lunch menu... pick a combo:"

    # 遍历选项列表，编号输出
    local i=1
    local choice
    for choice in ${LUNCH_MENU_CHOICES[@]}
    do
        echo "     $i. $choice"
        i=$(($i+1))
    done

    echo
}

# 选择编译产品（lunch命令）
function lunch()
{
    local answer

    # 如果传入参数，直接使用
    if [ "$1" ] ; then
        answer=$1
    else
        # 否则打印菜单并等待用户输入
        print_lunch_menu
        echo -n "Which would you like? [stb_wudangstick-eng] "
        read answer
    fi

    local selection=

    # 处理用户输入
    if [ -z "$answer" ]
    then
        # 空输入使用默认值
        selection=stb_wudangstick-eng
    elif (echo -n $answer | grep -q -e "^[0-9][0-9]*$")
    then
        # 数字输入选择对应序号的选项
        if [ $answer -le ${#LUNCH_MENU_CHOICES[@]} ]
        then
            selection=${LUNCH_MENU_CHOICES[$(($answer-1))]}
        fi
    else
        # 直接使用用户输入的选项名
        selection=$answer
    fi

    # 解析产品名和变体（格式：product-variant）
    local product product_device variant
    product=${selection%%-*} # 截取第一个-前的部分（产品名）
    variant=${selection#*-} # 截取第一个-后的部分（变体）

    # 检查产品名是否有效
    if [ -z "$product" ]
    then
        echo
        echo "Invalid lunch combo: $selection"
        return 1
    fi

    # 导出环境变量
    export TARGET_PRODUCT=$product
    export TARGET_BUILD_VARIANT=$variant

    # 打印配置信息
    printconfig
    # 设置环境变量
    set_stuff_for_environment
}

#!/bin/bash

# 设置脚本执行模式：遇到错误立即退出
set -e

# 显示脚本使用帮助信息
usage(){
    echo
    echo "使用方法"
    echo "       ./build.sh [--product-name <TARGET_PRODUCT>] [-h] [-j编译线程数]"
    echo "示例"
    echo "       请先执行source和lunch命令！然后执行 ./build.sh -h 或 ./build.sh -j8 或指定 --product-name 参数"
    exit 1

# 检查并解析命令行参数
check_parameter(){
    # 检查是否指定了产品名称参数
    if [ -n "$1" ] && [ "$1" == '--product-name' ]; then
        # 移除--product-name参数本身
        shift
        # 获取产品名称
        local product=$1
        # 移除产品名称参数，处理剩余参数
        shift
        
        # 定义支持的编译变体类型
        variants=("-eng" "-userdebug" "-user")
        # 遍历变体类型，检查产品名称中是否包含变体后缀
        for variant in ${variants[@]}; do
            if [[ "$product" =~ "$variant" ]]; then
                # 从产品名称中移除变体后缀
                product=${product/$variant/}
                # 导出编译变体环境变量（去掉前缀-）
                export TARGET_BUILD_VARIANT=${variant/-/}
                break
            fi
        done
        # 导出产品名称环境变量
        export TARGET_PRODUCT=${product}
    fi
    
    # 保存剩余的命令行参数
    command_args="$@"
    
    # 如果未指定目标产品，显示帮助信息并退出
    if [ -z "$TARGET_PRODUCT" ]; then usage;fi
    # 如果包含-h参数，显示帮助信息并退出
    if [ -n "$(echo "$1" | sed -n '/-h/p')" ];then usage;fi
}

# 设置编译参数
set_parameter(){
    # 如果未指定-j参数（编译线程数），自动设置为CPU核心数
    if [ -z "$(echo "$command_args" | sed -n '/-j/p')" ];then command_args+=" -j$(nproc)";fi

    # 提取参数中的键值对（格式如 KEY=VALUE）
    build_variants=$(echo "$command_args" |awk -F ' ' '{for(i=1;i<=NF;i++){print $i;}}' |awk '/=/{print}')
    # 遍历所有键值对，将其声明为全局变量
    for variant in $build_variants
    do
        echo "$variant"
        declare -g $variant
    done
}

# 清理编译中间文件
clear_intermediates(){
    # 获取编译中间文件输出目录
    INTERMEDIATES=$(get_build_var TARGET_OUT_INTERMEDIATES)
    # 获取芯片版本并转换为大写
    SUFFIX=$(echo $(get_build_var CHIP_REVISION) | tr a-z A-Z)

    # 删除内核配置文件和boot相关文件
    if [ -f "$INTERMEDIATES/KERNEL_OBJ_$SUFFIX"/.config ]; then rm -rf "$INTERMEDIATES/KERNEL_OBJ_$SUFFIX"/.config; fi
    if [ -d "$INTERMEDIATES/KERNEL_OBJ_$SUFFIX"/arch/arm64/boot ]; then rm -rf "$INTERMEDIATES/KERNEL_OBJ_$SUFFIX"/arch/arm64/boot; fi
    # 删除驱动模块文件(.ko)
    if [ -d "$INTERMEDIATES/KERNEL_OBJ_$SUFFIX"/drivers ]; then find -L "$INTERMEDIATES/KERNEL_OBJ_$SUFFIX"/drivers -name "*.ko" |xargs rm -f; fi
    
    # 删除recovery配置文件和boot相关文件
    if [ -f "$INTERMEDIATES/RECOVERY_OBJ_$SUFFIX"/.config ]; then rm -rf "$INTERMEDIATES/RECOVERY_OBJ_$SUFFIX"/.config; fi
    if [ -d "$INTERMEDIATES/RECOVERY_OBJ_$SUFFIX"/arch/arm64/boot ]; then rm -rf "$INTERMEDIATES/RECOVERY_OBJ_$SUFFIX"/arch/arm64/boot; fi
    
    # 删除bootloader相关二进制文件
    if [ -f "$INTERMEDIATES/EMMC_FIRSTBOOT_OBJ_$SUFFIX"/fastboot-burn.bin ]; then rm -rf "$INTERMEDIATES/EMMC_FIRSTBOOT_OBJ_$SUFFIX"/fastboot-burn.bin; fi
    if [ -f "$INTERMEDIATES/EMMC_SECONDBOOT_OBJ_$SUFFIX"/sbl.bin ]; then rm -rf "$INTERMEDIATES/EMMC_SECONDBOOT_OBJ_$SUFFIX"/sbl.bin; fi
    if [ -f "$INTERMEDIATES/EMMC_SLAVEBOOT_OBJ_$SUFFIX"/slaveboot-burn.bin ]; then rm -rf "$INTERMEDIATES/EMMC_SLAVEBOOT_OBJ_$SUFFIX"/slaveboot-burn.bin; fi
    
    # 删除preload目录
    if [ -d "$PRODUCT_OUT"/preload ]; then rm -rf "$PRODUCT_OUT"/preload; fi
    # 删除启动参数二进制文件
    if [ -f "$PRODUCT_OUT"/bootargs.bin ]; then rm -rf "$PRODUCT_OUT"/bootargs.bin; fi
    
    # 如果参数包含installclean，清理bootloader编译缓存
    if [ -n "$(echo "$command_args" | sed -n '/installclean/p')" ]; then
        rm -rf $INTERMEDIATES/EMMC_FIRSTBOOT_OBJ* $INTERMEDIATES/EMMC_SECONDBOOT_OBJ* $INTERMEDIATES/EMMC_SLAVEBOOT_OBJ*
        rm -rf $INTERMEDIATES/KERNEL_OBJ* $INTERMEDIATES/RECOVERY_OBJ* $INTERMEDIATES/SECURESIGN_OBJ
    fi
}

# 编译前预处理
__pre_process(){
    # 检查并解析命令行参数
    check_parameter "$@"
    # 设置编译参数
    set_parameter

    # 处理tvos-j平台特殊配置
    if [ -d "platform/tvos-j" ] && [ -z "${TVOS_PLATFORM}" ] || [ "${TVOS_PLATFORM}" = "tvos-j" ]; then
        # 保存当前工作目录
        WS_TOP=$(pwd)
        # 获取tvos-j平台的绝对路径
        WS=$(realpath platform/tvos-j)
        # 删除原有vendor链接并重新创建
        rm -vf $WS/vendor && ln -sv ../../vendor $WS/vendor
        # 切换到tvos-j目录
        cd ${WS}
    fi

    # 初始化编译环境
    if [ -n "$HUANGLONG_LINUX_ENV" ];then
        # 如果已初始化过环境，重新加载env.sh
        source build/env.sh
    else
        # 否则执行环境初始化脚本
        source build/envsetup.sh;
    fi
    
    # 执行lunch命令配置编译环境
    if [ -n "$TARGET_RELEASE" ]; then
        # 如果指定了版本号，使用完整的lunch参数
        lunch $TARGET_PRODUCT-$TARGET_RELEASE-$TARGET_BUILD_VARIANT
    else
        # 否则使用基础的lunch参数
        lunch $TARGET_PRODUCT-$TARGET_BUILD_VARIANT
    fi
    
    # 加载bep环境配置（非Android 9版本）
    if [ -f /usr/local/bin/bep_env.sh ] && [ "$(get_build_var PLATFORM_VERSION)" != "9" ];then
        if [ "$BUILD_CURRENT_TIMESTAMP" == 'true' ];then sh /usr/local/bin/bep_env.sh -c vendor/huanglong/build/tools/bep_env.conf;fi
        source /usr/local/bin/bep_env.sh -s vendor/huanglong/build/tools/bep_env.conf
    fi

    # 执行产品扩展预处理脚本（非installclean操作）
    if [ -f "$PRODUCT_EXT_PATH"/build/extra_process.sh ] && [ -z "$(echo "$command_args" | sed -n '/installclean/p')" ]; then
        source $PRODUCT_EXT_PATH/build/extra_process.sh;
        pre_process "$@"
    fi

    # 获取编译相关变量
    PRODUCT_USE_DYNAMIC_PARTITIONS=$(get_build_var PRODUCT_USE_DYNAMIC_PARTITIONS)
    PRODUCT_OUT=$(get_build_var PRODUCT_OUT)
    IMAGES_OUT=$PRODUCT_OUT/images  # 镜像输出目录

    SANITIZE_TARGET=$(get_build_var SANITIZE_TARGET)  # 编译检查目标
    
    # 删除已存在的镜像目录
    if [ -d "$IMAGES_OUT" ]; then rm -rf $IMAGES_OUT; fi
    # 清理编译中间文件
    clear_intermediates
}

# 执行编译操作
__do_make(){
    # 执行产品专属的编译前钩子函数（如果存在）
    if [ "$(type -t ${TARGET_PRODUCT}_pre_make)" == function ]; then
        "${TARGET_PRODUCT}_pre_make"
    fi

    # 如果不是installclean操作，先执行installclean清理
    if [ -z "$(echo "$command_args" | sed -n '/installclean/p')" ]; then make installclean; fi
    # 执行make编译
    make $command_args SANITIZE_TARGET=$SANITIZE_TARGET

    # 执行产品专属的编译后钩子函数（如果存在）
    if [ "$(type -t ${TARGET_PRODUCT}_post_make)" == function ]; then
        "${TARGET_PRODUCT}_post_make"
    fi
}

# 复制镜像文件到指定目录
copy_img(){
   # 遍历传入的镜像文件名
   for i in $@
   do
      # 如果文件存在，复制到镜像输出目录
      if [ -f "$PRODUCT_OUT"/"$i" ]; then cp $PRODUCT_OUT/$i $IMAGES_OUT/; fi
   done
}

# 编译后处理
__post_process(){
    # 创建镜像输出目录
    mkdir -p $IMAGES_OUT

    # 复制AOSP系统镜像
    if [ "${PRODUCT_USE_DYNAMIC_PARTITIONS}" == "true" ]; then
        # 动态分区模式：复制super.img
        copy_img super.img
    else
        # 非动态分区模式：复制system.img和vendor.img
        copy_img system.img vendor.img
    fi
    # 复制其他基础镜像文件
    copy_img cache.img userdata.img product.img vbmeta_system.img vbmeta_vendor.img

    # 复制OTA升级包（如果存在）
    if [ -f  "$PRODUCT_OUT"/"$TARGET_PRODUCT"-ota*.zip ]; then cp  $PRODUCT_OUT/$TARGET_PRODUCT-ota*.zip $IMAGES_OUT/update.zip; fi

    # 执行产品扩展后处理脚本（特定条件下）
    if [ -f "$PRODUCT_EXT_PATH"/build/extra_process.sh ] && [ -z "$(echo "$command_args" | sed -n '/installclean/p')" ] && [[ "$TVOS_VERSION" != "TVOS-H" ]] && [[ "$TVOS_PLATFORM" != "tvos-o" ]]; then
        post_process;
    fi
    
    # 如果切换过目录，切回原工作目录
    if [ -n "$WS_TOP" ]; then
        cd "$WS_TOP"
    fi
}

# 主构建函数
build(){
    # 记录开始时间
    start_time=$(date +%s)
    echo "开始编译..."
    # 执行预处理
    __pre_process "$@"
    # 执行编译
    __do_make
    # 执行后处理
    __post_process
    # 记录结束时间并计算耗时
    end_time=$(date +%s)
    echo "编译完成，总耗时: $(expr $end_time - $start_time) 秒"
}

# 脚本主入口
main(){
    local  result=0
    # 删除旧的编译日志文件
    if [ "$(ls $PWD/build_*.log 2> /dev/null | wc -l)" -ne 0 ]; then rm $(pwd)/build_*.log; fi
    # 执行编译并将输出重定向到日志文件（同时保留终端输出）
    build "$@" 2>&1 | tee -p build_$(date '+%Y%m%d-%H%M%S').log
    # 获取编译命令的返回值
    result=${PIPESTATUS[0]}
    # 如果编译失败，退出并返回错误码
    if [ $result -ne 0 ];then exit $result; fi
}

# 调用主函数，传入所有命令行参数
main "$@"

  # 检查是否在编译系统目录或输出目录中调用（禁止此类调用）

    $(error my-dir must be called before including any other makefile.) \
   , \
    # 提取并返回Makefile文件所在目录（去除末尾的/）
    $(patsubst %/,%,$(dir $(LOCAL_MODULE_MAKEFILE))) \
   ) \
 )
endef

# 产品配置文件继承函数（引入产品定义的Makefile）
# 参数说明：
# $(1): 产品Makefile路径
define inherit-product
  # 引入指定的产品配置文件
  $(eval include $(1))
endef

# 条件继承产品配置文件（文件存在时才引入）
# 参数说明：
# $(1): 产品Makefile路径
define inherit-product-if-exists
  # 检查文件是否存在，存在则调用inherit-product，否则不执行任何操作
  $(if $(wildcard $(1)),$(call inherit-product,$(1)),)
endef

# 引入编译系统的基础配置文件（核心编译规则）
include $(BUILD_SYSTEM)/base.mk
# 引入编译系统的构建规则文件（具体编译逻辑）
include $(BUILD_SYSTEM)/build.mk
~~~

## 1. 编译前置检查（base.mk 开头）

这是编译的**第一道门槛**，确保环境已正确初始化：

```
ifneq ($(HUANGLONG_LINUX_ENV),ok)
  $(error "Please run 'source env.sh' before building!")
endif
```

- 检查环境变量`HUANGLONG_LINUX_ENV`是否为`ok`，该变量由`source env.sh`初始化（之前的 shell 脚本中设置）。
- 如果未初始化，直接终止编译并报错，避免因环境缺失导致编译失败。
- 核心目的：保证编译所需的环境变量（如工具链路径、项目根目录）都已正确设置。

## 2. 系统信息与输出路径配置（base.mk）

完成环境检查后，配置编译的「基础运行环境」和「产物输出路径」：

### 2.1 主机系统信息识别

```makefile
# 标记系统平台为linux
OS_PLATFORM_VERSION := linux
# 获取主机信息（如 Linux x86_64）
UNAME := $(shell uname -sm)
# 识别主机系统为Linux
ifneq (,$(findstring Linux,$(UNAME)))
  HOST_OS := linux
endif
# 识别主机架构为x86_64
ifneq (,$(findstring x86_64,$(UNAME)))
  HOST_ARCH := x86_64
endif
```

- 核心作用：适配不同主机的编译工具（这里仅适配 Linux x86_64）。

### 2.2 编译输出路径配置

```makefile
# 默认输出目录为顶层out目录，可通过OUT_DIR自定义
ifeq (,$(strip $(OUT_DIR)))
  OUT_DIR := $(TOPDIR)out
endif
# 产品最终输出目录（核心路径）
PRODUCT_OUT := $(TARGET_PRODUCT_OUT_ROOT)/$(TARGET_DEVICE)
# 其他子路径：主机端产物、中间文件、镜像输出等
HOST_OUT := $(PRODUCT_OUT)/host          # 主机端可执行文件/库
TARGET_OUT_INTERMEDIATES := $(PRODUCT_OUT)/obj  # 编译中间文件
IMAGES_OUT := $(PRODUCT_OUT)/images      # 系统镜像输出目录
```

- 所有编译产物（中间文件、最终镜像、可执行文件）都集中在`OUT_DIR`下，便于管理和清理。
- `PRODUCT_OUT`是最核心的路径，**所有最终产物都输出到这个目录**。

## 3. 产品配置文件加载（base.mk 核心）

这是「定制化编译」的关键，加载指定产品的配置文件：

### 3.1 查找产品配置文件

```makefile
# 在device/vendor目录下查找TARGET_PRODUCT对应的.mk文件（最大深度4层）
product_name_mk := \
  $(strip $(sort $(wildcard \
    $(shell test -d device && find -L device -maxdepth 4 -path '*/$(TARGET_PRODUCT).mk') \
    $(shell test -d vendor && find -L vendor -maxdepth 4 -path '*/$(TARGET_PRODUCT).mk') \
  )))
# 校验：必须找到且只能找到1个配置文件
ifeq ($(product_name_mk),)
  $(error No config file found for TARGET_PRODUCT $(TARGET_PRODUCT))
endif
ifneq ($(words $(product_name_mk)),1)
  $(error Multiple board config files for TARGET_PRODUCT $(TARGET_PRODUCT): $(product_name_mk))
endif
# 引入产品配置文件
-include $(product_name_mk)
```

- `TARGET_PRODUCT`是编译的核心变量（由 lunch 命令或 --product-name 参数指定），对应具体的产品型号。
- 核心校验：避免「无配置」或「多配置冲突」，保证编译的唯一性。

### 3.2 产品路径变量初始化



```makefile
TARGET_DEVICE := $(PRODUCT_DEVICE)       # 设备型号
TARGET_DEVICE_DIR := $(PRODUCT_DEVICE_DIR) # 设备配置目录
TARGET_PRODUCT_OUT_ROOT := $(TARGET_OUT_ROOT)/product # 产品输出根目录
```

- 从产品配置文件中读取`PRODUCT_DEVICE`等变量，确定编译的具体设备参数。

## 4. 编译工具与编译选项配置（base.mk）

配置编译所需的「工具链」和「编译 / 链接规则」：

### 4.1 模块构建规则

```makefile
# 定义不同模块的编译规则文件路径
CLEAR_VARS := $(BUILD_SYSTEM)/clear_vars.mk          # 清空模块变量
BUILD_LINUX_LIBRARY := $(BUILD_SYSTEM)/library.mk    # 编译动态/静态库
BUILD_LINUX_EXECUTABLE := $(BUILD_SYSTEM)/executable.mk # 编译可执行文件
BUILD_LINUX_PREBUILT := $(BUILD_SYSTEM)/prebuilt.mk  # 处理预编译文件
```

- 这些是编译不同类型模块的「基础模板」，后续编译具体模块时会引入这些文件。

### 4.2 编译工具配置

```makefile
CC := gcc  # C编译器
# 交叉编译工具链（适配ARM/ARM64架构）
CROSS_COMPILE_TOOLCHAIN32 := arm-v100-linux
CROSS_COMPILE_TOOLCHAIN64 := aarch64-v100-linux
# 镜像制作工具（生成ext4/squashfs镜像）
MKEXT4FS := $(IMGTOOL_PATH)/make_ext4fs
MKSQUASHFS := $(IMGTOOL_PATH)/mksquashfs
hide := @  # 静默执行编译命令（不打印命令本身）
```

- 核心：适配嵌入式 Linux 的 ARM/ARM64 交叉编译，以及系统镜像的制作。

### 4.3 全局编译 / 链接选项

```makefile
# 全局编译选项（安全+优化+严格校验）
GLOBAL_CFLAGS := -fstack-protector-strong  # 栈保护（防溢出）
GLOBAL_CFLAGS += -Wall -Wextra -Werror     # 开启所有警告，且警告视为错误
GLOBAL_CFLAGS += -O2                        # 编译优化
# 不同ARM架构/CPU的专属编译选项（适配硬件）
cflags-arm-armv8-a := -march=armv8-a+crc -mfloat-abi=softfp -mfpu=neon-fp-armv8
# 链接选项（安全加固）
GLOBAL_LDFLAGS := -Wl,-z,relro,-z,now  # 重定位只读+立即绑定（防漏洞）
GLOBAL_EXECUTABLE_LDFLAGS := $(GLOBAL_LDFLAGS) -pie  # 位置无关可执行文件
```

- 核心目的：保证编译产物的**安全性**（栈保护、RELRO）、**兼容性**（适配不同 ARM 硬件）、**代码质量**（警告视为错误）。

## 5. 编译规则定义（build.mk 核心）

定义 make 命令的「具体执行规则」，包括文件复制、清理、帮助等目标：

### 5.1 产品文件复制规则（PRODUCT_COPY_FILES）

```makefile
# 处理产品定义的文件复制规则（格式：src:dest[:owner]）
# 1. 去重：过滤重复的src:dest对，忽略vendor/system等路径
# 2. 校验：避免同一目标文件被多个源文件覆盖
# 3. 复制：XML文件先校验合法性（xmllint），普通文件直接复制
$(foreach cf,$(unique_product_copy_files_pairs), \
    $(eval _src := $(call word-colon,1,$(cf))) \
    $(eval _dest := $(call word-colon,2,$(cf))) \
    $(if $(filter %.xml,$(_dest)),\
        $(eval $(call copy-xml-file-checked,$(_src),$(_fulldest))), \
        $(eval $(call copy-one-file,$(_src),$(_fulldest)))))
```

- `PRODUCT_COPY_FILES`是产品定制的核心：将配置文件、资源文件等复制到编译产物目录。
- 核心校验：XML 文件必须合法（避免配置错误），目标路径唯一（避免文件覆盖）。

### 5.2 清理类目标

|      目标      |                             作用                             |
| :------------: | :----------------------------------------------------------: |
| `installclean` | 清理中间文件（库、可执行文件、内核配置 / 镜像），保留产品配置，快速重编译 |
|    `clean`     |         彻底删除整个 OUT_DIR 目录，清理所有编译产物          |
|   `nothing`    |           仅验证 Makefile 语法，不执行任何编译操作           |
|     `help`     |                加载 help.mk，显示编译帮助信息                |

### 5.3 自定义规则加载

```
# 引入设备/厂商的自定义编译规则
-include $(TARGET_DEVICE_DIR)/components/build.mk
-include $(sort $(wildcard vendor/*/build/tasks/build.mk))
```

- 核心：支持厂商 / 设备级别的定制化编译逻辑（如内核编译、驱动编译）。

## 6. 编译执行与配置打印

### 6.1 配置打印

```
# 打印核心编译配置（方便调试）
$(info ============================================)
$(info   TARGET_PRODUCT=$(TARGET_PRODUCT))
$(info   TARGET_BUILD_VARIANT=$(TARGET_BUILD_VARIANT))
$(info   OUT_DIR=$(OUT_DIR))
$(info ============================================)
```

- 执行 make 时，会先打印核心配置（产品型号、编译变体、输出目录），便于确认编译参数。

### 6.2 变量查询（dumpvar）

```
# 支持查询编译变量（如 make dumpvar-PRODUCT_OUT）
$(dumpvar_target):
	@echo $(DUMPVAR_VALUE)
```

- 核心作用：通过`get_build_var`命令查询编译变量（如产物路径、架构），用于脚本调用。

### 总结（核心运行逻辑）

1. **前置校验**：检查环境初始化状态，确保`source env.sh`已执行。
2. **基础配置**：识别主机系统、配置输出路径，确定编译产物的存放位置。
3. **产品加载**：加载指定产品的配置文件，保证编译的定制化和唯一性。
4. **工具 / 选项**：配置交叉编译工具链、编译 / 链接选项，保证产物的安全性和兼容性。
5. **规则执行**：处理文件复制、执行清理 / 编译目标，加载厂商自定义规则。
6. **产物输出**：所有编译产物（镜像、可执行文件、配置文件）最终输出到`PRODUCT_OUT`目录。

### 关键要点

- `PRODUCT_OUT`是核心输出目录，所有最终产物都在这里（尤其是`images`子目录的系统镜像）。
- `TARGET_PRODUCT`决定编译的产品型号，是整个编译流程的「核心定制变量」。
- 编译选项开启严格校验（Werror、栈保护），保证产物的稳定性和安全性。
- 支持`clean/installclean`两种清理策略，适配不同的编译调试场景。
