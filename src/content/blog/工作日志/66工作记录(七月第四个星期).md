can_send

~~~c
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>
#include <sys/socket.h>
#include <sys/ioctl.h>
#include <net/if.h>
#include <linux/can.h>
#include <linux/can/raw.h>

int main() {
    int sockfd;

    // 创建CAN套接字
    if ((sockfd = socket(PF_CAN, SOCK_RAW, CAN_RAW)) < 0) 
    {
        printf("Socket creation failed");
        return -1;
    }

    // 设置CAN接口
    struct ifreq ifr;
    strcpy(ifr.ifr_name, "can0");
    if (ioctl(sockfd, SIOCGIFINDEX, &ifr) < 0) 
    {
        printf("SIOCGIFINDEX failed");
        close(sockfd);
        return -1;
    }

    // 绑定套接字到CAN接口
    struct sockaddr_can addr;
    addr.can_family = AF_CAN;
    addr.can_ifindex = ifr.ifr_ifindex;
    if (bind(sockfd, (struct sockaddr *)&addr, sizeof(addr)) < 0) 
    {
        printf("Bind failed");
        close(sockfd);
        return -1;
    }

    // 定义并初始化CAN帧
    struct can_frame frames[3] = 
    {
        {.can_id = 0x11, .can_dlc = 2, .data = {0x01, 0x02}},
        {.can_id = 0x11 | CAN_EFF_FLAG, .can_dlc = 2, .data = {0x01, 0x02}},
        {.can_id = 0x11 | CAN_RTR_FLAG, .can_dlc = 2}
    };

    // 发送CAN帧
    for (int i = 0; i < 3; i++) 
    {
        write(sockfd, &frames[i], sizeof(struct can_frame));
    }

    close(sockfd);
}
~~~



can_recv

~~~c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <net/if.h>
#include <sys/ioctl.h>
#include <sys/socket.h>
#include <linux/can.h>
#include <linux/can/raw.h>

int main() {
    int sockfd;
    struct sockaddr_can addr;
    struct ifreq ifr;
    struct can_frame frame;

    // 创建CAN套接字
    if ((sockfd = socket(PF_CAN, SOCK_RAW, CAN_RAW)) < 0) {
        perror("socket");
        return 1;
    }

    // 获取CAN接口索引
    strcpy(ifr.ifr_name, "can0");
    if (ioctl(sockfd, SIOCGIFINDEX, &ifr) < 0) {
        perror("ioctl");
        close(sockfd);
        return 1;
    }

    // 绑定套接字
    addr.can_family = AF_CAN;
    addr.can_ifindex = ifr.ifr_ifindex;
    if (bind(sockfd, (struct sockaddr *)&addr, sizeof(addr)) < 0) {
        perror("bind");
        close(sockfd);
        return 1;
    }

    // 循环接收CAN帧
	while (1) 
    {
        read(sockfd, &frame, sizeof(frame));

        // 输出ID（标准帧或扩展帧）
        if (frame.can_id & CAN_EFF_FLAG) 
        {
            printf("<0x%08x>", frame.can_id & CAN_EFF_MASK);
        } 
        else 
        {
            printf("<0x%03x>", frame.can_id & CAN_SFF_MASK);
        }

        // 输出数据长度
        printf("[%d]", frame.can_dlc);

        // 输出数据内容
        for (int i = 0; i < frame.can_dlc; i++) 
        {
            printf("%02x", frame.data[i]);
        }

        // 输出远程帧标志（如果有）
        if (frame.can_id & CAN_RTR_FLAG) 
        {
            printf(" remote request");
        }
        // 换行
        printf("\n");
    }

    close(sockfd);
    return 0;
}
~~~

行了，这次代码敲的差不多了，可以动手写手册了，先来来规划手册的具体内容。



| 结构体成员 | 详细说明                                                     |
| ---------- | ------------------------------------------------------------ |
| `can_id`   | 包含帧的 ID 及标志位（共 32 位），具体细分： <br />**标志位**：`CAN_EFF_FLAG`（`0x80000000`）：表示扩展帧（29 位 ID ）   `CAN_RTR_FLAG`（`0x40000000`）：表示远程帧（请求数据，无实际数据 ）   `CAN_ERR_FLAG`（`0x20000000`）：表示错误帧 - <br />**实际 ID**：标准帧（11 位 ID ）：低 11 位有效，需用 `CAN_SFF_MASK`（`0x7FF`）提取   扩展帧（29 位 ID ）：低 29 位有效，需用 `CAN_EFF_MASK`（`0x1FFFFFFF`）提取 |
| `can_dlc`  | 表示 `data` 数组中有效数据的长度，取值范围为 `0~8`（由 CAN 2.0 协议规定 ）。即使 `data` 数组长度固定为 8，也仅会传输 `can_dlc` 指定字节数的数据 |
| `data[8]`  | 存储 CAN 帧的实际数据，最多 8 字节，示例情况： - 若 `can_dlc = 3`，则 `data[0]`、`data[1]`、`data[2]` 为有效数据 - 远程帧（`CAN_RTR_FLAG` 置位 ）中，`data` 无效，仅 `can_id` 和 `can_dlc` 有意义 |

然后定义一个int 类型的变量socket_can，用来接收socket系统调用的返回值套接字描述符，然后添加一下socket系统调用要用到的几个头文件，socket系统调用的第一个参数为协议族，这里填写PF_CAN，第二个参数为套接字类型，这里填写SOCK_RAW，第三个参数为具体的协议，这里为CAN_RAW,套接字创建完成之后就要进行绑定套接字了，但绑定CAN套接字需要先获取到can的索引，所以这里需要定义一个ifreq类型的变量ifr，要用这个结构体需要先包含头文件为net/if.h,马上要用到字符串拷贝函数，所以这里也加上string.h,然后调用strcpy将can0拷贝到ifr的ifr_name成员，然后调用ioctl通过这里的name获取到can0的索引，ioctl的头文件先添加一下，第一个参数为文件描述符socket_can,第二个参数为SIOCGIFINDEX,第三个参数为ifr变量的地址，这时候就将can0的索引赋予到了ifr的ifr_ifindex中了，接下来会用到。

然后创建一个sockaddr_can类型的变量addr，用来存放can设备的地址属性，先来设置它的地址族，然后设置它的索引，也就是ifr变量的ifr_ifindex成员，然后就可以调用bind来进行绑定了，第一个参数为套接字描述符socke_can，第二个参数为刚刚创建的地址属性addr，但需要进行一下强制类型转换，第三个参数为地址属性addr的大小，这里使用sizeof来进行获取，这部分代码是CAN发送和CAN接收通用的，稍后编写第二个CAN接收程例程的时候，这部分代码就不再重复编写了。

然后创建一个can_frame类型的数组变量frame大小为3，它用来描述要发送的CAN帧，我们要定义三个帧，分别是标准数据帧、扩展数据帧，以及标准遥控帧，先来填充第一个标准数据帧，他的帧ID设置为0x123，dlc设置为2，表示要发送两个数据，要发送的数据data设置为，0x01和0x02，这样标准帧就填充完了，然后填充扩展数据帧，这里直接复制，不一样的地方在于can_id的标志位不同，需要使用按位或来使能CAN_EFF_FLAG,最后是标准遥控帧，先是标志位的不同，修改为CAN_RTR_FLAG，而遥控帧是用来请求数据的，所以他没有数据位，这里的data应该删掉。这样我们就定义完成了三个不同类型的帧。

最后就是帧的发送了，这里要发送三个帧，所以需要一个for循环，发送就用write系统调用即可，他的头文件为unistd.h，第一个参数为套接字描述符socket_can,第二个参数为要发送的帧，第三个参数为帧的大小，使用sizeof来获取，到这里CAN发送例程的代码就编写完成了，然后打开tabby终端，找到刚刚编写完成的代码，使用gcc命令进行编译，然后打开usb转can的上位机等待数据的接收，运行刚刚编译出来的可执行程序，可以看到，上位机这边已经成功打印出了发送来的三个CAN帧。至此，关于CAN发送例程就测试完成了，然后来进行最后一个实验，USB接收实验。

实时时钟有外部和内部两种常见实现方案，广泛用于嵌入式系统、物联网设备等需时间管理的场景，以下对比介绍这两种。



外部 RTC 是独立于主控芯片的专用芯片，通过 I2C、SPI 等接口连接，精度高，时间记录更准。它有独立电源管理电路，配纽扣电池可长时间独立运行，即便主控芯片断电或损坏，也能保证时间记录连续。但外部 RTC 成本较高，需额外采购芯片及相关元件，还会增加电路设计复杂性、开发难度，且占用主控芯片通信资源，因此更适合对时间精度、可靠性和功能要求较高的场景。iTOP-RK3568 开发板就集成了外部 RTC，使用的芯片是 RX8010，如下图所示：



内部 RTC 是主控芯片集成的实时时钟模块，硬件集成无需额外组件，设计简单、成本低且易用。但它依赖主控供电，断电需备用电池维持时间记录，还易受温度和电压波动影响，导致时间漂移大、可靠性较低，因此更适合对时间精度要求不高、成本敏感的场景，如家用电器、简单物联网设备或实时性要求低的系统。
