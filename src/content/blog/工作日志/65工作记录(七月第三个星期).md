~~~c
#include <stdio.h>
#include <fcntl.h>
#include <unistd.h>
#include <string.h>
#include <sys/socket.h>
#include <sys/ioctl.h>
#include <net/if.h>
#include <linux/can.h>
#include <linux/can/raw.h> // 包含 CAN_RAW 相关定义

int main(int argc, char *argv[]) {
    int sockfd; // 套接字描述符
    struct sockaddr_can addr; // CAN 地址结构体
    struct ifreq ifr; // 用于获取网络接口信息
    int ret; // 用于存储系统调用返回值

    // 创建 CAN 套接字
    sockfd = socket(PF_CAN, SOCK_RAW, CAN_RAW);
    if (sockfd < 0) {
        printf("Socket creation failed");
        return -1;
    }

    // 指定要绑定的 CAN 接口名称
    strncpy(ifr.ifr_name, "can0", IFNAMSIZ - 1);

    // 使用 ioctl 获取 CAN 接口的索引号
    if (ioctl(sockfd, SIOCGIFINDEX, &ifr) < 0) {
        printf("ioctl failed");
        close(sockfd);
        return -1;
    }

    // 设置 CAN 地址结构体
    addr.can_family = AF_CAN; // CAN 地址族
    addr.can_ifindex = ifr.ifr_ifindex; // CAN 接口索引号

    // 绑定套接字到指定的 CAN 接口
    if (bind(sockfd, (struct sockaddr *)&addr, sizeof(addr)) < 0) {
        printf("Bind failed");
        close(sockfd);
        return -1;
    }

    // 定义 CAN 帧数组
    struct can_frame frame[3]; // CAN 帧数组
    unsigned char data[2] = {0x01, 0x02}; // 数据内容

    // 初始化标准帧
    frame[0].can_id = 0x11; // 标准帧 ID
    frame[0].can_dlc = 2;   // 数据长度（2 字节）
    memcpy(frame[0].data, data, 2); // 填充数据

    // 初始化扩展帧
    frame[1].can_id = 0x11 | CAN_EFF_FLAG; // 扩展帧 ID
    frame[1].can_dlc = 2;                 // 数据长度（2 字节）
    memcpy(frame[1].data, data, 2);       // 填充数据

    // 初始化远程帧
    frame[2].can_id = 0x11 | CAN_RTR_FLAG; // 远程帧 ID
    frame[2].can_dlc = 2;                 // 数据长度（2 字节）

    // 循环发送 CAN 帧
    while (1) {
        // 发送标准帧
        ret = write(sockfd, &frame[0], sizeof(frame[0]));
        if (ret != sizeof(frame[0])) {
            printf("write (标准帧)");
            break;
        }
        printf("标准帧发送成功！\n");

        // 发送扩展帧
        ret = write(sockfd, &frame[1], sizeof(frame[1]));
        if (ret != sizeof(frame[1])) {
            printf("write (扩展帧)");
            break;
        }
        printf("扩展帧发送成功！\n");

        // 发送远程帧
        ret = write(sockfd, &frame[2], sizeof(frame[2]));
        if (ret != sizeof(frame[2])) {
            printf("write (远程帧)");
            break;
        }
        printf("远程帧发送成功！\n");

        // 延时 1 秒
        sleep(1);
    }

    // 关闭套接字
    close(sockfd);

    return 0;
}
~~~

can_recv

~~~c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <net/if.h>
#include <sys/ioctl.h>
#include <sys/socket.h>
#include <linux/can.h>
#include <linux/can/raw.h>

#define BUF_SIZE 128 // 缓冲区大小

int main() {
    int sockfd; // 套接字文件描述符
    struct sockaddr_can addr; // CAN 地址结构体
    struct ifreq ifr; // 网络接口请求结构体
    struct can_frame frame; // CAN 帧结构体
    char buf[BUF_SIZE]; // 格式化输出缓冲区
    int ret, n, i, err;
    FILE *out = stdout; // 输出文件流（标准输出）

    // 创建 CAN 套接字
    sockfd = socket(PF_CAN, SOCK_RAW, CAN_RAW);

    // 指定 CAN 接口名称
    strcpy(ifr.ifr_name, "can0");
    ioctl(sockfd, SIOCGIFINDEX, &ifr);

    // 设置地址信息
    addr.can_family = AF_CAN;
    addr.can_ifindex = ifr.ifr_ifindex;

    // 绑定 CAN 接口
    bind(sockfd, (struct sockaddr *)&addr, sizeof(addr));

    // 循环接收 CAN 帧
    while (1) {
        // 接收 CAN 帧
        ret = read(sockfd, &frame, sizeof(frame));
        if (ret <= 0) {
            printf("read");
            break;
        }

        // 格式化输出 CAN 帧信息
        if (frame.can_id & CAN_EFF_FLAG) {
            // 扩展帧
            n = snprintf(buf, BUF_SIZE, "<0x%08x>", frame.can_id & CAN_EFF_MASK);
        } else {
            // 标准帧
            n = snprintf(buf, BUF_SIZE, "<0x%03x>", frame.can_id & CAN_SFF_MASK);
        }

        // 添加数据长度
        n += snprintf(buf + n, BUF_SIZE - n, "[%d]", frame.can_dlc);

        // 添加数据内容
        for (i = 0; i < frame.can_dlc; i++) {
            n += snprintf(buf + n, BUF_SIZE - n, " %02x", frame.data[i]);
        }

        // 如果是远程帧
        if (frame.can_id & CAN_RTR_FLAG) {
            n += snprintf(buf + n, BUF_SIZE - n, " remote request");
        }

        // 输出到标准输出
        fprintf(out, "%s\n", buf);

        // 刷新输出缓冲区
        err = fflush(out);
        if (err < 0) {
            perror("fflush");
            break;
        }
    }

    // 关闭套接字
    close(sockfd);
    return 0;
}
~~~

**SIOCGIFINDEX =**

**S**et **I**nterface **O**peration **C**ontrol **G**et **I**nterface **INDEX**

> `struct ifreq` 是 Linux 系统中用于配置和获取网络接口信息的核心结构体，广泛应用于网络编程（如 socket 操作、接口配置等场景）。它的定义包含在 `<net/if.h>` 头文件中，设计目的是通过 **ioctl 系统调用** 与内核交互，传递或获取特定网络接口（如以太网 `eth0`、CAN 总线 `can0`、无线网卡 `wlan0` 等）的参数。
>
> ### 一、结构体的核心作用
>
> `struct ifreq` 是用户空间与内核空间之间传递网络接口信息的 “通用容器”，主要用于：
>
> 1. 获取网络接口的索引（如通过 `SIOCGIFINDEX` 命令，这是 CAN 编程中最常用的场景）。
> 2. 配置或获取接口的 MAC 地址、IP 地址、MTU（最大传输单元）等参数。
> 3. 启用 / 禁用接口（如通过 `SIOCSIFFLAGS` 命令设置接口为 “up” 状态）。
>
> ### 二、结构体的定义与关键成员
>
> `struct ifreq` 的定义因系统架构略有差异，但核心成员一致，简化后结构如下（来自 `<net/if.h>`）：
>
> ```c
> struct ifreq {
>     char ifr_name[IFNAMSIZ];  // 网络接口名称（如 "eth0", "can0"）
>     union {
>         struct sockaddr ifr_addr;       // 接口地址（如 IP 地址）
>         struct sockaddr ifr_dstaddr;    // 目的地址（用于点对点接口）
>         struct sockaddr ifr_broadaddr;  // 广播地址
>         struct sockaddr ifr_netmask;    // 子网掩码
>         struct sockaddr ifr_hwaddr;     // 硬件地址（MAC 地址）
>         short           ifr_flags;      // 接口标志（如 IFF_UP 表示接口启用）
>         int             ifr_ifindex;    // 接口索引（唯一标识接口的整数）
>         // 其他成员（根据系统配置可能包含更多字段）
>     } ifr_ifru;
> };
> 
> // 为方便访问 union 中的成员，通常定义宏别名：
> #define ifr_addr      ifr_ifru.ifr_addr
> #define ifr_dstaddr   ifr_ifru.ifr_dstaddr
> #define ifr_broadaddr ifr_ifru.ifr_broadaddr
> #define ifr_netmask   ifr_ifru.ifr_netmask
> #define ifr_hwaddr    ifr_ifru.ifr_hwaddr
> #define ifr_flags     ifr_ifru.ifr_flags
> #define ifr_ifindex   ifr_ifru.ifr_ifindex
> ```
>
> #### 关键成员解析：
>
> 1. **`ifr_name[IFNAMSIZ]`**
>    - 用于指定操作的网络接口名称（如 `"can0"`、`"eth0"`），是结构体的 “标识字段”。
>    - `IFNAMSIZ` 是宏定义（通常为 16），表示接口名称的最大长度（含终止符 `\0`）。
> 2. **`union ifr_ifru`**
>    - 一个联合体（共用体），根据不同的操作命令（如 `ioctl` 的请求码）存储不同类型的信息，节省内存空间。
>    - 常用子成员：
>      - `ifr_ifindex`：接口的整数索引（内核为每个接口分配的唯一标识，CAN 编程中通过 `SIOCGIFINDEX` 命令获取）。
>      - `ifr_flags`：接口的状态标志（如 `IFF_UP` 表示接口已启用，`IFF_RUNNING` 表示接口已连接）。
>      - `ifr_hwaddr`：硬件地址（如 MAC 地址，类型为 `struct sockaddr`，实际使用时需强制转换为 `struct sockaddr_ll` 操作）。
>
> ### 三、在 CAN 编程中的典型用法
>
> 在你提供的 CAN 发送代码中，`struct ifreq ifr` 的作用是 **获取 CAN 接口的索引**，具体步骤：
>
> 1. **指定接口名称**：
>    通过 `strncpy(ifr.ifr_name, "can0", IFNAMSIZ - 1)` 为 `ifr_name` 赋值，告诉内核要操作的是 `can0` 接口。
> 2. **通过 ioctl 获取索引**：
>    调用 `ioctl(sockfd, SIOCGIFINDEX, &ifr)` 向内核发送命令 `SIOCGIFINDEX`，内核会将 `can0` 的索引写入 `ifr.ifr_ifindex`。
> 3. **绑定接口时使用索引**：
>    CAN 套接字绑定接口时，需要通过索引指定接口（而非名称），因此将 `ifr.ifr_ifindex` 赋值给 `struct sockaddr_can` 的 `can_ifindex` 成员。
>
> ### 四、其他常见使用场景
>
> 除了 CAN 编程，`struct ifreq` 还用于：
>
> - **获取 MAC 地址**：
>
>   
>
>   ```c
>   struct ifreq ifr;
>   strncpy(ifr.ifr_name, "eth0", IFNAMSIZ-1);
>   ioctl(sockfd, SIOCGIFHWADDR, &ifr); // 发送获取硬件地址的命令
>   // ifr.ifr_hwaddr.sa_data 中存储 MAC 地址（前 6 字节）
>   ```
>
> - **启用接口**：
>
>   
>
>   ```c
>   struct ifreq ifr;
>   strncpy(ifr.ifr_name, "eth0", IFNAMSIZ-1);
>   ioctl(sockfd, SIOCGIFFLAGS, &ifr); // 获取当前标志
>   ifr.ifr_flags |= IFF_UP;          // 置位 "UP" 标志
>   ioctl(sockfd, SIOCSIFFLAGS, &ifr); // 应用标志（启用接口）
>   ```
>
> ### 五、总结
>
> `struct ifreq` 是网络接口操作的 “瑞士军刀”，通过 `ifr_name` 定位接口，通过 union 成员传递具体参数，配合 `ioctl` 命令实现对网络接口的灵活控制。在 CAN 编程中，它的核心作用是将 “接口名称（如 can0）” 转换为内核识别的 “接口索引”，为套接字绑定接口提供必要信息。



> 在 Linux 的 SocketCAN 编程中，`struct sockaddr_can` 是用于配置和操作 CAN（Controller Area Network）总线的核心结构体之一。它定义在 `<linux/can.h>` 头文件中，专门用于将 CAN 套接字绑定到特定的 CAN 接口。
>
> ### 一、结构体的定义与作用
>
> `struct sockaddr_can` 是 Linux 系统中表示 CAN 总线地址的结构体，主要用于：
>
> 1. **指定 CAN 套接字要绑定的接口**（通过接口索引）。
> 2. **设置 CAN 帧的传输类型**（如标准帧、扩展帧或远程帧）
>
> 其定义如下（简化版）：
>
> ```c
> struct sockaddr_can {
>     sa_family_t can_family;    // 地址族，固定为 AF_CAN
>     int         can_ifindex;   // CAN 接口的索引（通过 SIOCGIFINDEX 获取）
>     union {
>         struct { /* 未使用，保留 */ } tp;
>     } can_addr;
> };
> ```
>
> ### 二、关键成员解析
>
> 1. **`can_family`**
>    - 类型：`sa_family_t`（通常为 `unsigned short`）。
>    - 作用：指定地址族，固定为 `AF_CAN`（表示 CAN 总线协议）。
>    - 注意：必须设置为 `AF_CAN`，否则 `bind()` 会失败。
> 2. **`can_ifindex`**
>    - 类型：`int`。
>    - 作用：指定要绑定的 CAN 接口的**整数索引**（如 `can0` 的索引）。
>    - 获取方式：通过 `ioctl(sockfd, SIOCGIFINDEX, &ifr)` 从 `struct ifreq` 中获取（见上一轮对话的解释）。
> 3. **`can_addr` 联合体**
>    - 目前未被使用，保留用于未来扩展。
>    - 在实际编程中通常忽略，无需设置。
>
> ### 三、在 CAN 编程中的典型用法
>
> 在创建 CAN 套接字后，需要使用 `struct sockaddr_can` 将套接字绑定到特定的 CAN 接口（如 `can0`），步骤如下：
>
> #### 1. **创建 CAN 套接字**
>
> ```c
> int sockfd = socket(PF_CAN, SOCK_RAW, CAN_RAW);
> ```
>
> - `PF_CAN`：协议族（Protocol Family），表示 CAN 总线。
> - `SOCK_RAW`：原始套接字类型，直接操作底层协议。
> - `CAN_RAW`：CAN 原始套接字，可收发完整的 CAN 帧。
>
> #### 2. **获取 CAN 接口索引**
>
> ```c
> struct ifreq ifr;
> strncpy(ifr.ifr_name, "can0", IFNAMSIZ - 1);
> ioctl(sockfd, SIOCGIFINDEX, &ifr);  // 获取 can0 的索引
> ```
>
> #### 3. **配置 `struct sockaddr_can`**
>
> ```c
> struct sockaddr_can addr;
> addr.can_family = AF_CAN;          // 设置地址族为 CAN
> addr.can_ifindex = ifr.ifr_ifindex; // 使用 ioctl 获取的索引
> ```
>
> #### 4. **绑定套接字到 CAN 接口**
>
> ```c
> bind(sockfd, (struct sockaddr *)&addr, sizeof(addr));
> ```
>
> - 必须将 `struct sockaddr_can` 强制转换为通用的 `struct sockaddr*` 类型，这是 `bind()` 函数的要求。
>
> ### 四、对比其他网络协议的地址结构体
>
> `struct sockaddr_can` 的设计与其他网络协议（如 IPv4、IPv6）的地址结构体类似，但专门针对 CAN 总线优化：
>
> - **IPv4**：使用 `struct sockaddr_in`，包含 `sin_family`（`AF_INET`）、`sin_port` 和 `sin_addr`（IP 地址）。
> - **IPv6**：使用 `struct sockaddr_in6`，包含 `sin6_family`（`AF_INET6`）、`sin6_port` 和 `sin6_addr`。
> - **CAN**：使用 `struct sockaddr_can`，仅需 `can_family` 和 `can_ifindex`，结构更简洁。
>
> ### 五、总结
>
> `struct sockaddr_can` 是 SocketCAN 编程中的核心数据结构，其核心作用是：
>
> 1. **标识 CAN 总线协议**（通过 `can_family = AF_CAN`）。
> 2. **指定要操作的 CAN 接口**（通过 `can_ifindex`）。
> 3. **作为参数传递给 `bind()` 函数**，将 CAN 套接字与物理接口关联。
>
> 理解这个结构体是掌握 Linux 下 CAN 通信编程的基础。



> `struct can_frame frames[3]` 定义了一个包含 3 个 CAN（Controller Area Network）帧的数组。CAN 是一种工业标准的串行通信协议，常用于汽车、工业自动化等领域的设备间通信。下面详细解析这个结构体：
>
> ### **1. CAN 帧的基本概念**
>
> CAN 帧是 CAN 总线上传输的基本数据单元，包含：
>
> - **ID**：标识帧的优先级或来源（标准帧 11 位，扩展帧 29 位）。
> - **数据场**：0~8 字节的数据。
> - **控制位**：包括帧类型、数据长度等。
> - **CRC 校验**：确保数据传输的正确性。
>
> ### **2. `struct can_frame` 的典型定义**
>
> 在 Linux 系统中，`can_frame` 通常在 `<linux/can.h>` 中定义：
>
> ```c
> struct can_frame {
>     canid_t can_id;  // 帧ID（标准帧11位，扩展帧29位）
>     __u8    can_dlc; // 数据长度码（Data Length Code，0~8）
>     __u8    data[8]; // 数据字段（最多8字节）
>     // 某些实现可能包含额外字段（如时间戳、标志位等）
> };
> ```
>
> ### **3. 关键字段解析**
>
> - **`canid_t can_id`**：
>   - 标准帧：低 11 位有效（`0~0x7FF`）。
>   - 扩展帧：使用全部 29 位，需设置扩展帧标志位（如 `CAN_EFF_FLAG`）。
>   - 远程帧：需设置远程传输请求标志位（`CAN_RTR_FLAG`）。
> - **`__u8 can_dlc`**：
>   - 表示数据场的字节数（0~8）。
>   - 注意：CAN 协议规定数据长度只能是 0、1、2、3、4、5、6、7、8 字节。
> - **`__u8 data[8]`**：
>   - 实际传输的数据，长度由 `can_dlc` 决定。
>
> ### **4. 数组 `frames[3]` 的作用**
>
> `struct can_frame frames[3]` 创建了一个包含 3 个 CAN 帧的数组，可用于批量处理 CAN 消息。例如：
>
> ```c
> // 初始化3个CAN帧
> struct can_frame frames[3] = {
>     // 帧1：标准帧，ID=0x123，数据=0x01,0x02
>     { .can_id = 0x123, .can_dlc = 2, .data = {0x01, 0x02} },
>     // 帧2：扩展帧，ID=0x12345678，数据=0xAA,0xBB,0xCC
>     { .can_id = 0x12345678 | CAN_EFF_FLAG, .can_dlc = 3, .data = {0xAA, 0xBB, 0xCC} },
>     // 帧3：远程帧，请求ID=0x456的数据
>     { .can_id = 0x456 | CAN_RTR_FLAG, .can_dlc = 0 }
> };
> ```
>
> ### **5. 应用场景**
>
> - **批量发送 / 接收**：通过 CAN 总线一次性发送或接收多个消息。
> - **数据缓存**：临时存储 CAN 消息，等待后续处理。
> - **多节点通信**：与多个 CAN 节点同时交互时，分别存储不同节点的消息。
>
> ### **6. 相关注意事项**
>
> 1. **CAN 帧长度**：数据场最大为 8 字节，若需传输更多数据，需使用多帧协议（如 ISO 15765-2）。
> 2. **ID 冲突**：同一 CAN 网络中，不同节点的帧 ID 不能重复，否则会导致总线冲突。
> 3. **标志位处理**：设置扩展帧或远程帧时，需正确添加标志位（如 `CAN_EFF_FLAG`、`CAN_RTR_FLAG`）。
>
> ### **总结**
>
> `struct can_frame frames[3]` 是一个存储 3 个 CAN 帧的数组，常用于 CAN 总线通信中的批量数据处理。理解 CAN 帧的结构（ID、数据长度、数据内容）是实现 CAN 通信的基础。

~~~c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <net/if.h>
#include <sys/ioctl.h>
#include <sys/socket.h>
#include <linux/can.h>
#include <linux/can/raw.h>

int main() {
    int sockfd;
    struct sockaddr_can addr;
    struct ifreq ifr;
    struct can_frame frame;
    char buf[128];

    // 创建CAN套接字
    if ((sockfd = socket(PF_CAN, SOCK_RAW, CAN_RAW)) < 0) {
        perror("socket");
        return 1;
    }

    // 获取CAN接口索引
    strcpy(ifr.ifr_name, "can0");
    if (ioctl(sockfd, SIOCGIFINDEX, &ifr) < 0) {
        perror("ioctl");
        close(sockfd);
        return 1;
    }

    // 绑定套接字
    addr.can_family = AF_CAN;
    addr.can_ifindex = ifr.ifr_ifindex;
    if (bind(sockfd, (struct sockaddr *)&addr, sizeof(addr)) < 0) {
        perror("bind");
        close(sockfd);
        return 1;
    }

    // 循环接收CAN帧
    while (1) {
        if (read(sockfd, &frame, sizeof(frame)) < 0) {
            perror("read");
            break;
        }

        // 格式化并输出CAN帧
        int pos = 0;
        if (frame.can_id & CAN_EFF_FLAG) {
            pos = snprintf(buf, sizeof(buf), "<0x%08x>", frame.can_id & CAN_EFF_MASK);
        } else {
            pos = snprintf(buf, sizeof(buf), "<0x%03x>", frame.can_id & CAN_SFF_MASK);
        }

        pos += snprintf(buf + pos, sizeof(buf) - pos, "[%d]", frame.can_dlc);
        
        for (int i = 0; i < frame.can_dlc; i++) {
            pos += snprintf(buf + pos, sizeof(buf) - pos, " %02x", frame.data[i]);
        }

        if (frame.can_id & CAN_RTR_FLAG) {
            snprintf(buf + pos, sizeof(buf) - pos, " remote request");
        }

        printf("%s\n", buf);
    }

    close(sockfd);
    return 0;
}
~~~
