~~~shell
#!/bin/bash

# 定义变量
BACKUP_DIR="Image"              # 修改为你的备份目录
DEVICE="/dev/mmcblk0"       # eMMC 的主设备路径
ROOTFS_PART="p6"            # 根文件系统分区（rootfs）
ROOTFS_MOUNT="./binary"     # 临时挂载点


# 分区映射
PARTITIONS=(
  "p1:uboot.img"      # /dev/mmcblk0p1 -> uboot.img
  "p2:misc.img"       # /dev/mmcblk0p2 -> misc.img
  "p3:boot.img"       # /dev/mmcblk0p3 -> boot.img
  "p4:recovery.img"   # /dev/mmcblk0p4 -> recovery.img
  "p5:backup.img"     # /dev/mmcblk0p5 -> backup.img
)

# 创建备份目录
mkdir -p "$BACKUP_DIR"

# 通用错误处理函数
error_exit() {
  echo "[错误] $1"
  exit 1
}

# 检查挂载状态并卸载
ensure_unmounted() {
  local mount_point="$1"
  if mountpoint -q "$mount_point"; then
    echo "检测到 $mount_point 已挂载，正在解除挂载..."
    sudo umount "$mount_point" || error_exit "无法卸载 $mount_point"
  fi
}

# 备份分区函数
backup_partition() {
  local src="$1"
  local dest="$2"
  echo "正在备份 ${src} 到 ${dest}（稀疏模式）..."
  sudo dd if="$src" of="$dest" bs=4M status=progress conv=sparse || error_exit "备份 ${src} 失败"
}

# 清理并备份 rootfs 分区函数
backup_rootfs() {
  echo "开始备份 rootfs 分区..."
  
  # 创建挂载点
  mkdir -p "$ROOTFS_MOUNT"

  # 确保挂载点未被占用
  ensure_unmounted "$ROOTFS_MOUNT"
  
  # 挂载分区
  echo "挂载 ${DEVICE}${ROOTFS_PART} 到 ${ROOTFS_MOUNT}..."
  sudo mount "${DEVICE}${ROOTFS_PART}" "$ROOTFS_MOUNT" || error_exit "无法挂载 ${DEVICE}${ROOTFS_PART}"

  # 自定义清理操作
  echo "清理 rootfs 不需要的数据..."
  sudo rm -rf "${ROOTFS_MOUNT}/var/lib/misc/firstrun" || echo "[警告] 清理操作失败，继续执行"

  # 执行镜像制作脚本
  if [ -f "./mk-image.sh" ]; then
    echo "运行 mk-image.sh 制作 rootfs 镜像..."
    sudo ./mk-image.sh || error_exit "运行 mk-image.sh 失败"
  else
    echo "[警告] 找不到 mk-image.sh 脚本，跳过 rootfs 镜像制作"
  fi

  # 卸载分区
  echo "卸载 ${ROOTFS_MOUNT}..."
  sudo umount "$ROOTFS_MOUNT" || error_exit "无法卸载 ${ROOTFS_MOUNT}"

  # 清理挂载点
  rmdir "$ROOTFS_MOUNT"
}

# 选择 SOC 的函数，用于选择要构建的 SOC（系统芯片）
select_soc() {
    while true; do
        echo -e "\033[42;36m --------------------------------------------------------- \033[0m"
        echo -e "\033[42;36m 请输入要构建CPU的序号: \033[0m"
        echo -e "\033[42;36m [0] 退出 \033[0m"
        echo -e "\033[42;36m [1] rk3562 \033[0m"
        echo -e "\033[42;36m [2] rk3568_4.19 \033[0m"
        echo -e "\033[42;36m [3] rk3568_5.10 \033[0m"
        echo -e "\033[42;36m [4] rk3576 \033[0m"
        echo -e "\033[42;36m [5] rk3588 \033[0m"
        echo -e "\033[42;36m --------------------------------------------------------- \033[0m"
        read -p "选择: " input

        case $input in
            0) exit;;  # 退出
            1) SOC=rk3562; break;;  # 选择rk3562
            2) SOC=rk3568_4.19; break;;  # 选择rk3568_4.19
            3) SOC=rk3568_5.10; break;;  # 选择rk3588_5.10
            4) SOC=rk3576; break;;  # 选择rk3576
            5) SOC=rk3588; break;;  # 选择rk3588
            *) echo -e "\033[42;36m 无效输入，请重试。 \033[0m";;  # 输入错误提示
        esac
    done
    echo -e "\033[42;36m 设置 SOC=$SOC...... \033[0m"
}

post_rkbin() {
	
}
# 选择对应的SOC
select_soc

# 创建Image目录
mkdir -p $BACKUP_DIR 

# 开始备份
echo "开始备份分区..."
for PARTITION in "${PARTITIONS[@]}"; do
  PART=$(echo "$PARTITION" | cut -d: -f1)  # 提取分区号 (如 p1)
  IMG=$(echo "$PARTITION" | cut -d: -f2)   # 提取镜像文件名 (如 uboot.img)
  SRC="${DEVICE}${PART}"                   # 拼接分区路径 (如 /dev/mmcblk0p1)
  DEST="${BACKUP_DIR}/${IMG}"              # 目标备份文件路径

  backup_partition "$SRC" "$DEST"
done

# 备份 rootfs
backup_rootfs

echo "所有分区和文件系统备份完成！文件保存在 ${BACKUP_DIR}"
~~~

~~~shell
#!/bin/bash

# 获取显示器名称
display_name=$(xrandr | grep " connected" | awk '{ print $1 }')

# 判断是否存在触摸屏设备 ("goodix-ts" 或 "ft5x06")
touch_name=$(xinput list | grep -Ei "goodix-ts|ft5x06" | grep -oP '(?<=↳ ).*(?=\s+id)')
touch_id=$(xinput list | grep -Ei "goodix-ts|ft5x06" | grep -oP '(?<=id=)[0-9]+')

# 如果没有找到显示器，退出
if [ -z "$display_name" ]; then
    echo "没有找到连接的显示器！"
    exit 1
fi

# 提示是否启用触摸屏同步（如果没有找到触摸屏设备则跳过）
if [ -z "$touch_id" ]; then
    echo "警告：未检测到触摸屏设备，触摸屏同步将被跳过。"
    enable_touch=false
else
    enable_touch=true
fi

# 检查是否提供了参数
if [ -z "$1" ]; then
    echo "使用方法: $0 [选项]"
    echo "选项:"
    echo "1 - 正常显示"
    echo "2 - 向左旋转90度"
    echo "3 - 向右旋转90度"
    echo "4 - 旋转180度"
    exit 1
fi

# 判断输入参数
case $1 in
    1)
        # 正常显示
        xrandr --output "$display_name" --rotate normal
        if [ "$enable_touch" = true ]; then
            xinput set-prop "$touch_id" "Coordinate Transformation Matrix" 1 0 0 0 1 0 0 0 1
        fi
        ;;
    2)
        # 向左旋转90度
        xrandr --output "$display_name" --rotate left
        if [ "$enable_touch" = true ]; then
            xinput set-prop "$touch_id" "Coordinate Transformation Matrix" 0 -1 1 1 0 0 0 0 1
        fi
        ;;
    3)
        # 向右旋转90度
        xrandr --output "$display_name" --rotate right
        if [ "$enable_touch" = true ]; then
            xinput set-prop "$touch_id" "Coordinate Transformation Matrix" 0 1 0 -1 0 1 0 0 1
        fi
        ;;
    4)
        # 旋转180度
        xrandr --output "$display_name" --rotate inverted
        if [ "$enable_touch" = true ]; then
            xinput set-prop "$touch_id" "Coordinate Transformation Matrix" -1 0 1 0 -1 1 0 0 1
        fi
        ;;
    *)
        echo "无效选项！使用1、2、3或4作为参数。"
        exit 1
        ;;
esac

echo "显示器旋转完成。"
if [ "$enable_touch" = true ]; then
    echo "触摸屏旋转同步完成。"
else
    echo "未同步触摸屏旋转（未检测到触摸屏设备）。"
fi
~~~
