~~~c
static int dw_hdmi_rockchip_probe(struct platform_device *pdev)
{
	struct rockchip_hdmi *hdmi;
	const struct of_device_id *match;
	struct dw_hdmi_plat_data *plat_data;
	int id;

	hdmi = devm_kzalloc(&pdev->dev, sizeof(*hdmi), GFP_KERNEL);

	id = of_alias_get_id(pdev->dev.of_node, "hdmi");
	if (id < 0)
		id = 0;

	hdmi->id = id;
	hdmi->dev = &pdev->dev;

	match = of_match_node(dw_hdmi_rockchip_dt_ids, pdev->dev.of_node);
	plat_data = devm_kmemdup(&pdev->dev, match->data,
				 sizeof(*plat_data), GFP_KERNEL);
	if (!plat_data)
		return -ENOMEM;

	plat_data->id = hdmi->id;
	hdmi->plat_data = plat_data;
	hdmi->chip_data = plat_data->phy_data;

	platform_set_drvdata(pdev, hdmi);
	pm_runtime_enable(&pdev->dev);
	pm_runtime_get_sync(&pdev->dev);

	return component_add(&pdev->dev, &dw_hdmi_rockchip_ops);
}
~~~

~~~c
struct dw_hdmi_qp *dw_hdmi_qp_bind(struct platform_device *pdev,
				   struct drm_encoder *encoder,
				   struct dw_hdmi_plat_data *plat_data)
{
	struct dw_hdmi_qp *hdmi;
	int ret;

	hdmi = __dw_hdmi_probe(pdev, plat_data);
	if (IS_ERR(hdmi))
		return hdmi;

	if (!plat_data->first_screen) {
		ret = drm_bridge_attach(encoder, &hdmi->bridge, NULL, 0);
		if (ret) {
			__dw_hdmi_remove(hdmi);
			dev_err(hdmi->dev, "Failed to initialize bridge with drm\n");
			return ERR_PTR(ret);
		}

		plat_data->connector = &hdmi->connector;
		if (hdmi->skip_connector && hdmi->next_bridge)
			plat_data->bridge = hdmi->next_bridge;
		else
			plat_data->bridge = NULL;
	}

	if (plat_data->split_mode && !hdmi->plat_data->first_screen) {
		struct dw_hdmi_qp *secondary = NULL;

		if (hdmi->plat_data->left)
			secondary = hdmi->plat_data->left;
		else if (hdmi->plat_data->right)
			secondary = hdmi->plat_data->right;

		if (!secondary)
			return ERR_PTR(-ENOMEM);
		ret = drm_bridge_attach(encoder, &secondary->bridge, &hdmi->bridge,
					DRM_BRIDGE_ATTACH_NO_CONNECTOR);
		if (ret)
			return ERR_PTR(ret);
	}

	return hdmi;
}
~~~

~~~c
static struct dw_hdmi_qp *
__dw_hdmi_probe(struct platform_device *pdev,
		const struct dw_hdmi_plat_data *plat_data)
{
	struct device *dev = &pdev->dev;
	struct device_node *np = dev->of_node;
	struct device_node *ddc_node;
	struct dw_hdmi_qp *hdmi;
	struct dw_hdmi_qp_i2s_audio_data audio;
	struct platform_device_info pdevinfo;
	struct dw_hdmi_qp_cec_data cec;
	struct resource *iores = NULL;
	struct drm_panel *panel = NULL;
	struct drm_bridge *bridge = NULL;
	int irq;
	int ret;

	ret = drm_of_find_panel_or_bridge(np, 1, -1, &panel, &bridge);
	if (ret < 0 && ret != -ENODEV)
		return ERR_PTR(ret);

	hdmi = devm_kzalloc(dev, sizeof(*hdmi), GFP_KERNEL);
	if (!hdmi)
		return ERR_PTR(-ENOMEM);

	hdmi->panel = panel;
	hdmi->next_bridge = bridge;
	hdmi->connector.stereo_allowed = 1;
	hdmi->plat_data = plat_data;
	hdmi->dev = dev;
	hdmi->sample_rate = 48000;
	hdmi->disabled = true;

	mutex_init(&hdmi->mutex);
	mutex_init(&hdmi->audio_mutex);
	mutex_init(&hdmi->cec_notifier_mutex);

	ddc_node = of_parse_phandle(np, "ddc-i2c-bus", 0);
	if (ddc_node) {
		hdmi->ddc = of_get_i2c_adapter_by_node(ddc_node);
		of_node_put(ddc_node);
		if (!hdmi->ddc) {
			dev_dbg(hdmi->dev, "failed to read ddc node\n");
			return ERR_PTR(-EPROBE_DEFER);
		}

	} else {
		dev_dbg(hdmi->dev, "no ddc property found\n");
	}

	if (!plat_data->regm) {
		const struct regmap_config *reg_config;

		reg_config = &hdmi_regmap_config;

		iores = platform_get_resource(pdev, IORESOURCE_MEM, 0);
		hdmi->regs = devm_ioremap_resource(dev, iores);
		if (IS_ERR(hdmi->regs)) {
			ret = PTR_ERR(hdmi->regs);
			goto err_ddc;
		}

		hdmi->regm = devm_regmap_init_mmio(dev, hdmi->regs, reg_config);
		if (IS_ERR(hdmi->regm)) {
			dev_err(dev, "Failed to configure regmap\n");
			ret = PTR_ERR(hdmi->regm);
			goto err_ddc;
		}
	} else {
		hdmi->regm = plat_data->regm;
	}

	ret = dw_hdmi_detect_phy(hdmi);
	if (ret < 0)
		goto err_ddc;

	hdmi_writel(hdmi, 0, MAINUNIT_0_INT_MASK_N);
	hdmi_writel(hdmi, 0, MAINUNIT_1_INT_MASK_N);
	hdmi_writel(hdmi, 428571429, TIMER_BASE_CONFIG0);
	hdmi->logo_plug_out = false;
	if (hdmi->phy.ops->read_hpd(hdmi, hdmi->phy.data) == connector_status_connected &&
	    hdmi_readl(hdmi, I2CM_INTERFACE_CONTROL0)) {
		hdmi->initialized = true;
		hdmi->disabled = false;
	}

	hdmi->sink_is_hdmi = true;

	/* If DDC bus is not specified, try to register HDMI I2C bus */
	if (!hdmi->ddc) {
		hdmi->ddc = dw_hdmi_i2c_adapter(hdmi);
		if (IS_ERR(hdmi->ddc))
			hdmi->ddc = NULL;
		/*
		 * Read high and low time from device tree. If not available use
		 * the default timing scl clock rate is about 99.6KHz.
		 */
		if (of_property_read_u32(np, "ddc-i2c-scl-high-time-ns",
					 &hdmi->i2c->scl_high_ns))
			hdmi->i2c->scl_high_ns = 4708;
		if (of_property_read_u32(np, "ddc-i2c-scl-low-time-ns",
					 &hdmi->i2c->scl_low_ns))
			hdmi->i2c->scl_low_ns = 4916;
	}

	/* Reset HDMI DDC I2C master controller and mute I2CM interrupts */
	if (hdmi->i2c)
		dw_hdmi_i2c_init(hdmi);

	init_completion(&hdmi->flt_cmp);
	init_completion(&hdmi->earc_cmp);

	if (of_property_read_bool(np, "scramble-low-rates"))
		hdmi->scramble_low_rates = true;

	hdmi_init_clk_regenerator(hdmi);

	hdmi->bridge.driver_private = hdmi;
	hdmi->bridge.funcs = &dw_hdmi_bridge_funcs;
#ifdef CONFIG_OF
	hdmi->bridge.of_node = pdev->dev.of_node;
#endif

	if (hdmi->phy.ops->setup_hpd)
		hdmi->phy.ops->setup_hpd(hdmi, hdmi->phy.data);

	hdmi->connector.ycbcr_420_allowed = hdmi->plat_data->ycbcr_420_allowed;

	audio.hdmi	= hdmi;
	audio.eld	= hdmi->connector.eld;
	audio.write	= hdmi_writel;
	audio.read	= hdmi_readl;
	audio.mod	= hdmi_modb;
	hdmi->enable_audio = dw_hdmi_i2s_audio_enable;
	hdmi->disable_audio = dw_hdmi_i2s_audio_disable;

	memset(&pdevinfo, 0, sizeof(pdevinfo));
	pdevinfo.parent = dev;
	pdevinfo.id = PLATFORM_DEVID_AUTO;
	pdevinfo.name = "dw-hdmi-qp-i2s-audio";
	pdevinfo.data = &audio;
	pdevinfo.size_data = sizeof(audio);
	pdevinfo.dma_mask = DMA_BIT_MASK(32);
	hdmi->audio = platform_device_register_full(&pdevinfo);

	hdmi->extcon = devm_extcon_dev_allocate(hdmi->dev, dw_hdmi_cable);
	if (IS_ERR(hdmi->extcon)) {
		dev_err(hdmi->dev, "allocate extcon failed\n");
		ret = PTR_ERR(hdmi->extcon);
		goto err_aud;
	}

	ret = devm_extcon_dev_register(hdmi->dev, hdmi->extcon);
	if (ret) {
		dev_err(hdmi->dev, "failed to register extcon: %d\n", ret);
		goto err_aud;
	}

	ret = extcon_set_property_capability(hdmi->extcon, EXTCON_DISP_HDMI,
					     EXTCON_PROP_DISP_HPD);
	if (ret) {
		dev_err(hdmi->dev,
			"failed to set USB property capability: %d\n", ret);
		goto err_aud;
	}

	irq = platform_get_irq(pdev, 0);
	if (irq < 0) {
		ret = irq;
		goto err_aud;
	}

	hdmi->avp_irq = irq;
	ret = devm_request_threaded_irq(dev, hdmi->avp_irq,
					dw_hdmi_qp_avp_hardirq,
					dw_hdmi_qp_avp_irq, IRQF_ONESHOT,
					dev_name(dev), hdmi);
	if (ret)
		goto err_aud;

	irq = platform_get_irq(pdev, 1);
	if (irq < 0) {
		ret = irq;
		goto err_aud;
	}

	cec.irq = irq;

	if (of_property_read_bool(np, "cec-enable")) {
		hdmi->cec_enable = true;
		cec.hdmi = hdmi;
		cec.ops = &dw_hdmi_qp_cec_ops;
		pdevinfo.name = "dw-hdmi-qp-cec";
		pdevinfo.data = &cec;
		pdevinfo.size_data = sizeof(cec);
		pdevinfo.dma_mask = 0;
		hdmi->cec = platform_device_register_full(&pdevinfo);
	}

	irq = platform_get_irq(pdev, 2);
	if (irq < 0) {
		ret = irq;
		goto err_cec;
	}

	hdmi->earc_irq = irq;
	ret = devm_request_threaded_irq(dev, hdmi->earc_irq,
					dw_hdmi_qp_earc_hardirq,
					dw_hdmi_qp_earc_irq, IRQF_SHARED,
					dev_name(dev), hdmi);
	if (ret)
		goto err_cec;

	irq = platform_get_irq(pdev, 3);
	if (irq < 0) {
		ret = irq;
		goto err_cec;
	}

	hdmi->main_irq = irq;
	ret = devm_request_threaded_irq(dev, hdmi->main_irq,
					dw_hdmi_qp_main_hardirq, NULL,
					IRQF_SHARED, dev_name(dev), hdmi);
	if (ret)
		goto err_cec;

	dw_hdmi_register_debugfs(dev, hdmi);

	if (hdmi_readl(hdmi, CONFIG_REG) & CONFIG_HDCP14) {
		iores = platform_get_resource(pdev, IORESOURCE_MEM, 1);
		hdmi->hdcp14_mem = devm_ioremap_resource(dev, iores);

		if (IS_ERR(hdmi->hdcp14_mem)) {
			ret = PTR_ERR(hdmi->hdcp14_mem);
			goto err_cec;
		}

		ret = dw_hdmi_qp_register_hdcp(dev, hdmi);
		if (ret)
			goto err_cec;
	}

	return hdmi;

err_cec:
	if (!IS_ERR(hdmi->cec))
		platform_device_unregister(hdmi->cec);

err_aud:
	if (hdmi->audio && !IS_ERR(hdmi->audio))
		platform_device_unregister(hdmi->audio);

err_ddc:
	if (hdmi->i2c)
		i2c_del_adapter(&hdmi->i2c->adap);
	else
		i2c_put_adapter(hdmi->ddc);

	if (!hdmi->plat_data->first_screen) {
		dw_hdmi_destroy_properties(hdmi);
		hdmi->connector.funcs->destroy(&hdmi->connector);
	}

	if (hdmi->bridge.encoder && !hdmi->plat_data->first_screen)
		hdmi->bridge.encoder->funcs->destroy(hdmi->bridge.encoder);

	return ERR_PTR(ret);
}
~~~

~~~c
static int dw_hdmi_qp_bridge_attach(struct drm_bridge *bridge,
				    enum drm_bridge_attach_flags flags)
{
	struct dw_hdmi_qp *hdmi = bridge->driver_private;
	struct drm_encoder *encoder = bridge->encoder;
	struct drm_connector *connector = &hdmi->connector;
	struct cec_connector_info conn_info;
	struct cec_notifier *notifier;
	bool skip_connector = false;

	if (hdmi->next_bridge) {
		struct drm_bridge *next_bridge = hdmi->next_bridge;
		int ret;

		ret = drm_bridge_attach(bridge->encoder, next_bridge, bridge,
					next_bridge->ops & DRM_BRIDGE_OP_MODES ?
					DRM_BRIDGE_ATTACH_NO_CONNECTOR : 0);
		if (ret) {
			DRM_ERROR("failed to attach next bridge: %d\n", ret);
			return ret;
		}

		skip_connector = !(next_bridge->ops & DRM_BRIDGE_OP_MODES);
	}

	hdmi->skip_connector = skip_connector;
	if (flags & DRM_BRIDGE_ATTACH_NO_CONNECTOR || skip_connector)
		return 0;

	connector->interlace_allowed = 1;
	connector->polled = DRM_CONNECTOR_POLL_HPD;
	if (hdmi->next_bridge && hdmi->next_bridge->ops & DRM_BRIDGE_OP_DETECT)
		connector->polled = DRM_CONNECTOR_POLL_CONNECT | DRM_CONNECTOR_POLL_DISCONNECT;
	drm_connector_helper_add(connector, &dw_hdmi_connector_helper_funcs);

	drm_connector_init(bridge->dev, connector, &dw_hdmi_connector_funcs,
			   DRM_MODE_CONNECTOR_HDMIA);

	drm_connector_attach_encoder(connector, encoder);
	dw_hdmi_attach_properties(hdmi);

	if (hdmi->cec_enable) {
		cec_fill_conn_info_from_drm(&conn_info, connector);
		notifier = cec_notifier_conn_register(hdmi->dev, NULL, &conn_info);
		if (!notifier)
			return -ENOMEM;

		mutex_lock(&hdmi->cec_notifier_mutex);
		hdmi->cec_notifier = notifier;
		mutex_unlock(&hdmi->cec_notifier_mutex);
	}

	return 0;
}
~~~

这里的get_modes 指针函数dw_hdmi_connector_get_modes就是最终的edid模式获取，具体内容如下所示：

~~~c
static int dw_hdmi_connector_get_modes(struct drm_connector *connector)
{
    // 从 connector 获取 dw_hdmi_qp 结构体指针
    struct dw_hdmi_qp *hdmi =
        container_of(connector, struct dw_hdmi_qp, connector);
    
    // 获取 HDR 静态元数据指针
    struct hdr_static_metadata *metedata =
        &connector->hdr_sink_metadata.hdmi_type1;
    
    // EDID 数据指针
    struct edid *edid;
    
    // 显示模式指针
    struct drm_display_mode *mode;
    
    // 连接器显示信息
    struct drm_display_info *info = &connector->display_info;
    
    // 平台数据（PHY 相关数据）
    void *data = hdmi->plat_data->phy_data;
    
    // EDID blob 指针（用于缓存EDID数据）
    struct drm_property_blob *edid_blob_ptr = connector->edid_blob_ptr;
    
    // 循环计数器
    int i, ret = 0;

    // 处理 HDMI 分屏模式（右侧设备）
    if (hdmi->plat_data->right && hdmi->plat_data->right->next_bridge) {
        struct drm_bridge *bridge = hdmi->plat_data->right->next_bridge;

        // 如果桥接器支持获取模式，则调用其获取模式函数
        if (bridge->ops & DRM_BRIDGE_OP_MODES) {
            if (!drm_bridge_get_modes(bridge, connector))
                return 0;
        }
    }

    // 如果有面板（如LCD面板），直接获取面板模式
    if (hdmi->panel)
        return drm_panel_get_modes(hdmi->panel, connector);

    // 如果有下一个桥接器，调用其获取模式函数
    if (hdmi->next_bridge && hdmi->next_bridge->ops & DRM_BRIDGE_OP_MODES)
        return drm_bridge_get_modes(hdmi->next_bridge, connector);

    // 如果没有DDC接口（I2C），直接返回
    if (!hdmi->ddc)
        return 0;

    // 清空HDR元数据
    memset(metedata, 0, sizeof(*metedata));

    // 检查是否已有EDID缓存（从EDID blob获取）
    if (edid_blob_ptr && edid_blob_ptr->length) {
        edid = kmalloc(edid_blob_ptr->length, GFP_KERNEL);
        if (!edid)
            return -ENOMEM;
        memcpy(edid, edid_blob_ptr->data, edid_blob_ptr->length);
    } else {
        // 通过DDC接口从显示器读取EDID
        edid = drm_get_edid(connector, hdmi->ddc);
        hdmi->hdcp_caps = dw_hdmi_qp_hdcp_capable(hdmi); // 检查HDCP支持能力
    }

    // 以下代码用于调试，实际生产中可能移除
    printk("hdmi 11111111111\n");
    if (edid) {
        printk("hdmi 222222222222222\n");
        dev_dbg(hdmi->dev, "got edid: width[%d] x height[%d]\n",
            edid->width_cm, edid->height_cm);

        // 检测显示器是否支持HDMI
        hdmi->support_hdmi = drm_detect_hdmi_monitor(edid);
        // 检测显示器是否支持音频
        hdmi->sink_has_audio = drm_detect_monitor_audio(edid);
        
        // 将EDID数据更新到连接器属性
        drm_connector_update_edid_property(connector, edid);
        
        // 如果有CEC通知器，设置物理地址
        if (hdmi->cec_notifier)
            cec_notifier_set_phys_addr_from_edid(hdmi->cec_notifier, edid);
        
        // 调用平台特定函数处理EDID中的DSC信息
        if (hdmi->plat_data->get_edid_dsc_info)
            hdmi->plat_data->get_edid_dsc_info(data, edid);
        
        // 将EDID解析为显示模式并添加到连接器
        ret = drm_add_edid_modes(connector, edid);
        
        // 调用平台特定函数获取色彩信息
        if (hdmi->plat_data->get_colorimetry)
            hdmi->plat_data->get_colorimetry(data, edid);
        
        // 调用平台特定函数获取YUV422格式信息
        if (hdmi->plat_data->get_yuv422_format)
            hdmi->plat_data->get_yuv422_format(connector, edid);
        
        // 更新HDR属性
        dw_hdmi_update_hdr_property(connector);
        
        // 重新检查HDCP支持能力
        hdmi->hdcp_caps = dw_hdmi_qp_hdcp_capable(hdmi);
        
        // 如果是分屏模式（双屏），处理第二个屏幕
        if (ret > 0 && hdmi->plat_data->split_mode) {
            struct dw_hdmi_qp *secondary = NULL;
            void *secondary_data;

            // 确定第二个屏幕（左侧或右侧）
            if (hdmi->plat_data->left)
                secondary = hdmi->plat_data->left;
            else if (hdmi->plat_data->right)
                secondary = hdmi->plat_data->right;

            // 如果没有找到第二个屏幕，返回错误
            if (!secondary) {
                kfree(edid);
                return -ENOMEM;
            }
            secondary_data = secondary->plat_data->phy_data;

            // 为所有模式应用分屏转换
            list_for_each_entry(mode, &connector->probed_modes, head)
                hdmi->plat_data->convert_to_split_mode(mode);

            // 设置第二个屏幕的HDMI/音频支持
            secondary->sink_is_hdmi = drm_detect_hdmi_monitor(edid);
            secondary->sink_has_audio = drm_detect_monitor_audio(edid);
            
            // 如果有CEC通知器，设置第二个屏幕的物理地址
            if (secondary->cec_notifier)
                cec_notifier_set_phys_addr_from_edid(secondary->cec_notifier,
                                                     edid);
            
            // 调用平台特定函数处理第二个屏幕的DSC信息
            if (secondary->plat_data->get_edid_dsc_info)
                secondary->plat_data->get_edid_dsc_info(secondary_data, edid);
        }
        // 释放EDID内存
        kfree(edid);
    } else {
        // 如果EDID获取失败（没有有效EDID）
        printk("hdmi 3333333333333333333333333\n");
        hdmi->support_hdmi = true; // 默认认为支持HDMI
        hdmi->sink_has_audio = true; // 默认认为支持音频

        // 如果是分屏模式，设置第二个屏幕的默认支持
        if (hdmi->plat_data->split_mode) {
            if (hdmi->plat_data->left) {
                hdmi->plat_data->left->sink_is_hdmi = true;
                hdmi->plat_data->left->sink_has_audio = true;
            } else if (hdmi->plat_data->right) {
                hdmi->plat_data->right->sink_is_hdmi = true;
                hdmi->plat_data->right->sink_has_audio = true;
            }
        }

        // 添加默认显示模式（当EDID获取失败时）
        for (i = 0; i < ARRAY_SIZE(dw_hdmi_default_modes); i++) {
            const struct drm_display_mode *ptr =
                &dw_hdmi_default_modes[i];

            mode = drm_mode_duplicate(connector->dev, ptr);
            if (mode) {
                if (!i)
                    mode->type = DRM_MODE_TYPE_PREFERRED; // 设置首选模式
                drm_mode_probed_add(connector, mode);
                ret++;
            }
        }

        // 如果是分屏模式，为所有模式应用分屏转换
        if (ret > 0 && hdmi->plat_data->split_mode) {
            struct drm_display_mode *mode;
            list_for_each_entry(mode, &connector->probed_modes, head)
                hdmi->plat_data->convert_to_split_mode(mode);
        }

        // 设置显示信息
        info->edid_hdmi_dc_modes = 0;
        info->hdmi.y420_dc_modes = 0;
        info->color_formats = 0;

        // 打印错误日志
        dev_info(hdmi->dev, "failed to get edid\n");
    }

    // 如果EDID获取失败，添加默认模式
    if (!edid) {
        mode = drm_mode_duplicate(connector->dev, &dw_hdmi_default_modes[0]);
        if (mode) {
            drm_mode_probed_add(connector, mode);
            dev_info(hdmi->dev, "Force add mode: %s\n", mode->name);
        }
    }

    // 返回成功获取的模式数量
    return ret;
}
~~~

以下是对 OV13850 驱动代码的**分层结构分析**，按驱动架构的典型层次进行组织，便于理解其设计逻辑和关键组件：

------

### **1. 硬件抽象层（Hardware Abstraction Layer）**

***作用：** 封装传感器寄存器操作、硬件参数和配置，屏蔽底层差异。*

#### **关键组件：**

**寄存器定义与位字段**

```c
#define OV13850_REG_CHIP_ID     0x300a  // 传感器ID寄存器
#define OV13850_MODE_STREAMING  BIT(0)  // 流模式控制位
#define OV13850_GAIN_H_MASK     0x07    // 增益高字节掩码
#define OV13850_XVCLK_FREQ      24000000  // XCLK时钟频率
```

定义了传感器核心寄存器地址、控制位、关键硬件参数（如时钟频率、像素率）。

**传感器模式配置表**

```c
static const struct ov13850_mode supported_modes[] = {
    {
        .width = 4224, .height = 3136, .max_fps = {10000, 150000}, // 20fps
        .reg_list = ov13850_4224x3136_regs,  // 该分辨率配置寄存器列表
    },
    { ... } // 2112x1568 30fps模式
};
```

定义了传感器支持的分辨率、帧率、默认寄存器配置。

**寄存器配置表（全局/模式级）**

```c
static const struct regval ov13850_global_regs_r1a[] = { ... }; // 芯片版本R1A的全局寄存器
static const struct regval ov13850_4224x3136_regs[] = { ... }; // 4224x3136模式寄存器
```

- `ov13850_global_regs_*`：初始化传感器的基础配置（根据芯片版本选择）。
- `ov13850_*_regs`：针对特定分辨率的寄存器配置。

**寄存器操作函数**

```c
static int ov13850_write_reg(struct i2c_client *client, u16 reg, u32 len, u32 val); // 写寄存器
static int ov13850_read_reg(struct i2c_client *client, u16 reg, unsigned int len, u32 *val); // 读寄存器
static int ov13850_write_array(struct i2c_client *client, const struct regval *regs); // 批量写寄存器
```

封装 I2C 通信细节，支持 8/16/24 位寄存器操作。

------

### **2. 设备驱动层（Device Driver Layer）**

***作用：** 管理硬件资源（电源、时钟、GPIO）、设备初始化/移除、电源管理。*

#### **关键组件：**

- **设备结构体**

  ```c
  struct ov13850 {
      struct i2c_client *client;
      struct clk *xvclk;          // XCLK时钟
      struct gpio_desc *power_gpio; // 电源控制GPIO
      struct regulator_bulk_data supplies[OV13850_NUM_SUPPLIES]; // 电源管理
      struct pinctrl *pinctrl;    // 引脚控制
      const struct ov13850_mode *cur_mode; // 当前工作模式
      // ... 其他硬件资源
  };
  ```

- **设备探测（`ov13850_probe`）**

  从设备树获取配置

  ```c
  of_property_read_string(node, RKMODULE_CAMERA_MODULE_NAME, &ov13850->module_name);
  ```

  初始化硬件资源

  - 获取时钟、GPIO、电源（`devm_clk_get`, `devm_gpiod_get`）。
  - 初始化电源管理（`devm_regulator_bulk_get`）。

  上电序列（`__ov13850_power_on`）

  ```c
  clk_prepare_enable(ov13850->xvclk); // 使能时钟
  gpiod_set_value_cansleep(ov13850->power_gpio, 1); // 使能电源
  regulator_bulk_enable(...); // 使能电源
  ```

  传感器ID检测

  ```c
  ov13850_check_sensor_id(ov13850, client); // 读取ID并选择寄存器表
  ```

  注册V4L2子设备

  ```c
  v4l2_async_register_subdev_sensor_common(sd);
  ```

- **设备移除（`ov13850_remove`）**

  - 注销子设备、清理媒体实体、下电（`__ov13850_power_off`）。

- **电源管理（Runtime PM）**

  ```c
  static int ov13850_runtime_suspend(struct device *dev); // 挂起时下电
  static int ov13850_runtime_resume(struct device *dev);  // 恢复时上电
  ```

  - 通过 `pm_runtime` 优化功耗（仅在需要时激活传感器）。

------

### **3. V4L2 子设备接口层（V4L2 Subdev Interface Layer）**

***作用：** 实现 V4L2 标准接口，供上层应用（如 Camera HAL）调用。*

#### **关键操作函数表：**

```c
static const struct v4l2_subdev_ops ov13850_subdev_ops = {
    .core   = &ov13850_core_ops,
    .video  = &ov13850_video_ops,
    .pad    = &ov13850_pad_ops,
};
```

##### **(1) 核心操作（`.core`）**

| 函数                     | 功能                                           |
| ------------------------ | ---------------------------------------------- |
| `ov13850_s_power`        | 电源开关（上电/下电）                          |
| `ov13850_ioctl`          | 自定义 ioctl（如 `RKMODULE_SET_QUICK_STREAM`） |
| `ov13850_compat_ioctl32` | 32位兼容 ioctl（适配旧用户空间）               |

> **关键点**：
>
> - `RKMODULE_SET_QUICK_STREAM` 直接写 `OV13850_REG_CTRL_MODE`，跳过完整配置，实现快速启动/停止。

##### **(2) 视频操作（`.video`）**

| 函数                       | 功能                                                         |
| -------------------------- | ------------------------------------------------------------ |
| `ov13850_s_stream`         | 启动/停止视频流（调用 `__ov13850_start_stream`/`__ov13850_stop_stream`） |
| `ov13850_g_frame_interval` | 获取当前帧率（根据当前模式返回 `max_fps`）                   |

> **关键点**：
>
> - `__ov13850_start_stream`：写入模式寄存器配置 + 设置流模式。

##### **(3) Pad 操作（`.pad`）**

| 函数                          | 功能                                         |
| ----------------------------- | -------------------------------------------- |
| `ov13850_set_fmt`             | 设置分辨率（选择最接近的 `supported_modes`） |
| `ov13850_get_fmt`             | 获取当前分辨率                               |
| `ov13850_enum_frame_sizes`    | 枚举支持的分辨率（4224x3136 / 2112x1568）    |
| `ov13850_enum_frame_interval` | 枚举支持的帧率（20fps / 30fps）              |
| `ov13850_g_mbus_config`       | 获取 MIPI 配置（`V4L2_MBUS_CSI2_DPHY`）      |

> **关键点**：
>
> - `ov13850_set_fmt` 通过 `ov13850_find_best_fit` 选择最接近的分辨率。

------

### **4. 控制接口层（Control Interface Layer）**

***作用：** 实现 V4L2 控制（如曝光、增益），通过 `v4l2_ctrl_handler` 管理。*

#### **关键组件：**

- **控制初始化（`ov13850_initialize_controls`）**

  ```c
  ov13850->exposure = v4l2_ctrl_new_std(..., V4L2_CID_EXPOSURE, ...);
  ov13850->anal_gain = v4l2_ctrl_new_std(..., V4L2_CID_ANALOGUE_GAIN, ...);
  ```

- **控制处理（`ov13850_set_ctrl`）**

  ```c
  switch (ctrl->id) {
      case V4L2_CID_EXPOSURE:
          ov13850_write_reg(OV13850_REG_EXPOSURE, ...); // 写曝光寄存器
          break;
      case V4L2_CID_ANALOGUE_GAIN:
          ov13850_write_reg(OV13850_REG_GAIN_H, ...);  // 写增益高/低字节
          break;
      // ... 其他控制
  }
  ```

> **关键点**：
>
> - 曝光值（`V4L2_CID_EXPOSURE`）：4 位小数部分（`ctrl->val << 4`）。
> - 增益值（`V4L2_CID_ANALOGUE_GAIN`）：拆分为高/低字节写入寄存器。

------

### **5. 应用层接口（User Space Interface）**

***作用：** 通过 ioctl 供用户空间调用，实现设备特定功能。*

#### **关键 ioctl 命令：**

| 命令                        | 作用                               | 实现函数                 |
| --------------------------- | ---------------------------------- | ------------------------ |
| `RKMODULE_GET_MODULE_INFO`  | 获取模块信息（传感器/镜头/模块名） | `ov13850_ioctl`          |
| `RKMODULE_SET_QUICK_STREAM` | 快速启动/停止流（避免完整配置）    | `ov13850_ioctl`          |
| `RKMODULE_AWB_CFG`          | 适配白平衡配置（兼容层）           | `ov13850_compat_ioctl32` |

> **设计亮点**：
>
> - `RKMODULE_SET_QUICK_STREAM` 直接操作 `OV13850_REG_CTRL_MODE`，**避免重复写入模式寄存器**（提升启动速度）。

------

### **6. 关键设计亮点总结**

| 特性               | 说明                                       |
| ------------------ | ------------------------------------------ |
| **双分辨率支持**   | 4224x3136 (20fps) 和 2112x1568 (30fps)     |
| **快速流控制**     | `RKMODULE_SET_QUICK_STREAM` 跳过寄存器配置 |
| **运行时电源管理** | `pm_runtime` 仅在需要时激活传感器          |
| **设备树兼容**     | 通过 `of_match_table` 匹配设备树节点       |
| **硬件抽象**       | 寄存器操作封装（`ov13850_write_reg`）      |
| **多版本支持**     | 根据芯片版本（R1A/R2A）选择寄存器配置      |

------

### **总结**

- **硬件抽象层**：定义寄存器、分辨率、配置表，屏蔽硬件细节。
- **设备驱动层**：管理电源/时钟/GPIO，实现设备初始化/移除。
- **V4L2 接口层**：提供标准子设备接口（`set_fmt`/`s_stream`）。
- **控制接口层**：实现 V4L2 控制（曝光/增益）。
- **应用层接口**：通过 ioctl 提供快速流控制等特性。

此驱动**高度模块化**，符合 Linux V4L2 驱动框架规范，同时针对 OV13850 传感器特性（如快速启动）做了优化，是典型的高质量嵌入式摄像头驱动实现。

### **0) `svp_acl_resnet50`**

- **说明**：加载一个预编译好的 ResNet50 `.om` 模型，在 NPU 上执行一次完整的前向推理。

### **1) `svp_acl_resnet50_multi_thread`**
- **用途**：多线程并发执行 ResNet50 推理。
- **说明**：启动多个线程，每个线程独立运行 ResNet50 推理任务，用于测试 NPU 在多任务并发下的性能与稳定性。

### **2) `svp_acl_resnet50_dynamic_batch_with_mem_cached`**
- **用途**：支持动态 batch size 的 ResNet50 推理，并启用模型内存缓存。
- **说明**：
  - **动态 batch**：允许一次推理处理不同数量的输入图像（如 batch=1,2,4 等），提升吞吐。
  - **内存缓存**：避免每次推理都重新加载模型到设备内存，提升效率。
- **典型场景**：高吞吐 AI 推理服务，需灵活应对不同负载。

### **3) `svp_acl_lstm`**
- **用途**：测试 LSTM（长短期记忆网络）模型在 NPU 上的推理能力。

### **4) `svp_acl_preemption`**
- **用途**：测试 NPU 任务抢占（Preemption）机制。
- **说明**：

  - 验证系统在资源紧张时能否正确调度 NPU 任务。
- **典型场景**：实时性要求高的系统（如安防告警）中，确保关键任务优先执行。

~~~c
/**
 * @brief 初始化 SVP NPU 的 ACL（Ascend Computing Language）运行环境
 * 
 * 该函数完成以下关键步骤：
 * 1. 检查系统 MPI（媒体处理接口）是否已初始化；
 * 2. 初始化 ACL 运行时；
 * 3. 指定并打开 NPU 设备；
 * 4. 验证当前运行模式是否为设备模式（Device Mode）。
 * 
 * @param acl_config_path ACL 配置文件路径（可为 NULL 或空字符串，表示使用默认配置）
 * @return TD_SUCCESS: 成功；TD_FAILURE: 失败
 */
static td_s32 sample_svp_npu_acl_init(const td_char *acl_config_path)
{
    /* 定义变量 */
    svp_acl_rt_run_mode run_mode;  // 用于存储当前 ACL 运行模式（如 Device / Host）
    svp_acl_error ret;             // ACL 接口返回的错误码
    td_bool is_mpi_init;           // 标记 MPI 系统是否已初始化

    /* 1. 检查 MPI（媒体处理接口）是否已初始化 */
    is_mpi_init = sample_common_svp_check_sys_init();
    // 如果 MPI 未初始化，则返回失败
    sample_svp_check_exps_return(
        is_mpi_init != TD_TRUE, 
        TD_FAILURE, 
        SAMPLE_SVP_ERR_LEVEL_ERROR, 
        "mpi init failed!\n"
    );

    /* 2. 初始化 ACL 运行时环境 */
    ret = svp_acl_init(acl_config_path);
    // 若初始化失败，返回错误
    sample_svp_check_exps_return(
        ret != SVP_ACL_SUCCESS, 
        TD_FAILURE, 
        SAMPLE_SVP_ERR_LEVEL_ERROR, 
        "acl init failed!\n"
    );
    sample_svp_trace_info("svp acl init success!\n");  // 打印成功日志

    /* 3. 指定并打开 NPU 设备（g_svp_npu_dev_id 通常是 0，表示第一个 NPU 设备） */
    ret = svp_acl_rt_set_device(g_svp_npu_dev_id);
    if (ret != SVP_ACL_SUCCESS) {
        // 若打开设备失败，需先调用 finalize 清理 ACL 环境
        (td_void)svp_acl_finalize();
        sample_svp_trace_err("svp acl open device %d failed!\n", g_svp_npu_dev_id);
        return TD_FAILURE;
    }
    sample_svp_trace_info("open device %d success!\n", g_svp_npu_dev_id);

    /* 4. 获取当前运行模式，并验证是否为设备模式（SVP_ACL_DEVICE） */
    ret = svp_acl_rt_get_run_mode(&run_mode);
    if ((ret != SVP_ACL_SUCCESS) || (run_mode != SVP_ACL_DEVICE)) {
        // 若获取失败或不是设备模式，需重置设备并清理 ACL 环境
        (td_void)svp_acl_rt_reset_device(g_svp_npu_dev_id);
        (td_void)svp_acl_finalize();
        sample_svp_trace_err("acl get run mode failed!\n");
        return TD_FAILURE;
    }
    sample_svp_trace_info("get run mode success!\n");

    /* 所有步骤成功，返回成功状态 */
    return TD_SUCCESS;
}
~~~
