~~~c
#include <stdint.h>               // 包含整数类型定义
#include <stdio.h>                // 包含标准输入输出函数
#include <stdlib.h>               // 包含标准库函数
#include <string.h>               // 包含字符串处理函数

#include "yolov5.h"               // YOLOv5 模型头文件
#include "image_utils.h"          // 图像处理工具头文件
#include "file_utils.h"           // 文件操作工具头文件
#include "image_drawing.h"        // 图像绘制工具头文件
#include <opencv2/opencv.hpp>     // OpenCV库头文件

/*-------------------------------------------
                  主函数
-------------------------------------------*/
int main(int argc, char **argv)
{
    // 检查命令行参数是否正确
    if (argc != 2)
    {
        printf("%s <model_path> \n", argv[0]); // 输出使用说明
        return -1; // 返回错误代码
    }

    const char *model_path = argv[1]; // 获取模型路径

    int ret;                             // 返回值
    rknn_app_context_t rknn_app_ctx;    // RKNN应用程序上下文
    memset(&rknn_app_ctx, 0, sizeof(rknn_app_context_t)); // 初始化上下文

    init_post_process(); // 初始化后处理函数

    // 初始化 YOLOv5 模型
    ret = init_yolov5_model(model_path, &rknn_app_ctx);
    if (ret != 0) // 如果初始化失败
    {
        printf("init_yolov5_model fail! ret=%d model_path=%s\n", ret, model_path); // 输出错误信息
    }

    image_buffer_t src_image;  // 源图像缓冲区
    memset(&src_image, 0, sizeof(image_buffer_t)); // 初始化源图像缓冲区
    object_detect_result_list od_results; // 检测结果列表

    // 打开视频捕获设备
    cv::VideoCapture cap("/dev/video25");
    if (!cap.isOpened()) {
        std::cerr << "Failed to open the camera." << std::endl; // 如果打开设备失败，输出错误信息
        return -1; // 返回错误代码
    }

    cv::Mat src_frame; // 存储捕获的帧
    while(true){ // 循环捕获视频帧
        cap >> src_frame; // 从视频捕获设备读取一帧
        if (src_frame.empty()) { // 如果帧为空（读取失败）
            std::cerr << "Failed to grab frame from the camera." << std::endl; // 输出错误信息
            break; // 退出循环
        }

        // 设置源图像的属性
        src_image.height = src_frame.rows; // 设置图像高度
        src_image.width = src_frame.cols;  // 设置图像宽度
        src_image.width_stride = src_frame.step[0]; // 设置行跨度（每行字节数）
        src_image.virt_addr = src_frame.data; // 设置图像数据指针
        src_image.format = IMAGE_FORMAT_RGB888; // 设置图像格式为 RGB888
        src_image.size = src_frame.total() * src_frame.elemSize(); // 计算图像总大小

        // 执行 YOLOv5 推理
        ret = inference_yolov5_model(&rknn_app_ctx, &src_image, &od_results);
        if (ret != 0) // 如果推理失败
        {
            printf("inference_yolov5_model fail! ret=%d\n", ret); // 输出错误信息
        }

        // 画框和文本
        char text[256]; // 存储文本信息
        for (int i = 0; i < od_results.count; i++) // 遍历检测结果
        {
            object_detect_result *det_result = &(od_results.results[i]); // 获取检测结果
            printf("%s @ (%d %d %d %d) %.3f\n", coco_cls_to_name(det_result->cls_id),
                det_result->box.left, det_result->box.top,
                det_result->box.right, det_result->box.bottom,
                det_result->prop); // 输出检测对象的信息

            int x1 = det_result->box.left;  // 左上角 x 坐标
            int y1 = det_result->box.top;   // 左上角 y 坐标
            int x2 = det_result->box.right; // 右下角 x 坐标
            int y2 = det_result->box.bottom; // 右下角 y 坐标

            // 在图像上画矩形框
            draw_rectangle(&src_image, x1, y1, x2 - x1, y2 - y1, COLOR_BLUE, 3);

            // 格式化文本信息
            sprintf(text, "%s %.1f%%", coco_cls_to_name(det_result->cls_id), det_result->prop * 100);
            // 在图像上绘制文本
            draw_text(&src_image, text, x1, y1 - 20, COLOR_GREEN, 10);
        }

        // 创建 OpenCV 决策的 Mat 对象并显示结果
        cv::Mat result_mat = cv::Mat(src_image.height, src_image.width, CV_8UC3, src_image.virt_addr, src_image.width_stride);
        cv::imshow("img", result_mat); // 显示图像
        int key = cv::waitKey(1); // 等待按键
        if (key == 'q' || key == 27) { // 如果按下 'q' 或 Esc 键，退出循环
            break;
        }
    }

    deinit_post_process(); // 清理后处理资源

    // 释放 YOLOv5 模型
    ret = release_yolov5_model(&rknn_app_ctx);
    if (ret != 0) // 如果释放失败
    {
        printf("release_yolov5_model fail! ret=%d\n", ret); // 输出错误信息
    }

    // 如果虚拟地址不为 NULL，释放内存
    if (src_image.virt_addr != NULL)
    {
        free(src_image.virt_addr); // 释放虚拟地址内存
    }

    return 0; // 返回成功代码
}
~~~

~~~c
void doInference(IExecutionContext& context, float* input, float* output, const int output_size, cv::Size input_shape,const char *INPUT_BLOB_NAME,const char *OUTPUT_BLOB_NAME) {
    const ICudaEngine& engine = context.getEngine();

    // Pointers to input and output device buffers to pass to engine.
    // Engine requires exactly IEngine::getNbBindings() number of buffers.
    assert(engine.getNbBindings() == 2);
    void* buffers[2];

    // In order to bind the buffers, we need to know the names of the input and output tensors.
    // Note that indices are guaranteed to be less than IEngine::getNbBindings()
    const int inputIndex = engine.getBindingIndex(INPUT_BLOB_NAME);

    assert(engine.getBindingDataType(inputIndex) == nvinfer1::DataType::kFLOAT);
    const int outputIndex = engine.getBindingIndex(OUTPUT_BLOB_NAME);
    assert(engine.getBindingDataType(outputIndex) == nvinfer1::DataType::kFLOAT);
    int mBatchSize = engine.getMaxBatchSize();

    // Create GPU buffers on device
    CHECK(cudaMalloc(&buffers[inputIndex], 3 * input_shape.height * input_shape.width * sizeof(float)));
    CHECK(cudaMalloc(&buffers[outputIndex], output_size*sizeof(float)));

    // Create stream
    cudaStream_t stream;
    CHECK(cudaStreamCreate(&stream));

    // DMA input batch data to device, infer on the batch asynchronously, and DMA output back to host
    CHECK(cudaMemcpyAsync(buffers[inputIndex], input, 3 * input_shape.height * input_shape.width * sizeof(float), cudaMemcpyHostToDevice, stream));
    context.enqueue(1, buffers, stream, nullptr);
    // context.enqueueV2( buffers, stream, nullptr);
    CHECK(cudaMemcpyAsync(output, buffers[outputIndex], output_size * sizeof(float), cudaMemcpyDeviceToHost, stream));
    cudaStreamSynchronize(stream);

    // Release stream and buffers
    cudaStreamDestroy(stream);
    CHECK(cudaFree(buffers[inputIndex]));
    CHECK(cudaFree(buffers[outputIndex]));
}
~~~
