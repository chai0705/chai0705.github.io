~~~c
#include <stdlib.h>
#include <fcntl.h>
#include <unistd.h>
#include <linux/input.h>
#include <string.h>
#include <errno.h>

#define DEVICE_NAME "icm4260x" // 这里匹配 input device 名称关键字

// 在 /dev/input/ 下查找设备文件
int find_input_device(char *event_path, size_t size)
{
    char path[64];
    char name[256];
    int fd;

    for (int i = 0; i < 32; i++) {
        snprintf(path, sizeof(path), "/dev/input/event%d", i);
        fd = open(path, O_RDONLY);
        if (fd < 0)
            continue;

        if (ioctl(fd, EVIOCGNAME(sizeof(name)), name) >= 0) {
            if (strstr(name, DEVICE_NAME)) {
                strncpy(event_path, path, size - 1);
                event_path[size - 1] = '\0';
                close(fd);
                return 0;
            }
        }
        close(fd);
    }
    return -1;
}

int main(void)
{
    char event_path[64];
    struct input_event ev;
    int fd;

    if (find_input_device(event_path, sizeof(event_path)) < 0) {
        fprintf(stderr, "未找到包含 '%s' 的 input 设备\n", DEVICE_NAME);
        return 1;
    }

    printf("找到 ICM4260x 设备: %s\n", event_path);

    fd = open(event_path, O_RDONLY);
    if (fd < 0) {
        perror("打开 input 设备失败");
        return 1;
    }

    printf("开始读取加速度数据 (Ctrl+C 退出)...\n");

    while (1) {
        ssize_t n = read(fd, &ev, sizeof(ev));
        if (n == (ssize_t)sizeof(ev)) {
            if (ev.type == EV_ABS) {
                switch (ev.code) {
                    case ABS_X:
                        printf("X: %d\n", ev.value);
                        break;
                    case ABS_Y:
                        printf("Y: %d\n", ev.value);
                        break;
                    case ABS_Z:
                        printf("Z: %d\n", ev.value);
                        break;
                }
            }
        } else if (n < 0) {
            perror("读取事件失败");
            break;
        }
    }

    close(fd);
    return 0;
}
~~~

~~~c
static int find_input_device(char *event_path, size_t size)
{
    char path[64];
    char name[256];
    int fd;

    for (int i = 0; i < 32; i++) {
        snprintf(path, sizeof(path), "/dev/input/event%d", i);
        fd = open(path, O_RDONLY);
        if (fd < 0)
            continue;

        if (ioctl(fd, EVIOCGNAME(sizeof(name)), name) >= 0) {

            }
        }
        close(fd);
    }
    return -1;

int main(void)
{
    char event_path[64];
    int fd;
    struct input_event ev;
    struct pollfd pfd;
    int gyro_x = 0, gyro_y = 0, gyro_z = 0;

    if (find_input_device(event_path, sizeof(event_path)) < 0) {
        fprintf(stderr, "未找到包含 '%s' 的 input 设备\n", DEVICE_KEYWORD);
        return 1;
    }

    printf("找到 ICM4260x 陀螺仪设备: %s\n", event_path);

    fd = open(event_path, O_RDONLY);
    if (fd < 0) {
        perror("打开 input 设备失败");
        return 1;
    }

    pfd.fd = fd;
    pfd.events = POLLIN;

    printf("开始读取陀螺仪数据 (Ctrl+C 退出)...\n");

    while (1) {
        int ret = poll(&pfd, 1, 1000); // 1秒超时
        if (ret < 0) {
            perror("poll 失败");
            break;
        } else if (ret == 0) {
            printf("等待数据超时...\n");
            continue;
        }

        if (pfd.revents & POLLIN) {
            ssize_t n = read(fd, &ev, sizeof(ev));
            if (n == sizeof(ev)) {
                if (ev.type == EV_ABS) {
                    switch (ev.code) {
                        case ABS_RX: gyro_x = ev.value; break;
                        case ABS_RY: gyro_y = ev.value; break;
                        case ABS_RZ: gyro_z = ev.value; break;
                    }
                } else if (ev.type == EV_SYN && ev.code == SYN_REPORT) {
                    printf("Gyro X=%d  Y=%d  Z=%d\n", gyro_x, gyro_y, gyro_z);
                }
            }
        }
    }

    close(fd);
    return 0;
}
~~~

~~~c
#include <stdio.h>
#include <stdlib.h>
#include <stdint.h>
#include <errno.h>
#include <fcntl.h>
#include <unistd.h>
#include <signal.h>
#include <string.h>
#include <dirent.h>
#include <sys/ioctl.h>
#include <linux/input.h>

/* IOCTL命令定义 */
#define L3G4200D_IOCTL_BASE        77
#define L3G4200D_IOCTL_SET_ENABLE  _IOW(L3G4200D_IOCTL_BASE, 2, int)

/* 传感器状态定义 */
#define SENSOR_ON   1   // 传感器开启
#define SENSOR_OFF  0   // 传感器关闭

/* 数据转换参数 */
#define DPS_SCALE_FACTOR   16.384  // 32768/2000，±2000 dps范围的转换因子
#define RAW_TO_DPS(raw)   ((double)(raw) / DPS_SCALE_FACTOR)

/* 功能探测辅助宏 */
#ifndef BITS_PER_LONG
#define BITS_PER_LONG (sizeof(unsigned long) * 8)
#endif
#define BIT_WORD(bit)    ((bit) / BITS_PER_LONG)
#define BIT_MASK(bit)    (1UL << ((bit) % BITS_PER_LONG))
#define TEST_BIT(bit, arr)  ((arr)[BIT_WORD(bit)] & BIT_MASK(bit))

/* 全局变量 */
static volatile int g_running = 1;  // 程序运行标志
static int g_misc_fd = -1;          // 设备文件描述符
static int g_ev_fd = -1;            // 事件文件描述符

/**
 * 信号处理函数 - 处理Ctrl+C中断
 * @param sig 信号编号
 */
static void on_sigint(int sig)
{
    (void)sig;
    g_running = 0;  // 停止主循环
}

/**
 * 启用或禁用传感器
 * @param on 1=开启, 0=关闭
 * @return 0=成功, -1=失败
 */
static int enable_sensor(int on)
{
    if (g_misc_fd < 0) {
        errno = EBADF;
        return -1;
    }
    
    int flag = on ? SENSOR_ON : SENSOR_OFF;
    if (ioctl(g_misc_fd, L3G4200D_IOCTL_SET_ENABLE, &flag) < 0) {
        perror("ioctl(设置启用状态)");
        return -1;
    }
    
    return 0;
}

/**
 * 检查设备是否支持ABS_RX/ABS_RY/ABS_RZ轴
 * @param fd 设备文件描述符
 * @return 1=支持, 0=不支持
 */
static int has_abs_xyz(int fd)
{
    unsigned long evbit[(EV_MAX + BITS_PER_LONG) / BITS_PER_LONG];
    unsigned long absbit[(ABS_MAX + BITS_PER_LONG) / BITS_PER_LONG];
    
    memset(evbit, 0, sizeof(evbit));
    memset(absbit, 0, sizeof(absbit));

    // 检查是否支持绝对事件
    if (ioctl(fd, EVIOCGBIT(0, sizeof(evbit)), evbit) < 0)
        return 0;
    if (!TEST_BIT(EV_ABS, evbit))
        return 0;
    
    // 检查是否支持所需的轴
    if (ioctl(fd, EVIOCGBIT(EV_ABS, sizeof(absbit)), absbit) < 0)
        return 0;
        
    return TEST_BIT(ABS_RX, absbit) && 
           TEST_BIT(ABS_RY, absbit) && 
           TEST_BIT(ABS_RZ, absbit);
}

/**
 * 打开陀螺仪事件设备
 * @return 成功返回文件描述符，失败返回-1
 */
static int open_gyro_event(void)
{
    DIR *dir = opendir("/dev/input");
    if (!dir) {
        perror("打开目录/dev/input失败");
        return -1;
    }

    int fd = -1;
    struct dirent *de;
    char name[128];

    /* 第一步: 查找名称正好为"gyro"的设备 */
    rewinddir(dir);
    while ((de = readdir(dir))) {
        if (strncmp(de->d_name, "event", 5)) continue;
        
        int tfd = openat(dirfd(dir), de->d_name, O_RDONLY | O_CLOEXEC);
        if (tfd < 0) continue;
        
        memset(name, 0, sizeof(name));
        if (ioctl(tfd, EVIOCGNAME(sizeof(name) - 1), name) < 1) 
            name[0] = ' ';
            
        if (strcmp(name, "gyro") == 0) { 
            fd = tfd; 
            break; 
        }
        
        close(tfd);
    }
    if (fd >= 0) { 
        closedir(dir); 
        return fd; 
    }

    /* 第二步: 查找名称包含"icm4260"或"icm42607"的设备 */
    rewinddir(dir);
    while ((de = readdir(dir))) {
        if (strncmp(de->d_name, "event", 5)) continue;
        
        int tfd = openat(dirfd(dir), de->d_name, O_RDONLY | O_CLOEXEC);
        if (tfd < 0) continue;
        
        memset(name, 0, sizeof(name));
        if (ioctl(tfd, EVIOCGNAME(sizeof(name) - 1), name) < 1) 
            name[0] = ' ';
            
        if (strcasestr(name, "icm4260") || strcasestr(name, "icm42607")) { 
            fd = tfd; 
            break; 
        }
        
        close(tfd);
    }
    if (fd >= 0) { 
        closedir(dir); 
        return fd; 
    }

    /* 第三步: 基于功能的备用方案（存在ABS_RX/RY/RZ） */
    rewinddir(dir);
    while ((de = readdir(dir))) {
        if (strncmp(de->d_name, "event", 5)) continue;
        
        int tfd = openat(dirfd(dir), de->d_name, O_RDONLY | O_CLOEXEC);
        if (tfd < 0) continue;
        
        if (has_abs_xyz(tfd)) { 
            fd = tfd; 
            break; 
        }
        
        close(tfd);
    }
    
    closedir(dir);
    return fd; /* 未找到返回-1 */
}

/**
 * 读取陀螺仪校准数据
 * @param ox x轴偏移量输出
 * @param oy y轴偏移量输出
 * @param oz z轴偏移量输出
 * @return 0=成功, -1=失败
 */
static int read_calibration(int *ox, int *oy, int *oz)
{
    int fd = open("/sys/class/sensor_class/gyro_calibration", O_RDWR | O_CLOEXEC);
    if (fd < 0) {
        perror("打开gyro_calibration失败");
        return -1;
    }
    
    // 触发校准
    if (write(fd, "1", 1) != 1) {
        perror("写入校准命令失败");
        close(fd);
        return -1;
    }
    
    usleep(100 * 1000);  // 等待校准完成

    // 读取校准数据
    char buf[128] = {0};
    int r = pread(fd, buf, sizeof(buf) - 1, 0);
    close(fd);
    
    if (r <= 0) {
        fprintf(stderr, "读取校准数据失败\n");
        return -1;
    }

    // 解析校准数据
    long vals[3] = {0};
    int cnt = 0; 
    char *p = buf;
    
    while (*p && cnt < 3) {
        if (*p == '-' || (*p >= '0' && *p <= '9')) {
            char *endp = NULL; 
            long v = strtol(p, &endp, 10);
            
            if (endp != p) { 
                vals[cnt++] = v; 
                p = endp; 
                continue; 
            }
        }
        p++;
    }
    
    if (cnt != 3) {
        fprintf(stderr, "校准数据格式无效: %s\n", buf);
        return -1;
    }
    
    *ox = (int)vals[0]; 
    *oy = (int)vals[1]; 
    *oz = (int)vals[2];
    
    return 0;
}

/**
 * 清理系统资源
 */
static void cleanup_resources(void)
{
    // 禁用传感器
    if (g_misc_fd >= 0) {
        enable_sensor(SENSOR_OFF);
    }
    
    // 关闭文件描述符
    if (g_ev_fd >= 0) {
        close(g_ev_fd);
        g_ev_fd = -1;
    }
    
    if (g_misc_fd >= 0) {
        close(g_misc_fd);
        g_misc_fd = -1;
    }
}

/**
 * 主函数 - 程序入口点
 * @return 0=成功, 非0=失败
 */
int main(void)
{
    setbuf(stdout, NULL);
    signal(SIGINT, on_sigint);  // 注册信号处理函数

    /* 1) 执行校准 */
    printf("开始陀螺仪校准...\n");
    int x_off = 0, y_off = 0, z_off = 0;
    if (read_calibration(&x_off, &y_off, &z_off) < 0) {
        return EXIT_FAILURE;
    }
    printf("校准偏移量: [%d, %d, %d]\n", x_off, y_off, z_off);

    /* 2) 打开misc设备节点 */
    g_misc_fd = open("/dev/gyrosensor", O_RDWR | O_CLOEXEC);
    if (g_misc_fd < 0) {
        perror("打开/dev/gyrosensor失败");
        return EXIT_FAILURE;
    }

    /* 3) 查找输入事件设备 */
    g_ev_fd = open_gyro_event();
    if (g_ev_fd < 0) {
        fprintf(stderr, "未找到合适的陀螺仪输入事件\n");
        cleanup_resources();
        return EXIT_FAILURE;
    }

    /* 4) 启用传感器数据流 */
    if (enable_sensor(SENSOR_ON) < 0) {
        cleanup_resources();
        return EXIT_FAILURE;
    }

    printf("正在流式输出数据 (按Ctrl+C停止)\n");
    printf("%-10s %-10s %-10s\n", "X(dps)", "Y(dps)", "Z(dps)");
    printf("=============================\n");

    struct input_event ev;
    int raw_x = 0, raw_y = 0, raw_z = 0;

    /* 5) 数据采集主循环 */
    while (g_running) {
        ssize_t n = read(g_ev_fd, &ev, sizeof(ev));
        if (n != (ssize_t)sizeof(ev)) {
            if (errno == EINTR) continue;
            perror("读取事件失败");
            break;
        }

        // 处理加速度数据
        if (ev.type == EV_ABS) {
            if (ev.code == ABS_RX)      raw_x = ev.value;
            else if (ev.code == ABS_RY) raw_y = ev.value;
            else if (ev.code == ABS_RZ) raw_z = ev.value;
        }
        // 处理同步事件，输出数据
        else if (ev.type == EV_SYN && ev.code == SYN_REPORT) {
            // 计算并输出经过校准和转换的角速度值
            double gx = RAW_TO_DPS(raw_x - x_off);
            double gy = RAW_TO_DPS(raw_y - y_off);
            double gz = RAW_TO_DPS(raw_z - z_off);
            
            printf("%-10.3f %-10.3f %-10.3f\n", gx, gy, gz);
            fflush(stdout);
~~~

~~~c
#define CALIBRATION_DELAY   2   // 校准前延迟时间(秒)
#define CALIBRATION_WAIT    100 // 校准等待时间(毫秒)
#define SAMPLE_SKIP_COUNT   5   // 初始采样跳过数量
#define G_SCALE_FACTOR      16384.0  // 重力加速度转换因子

/* 全局变量 */
static volatile sig_atomic_t system_running = 1;  // 程序运行标志
static int sensor_control_fd = -1;                // 传感器控制文件描述符
static int event_data_fd = -1;                    // 事件数据文件描述符

/**

/**
 * 控制传感器的激活状态

/**

 * @param fd 文件描述符
 * @param buffer 存储读取数据的缓冲区
 * @param size 缓冲区大小
 * @return 0=成功, -1=失败
 */
static int read_calibration_data(int fd, char *buffer, size_t size)
{
    ssize_t bytes_read = pread(fd, buffer, size - 1, 0);
    if (bytes_read <= 0) return -1;
    
    // 清除换行符
    for (ssize_t i = 0; i < bytes_read; i++) {
        if (buffer[i] == '\n' || buffer[i] == '\r') {
            buffer[i] = '\0';
            break;
        }
    }
    
    return 0;
}

/**
 * 解析校准数据中的偏移量
 * @param data 校准数据字符串
 * @param x_offset x轴偏移量输出
 * @param y_offset y轴偏移量输出
 * @param z_offset z轴偏移量输出
 * @return 0=成功, -1=失败
 */
static int parse_calibration_offsets(const char *data, int *x_offset, 
                                    int *y_offset, int *z_offset)
{
    // 尝试从"offset:"标记后解析
    const char *offset_marker = strstr(data, "offset:");
    if (offset_marker) {
        offset_marker += 7;
        // 跳过非数字字符
        while (*offset_marker && !isdigit(*offset_marker) && 
               *offset_marker != '-' && *offset_marker != '[') {
            offset_marker++;
        }
        
        if (sscanf(offset_marker, "[%d, %d, %d]", x_offset, y_offset, z_offset) == 3)
            return 0;
    }
    
    // 通用格式解析
    int values[3] = {0}, count = 0;
    const char *ptr = data;
    
    while (*ptr && count < 3) {
        // 跳过非数字字符
        while (*ptr && !isdigit((unsigned char)*ptr) && *ptr != '-') ptr++;
        if (!*ptr) break;
        
        char *end_ptr;
        long value = strtol(ptr, &end_ptr, 10);
        if (end_ptr > ptr) {
            values[count++] = (int)value;
            ptr = end_ptr;
        } else {
            ptr++;
        }
    }
    
    if (count == 3) {
        *x_offset = values[0];
        *y_offset = values[1];
        *z_offset = values[2];
        return 0;
    }
    
    return -1;
}

/**
 * 查找并打开加速度计设备
 * @param exact_name 精确匹配的设备名
 * @param keyword 关键字匹配
 * @return 成功返回文件描述符，失败返回-1
 */
static int discover_accelerometer(const char *exact_name, const char *keyword)
{
    DIR *input_dir = opendir("/dev/input");
    if (!input_dir) {
        perror("无法打开输入设备目录");
        return -1;
    }
    
    int device_fd = -1;
    struct dirent *entry;
    char device_name[128];
    
    while ((entry = readdir(input_dir)) != NULL) {
        // 只处理event开头的设备
        if (strncmp(entry->d_name, "event", 5) != 0) continue;
        
        char device_path[64];
        snprintf(device_path, sizeof(device_path), "/dev/input/%s", entry->d_name);
        
        int fd = open(device_path, O_RDONLY | O_CLOEXEC);
        if (fd < 0) continue;
        
        // 获取设备名称
        memset(device_name, 0, sizeof(device_name));
        if (ioctl(fd, EVIOCGNAME(sizeof(device_name) - 1), device_name) < 0) {
            close(fd);
            continue;
        }
        
        // 去除换行符
        device_name[strcspn(device_name, "\r\n")] = '\0';
        
        // 检查名称匹配
        if ((exact_name && strcasecmp(device_name, exact_name) == 0) ||
            (keyword && strcasestr(device_name, keyword))) {
            printf("匹配到加速度计: '%s' (%s)\n", device_name, entry->d_name);
            device_fd = fd;
            break;
        }
        
        close(fd);
    }
    
    closedir(input_dir);
    
    if (device_fd < 0) {
        fprintf(stderr, "未找到匹配的加速度计设备\n");
    }
    
    return device_fd;
}

/**
 * 执行传感器校准
 * @param x_offset x轴偏移量输出
 * @param y_offset y轴偏移量输出
 * @param z_offset z轴偏移量输出
 * @return 0=成功, -1=失败

static int execute_calibration(int *x_offset, int *y_offset, int *z_offset)
{

    sleep(CALIBRATION_DELAY);

        perror("无法打开校准接口");
        return -1;
    }
    
    // 触发校准
    if (write(cal_fd, "1", 1) != 1) {
        perror("触发校准失败");
        close(cal_fd);
        return -1;
    }
    
    // 等待校准完成
    usleep(CALIBRATION_WAIT * 1000);
    
    // 读取校准数据
    char cal_data[256] = {0};
    if (read_calibration_data(cal_fd, cal_data, sizeof(cal_data)) < 0) {
        fprintf(stderr, "读取校准数据失败: %s\n", strerror(errno));
        close(cal_fd);
        return -1;
    }
    
    close(cal_fd);
    
    // 解析校准数据
    if (parse_calibration_offsets(cal_data, x_offset, y_offset, z_offset) < 0) {
        fprintf(stderr, "解析校准数据失败\n");
        return -1;
    }
    
    printf("校准完成: X=%d Y=%d Z=%d\n", *x_offset, *y_offset, *z_offset);

/**

 * @param x_offset x轴校准偏移量

 * @param z_offset z轴校准偏移量
 * @return 0=成功, -1=失败
 */
static int run_data_acquisition(int x_offset, int y_offset, int z_offset)
{
    printf("开始采集数据 (Ctrl+C退出)\n");
    printf("%-12s %-12s %-12s\n", "X(g)", "Y(g)", "Z(g)");
    printf("================================\n");
    
    struct input_event ev;
    int ax = 0, ay = 0, az = 0, sample_count = 0;
    
    while (system_running) {
        ssize_t bytes_read = read(event_data_fd, &ev, sizeof(ev));
        if (bytes_read != sizeof(ev)) {
            if (errno == EINTR) continue;
            perror("读取事件失败");
            break;
        }
        
        // 处理加速度数据
        if (ev.type == EV_ABS) {
            if (ev.code == ABS_X) ax = ev.value;
            else if (ev.code == ABS_Y) ay = ev.value;
            else if (ev.code == ABS_Z) az = ev.value;
        }
        // 处理同步事件，输出数据
        else if (ev.type == EV_SYN && ev.code == SYN_REPORT) {
            sample_count++;
            if (sample_count <= SAMPLE_SKIP_COUNT) continue;
            
            // 计算并转换为重力加速度值
            double gx = (ax - x_offset) / G_SCALE_FACTOR;
            double gy = (ay - y_offset) / G_SCALE_FACTOR;
            double gz = (az - z_offset) / G_SCALE_FACTOR;
            
            printf("%-12.4f %-12.4f %-12.4f\n", gx, gy, gz);
            fflush(stdout);
            
            usleep(50 * 1000);  // 50ms采样间隔
        }
    }
    
    return 0;
}
~~~

### 陀螺仪和加速度计的英文名称

- 陀螺仪：Gyroscope（常简称为 Gyro）
- 加速度计：Accelerometer

### 陀螺仪的主要应用领域

陀螺仪是一种用于测量或维持方向（角运动）的装置，基于角动量守恒原理工作，主要应用在需要感知旋转、倾斜或方向变化的场景，例如：

- **导航与定位**：如无人机、自动驾驶汽车、船舶、航天器等，用于在 GPS 信号弱或丢失时维持导航精度。
- **消费电子**：智能手机（屏幕旋转、游戏体感控制）、智能手表（运动姿态识别）、VR/AR 设备（头部追踪，提升沉浸感）。
- **工业设备**：机器人（运动姿态控制）、机床（振动监测与稳定控制）、工程机械（倾斜角度监测，防止倾覆）。
- **航空航天**：飞机自动驾驶系统、卫星姿态控制，确保设备按预定轨迹运行。
- **军事领域**：导弹制导、坦克火控系统，提高武器命中精度。

### 加速度计的主要应用领域

加速度计用于测量物体的线性加速度（包括重力加速度），可感知运动速度变化、倾斜角度等，应用场景包括：

- **消费电子**：智能手机（计步、跌落检测自动锁屏）、平板电脑（横竖屏切换）、游戏手柄（动作感应）。
- **运动与健康**：智能手环 / 手表（步数统计、运动模式识别）、运动相机（防抖功能）。
- **汽车工业**：安全气囊触发（检测碰撞加速度）、电子稳定程序（ESP，监测车身倾斜与滑动）、胎压监测系统。
- **工业与机器人**：机械臂运动控制、振动监测（预测设备故障）、电梯限速器（检测异常加速度）。
- **航空航天**：航天器发射与着陆时的加速度监测、飞机过载检测（保障飞行员安全）。

两者常结合使用（如构成 IMU 惯性测量单元），在需要同时感知线性运动和旋转运动的场景中发挥协同作用，例如无人机的稳定控制、VR 设备的全姿态追踪等。

## 瑞芯微uboot启动流程

瑞芯微 U-Boot 的启动流程是一个多阶段、多模块协同工作的复杂过程，涉及硬件初始化、镜像加载、设备树解析和内核启动等关键环节。以下结合瑞芯微平台特性和源码实现，详细解析其启动流程：

### 一、启动前的硬件准备

芯片上电后，片内**BootROM**（只读代码）首先执行，负责完成以下关键操作：

1. **启动模式检测**：通过硬件引脚（如 GPIO）或寄存器配置，确定从哪个存储设备（如 SPI Flash、eMMC、SD 卡）启动。
2. **基础硬件初始化**：初始化最基本的时钟、串口和存储控制器，确保能从外部存储读取数据。
3. **加载第一阶段引导程序**：根据启动模式，从存储设备的固定偏移地址（如 SPI Flash 的 0x0）加载**Miniloader**或**SPL**（Secondary Program Loader）到 DDR 中。

### 二、第一阶段引导程序（Miniloader/SPL）

#### 1. **Miniloader（闭源方案）**

- **代码位置**：通常位于`board/rockchip/common/rkloader/`目录下，由瑞芯微提供闭源二进制文件（如`MiniLoaderAll.bin`）。
- 核心功能：
  - **DDR 初始化**：通过寄存器配置 DDR 控制器，确保内存可用。
  - **分区表解析**：读取`parameter.txt`文件，解析出内核、设备树、根文件系统等镜像的存储位置。
  - **镜像加载**：根据分区信息，从存储设备加载`trust.img`（可信执行环境）和`uboot.img`到 DDR。
  - **安全启动校验**：验证`trust.img`和`uboot.img`的数字签名，确保镜像未被篡改。

#### 2. **SPL（开源方案）**

代码位置**：位于`arch/arm/cpu/armv8/rockchip/`目录下，使用开源代码实现。

核心功能：

- **CPU 初始化**：配置 CPU 核心（如大核 / 小核）、关闭中断、设置栈指针。
- **存储设备驱动**：初始化 SPI NAND、eMMC、SD 卡等存储控制器。
- **镜像加载**：直接从存储设备加载`uboot.img`到 DDR，并跳转到 U-Boot 主程序入口。

### 三、U-Boot 主程序启动

#### 1. **入口函数与异常向量表**

**入口代码**：位于`arch/arm/cpu/armv8/rockchip/start.S`，初始化异常向量表，处理复位、中断等异常。

关键操作：

- **关闭 MMU 和 Cache**：确保初始化阶段直接访问物理地址。
- **设置栈指针**：为后续 C 代码执行准备栈空间。
- **跳转到 C 语言入口**：调用`_main`函数（位于`arch/arm/lib/crt0_64.S`），进入第二阶段初始化。

#### 2. **第二阶段初始化（C 语言实现）**

- **核心函数**：`board_init_f`和`board_init_r`（位于`common/board_f.c`和`common/board_r.c`）。

- 初始化流程：

  基本硬件初始化：

  - 串口初始化（用于调试输出）。
  - 时钟初始化（设置 CPU 和外设的工作频率）。
  - GPIO 初始化（配置按键、LED 等外设）。

  环境变量加载**：从存储设备（如 eMMC 的`env`分区）读取 U-Boot 环境变量，包括启动命令（`bootcmd`）、内核参数（`bootargs`）等。

  设备树加载：

  从`resource.img`或分区中查找`rk-kernel.dtb`文件。

  若未找到，根据硬件引脚状态或配置文件选择默认设备树。

  **内核镜像加载**：根据`bootcmd`命令，从存储设备（如 eMMC 的`kernel`分区）加载内核镜像到内存指定地址。

#### 3. **设备树解析与内核参数传递**

设备树解析：

使用`libfdt`库解析设备树文件，提取硬件信息（如内存布局、外设寄存器地址）。

将解析后的设备树地址通过`r2`寄存器传递给内核。

内核参数传递：

通过设备树的chosen

节点设置bootargs参数，例如：

```dts
chosen {
    bootargs = "console=ttyFIQ0,1500000 root=/dev/mmcblk0p2 rw";
};
```

U-Boot 在启动内核时，将`bootargs`参数写入环境变量，并通过`bootm`命令传递给内核。

### 四、内核启动与控制权移交

#### 1. **启动命令执行**

- **命令触发**：U-Boot 启动后，默认等待用户输入命令。若超时未输入，执行`bootcmd`环境变量中的命令（如`bootm 0x80200000 - 0x83000000`）。
- **参数解析**：`bootm`命令解析出内核镜像地址（`0x80200000`）、initrd 地址（`-`表示无）和设备树地址（`0x83000000`）。

#### 2. **内核启动流程**

- **内核入口**：U-Boot 通过`armv8_cpu_boot`函数（位于`arch/arm/lib/bootm.c`）跳转到内核镜像入口地址。
- 参数传递：
  - **寄存器传递**：将设备树地址存入`r2`寄存器，内核参数通过`r0`和`r1`寄存器传递。
  - **环境变量传递**：内核启动后，从`/proc/cmdline`读取`bootargs`参数，用于初始化根文件系统、串口等。

#### 3. **控制权移交**

- **内核初始化**：内核开始执行`start_kernel`函数，初始化内存管理、进程调度、设备驱动等子系统。
- **U-Boot 退出**：内核启动成功后，U-Boot 进入休眠状态，或根据配置重启系统。

### 五、瑞芯微平台特性与优化

#### 1. **多启动模式支持**

- **启动模式检测**：U-Boot 在`board_late_init`函数中检测硬件状态（如按键、USB 连接），动态选择启动路径。
- 支持的模式：
  - **Normal 模式**：正常启动内核。
  - **Loader 模式**：进入 USB 刷机模式，支持通过`rkdeveloptool`工具烧写镜像。
  - **Maskrom 模式**：用于救砖，通过短接特定引脚强制进入 BootROM 模式。

#### 2. **设备树动态选择**

- **硬件引脚检测**：根据 GPIO 电平或 ADC 输入值，选择不同的设备树文件（如`rk3399-firefly.dtb`或`rk3399-nanopi.dtb`）。
- **镜像内动态加载**：从`resource.img`中提取多个设备树，根据硬件 ID 自动匹配最适合的设备树。

#### 3. **安全启动机制**

- **镜像签名校验**：`trust.img`和`uboot.img`需通过瑞芯微提供的私钥签名，U-Boot 在启动时使用公钥验证签名。
- **加密存储**：敏感数据（如序列号、MAC 地址）存储在`Vendor Storage`区域，通过硬件加密保护。

### 六、源码结构与关键文件

#### 1. **核心目录**

- **架构相关代码**：`arch/arm/`和`arch/arm64/`目录下，包含 CPU、中断、时钟等底层驱动。
- **平台相关代码**：`board/rockchip/`目录下，包含瑞芯微平台特有的初始化代码和设备树文件。
- **通用工具代码**：`common/`目录下，包含命令行解析、环境变量管理、设备树操作等通用函数。

#### 2. **关键文件**

- **启动入口**：`arch/arm/cpu/armv8/rockchip/start.S`。
- **设备树解析**：`lib/fdt/fdt_support.c`和`lib/fdt/fdt.c`。
- **内核启动**：`arch/arm/lib/bootm.c`和`common/bootm_common.c`。

### 七、调试与优化建议

1. **串口调试**：通过串口输出 U-Boot 日志（如`printenv`、`bootm`命令执行过程），定位启动失败原因。
2. **代码跟踪**：使用`gdb`调试工具，在`start.S`和`_main`函数中设置断点，跟踪初始化流程。
3. 性能优化：
   - **启动时间优化**：减少不必要的初始化步骤，或通过`bootdelay`环境变量缩短等待用户输入的时间。
   - **内存优化**：合理配置 DDR 参数，减少初始化时间；使用`CONFIG_SPL_LIBCOMMON_SUPPORT`等配置选项精简 SPL 代码。

### 总结

瑞芯微 U-Boot 的启动流程是一个从硬件初始化到内核启动的完整链条，涉及闭源与开源方案的混合使用、多阶段镜像加载、设备树动态解析等复杂机制。通过深入理解各阶段的核心功能和源码实现，开发者可以灵活定制 U-Boot 以满足不同应用场景的需求，同时利用瑞芯微提供的安全启动、多模式支持等特性，构建稳定可靠的嵌入式系统。

## 瑞芯微uboot的功能

瑞芯微 U-Boot（Universal Boot Loader）是适配瑞芯微（Rockchip）芯片平台的嵌入式引导程序，是连接硬件初始化与操作系统启动的核心桥梁。它在系统上电后接管硬件控制权，完成底层初始化、镜像加载、环境管理等关键操作，最终将控制权移交操作系统内核。其功能覆盖硬件适配、启动管理、安全防护、开发调试等多个维度，以下从**核心作用**和**具体功能**两方面详细解析：

### 一、瑞芯微 U-Boot 的核心作用

瑞芯微 U-Boot 的本质是 **“硬件初始化器”**和**“操作系统启动器”**，在整个系统生命周期中承担三大核心角色：

1. **硬件到软件的 “翻译官”**：将底层硬件（CPU、内存、外设等）的物理特性抽象为操作系统可识别的接口（如设备树、内存布局），让内核无需关心具体硬件细节即可运行。
2. **启动流程的 “指挥官”**：根据硬件状态（如启动引脚、按键）和配置（如环境变量），动态选择启动路径（如正常启动、刷机、救砖），并按顺序加载所需镜像（信任区、内核、设备树等）。
3. **系统安全的 “守门人”**：通过签名校验、加密存储等机制，确保加载的镜像未被篡改，保护设备免受恶意固件攻击。

### 二、瑞芯微 U-Boot 承载的具体功能（按流程与场景划分）

#### （一）硬件初始化：为系统运行 “搭建基础环境”

硬件初始化是 U-Boot 最核心的功能之一，目标是将上电后处于 “原始状态” 的硬件配置为 “可运行程序” 的状态。瑞芯微 U-Boot 针对其芯片（如 RK3399、RK3588、RK3288 等）的硬件架构，完成以下初始化：

1. **CPU 与核心架构初始化**
   - **复位与模式设置**：从芯片复位向量（Reset Vector）开始执行，关闭 CPU 中断、MMU（内存管理单元）和 Cache，确保初始状态干净（代码位于`arch/arm/cpu/armv8/rockchip/start.S`）。
   - **核心配置**：根据芯片型号（如大核 Cortex-A72、小核 Cortex-A53）设置 CPU 工作模式（如 SVC 模式）、栈指针（SP），为后续 C 语言代码执行分配栈空间。
   - **多核管理**：对于多核芯片（如 RK3399 的 6 核），仅启动主核（Primary Core），其他核（Secondary Core）进入休眠，待内核启动后由内核唤醒。
2. **内存（DDR）初始化**
   - 瑞芯微芯片依赖外部 DDR（如 LPDDR4、DDR4）作为运行内存，U-Boot 必须先初始化 DDR 控制器才能加载后续程序。
   - **流程**：通过寄存器配置 DDR 的时钟（PLL）、时序参数（行地址、列地址、刷新周期）、电压等，最终执行 DDR 自检测试（如读写验证），确保内存可用。
   - **瑞芯微特性**：针对不同 DDR 型号（如三星、镁光），U-Boot 通过`ddrbin`（闭源 DDR 初始化二进制）或开源代码（`arch/arm/cpu/armv8/rockchip/ddr/`）适配，支持自动识别 DDR 容量和规格。
3. **时钟与电源管理初始化**
   - **时钟树配置**：初始化芯片内部 PLL（锁相环），为 CPU、DDR、外设（如 UART、eMMC）分配合适的时钟频率（如 RK3588 的 CPU 大核可超频至 2.4GHz）。
   - **PMIC（电源管理芯片）配置**：通过 I2C 接口与外部 PMIC（如 RK809、RK817）通信，设置各硬件模块的供电电压（如 CPU 核心电压 1.0V、DDR 电压 1.2V），确保硬件稳定工作。
4. **外设控制器初始化**
   - **串口（UART）**：优先初始化调试串口（如 RK3399 的 UART2），用于输出启动日志（打印 “U-Boot” 标志、初始化步骤等），是开发调试的核心工具。
   - **存储控制器**：初始化 eMMC、SD 卡、SPI Flash、NAND Flash 等存储设备的控制器，确保 U-Boot 能读取存储中的镜像文件（代码位于`drivers/mmc/`、`drivers/spi/`）。
   - **GPIO 与基础外设**：配置启动引脚（用于检测启动模式）、LED（指示启动状态）、按键（如复位键、进入刷机模式的按键）等。

#### （二）启动管理：“指挥” 系统从存储到内存的加载流程

瑞芯微 U-Boot 需根据硬件状态和配置，确定启动路径并加载必要的镜像文件（如信任区、内核、设备树），具体包括：

1. **启动模式检测与选择**

   - 瑞芯微芯片支持多种启动模式，U-Boot 通过检测硬件引脚（GPIO）或寄存器状态，决定从哪个设备启动：
     - **正常启动**：从 eMMC/SD 卡的内核分区加载镜像，适用于设备正常运行。
     - **Loader 模式**：检测到特定按键（如 “Recovery 键”）或 USB 连接时，进入刷机模式，支持通过`rkdeveloptool`工具烧写固件。
     - **Maskrom 模式**：当外部存储无有效镜像时，强制进入 BootROM 模式，用于救砖（需通过专用工具烧写 Miniloader）。

2. **分区表与镜像解析**

   瑞芯微平台通常通过`parameter.txt`文件定义存储分区（如`boot`、`system`、`kernel`、`dtb`），U-Boot 在启动时读取该文件，解析各镜像的存储位置（偏移地址、大小）。例如，parameter.txt可能包含：

   ```txt
   FEX_VERSION:1.1
   MACHINE:RK3399
   PARTITION: bootloader  : 0x00000000 0x00080000;  // 存放Miniloader
   PARTITION: trust      : 0x00080000 0x00100000;  // 存放trust.img
   PARTITION: uboot      : 0x00180000 0x00100000;  // 存放uboot.img
   PARTITION: kernel     : 0x00280000 0x01000000;  // 存放内核镜像
   ```

3. **多阶段镜像加载**

   瑞芯微启动流程依赖多个镜像，U-Boot 按顺序加载并验证：

   1. **信任区镜像（trust.img）**：包含 OP-TEE（可信执行环境），负责安全相关操作（如加密、签名校验），U-Boot 先加载其到指定内存地址并启动。
   2. **设备树（dtb）**：从`dtb`分区或`resource.img`中读取设备树文件（如`rk3399-firefly.dtb`），解析后传递给内核（内核依赖设备树识别硬件）。
   3. **内核镜像（zImage/Image）**：从`kernel`分区加载内核到内存（如 RK3399 的内核加载地址`0x80800000`），并准备内核启动参数。

4. **启动命令执行**

   U-Boot 通过环境变量bootcmd定义默认启动命令，例如：

   ```bash
   bootcmd=ext4load mmc 0:6 0x80800000 /boot/Image;  // 从eMMC分区6加载内核
           ext4load mmc 0:6 0x83000000 /boot/rk3399.dtb;  // 加载设备树
           booti 0x80800000 - 0x83000000;  // 启动内核
   ```

   若用户在启动倒计时（`bootdelay`，默认 3 秒）内按下按键，可进入 U-Boot 命令行，手动修改`bootcmd`或执行其他命令（如`tftp`从网络加载镜像）。

#### （三）环境变量管理：“保存” 系统配置的动态参数

环境变量是 U-Boot 的 “配置中心”，用于存储启动参数、硬件信息等动态配置，支持掉电保存（通常存储在 eMMC 的`env`分区或 SPI Flash 中）。瑞芯微 U-Boot 的环境变量功能包括：

1. **核心环境变量**

   - `bootcmd`：默认启动命令（如加载内核的脚本）。
   - `bootargs`：传递给内核的参数（如`console=ttyFIQ0,1500000 root=/dev/mmcblk0p2 rw`，指定控制台和根文件系统）。
   - `bootdelay`：启动倒计时时间（秒），超时后自动执行`bootcmd`。
   - `ethaddr`/`wlanaddr`：以太网 / MAC 地址（硬件唯一标识）。
   - `fdtfile`：指定设备树文件名（如`rk3399-nanopi4.dtb`）。

2. **环境变量操作**

   U-Boot 命令行支持printenv（查看）、setenv（修改）、saveenv（保存到存储）等命令，例如：

   ```bash
   => setenv bootdelay 5  // 将启动倒计时改为5秒
   => saveenv  // 保存配置（写入eMMC的env分区）
   ```

   瑞芯微 U-Boot 会在初始化时从存储加载环境变量，若未找到则使用默认配置（定义在`include/configs/rockchip-common.h`）。

#### （四）设备树处理：“传递” 硬件信息给内核

设备树（Device Tree Blob, DTB）是描述硬件的结构化数据（替代传统的硬编码），瑞芯微 U-Boot 负责设备树的加载、修改和传递，确保内核能正确识别硬件：

**设备树加载**

从存储中读取设备树文件（如`rk3399.dtb`）到内存（如`0x83000000`），支持从多个候选设备树中自动选择（根据硬件型号或引脚状态）。

**设备树动态修改**

U-Boot 可在启动前修改设备树，适配硬件动态状态，例如：

- 将`bootargs`参数写入设备树的`chosen`节点（内核启动时从该节点读取参数）。
- 根据检测到的硬件（如是否插了 HDMI 线），动态禁用 / 启用设备树中的对应节点（如`hdmi`节点）。
- 调整内存节点（`memory`）的大小（根据实际检测到的 DDR 容量）。

**传递给内核**

- 启动内核时，U-Boot 通过寄存器（如 ARM64 的`x2`寄存器）将设备树在内存中的地址传递给内核，内核解析后即可获取完整的硬件信息。

#### （五）安全机制：“防护” 系统免受恶意攻击

瑞芯微 U-Boot 集成了安全启动（Secure Boot）机制，防止未授权的固件被加载，保护设备安全：

1. **镜像签名校验**
   - 瑞芯微芯片支持 “链式校验”：BootROM 校验 Miniloader，Miniloader 校验 trust.img，trust.img 校验 U-Boot 主程序，U-Boot 校验内核。
   - 校验通过私钥签名的镜像（如`uboot.img.signed`），若签名无效则拒绝启动（防止篡改或替换固件）。
2. **加密存储与访问控制**
   - 敏感数据（如设备序列号、MAC 地址、加密密钥）存储在芯片的 “Vendor Storage” 区域（硬件加密分区），U-Boot 通过专用接口（`rockchip_vendor_storage`驱动）读取，防止直接访问。
   - 限制对关键寄存器（如安全配置寄存器）的修改，防止攻击者绕过安全机制。

#### （六）开发与调试支持：“辅助” 开发者调试与维护

瑞芯微 U-Boot 提供丰富的工具和命令，方便开发者调试硬件、更新固件：

1. **命令行交互**

   提供数十种命令，覆盖存储操作（mmc、spi）、网络（tftp、ping）、内存操作（md、mw）、镜像启动（bootm、booti）等，例如：

   `mmc list`：列出所有 eMMC/SD 设备。

   `tftp 0x80800000 kernel.img`：通过 TFTP 从网络下载内核到内存。

   `md 0x80800000 100`：查看内存`0x80800000`开始的 256 字节数据。

2. **刷机与固件更新**

   - 支持通过 USB（`rkdeveloptool`）、TFTP、SD 卡等方式更新固件，例如进入 Loader 模式后，可通过工具烧写`bootloader`、`system`等分区。

3. **日志输出与调试**

   - 通过串口输出详细的初始化日志（如 “DDR initialized: 4GB”“MMC: rockchip_mmc: 0, rockchip_mmc: 1”），帮助定位硬件初始化失败原因（如 DDR 不识别、MMC 控制器错误）。
   - 支持`debug`配置（`CONFIG_DEBUG`），输出更详细的函数调用和寄存器操作日志。

### 三、瑞芯微 U-Boot 与通用 U-Boot 的差异

瑞芯微 U-Boot 基于通用 U-Boot 框架开发，但针对其芯片特性做了深度定制：

1. **硬件适配**：针对瑞芯微独有的 CPU 架构（如 RK3588 的四核 A76 + 四核 A55）、外设控制器（如 RGA2 图形加速器）开发专用驱动。
2. **闭源组件集成**：部分核心功能（如 DDR 初始化、安全启动校验）依赖瑞芯微提供的闭源二进制（如`ddrbin`、`miniloader`），开源代码仅负责调用。
3. **工具链适配**：支持瑞芯微定制的交叉编译工具链（如`aarch64-rockchip-linux-gnu-`），确保代码兼容性。

### 总结

瑞芯微 U-Boot 是连接瑞芯微硬件与操作系统的 “桥梁”，其核心功能可概括为：**“初始化硬件、管理启动、传递信息、保障安全、辅助开发”**。它不仅需要完成从 CPU 到外设的全链路硬件初始化，还要根据配置动态选择启动路径，最终可靠地将控制权移交内核。对于开发者而言，理解瑞芯微 U-Boot 的功能有助于定制启动流程、调试硬件问题、优化启动时间；对于设备而言，U-Boot 是确保系统稳定、安全启动的核心组件。

> ### **1. 启动入口与硬件初始化**
>
> - 链接脚本指定入口：u-boot.lds（arch/arm/cpu/u-boot.lds）定义入口为_start

>           - `cpu_init_crit`（同文件）：关键 CPU 初始化

>
> ### **2. 早期初始化（重定位前）**
>
> - `_main`（arch/arm/lib/crt0.S）：进入 C 语言环境入口
>   - **`board_init_f_alloc_reserve`**（common/init/board_init.c）：为全局数据结构（`gd`）分配内存空间
>   - **`board_init_f_init_reserve`**（同文件）：清零`gd`结构体，初始化基础字段
>   - `board_init_f`（common/board_f.c）：重定位前的板级初始化
>     - `initcall_run_list`（include/initcall.h）：执行早期初始化序列
>       - `init_sequence_f[]`（common/board_f.c）：早期初始化函数数组，包含：

>         - `init_baud_rate`（common/board_f.c）：从环境变量获取并设置波特率
>         - `serial_init`（drivers/serial/serial.c）：初始化串口控制器
>         - `console_init_f`（common/console.c）：初始化控制台（早期输出）

>

>
> ### **4. 后期初始化（重定位后）**
>
> - `board_init_r`（common/board_r.c）：重定位后的板级初始化（依赖重定位完成）
>   - `initcall_run_list`（include/initcall.h）：执行后期初始化序列
>     - `init_sequence_r[]`（common/board_r.c）：后期初始化函数数组，包含：
>       - `initr_reloc`（common/board_r.c）：标记重定位完成（设置`gd->flags`）
>       - serial_initialize（drivers/serial/serial-uclass.c）：初始化串口设备（驱动模型）
>         - `serial_init`（同文件）：完成串口最终配置
>       - initr_mmc（common/board_r.c）：初始化 eMMC/SD 卡
>         - mmc_initialize（drivers/mmc/mmc.c）：扫描并初始化 MMC 设备
>           - `mmc_do_preinit` -> `mmc_start_init`（同文件）：执行 MMC 初始化流程
>       - `console_init_r`（common/console.c）：初始化完整控制台（支持输入输出）
>       - `interrupt_init`（arch/arm/lib/interrupts.c）：初始化中断控制器
>       - initr_net（common/board_r.c）：初始化网络设备
>         - eth_initialize（net/eth-uclass.c）：扫描并初始化以太网设备
>           - eth_common_init（net/eth_common.c）：通用以太网初始化
>             - `phy_init`（drivers/net/phy/phy.c）：初始化 PHY 芯片
>           - uclass_first_device_check（drivers/core/uclass.c）：查找并探测以太网设备
>             - device_probe（drivers/core/device.c）：设备探测（驱动匹配与初始化）
>               - `device_of_to_plat`：解析设备树信息到平台数据（如`fecmxc_of_to_plat`）
>               - drv->probe（驱动 probe 函数，如fecmxc_probe）：初始化以太网控制器
>                 - `fec_get_miibus`：初始化 MDIO 总线（管理 PHY）
>                 - fec_phy_init：初始化 PHY 芯片
>                   - `phy_connect`：连接 PHY 与 MAC 控制器
>                   - `phy_config`：配置 PHY（如自协商，调用`genphy_config_aneg`）
>               - `uclass_post_probe_device`：执行设备探测后操作（如`eth_post_probe`设置 MAC 地址）
>
> ### **5. 进入命令交互**
>
> - `run_main_loop`（common/board_r.c）：进入 U-Boot 主循环
>   - `main_loop`（common/main.c）：主循环核心逻辑
>     - **`bootdelay_process`**（common/autoboot.c）：读取环境变量`bootdelay`（启动延时）和`bootcmd`（自动启动命令）
>     - `autoboot_command`（同文件）：自动启动处理
>       - `abortboot`：倒计时提示（如 "Hit any key to stop autoboot: 3"），超时执行`bootcmd`
>     - **`cli_loop`**（common/cli.c）：若用户按键中断自动启动，进入命令行交互模式
>
> 以上流程清晰展现了 U-Boot 从硬件复位到命令交互的完整启动过程，涵盖了初始化、重定位、设备驱动加载等关键环节。

## 流程分析

### 入口确定

首先根据uboot顶层Makefile中的这段代码就决定了uboot的入口，即u-boot-init对应的head-y变量

~~~Makefile
u-boot-init := $(head-y)
u-boot-main := $(libs-y)

u-boot:	$(u-boot-init) $(u-boot-main) u-boot.lds FORCE
    +$(call if_changed,u-boot__)
~~~

head-y定义在具体架构目录的Makefile中，架构由config.mk所得，也就是CONFIG_SYS_ARCH

~~~c
ARCH := $(CONFIG_SYS_ARCH:"%"=%)
~~~

CONFIG_SYS_ARCH具体定义在config文件里面，具体内容如下所示：

~~~c
CONFIG_SYS_ARCH="arm"
~~~

RK3588是arm架构，head-y 就定义在arch/arm/Makefile中，具体内容如下所示：

~~~makefile
head-y := arch/arm/cpu/$(CPU)/start.o
~~~

这里又有了一个名为CPU的变量，他是根目录的config.mk中由CONFIG_SYS_CPU配置的，具体内容如下所示：

~~~C
CPU := $(CONFIG_SYS_CPU:"%"=%) 
~~~

CONFIG_SYS_CPU也定义在config文件里面，具体内容如下所示：

~~~c
CONFIG_SYS_CPU="armv8"
~~~

所以head-y的最终值为arch/arm/cpu/armv8/start.o，从而确定入口为arch/arm/cpu/armv8/start.S。

### crt0.S汇编分析

==这里有些疑问，不知道这个_main是如何定位到arch/arm/lib/crt0.S的，这个具体定位先不管，先来继续分析了==

crt0.S的运行逻辑是 U-Boot 启动后从汇编世界切换到 C 世界的关键桥梁，具体运行流程如下

**1. 设置初始栈指针，根据不同的构建类型（TPL/SPL/U-Boot），选择合适的栈地址，并保证 8 字节对齐：**

~~~asm
#if defined(CONFIG_TPL_BUILD) && defined(CONFIG_TPL_NEEDS_SEPARATE_STACK)
	ldr	r0, =(CONFIG_TPL_STACK)
#elif defined(CONFIG_SPL_BUILD) && defined(CONFIG_SPL_STACK)
	ldr	r0, =(CONFIG_SPL_STACK)
#else
	ldr	r0, =(CONFIG_SYS_INIT_SP_ADDR)
#endif
	bic	r0, r0, #7	/* 8-byte alignment for ABI compliance */
	mov	sp, r0
~~~

**2.初始化全局数据（gd）调用一系列 board_init_f 相关的函数，分配和初始化 gd（global data）结构，并设置启动标志，此时 gd 结构体已准备好，硬件初始化工作开始。**

~~~asm
bl	board_init_f_alloc_reserve
mov	sp, r0
mov	r9, r0
bl	board_init_f_init_reserve
bl	board_init_f_boot_flags
bl	board_init_f
~~~

**3.代码重定位（U-Boot主程序），如果不是 SPL，设置新的栈和 gd，然后进行代码重定位，重定位后，代码和数据都在新的内存地址。**

~~~asm
	ldr	r0, [r9, #GD_START_ADDR_SP]	/* sp = gd->start_addr_sp */
	bic	r0, r0, #7	/* 8-byte alignment for ABI compliance */
	mov	sp, r0
	ldr	r9, [r9, #GD_BD]		/* r9 = gd->bd */
	sub	r9, r9, #GD_SIZE		/* new GD is below bd */

#ifndef CONFIG_SKIP_RELOCATE_UBOOT
	adr	lr, here
	ldr	r0, [r9, #GD_RELOC_OFF]		/* r0 = gd->reloc_off */
	add	lr, lr, r0
#if defined(CONFIG_CPU_V7M)
	orr	lr, #1				/* As required by Thumb-only */
#endif
	ldr	r0, [r9, #GD_RELOCADDR]		/* r0 = gd->relocaddr */
	b	relocate_code
#endif
here:
~~~

**4.重定位异常向量表，保证异常向量表在新地址可用。**

~~~asm
bl relocate_vectors
~~~

**5.设置最终运行环境**

调用 `c_runtime_cpu_setup`，做 CPU 相关的最终设置。

**6.清零 BSS 段，将 BSS 段（未初始化的全局变量区）全部清零：**

~~~asm
ldr r0, =__bss_start
ldr r1, =__bss_end
mov r2, #0x00000000
clbss_l:cmp r0, r1
strlo  r2, [r0]
addlo  r0, r0, #4
blo clbss_l
~~~

**7.板级初始化和主流程跳转，调用板级初始化和主入口，进入 C 语言主流程，后续就是 U-Boot 的正常运行。：**

~~~asm
mov   r0, r9         /* gd_t */
ldr r1, [r9, #GD_RELOCADDR] /* dest_addr */
ldr pc, =board_init_r
~~~

> 设置初始栈和 gd
>
> 调用 board_init_f 做早期硬件初始化
>
> 代码重定位（U-Boot主程序）
>
> 重定位异常向量表
>
> 清零 BSS
>
> 跳转到 board_init_r，进入 C 世界
>
> 这是 U-Boot 启动从汇编到 C 的完整桥梁流程。

然后对start.S和crt0.S做一个总结

`start.S` 负责U-Boot的最底层启动流程，从CPU复位后开始执行，完成异常向量表设置、缓存和多核相关配置、硬件初始化等底层准备工作，最终将主核引导到C语言世界的入口 `_main`。它确保所有硬件和运行环境达到安全状态，为后续的C环境初始化和U-Boot主流程打下基础。

crt0.S 则承接了汇编入口的工作，负责建立C运行环境，包括设置栈指针、初始化全局数据结构gd、清零BSS段、代码重定位等，并最终跳转到 `board_init_r`，进入U-Boot的主功能流程。它是U-Boot从底层汇编到高级C代码的桥梁，确保系统能够顺利进入主程序并开始正常运行。

### board_f.c分析

在初始化了堆和栈之后，具体会执行下面的这四个函数，下面一点点的分析

~~~asm
bl	board_init_f_alloc_reserve  ; 跳转到该函数，为全局数据结构(gd)分配内存空间
bl	board_init_f_init_reserve  ; 跳转到该函数，初始化已分配的gd内存（如清零操作）
bl	board_init_f_boot_flags    ; 跳转到该函数，设置启动相关的标志位（如初始化状态标记）
bl	board_init_f               ; 跳转到该函数，执行重定位前的早期板级初始化（如串口、定时器等基础初始化）
~~~

board_init_f_alloc_reserve定义在common/init/board_init.c中，具体内容如下所示：

~~~c
/* 为早期初始化分配内存（从内存顶部向下分配） */
ulong board_init_f_alloc_reserve(ulong top)
{
	/* 预留早期malloc所需空间 */
#if CONFIG_VAL(SYS_MALLOC_F_LEN)
	top -= CONFIG_VAL(SYS_MALLOC_F_LEN);  // 减去早期内存池大小
#endif

	/* 预留全局数据结构(GD)空间，16字节对齐 */
	top = rounddown(top - sizeof(struct global_data), 16);

	return top;  // 返回分配后剩余内存的顶部地址
}
~~~

这个函数就是用来分配内存的，然后来看第二个函数board_init_f_init_reserve，他是用来初始化已经分配的gd内存的，具体代码如下所示：

~~~c
/* 初始化预留的内存区域（全局数据和早期malloc池） */
void board_init_f_init_reserve(ulong base)
{
	struct global_data *gd_ptr;  // 全局数据结构指针

	/* 初始化全局数据结构(GD) */
	gd_ptr = (struct global_data *)base;  // 指向GD指针指向预留的内存起始地址
	memset(gd_ptr, '\0', sizeof(*gd));    // 清空GD区域（初始化）
#if !defined(CONFIG_ARM)
	arch_setup_gd(gd_ptr);  // 非ARM架构：设置GD（ARM在汇编中已处理）
#endif

	// 计算早期malloc池的起始地址（GD大小向上取整16字节对齐）
	base += roundup(sizeof(struct(struct global_data), 16);

	/* 初始化早期malloc内存池 */
#if CONFIG_VAL(SYS_MALLOC_F_LEN)
	gd->malloc_base = base;  // 记录malloc池起始地址
	base += CONFIG_VAL(SYS_MALLOC_F_LEN);  // 移动到malloc池结束位置
#endif
}
~~~

第三个函数为board_init_f_boot_flags，看名称他的作用是初始化标志位，具体内容如下所示：

~~~c
/*
 * 初始化启动相关标志（弱定义函数，可被板级代码重写）
 * 返回值：0表示成功
 */
__weak int board_init_f_boot_flags(void)
{
	gd->baudrate = CONFIG_BAUDRATE;               // 设置全局数据中的波特率（来自配置）
	gd->serial.baudrate = CONFIG_BAUDRATE;        // 设置串口波特率（与全局一致）
	gd->serial.addr = CONFIG_DEBUG_UART_BASE;     // 设置调试串口的基地址（来自配置）
	gd->serial.using_pre_serial = 0;              // 标记未使用早期串口（初始化为0）

	return 0;
}
~~~

最后一个函数为board_init_f，他定义在common/board_r.c，他用来执行重定位前的早期板级初始化（如串口、定时器等基础初始化），具体内容如下所示：

~~~c
/* 重定位前的板级初始化函数 */
void board_init_f(ulong boot_flags)
{
	gd->flags = boot_flags;       // 保存启动标志到全局数据结构
	gd->have_console = 0;         // 标记控制台尚未初始化

	// 执行重定位前的初始化序列，失败则挂起
	if (initcall_run_list(init_sequence_f))
		hang();

	// 对于非ARM、非沙箱、非EFI应用和非x86_64架构
	// 执行到此处会挂起（因为jump_to_copy()不会返回）
#if !defined(CONFIG_ARM) && !defined(CONFIG_SANDBOX) && \
		!defined(CONFIG_EFI_APP) && !CONFIG_IS_ENABLED(X86_64)
	hang();
#endif
~~~

### 重定向代码分析

在crt0.S的第3和第4步骤，分别调用了relocate_code和relocate_vectors进行了代码重定位和异常向量表重定位。

relocate_code定义在arch/arm/lib/relocate_64.S中，具体内容如下所示：

~~~asm
ENTRY(relocate_code)                  ; 重定位代码的入口点
    stp     x29, x30, [sp, #-32]!     ; 保存x29（帧指针）和x30（返回地址）到栈，同时栈指针减32字节（创建栈帧）
    mov     x29, sp                   ; 更新帧指针为当前栈指针
    str     x0, [sp, #16]             ; 将输入参数x0（重定位目标地址）存入栈偏移16字节处

    /* 从flash拷贝u-boot到RAM */
    adrp    x1, __image_copy_start    ; x1获取__image_copy_start的高20位地址（页对齐部分）
    add     x1, x1, :lo12:__image_copy_start  ; 补充低12位，x1指向__image_copy_start（代码段起始地址）
    subs    x9, x0, x1                ; 计算目标地址与当前代码段起始地址的偏移（x9 = 目标地址 - 当前起始地址）
    b.eq    relocate_done             ; 如果偏移为0，说明无需重定位，跳转到relocate_done

    /*
     * 不能直接用ldr x1, __image_copy_start，因为如果代码已在非链接地址运行，
     * 该指令会加载重定位后的__image_copy_start值。为正确重定位，需知道链接时的地址。
     * 链接时的&__image_copy_start存于_TEXT_BASE，它是非重定位值（不是符号引用）
     */
    ldr     x1, _TEXT_BASE            ; x1加载链接时的代码段起始地址（CONFIG_SYS_TEXT_BASE）
    subs    x9, x0, x1                ; 计算目标地址与链接地址的偏移（x9 = 目标地址 - 链接地址）

    adrp    x1, __image_copy_start    ; 重新获取当前__image_copy_start的高20位
    add     x1, x1, :lo12:__image_copy_start  ; x1指向当前代码段起始地址
    adrp    x2, __image_copy_end      ; x2获取__image_copy_end的高20位地址
    add     x2, x2, :lo12:__image_copy_end    ; 补充低12位，x2指向代码段结束地址（__image_copy_end）
~~~

第二个relocate_vectors这个就不找了，在重定向完成就到了board_init_r了。

### 命令交互run_main_loop

init列表里的最后一行代码是run_main_loop，这就会进入uboot的最后一步主循环函数

~~~c
/* 运行U-Boot主循环的函数 */
static int run_main_loop(void)
{
#ifdef CONFIG_SANDBOX
	sandbox_main_loop_init();  // 沙箱模式下初始化主循环（用于模拟环境）
#endif

	/* main_loop()可能返回以重试自动启动，因此需要循环调用 */
	for (;;)  // 无限循环
		main_loop();  // 调用主循环函数（处理命令交互、自动启动等）

	return 0;  // 理论上不会执行到这里（循环不会退出）
}
~~~

该函数的重点在第10行main_loop，它定义在common/main.c目录下，具体代码为

~~~c
void main_loop(void)
{
	const char *s;  // 用于存储自动启动命令字符串

	// 标记启动阶段为"main_loop"，用于跟踪启动进度
	bootstage_mark_name(BOOTSTAGE_ID_MAIN_LOOP, "main_loop");

#ifdef CONFIG_VERSION_VARIABLE
	// 设置环境变量"ver"为U-Boot版本信息
	env_set("ver", version_string);  /* 设置版本变量 */
#endif /* CONFIG_VERSION_VARIABLE */

	// 初始化命令行接口（CLI）
	cli_init();

	// 执行预启动环境变量中定义的命令（如preboot）
	run_preboot_environment_command();

#if defined(CONFIG_UPDATE_TFTP)
	// 通过TFTP更新U-Boot（如果配置了该功能）
	update_tftp(0UL, NULL, NULL);
#endif /* CONFIG_UPDATE_TFTP */

	// 处理启动延时（bootdelay），返回自动启动命令（bootcmd）
	s = bootdelay_process();
	// 处理设备树中的命令，若启用安全启动则执行安全相关命令
	if (cli_process_fdt(&s))
		cli_secure_boot_cmd(s);

	// 执行自动启动命令（倒计时结束无操作时）
	autoboot_command(s);

	// 进入命令行交互循环（用户中断自动启动后）
	cli_loop();

	// 如果命令行循环意外退出，触发恐慌提示
~~~

~~~c
/**
 * 瑞芯微下载模式（dnl）的按键检测函数（弱定义，可被重写）
 * 返回值：1表示检测到按键按下，0表示未按下
 */
__weak int rockchip_dnl_key_pressed(void)
{
#if defined(CONFIG_DM_KEY)
    // 若启用驱动模型（DM）按键功能，直接读取音量加键状态
    return key_is_pressed(key_read(KEY_VOLUMEUP));  // 读取并判断音量加键是否按下

#elif defined(CONFIG_ADC)
    // 若通过ADC（模数转换）检测按键（如某些按键通过分压电阻连接ADC）
    const void *blob = gd->fdt_blob;  // 获取设备树 blob 指针
    int node, ret, channel = 1;       // node：设备树节点，ret：返回值，channel：默认ADC通道
    u32 val, chns[2];                 // val：ADC读取值，chns：存储设备树中的ADC通道信息

    // 在设备树中查找"adc-keys"兼容的节点（ADC按键节点）
    node = fdt_node_offset_by_compatible(blob, 0, "adc-keys");
    if (node >= 0) {
        // 从设备树读取"io-channels"属性（ADC控制器和通道），获取实际通道号
        if (!fdtdec_get_int_array(blob, node, "io-channels", chns, 2))
            channel = chns[1];  // 第二个元素为通道号
    }

    // 通过ADC读取指定通道的电压值（saradc为ADC控制器名称）
    ret = adc_channel_single_shot("saradc", channel, &val);
    if (ret) {
        printf("%s: 读取saradc失败，返回值=%d\n", __func__, ret);
        return 0;  // 读取失败视为未按下
    }
	
    // 判断ADC值是否在按键按下的范围内（KEY_DOWN_MIN_VAL/MAX_VAL为预设阈值）
    return ((val >= KEY_DOWN_MIN_VAL) && (val <= KEY_DOWN_MAX_VAL));
#endif

    return 0;  // 未启用上述配置时，默认返回未按下
}
~~~

### USB 节点配置说明

RK3568 集成多路 USB 控制器及配套 PHY，硬件上支持 1 路 USB3.0 OTG、1 路 USB3.0 HOST 和 2 路 USB2.0 HOST。其功能配置通过设备树节点使能及参数设置实现，核心节点对应关系如下：



- **PHY 节点**：
  - `combphy0_us`：USB3.0 OTG 与 SATA 共用的 Combo PHY（编号 0）
  - `combphy1_usq`：USB3.0 HOST、SATA 及 QSGMII 共用的 Combo PHY（编号 1）
  - `usb2phy0`：USB OTG 与 USB Host 1 共用的 USB2.0 Combo PHY（编号 0）
  - `usb2phy1`：USB Host 1 与 USB Host 2 共用的 USB2.0 Combo PHY（编号 1）
- **控制器节点**：
  - `usbdrd30`：USB3.0 OTG 主控制器，包含 `usbdrd_dwc3` 子节点（DWC3 核）
  - `usbhost30`：USB3.0 Host 控制器，包含 `usbhost_dwc3` 子节点（DWC3 核）
  - `usb_host0_ehci`/`usb_host0_ohci`：USB2.0 Host 2 控制器（EHCI 负责高速，OHCI 负责全速 / 低速）
  - `usb_host1_ehci`/`usb_host1_ohci`：USB2.0 Host 3 控制器（EHCI 负责高速，OHCI 负责全速 / 低速）

#### 1. USB3.0 OTG 功能配置

USB3.0 OTG 功能依赖 `usbdrd30` 控制器、`usbdrd_dwc3` 子节点及配套的 `combphy0_us` PHY。配置示例如下：

```dts
&usbdrd_dwc3 {
    dr_mode = "otg";  // 工作模式设置为 OTG（支持主机/从机切换）
    status = "okay";  // 使能 DWC3 核
};
&usbdrd30 {
    status = "okay";  // 使能 USB3.0 OTG 主控制器
};
&combphy0_us {  // 复用为 USB3.0 OTG 的 PHY（与 SATA0 共用，需确保 SATA0 未占用）
    status = "okay";  // 使能 Combo PHY
};
```

#### 2. USB3.0 Host 功能配置（HOST1）

HOST1 配置为 USB3.0 Host 时，需使能 `usbhost30` 控制器、`usbhost_dwc3` 子节点及 `combphy1_usq` PHY。配置示例如下：

```dts
&usbhost30 {
    status = "okay";  // 使能 USB3.0 Host 主控制器
};
&usbhost_dwc3 {
    status = "okay";  // 使能 DWC3 核
};
&combphy1_usq {  // 复用为 USB3.0 Host 的 PHY（与 SATA1、QSGMII 共用，需确保未被占用）
    status = "okay";  // 使能 Combo PHY
};
```





#### 3. USB2.0 Host 功能配置（HOST1）

HOST1 降级为 USB2.0 Host 时，需禁用 USB3.0 相关 PHY，改用 USB2.0 PHY 并限制速率。配置示例如下

```dts
&usbhost30 {
    status = "okay";  // 使能主控制器
};
&usbhost_dwc3 {
    phys = <&u2phy0_host>;  // 绑定 USB2.0 PHY
    phy-names = "usb2-phy";
    maximum-speed = "high-speed";  // 限制最高速率为 USB2.0 高速（480Mbps）
    status = "okay";
};
&combphy1_usq {
    rockchip,dis-u3otg1-port;  // 禁用 USB3.0 端口功能
    status = "okay";  // 保持 PHY 使能（硬件关联需求）
};
&usb2phy0 {
    status = "okay";  // 使能 USB2.0 Combo PHY
};
&u2phy0_host {  // USB2.0 Host 模式专用 PHY 子节点
    phy-supply = <&vcc5v0_host>;  // 绑定电源（5V 供电）
    status = "okay";
};
```



以上配置通过使能对应控制器、PHY 节点并设置工作模式 / 速率，实现不同 USB 功能的切换。实际应用中需注意：

- Combo PHY 为多功能复用硬件，需确保同一时刻仅被一个功能占用；
- USB2.0 模式需显式绑定 USB2.0 PHY 并限制速率，避免与 USB3.0 配置冲突；
- 电源节点（如 `vcc5v0_host`）需根据硬件设计正确关联，确保 PHY 供电正常。



更详细的配置规范可参考瑞芯微官方文档《Rockchip_Developer_Guide_USB_CN.pdf》。

下面我将详细讲解 U-Boot 与内核双向绑定的实现步骤，包括具体代码修改位置、验证流程和关键细节，确保方案可落地实施。

### **整体实现逻辑**

实现双向绑定需要完成两个核心验证：

1. **U-Boot 验证内核**：U-Boot 启动时，先验证内核是否为 "自家" 签名的合法镜像，不合法则拒绝启动。
2. **内核验证 U-Boot**：内核启动初期，验证启动它的 U-Boot 是否为 "自家" 版本，不合法则立即终止启动。

两者通过**非对称加密算法（如 RSA）** 实现安全验证，避免密钥泄露风险。

### **前置准备**

1. **生成密钥对**：
   使用 RSA 算法生成公钥（public key）和私钥（private key），私钥严格保管（仅用于签名自家镜像），公钥嵌入 U-Boot 和内核中用于验证。

   ```bash
   # 生成2048位RSA密钥对
   openssl genrsa -out private_key.pem 2048
   openssl rsa -in private_key.pem -pubout -out public_key.pem
   ```

2. **集成加密库**：
   U-Boot 和内核需要 RSA 验证功能，可集成轻量级加密库（如`mbedTLS`）或使用内核自带的`crypto`模块。

   - U-Boot：在`Kconfig`中开启`CONFIG_MBEDTLS`，并在`Makefile`中添加相关编译选项。
   - 内核：开启`CONFIG_CRYPTO_RSA`和`CONFIG_CRYPTO_SHA256`（用于哈希计算）。

### **步骤 1：U-Boot 验证内核镜像（防止非自家内核启动）**

#### 1.1 给内核镜像添加签名

内核编译完成后，用私钥对内核镜像（如`zImage`或`uImage`）的哈希值签名，生成签名文件（随内核镜像一起存放）。

**编写签名工具（Python 示例）**：

```python
import rsa
import hashlib

# 读取私钥和内核镜像
with open("private_key.pem", "rb") as f:
    private_key = rsa.PrivateKey.load_pkcs1(f.read())
with open("zImage", "rb") as f:
    kernel_data = f.read()

# 计算内核镜像的SHA256哈希
hash_obj = hashlib.sha256(kernel_data)
kernel_hash = hash_obj.digest()

# 用私钥签名哈希值
signature = rsa.sign(kernel_hash, private_key, "SHA-256")

# 保存签名（后续会被U-Boot读取）
with open("kernel_signature.bin", "wb") as f:
    f.write(signature)
```

#### 1.2 修改 U-Boot 代码，添加内核验证逻辑

U-Boot 在加载内核前（如`do_bootm`函数），需读取内核镜像和签名，用公钥验证签名合法性。

**修改位置**：U-Boot 源码`common/bootm.c`（不同版本可能在`bootm_os.c`）

**核心代码**：

```c
#include <mbedtls/rsa.h>
#include <mbedtls/sha256.h>

// 嵌入公钥（从public_key.pem转换为C数组，可通过工具生成）
const unsigned char kernel_pub_key[] = {
    0x30, 0x82, 0x01, 0x22, 0x30, 0x0d, ... // 公钥二进制数据
};
const int kernel_pub_key_len = sizeof(kernel_pub_key);

// 验证内核签名
int verify_kernel_signature(const void *kernel_data, int kernel_len, 
                           const void *signature, int sig_len) {
    mbedtls_rsa_context rsa;
    mbedtls_sha256_context sha256;
    unsigned char hash[32];
    int ret = -1;

    // 初始化加密上下文
    mbedtls_rsa_init(&rsa, MBEDTLS_RSA_PKCS_V15, 0);
    mbedtls_sha256_init(&sha256);

    // 解析公钥
    if (mbedtls_rsa_parse_public_key(&rsa, kernel_pub_key, kernel_pub_key_len) != 0)
        goto exit;

    // 计算内核镜像的SHA256哈希
    mbedtls_sha256_update(&sha256, kernel_data, kernel_len);
    mbedtls_sha256_finish(&sha256, hash);

    // 用公钥验证签名（签名是对哈希的加密）
    ret = mbedtls_rsa_verify(&rsa, NULL, NULL, MBEDTLS_RSA_PUBLIC, 
                            MBEDTLS_MD_SHA256, 32, hash, signature);

exit:
    mbedtls_rsa_free(&rsa);
    mbedtls_sha256_free(&sha256);
    return (ret == 0) ? 0 : -1; // 0表示验证成功
}

// 在启动内核前调用验证函数
int do_bootm(cmd_tbl_t *cmdtp, int flag, int argc, char *const argv[]) {
    // ... 原有代码（加载内核镜像到内存）

    // 读取内核镜像和签名（假设签名存放在内核地址+内核长度的位置）
    void *kernel_data = (void *)load_addr;
    int kernel_len = ...; // 内核大小
    void *signature = kernel_data + kernel_len;
    int sig_len = 256; // RSA2048签名长度为256字节

    // 验证内核合法性
    if (verify_kernel_signature(kernel_data, kernel_len, signature, sig_len) != 0) {
        printf("Error: Invalid kernel signature! Refuse to boot.\n");
        return -1; // 验证失败，终止启动
    }

    // ... 继续启动内核（原有代码）
}
```

#### 2.2 内核验证 U-Boot 令牌

内核启动初期（越早越好，如`start_kernel`之前的汇编阶段或`setup_arch`函数），读取 U-Boot 传递的令牌，用 U-Boot 公钥验证合法性。

**修改内核代码（例如在`arch/arm/kernel/head.S`或`setup.c`）**：

```c
#include <crypto/rsa.h>
#include <crypto/hash.h>

// 嵌入U-Boot公钥（用于验证令牌）
const unsigned char uboot_pub_key[] = {
    0x30, 0x82, 0x01, 0x22, 0x30, 0x0d, ... // U-Boot公钥二进制数据
};

// 从设备树读取U-Boot令牌
static unsigned char *get_uboot_token(void) {
    struct device_node *root = of_find_node_by_path("/");
    const void *token;
    int len;

    token = of_get_property(root, "uboot_auth_token", &len);
    if (!token || len != 256)
        return NULL;

    return (unsigned char *)token;
}

// 验证U-Boot令牌
static int verify_uboot_token(void) {
    unsigned char *token = get_uboot_token();
    if (!token)
        return -1;

    // 读取U-Boot传递的原始信息（auth_info）
    // （实际中需和令牌一起传递，或从硬件ID反推验证）
    char auth_info[128];
    snprintf(auth_info, sizeof(auth_info), "hw_id=%s,timestamp=%lu",
             get_chip_unique_id(), get_uboot_timestamp()); // 内核侧获取相同信息

    // 用U-Boot公钥验证令牌（令牌是auth_info的签名）
    struct shash_desc *desc;
    unsigned char hash[32];
    int ret;

    // 计算auth_info的SHA256哈希
    desc = kzalloc(sizeof(*desc) + crypto_shash_descsize(sha256_tfm), GFP_KERNEL);
    desc->tfm = crypto_alloc_shash("sha256", 0, 0);
    crypto_shash_init(desc);
    crypto_shash_update(desc, auth_info, strlen(auth_info));
    crypto_shash_final(desc, hash);

    // 验证签名
    ret = rsa_verify(uboot_pub_key, sizeof(uboot_pub_key),
                    hash, 32, token, 256);

    crypto_free_shash(desc->tfm);
    kfree(desc);
    return ret;
}

// 内核启动初期调用验证
void __init setup_arch(char **cmdline_p) {
    // ... 原有代码

    // 验证U-Boot合法性，失败则死机
    if (verify_uboot_token() != 0) {
        pr_crit("Error: Invalid U-Boot! Kernel startup aborted.\n");
        while (1); // 终止启动
    }

    // ... 继续启动流程
}
```

#### **1.2 实现校验值计算和硬件 ID 读取（board_auth.c）**

这部分代码需要在 U-Boot 和内核中分别编译（逻辑完全相同，但编译环境不同，需注意适配）。

~~~c
#include "board_auth.h"
#include <string.h>
#include <stdio.h>

// 计算校验值：将密钥和硬件ID通过简单算法混合，生成唯一值
// 算法说明：通过移位、异或、累加组合，确保输入不同则输出不同
unsigned int calculate_checksum(const char *hardware_id) {
    unsigned int checksum = 0x5A5A5A5A; // 初始种子值（自定义）
    int i;

    // 第一步：混合密钥（SECRET_KEY）
    for (i = 0; i < strlen(SECRET_KEY); i++) {
        // 移位+异或：增强随机性（可根据需要调整算法）
        checksum = (checksum << 4) | (checksum >> 28); // 循环左移4位
        checksum ^= (unsigned int)SECRET_KEY[i];       // 与密钥字符异或
        checksum += (checksum >> 10);                  // 累加高位，增加混淆
    }

    // 第二步：混合硬件ID（确保不同硬件生成的校验值不同）
    for (i = 0; i < strlen(hardware_id); i++) {
        checksum = (checksum << 3) | (checksum >> 29); // 循环左移3位（与上面移位不同，增加复杂度）
        checksum ^= (unsigned int)hardware_id[i];      // 与硬件ID字符异或
        checksum += (checksum >> 8);                   // 累加高位
    }

    return checksum;
}

// 读取硬件唯一标识（重点！必须选择自家板子独有的特征）
// 示例：读取板载EEPROM中的定制数据（竞品没有这个EEPROM或数据不同）
char *get_hardware_id(void) {
    static char hardware_id[64]; // 存储硬件ID字符串
    unsigned char eeprom_data[8]; // 从EEPROM读取8字节数据

    // 实际开发时，需根据硬件手册修改以下逻辑：
    // 1. 确定硬件特征位置（如板载EEPROM地址、专用寄存器等）
    // 2. 读取数据（确保竞品无法读取到相同数据）

    // 示例：假设板载EEPROM在0x02000000地址，存储了8字节定制ID
    // 注意：真实场景中需通过I2C/SPI驱动读取EEPROM，此处简化为直接读内存
    volatile unsigned char *eeprom_addr = (unsigned char *)0x02000000;
    for (int i = 0; i < 8; i++) {
        eeprom_data[i] = eeprom_addr[i]; // 读取8字节数据
    }

    // 将8字节数据格式化为字符串（如"a1b2c3d4e5f67890"）
    // 格式化为字符串便于处理，也可直接用二进制计算
    snprintf(hardware_id, sizeof(hardware_id),
             "%02x%02x%02x%02x%02x%02x%02x%02x",
             eeprom_data[0], eeprom_data[1], eeprom_data[2], eeprom_data[3],
             eeprom_data[4], eeprom_data[5], eeprom_data[6], eeprom_data[7]);

    return hardware_id;
}
~~~

- 这是**演示级**绑定，**不安全**（任意人可仿造 `/chosen/yourco,bind`）。

~~~c
        ret = fdt_setprop_u32(new_fdt, chosen, "yourco,bind-version", info->version);
~~~

~~~c
#include <linux/init.h>
#include <linux/of.h>
#include <linux/printk.h>
#include <linux/kernel.h>
#include <linux/string.h>
#include <linux/yourco_bind.h>

/* 模块参数，允许通过启动参数控制行为 */
static bool yourco_bind_enforce = true;
module_param_named(yourco_bind_enforce, yourco_bind_enforce, bool, 0444);
MODULE_PARM_DESC(yourco_bind_enforce, "Enforce yourco binding check (default: true)");

/* 检查令牌是否为已知的有效版本 */
static bool yourco_validate_token(const char *token)
{
    if (!token)
        return false;

    /* 支持多版本检查 */
    if (strcmp(token, YOURCO_BIND_TOKEN_V1) == 0)
        return true;

    /* 可以添加更多版本检查 */
    // if (strcmp(token, YOURCO_BIND_TOKEN_V2) == 0)
    //     return true;

    return false;
}

/* 检查版本兼容性 */
static bool yourco_check_version_compatibility(u32 version)
{
    /*
     * 简单的兼容性检查示例:
     * - 主版本号必须匹配
     * - 次版本号可以兼容旧版本
     */
    return (version == YOURCO_BIND_VERSION_CURRENT);
}

static int __init yourco_bind_check(void)
{
    struct device_node *chosen;
    const char *token;
    u32 version;
    u32 timestamp;  // 移到代码块开头声明
    int ret;

    /* 如果不强制检查且通过启动参数禁用，则直接返回成功 */
    if (!yourco_bind_enforce && !YOURCO_BIND_PANIC_ON_FAILURE) {
        pr_warn("yourco: binding check is disabled\n");
        return 0;
    }

    chosen = of_find_node_by_path("/chosen");
    if (!chosen) {
        pr_emerg("yourco: /chosen node not found in device tree\n");
        if (YOURCO_BIND_PANIC_ON_FAILURE)
            panic("yourco: critical binding information missing");
        return -ENODEV;
    }

    /* 读取并验证令牌 */
    ret = of_property_read_string(chosen, "yourco,bind-token", &token);
    if (ret) {
        pr_emerg("yourco: failed to read yourco,bind-token (ret=%d)\n", ret);
        if (YOURCO_BIND_PANIC_ON_FAILURE)
            panic("yourco: binding token missing");
        return ret;
    }

    if (!yourco_validate_token(token)) {
        pr_emerg("yourco: invalid binding token: '%s'\n", token);
        if (YOURCO_BIND_PANIC_ON_FAILURE)
            panic("yourco: invalid binding token");
        return -EINVAL;
    }

    /* 读取并验证版本 */
    ret = of_property_read_u32(chosen, "yourco,bind-version", &version);
    if (ret) {
        pr_emerg("yourco: failed to read yourco,bind-version (ret=%d)\n", ret);
        if (YOURCO_BIND_PANIC_ON_FAILURE)
            panic("yourco: version information missing");
        return ret;
    }

    if (!yourco_check_version_compatibility(version)) {
        pr_emerg("yourco: version incompatibility (found v%d, expected v%d)\n",
                version, YOURCO_BIND_VERSION_CURRENT);
        if (YOURCO_BIND_PANIC_ON_FAILURE)
            panic("yourco: version incompatibility");
        return -EINVAL;
    }

    /* 读取并记录时间戳 */
    if (!of_property_read_u32(chosen, "yourco,build-timestamp", &timestamp)) {
        pr_info("yourco: U-Boot build timestamp: %u\n", timestamp);
    }

    pr_info("yourco: binding check passed - token: '%s', version: %d\n",
            token, version);
    return 0;
}

/* 使用earlier initcall确保在系统关键操作前完成检查 */
core_initcall(yourco_bind_check);
~~~

~~~c
/* 常量时间比较函数（兼容C89） */
static inline int yourco_consttime_memcmp(const void *a, const void *b, size_t len)
{
    const unsigned char *p = a;
    const unsigned char *q = b;
    int diff = 0;
    size_t i;  // 在循环外声明变量

    for (i = 0; i < len; i++) {  // 循环中不声明变量
        diff |= (*p++ ^ *q++);
    }
    return diff;
}
~~~

~~~shell
# 从配置文件中提取指定键的值
function filt_val()
{
	# 查找包含"键="的行，移除"键="部分，去除回车和引号
	sed -n "/${1}=/s/${1}=//p" $2 | tr -d '\r' | tr -d '"'
}

# 准备工作：检查rkbin工具路径并初始化平台类型
function prepare()
{
	if [ -d ${RKBIN_TOOLS} ]; then
		# 获取rkbin工具的绝对路径
		absolute_path=$(cd `dirname ${RKBIN_TOOLS}`; pwd)
		RKBIN=${absolute_path}
	else
		echo "错误: 未找到../rkbin仓库"
		exit 1
	fi

	# 检查是否启用ARM64信任区
	if grep -Eq ''^CONFIG_ARM64=y'|'^CONFIG_ARM64_BOOT_AARCH32=y'' .config ; then
		ARM64_TRUSTZONE="y"
	fi

	# 确定平台类型
	if grep  -q '^CONFIG_ROCKCHIP_FIT_IMAGE_PACK=y' .config ; then
		PLAT_TYPE="FIT"  # FIT镜像格式
	elif grep  -q '^CONFIG_SPL_DECOMP_HEADER=y' .config ; then
		PLAT_TYPE="DECOMP"  # 带解压头的格式
	fi
}

# 处理命令行参数
function process_args()
{
	while [ $# -gt 0 ]; do
		case $1 in
			*help|--h|-h)
				help
				exit 0
				;;
			CROSS_COMPILE=*)  # 设置交叉编译器
				ARG_COMPILE="y"
				CROSS_COMPILE_ARM32=${1#*=}
				CROSS_COMPILE_ARM64=${1#*=}
				if [ ${CMD_ARGS} == $1 ]; then
					shift 1
					CMD_ARGS=$1
				else
					shift 1
				fi
				;;
			# 空参数: 使用现有.config构建
			# loader|trust|uboot: 打包镜像
			# debug*|map|sym|elf*|nm: 调试命令
			# env: 构建环境工具
			# itb: 打包itb镜像
			''|loader|trust|uboot|debug*|itb|env|fit|map|sym|elf*|nm)
				if [ "$2" == "spl" -o "$2" == "tpl" ]; then
					ARG_TSPL=$2
					shift 1
				fi
				shift 1
				;;
			# trust/loader的ini配置文件
			*.ini|*.INI)
				if [ ! -f $1 ]; then
					echo "错误: 未找到$1"
				fi
				# 判断是loader还是trust的配置文件
				if grep -q 'CODE471_OPTION' $1 ; then
					ARG_INI_LOADER=$1
				elif grep -Eq ''BL31_OPTION'|'TOS'' $1 ; then
					ARG_INI_TRUST=$1
				fi
				shift 1
				;;
			--sz-trust) # 设置trust大小
				ARG_TRUST_SIZE="--size $2 $3"
				shift 3
				;;
			--sz-uboot) # 设置uboot大小
				ARG_UBOOT_SIZE="--size $2 $3"
				shift 3
				;;
			--raw-compile)  # FIT: 只构建不打包镜像
				ARG_RAW_COMPILE="y"
				shift 1
				;;
			--no-uboot) # FIT: 打包不包含u-boot的uboot.img
				ARG_NO_UBOOT="y"
				shift 1
				;;
			--idblock)  # 打包idblock.bin
				shift 1
				;;
			--tpl|tpl)  # 使用tpl文件
				ARG_TPL_BIN="tpl/u-boot-tpl.bin"
				shift 1
				;;
			--spl|spl*) # 使用spl文件
				ARG_SPL_BIN="spl/u-boot-spl.bin"
				shift 1
				;;
			--uboot|--fdt|--optee|--mcu|--bl31) # uboot.img组件
				mkdir -p ${REP_DIR}
				if [ ! -f $2 ]; then
					echo "错误: 未找到$2"
					exit 1
				fi
				# 根据参数类型复制相应文件到临时目录
				if [ "$1" == "--uboot" ]; then
					cp $2 ${REP_DIR}/u-boot-nodtb.bin
				elif [ "$1" == "--fdt" ]; then
					cp $2 ${REP_DIR}/u-boot.dtb
				elif [ "$1" == "--optee" ]; then
					cp $2 ${REP_DIR}/tee.bin
				elif [ "$1" == "--mcu" ]; then
					cp $2 ${REP_DIR}/mcu.bin
				elif [ "$1" == "--bl31" ]; then
					if ! file $2 | grep 'ELF ' >/dev/null 2>&1 ; then
						echo "错误: $2不是bl31.elf文件"
						exit 1
					fi
					cp $2 ${REP_DIR}/bl31.elf
				fi
				shift 2
				;;
			*)
				# 1. 处理FIT脚本参数
				NUM=$(${SCRIPT_FIT} --args $1)
				if  [ ${NUM} -ne 0 ]; then
					[ ${NUM} -eq 1 ] && ARG_LIST_FIT="${ARG_LIST_FIT} $1"
					[ ${NUM} -eq 2 ] && ARG_LIST_FIT="${ARG_LIST_FIT} $1 $2"
					shift ${NUM}
					continue
				# 2. 解析函数地址
				elif [ -z $(echo $1 | sed 's/[0-9,a-f,A-F,x,X,-]//g') ]; then
					ARG_FUNCADDR=$1
				# 3. 处理defconfig
				else
					ARG_BOARD=$1
					# 检查配置文件是否存在
					if [ ! -f configs/${ARG_BOARD}_defconfig -a ! -f configs/${ARG_BOARD}.config ]; then
						echo -e "\n${SUPPORT_LIST}\n"
						echo "错误: 未找到configs/${ARG_BOARD}_defconfig"
						exit 1
					elif [ -f configs/${ARG_BOARD}.config ]; then
						# 处理多层继承的配置
						BASE1_DEFCONFIG=`filt_val "CONFIG_BASE_DEFCONFIG" configs/${ARG_BOARD}.config`
						BASE0_DEFCONFIG=`filt_val "CONFIG_BASE_DEFCONFIG" configs/${BASE1_DEFCONFIG}`
						MAKE_CMD="make ${BASE0_DEFCONFIG} ${BASE1_DEFCONFIG} ${ARG_BOARD}.config -j${JOB}"
						echo "## ${MAKE_CMD}"
						make ${BASE0_DEFCONFIG} ${BASE1_DEFCONFIG} ${ARG_BOARD}.config ${OPTION}
						rm -f ${CC_FILE}
					else
						MAKE_CMD="make ${ARG_BOARD}_defconfig -j${JOB}"
						echo "## ${MAKE_CMD}"
						make ${ARG_BOARD}_defconfig ${OPTION}
						rm -f ${CC_FILE}
					fi
				fi
				shift 1
				;;
		esac
	done

	# 检查.config是否存在
	if [ ! -f .config ]; then
		echo
		echo "错误: 未找到.config"
		help
		exit 1
	fi
}

# 选择合适的交叉编译器
function select_toolchain()
{
	# 如果没有外部指定CROSS_COMPILE，从CC_FILE中读取
	if [ "${ARG_COMPILE}" != "y" ]; then
		if [ -f ${CC_FILE} ]; then
			CROSS_COMPILE_ARM32=`cat ${CC_FILE}`
			CROSS_COMPILE_ARM64=`cat ${CC_FILE}`
		else
			# 根据配置选择32位或64位编译器
			if grep -q '^CONFIG_ARM64=y' .config ; then
				CROSS_COMPILE_ARM64=$(cd `dirname ${CROSS_COMPILE_ARM64}`; pwd)"/aarch64-linux-gnu-"
			else
				CROSS_COMPILE_ARM32=$(cd `dirname ${CROSS_COMPILE_ARM32}`; pwd)"/arm-linux-gnueabihf-"
			fi
		fi
	fi

	# 设置工具链路径
	if grep -q '^CONFIG_ARM64=y' .config ; then
		TOOLCHAIN=${CROSS_COMPILE_ARM64}
		TOOLCHAIN_NM=${CROSS_COMPILE_ARM64}nm
		TOOLCHAIN_OBJDUMP=${CROSS_COMPILE_ARM64}objdump
		TOOLCHAIN_ADDR2LINE=${CROSS_COMPILE_ARM64}addr2line
	else
		TOOLCHAIN=${CROSS_COMPILE_ARM32}
		TOOLCHAIN_NM=${CROSS_COMPILE_ARM32}nm
		TOOLCHAIN_OBJDUMP=${CROSS_COMPILE_ARM32}objdump
		TOOLCHAIN_ADDR2LINE=${CROSS_COMPILE_ARM32}addr2line
	fi

	# 检查编译器是否存在
	if [ ! `which ${TOOLCHAIN}gcc` ]; then
		echo "错误: 未找到${TOOLCHAIN}gcc"
		exit 1
	fi

	# 保存编译器路径到文件
	if [ "${ARG_COMPILE}" == "y" ]; then
		echo "${TOOLCHAIN}" > ${CC_FILE}
	fi
}

#
# 选择芯片信息用于:
#	1. RKCHIP:        修复平台配置
#	2. RKCHIP_LOADER: 搜索用于打包loader的ini文件
#	3. RKCHIP_TRUST:  搜索用于打包trust的ini文件
#	4. RKCHIP_LABEL:  显示构建信息
#
function select_chip_info()
{
	# 从.config中读取RKCHIP
	# 匹配模式:
	#  - PX30, PX3SE
	#  - RK????, RK????X
	#  - RV????
	CHIP_PATTERN='^CONFIG_ROCKCHIP_[R,P][X,V,K][0-9ESX]{1,5}'
	RKCHIP=`egrep -o ${CHIP_PATTERN} .config`

	# 默认值设置
	RKCHIP=${RKCHIP##*_}
	RKCHIP_LOADER=${RKCHIP}
	RKCHIP_TRUST=${RKCHIP}
	RKCHIP_LABEL=`filt_val "CONFIG_CHIP_NAME" .config`
	if [ -z "${RKCHIP_LABEL}" ]; then
		RKCHIP_LABEL=${RKCHIP}
	fi
}

# 优先级: 默认 < CHIP_CFG_FIXUP_TABLE() < make.sh参数
function fixup_platform_configure()
{
	# 从.config读取配置值
	U_KB=`filt_val "CONFIG_UBOOT_SIZE_KB" .config`
	U_NUM=`filt_val "CONFIG_UBOOT_NUM" .config`
	T_KB=`filt_val "CONFIG_TRUST_SIZE_KB" .config`
	T_NUM=`filt_val "CONFIG_TRUST_NUM" .config`
	SHA=`filt_val "CONFIG_TRUST_SHA_MODE" .config`
	RSA=`filt_val "CONFIG_TRUST_RSA_MODE" .config`

	# 从.config设置平台参数
	PLAT_UBOOT_SIZE="--size ${U_KB} ${U_NUM}"
	PLAT_TRUST_SIZE="--size ${T_KB} ${T_NUM}"
	PLAT_SHA="--sha ${SHA}"
	PLAT_RSA="--rsa ${RSA}"

	# 用命令行参数覆盖配置
	if [ ! -z "${ARG_UBOOT_SIZE}" ]; then
		PLAT_UBOOT_SIZE=${ARG_UBOOT_SIZE}
	fi
	if [ ! -z "${ARG_TRUST_SIZE}" ]; then
		PLAT_TRUST_SIZE=${ARG_TRUST_SIZE}
	fi
}

# 选择ini配置文件
function select_ini_file()
{
	# 默认配置文件
	INI_LOADER=${RKBIN}/RKBOOT/${RKCHIP_LOADER}MINIALL.ini
	if [ "${ARM64_TRUSTZONE}" == "y" ]; then
		INI_TRUST=${RKBIN}/RKTRUST/${RKCHIP_TRUST}TRUST.ini
	else
		INI_TRUST=${RKBIN}/RKTRUST/${RKCHIP_TRUST}TOS.ini
	fi

	# 从defconfig读取配置文件
	NAME=`filt_val "CONFIG_LOADER_INI" .config`
	if [ ! -z "${NAME}" ]; then
		INI_LOADER=${RKBIN}/RKBOOT/${NAME}
	fi
	NAME=`filt_val "CONFIG_TRUST_INI" .config`
	if [ ! -z "${NAME}" ]; then
		INI_TRUST=${RKBIN}/RKTRUST/${NAME}
	fi

	# 用命令行参数指定的配置文件覆盖
	if [ ! -z "${ARG_INI_TRUST}" ]; then
		INI_TRUST=${ARG_INI_TRUST}
	fi
	if [ ! -z "${ARG_INI_LOADER}" ]; then
		INI_LOADER=${ARG_INI_LOADER}
	fi
}

# 处理子命令
function sub_commands()
{
	# 跳过"--"开头的参数，如"--rollback-index-..."
	if [[ ${CMD_ARGS} != --* ]]; then
		CMD=${CMD_ARGS%-*}
		ARG=${CMD_ARGS#*-}
	else
		CMD=${CMD_ARGS}
	fi

	# 根据参数选择要处理的ELF文件
	if [ "${ARG_TSPL}" == "tpl" -o "${ARG_TSPL}" == "spl" ]; then
		ELF=`find -name u-boot-${ARG_TSPL}`
		MAP=`find -name u-boot-${ARG_TSPL}.map`
		SYM=`find -name u-boot-${ARG_TSPL}.sym`
	else
		ELF=u-boot
		MAP=u-boot.map
		SYM=u-boot.sym
	fi

	# 根据命令执行相应操作
	case ${CMD} in
		elf|nm)
			if [ "${CMD}" == "nm" ]; then
				echo -e "\n${ELF}:     文件格式 elf\n"
				${TOOLCHAIN_NM} -r --size ${ELF} | grep -iv 'b' | less
			else
				if [ "${CMD}" == "elf" -a "${ARG}" == "elf" ]; then
					ARG=D # 默认参数
				fi
				${TOOLCHAIN_OBJDUMP} -${ARG} ${ELF} | less
			fi
			exit 0
			;;
		map|sym)
			if [ ${CMD} == "map" ]; then
				cat ${MAP} | less
			else
				cat ${SYM} | less
			fi
			exit 0
			;;
		debug)
			./scripts/rkpatch.sh ${ARG}
			exit 0
			;;
		fit)
			# 非安全模式
			${SCRIPT_FIT} --boot_img_dir images/ ${ARG_LIST_FIT}
			exit 0
			;;
		uboot)
			pack_uboot_image
			exit 0
			;;
		trust)
			pack_trust_image
			exit 0
			;;
		loader)
			pack_loader_image
			exit 0
			;;
		itb)
			pack_uboot_itb_image
			exit 0
			;;
		env)
			make CROSS_COMPILE=${TOOLCHAIN} envtools
			exit 0
			;;
		--idblock)
			pack_idblock
			exit 0
			;;
		--tpl|--spl|tpl|spl)
			pack_spl_loader_image
			exit 0
			;;
		*)
			unwind_addr_or_continue
			;;
	esac
}

# 解析地址或继续执行
function unwind_addr_or_continue()
{
	FUNCADDR=${ARG_FUNCADDR}
	RELOCOFF=${FUNCADDR#*-}
	FUNCADDR=${FUNCADDR%-*}

	# 检查是否是有效的地址格式
	if [ -z $(echo ${FUNCADDR} | sed 's/[0-9,a-f,A-F,x,X,-]//g') ] && [ ${FUNCADDR} ]; then
		# 处理带0x或0X前缀的地址
		if [ `echo ${FUNCADDR} | sed -n "/0[x,X]/p" | wc -l` -ne 0 ]; then
			FUNCADDR=`echo ${FUNCADDR} | awk '{ print strtonum($0) }'`
			FUNCADDR=`echo "obase=16;${FUNCADDR}"|bc | tr '[A-Z]' '[a-z]'`
		fi
		if [ `echo ${RELOCOFF} | sed -n "/0[x,X]/p" | wc -l` -ne 0 ] && [ ${RELOCOFF} ]; then
			RELOCOFF=`echo ${RELOCOFF} | awk '{ print strtonum($0) }'`
			RELOCOFF=`echo "obase=16;${RELOCOFF}"|bc | tr '[A-Z]' '[a-z]'`
		fi

		# 如果指定了重定位偏移，计算实际地址
		if [ "${FUNCADDR}" != "${RELOCOFF}" ]; then
			# 十六进制->十进制->减法->十六进制
			FUNCADDR=`echo $((16#${FUNCADDR}))`
			RELOCOFF=`echo $((16#${RELOCOFF}))`
			FUNCADDR=$((FUNCADDR-RELOCOFF))
			FUNCADDR=$(echo "obase=16;${FUNCADDR}"|bc | tr '[A-Z]' '[a-z]')
		fi

		echo
		sed -n "/${FUNCADDR}/p" ${SYM}
		${TOOLCHAIN_ADDR2LINE} -e ${ELF} ${FUNCADDR}
		exit 0
	fi
}

# 打包uboot.itb镜像
function pack_uboot_itb_image()
{
	INI=${INI_TRUST}
	if [ ! -f ${INI} ]; then
		echo "错误: 未找到${INI}"
		exit 1
	fi

	# 处理ARM64信任区
	if [ "${ARM64_TRUSTZONE}" == "y" ]; then
		BL31_ELF=`sed -n '/_bl31_/s/PATH=//p' ${INI} | tr -d '\r'`
		BL32_BIN=`sed -n '/_bl32_/s/PATH=//p' ${INI} | tr -d '\r'`
		rm bl31.elf tee.bin -rf
		cp ${RKBIN}/${BL31_ELF} bl31.elf
		if grep BL32_OPTION -A 1 ${INI} | grep SEC=1 ; then
			cp ${RKBIN}/${BL32_BIN} tee.bin
			TEE_OFFSET=`grep BL32_OPTION -A 3 ${INI} | grep ADDR= | awk -F "=" '{ printf $2 }' | tr -d '\r'`
			TEE_ARG="-t ${TEE_OFFSET}"
		fi
	else
		# 处理TOS
		TOS=`filt_val "TOS" ${INI}`
		TOSTA=`filt_val "TOSTA" ${INI}`
		if [ ! -z "${TOSTA}" ]; then
			cp ${RKBIN}/${TOSTA} tee.bin
		elif [ ! -z "${TOS}" ]; then
			cp ${RKBIN}/${TOS}   tee.bin
		else
			echo "警告: 未找到tee二进制文件"
		fi
		if [ ! -z "${TOSTA}" -o ! -z "${TOS}" ]; then
			TEE_OFFSET=`filt_val "ADDR" ${INI}`
			if [ "${TEE_OFFSET}" == "" ]; then
				TEE_OFFSET=0x8400000
			fi
			TEE_ARG="-t ${TEE_OFFSET}"
		fi
	fi

	# 处理MCU
	for ((i=0; i<5; i++))
	do
		MCU_BIN="mcu${i}.bin"
		MCU_IDX="MCU${i}"

		# 兼容性处理: 如果存在"MCU"则用"MCU"替换"MCU0"
		ENABLED=`awk -F"," '/MCU=/  { printf $3 }' ${INI} | tr -d ' '`
		if [ ${i} -eq 0 ]; then
			ENABLED=`awk -F"," '/MCU=/  { printf $3 }' ${INI} | tr -d ' '`
			if [ ! -z ${ENABLED} ]; then
				MCU_IDX="MCU"
			fi
		fi

		ENABLED=`awk -F "," '/'${MCU_IDX}'=/  { printf $3 }' ${INI} | tr -d ' '`
		if [ "${ENABLED}" == "enabled" -o "${ENABLED}" == "okay" ]; then
			NAME=`awk -F "," '/'${MCU_IDX}'=/ { printf $1 }' ${INI} | tr -d ' ' | awk -F "=" '{ print $2 }'`
			OFFS=`awk -F "," '/'${MCU_IDX}'=/ { printf $2 }' ${INI} | tr -d ' '`
			cp ${RKBIN}/${NAME} ${MCU_BIN}
			if [ -z ${OFFS} ]; then
				echo "错误: 在${INI}中未找到${MCU_BIN}的地址"
				exit 1
			fi
			MCU_ARG=${MCU_ARG}" -m${i} ${OFFS}"
		fi
	done

	# 处理可加载项
	for ((i=0; i<5; i++))
	do
		LOAD_BIN="load${i}.bin"
		LOAD_IDX="LOAD${i}"
		ENABLED=`awk -F "," '/'${LOAD_IDX}'=/  { printf $3 }' ${INI} | tr -d ' '`
		if [ "${ENABLED}" == "enabled" -o "${ENABLED}" == "okay" ]; then
			NAME=`awk -F "," '/'${LOAD_IDX}'=/ { printf $1 }' ${INI} | tr -d ' ' | awk -F "=" '{ print $2 }'`
			OFFS=`awk -F "," '/'${LOAD_IDX}'=/ { printf $2 }' ${INI} | tr -d ' '`
			cp ${RKBIN}/${NAME} ${LOAD_BIN}
			if [ -z ${OFFS} ]; then
				echo "错误: 在${INI}中未找到${LOAD_BIN}的地址"
				exit 1
			fi
			LOAD_ARG=${LOAD_ARG}" -l${i} ${OFFS}"
		fi
	done

	# 处理压缩选项
	COMPRESSION=`awk -F"," '/COMPRESSION=/  { printf $1 }' ${INI} | tr -d ' ' | cut -c 13-`
	if [ ! -z "${COMPRESSION}" -a "${COMPRESSION}" != "none" ]; then
		COMPRESSION_ARG="-c ${COMPRESSION}"
	fi

	if [ -d ${REP_DIR} ]; then
		mv ${REP_DIR}/* ./
	fi

	SPL_FIT_SOURCE=`filt_val "CONFIG_SPL_FIT_SOURCE" .config`
	if [ ! -z ${SPL_FIT_SOURCE} ]; then
		cp ${SPL_FIT_SOURCE} u-boot.its
	else
		SPL_FIT_GENERATOR=`filt_val "CONFIG_SPL_FIT_GENERATOR" .config`
		# *.py是旧版本的生成器
		if [[ ${SPL_FIT_GENERATOR} == *.py ]]; then
			${SPL_FIT_GENERATOR} u-boot.dtb > u-boot.its
		else
			${SPL_FIT_GENERATOR} ${TEE_ARG} ${COMPRESSION_ARG} ${MCU_ARG} ${LOAD_ARG} > u-boot.its
		fi
	fi

	./tools/mkimage -f u-boot.its -E u-boot.itb >/dev/null 2>&1
	echo "已成功打包u-boot.itb! 输入文件: ${INI}"
	echo
}
~~~

~~~shell
            echo "pack loader with new: spl/u-boot-spl.bin"
        else
            ./make.sh loader ${ARG_INI_LOADER}
        fi
    else
        # 检查 RSA 密钥文件是否存在
        check_rsa_keys

        # 检查配置是否启用了 SPL FIT 签名
        if ! grep -q '^CONFIG_SPL_FIT_SIGNATURE=y' .config ; then
            echo "ERROR: CONFIG_SPL_FIT_SIGNATURE is disabled"
            exit 1
        fi

        # 如果启用了回滚保护，检查参数
        if grep -q '^CONFIG_SPL_FIT_ROLLBACK_PROTECT=y' .config ; then
            ARG_SPL_ROLLBACK_PROTECT="y"
            if [ -z ${ARG_ROLLBACK_IDX_UBOOT} ]; then
                echo "ERROR: No arg \"--rollback-index-uboot <n>\""
                exit 1
            fi
        fi

        # 设置回滚保护索引
        if [ "${ARG_SPL_ROLLBACK_PROTECT}" == "y" ]; then
            VERSION=`grep 'rollback-index' ${ITS_UBOOT} | awk -F '=' '{ printf $2 }' | tr -d ' '`
            sed -i "s/rollback-index = ${VERSION}/rollback-index = <${ARG_ROLLBACK_IDX_UBOOT}>;/g" ${ITS_UBOOT}
        fi

        # 通常 boot.img 先签名，RSA key 已在 u-boot.dtb 中，否则插入公钥
        if ! fdtget -l ${UBOOT_DTB} /signature >/dev/null 2>&1 ; then
            ${MKIMAGE} -f ${ITS_UBOOT} -k ${KEY_DIR} -K ${UBOOT_DTB} -E -p ${OFFS_DATA} -r ${ITB_UBOOT} -v ${ARG_VER_UBOOT}
            echo "## Adding RSA public key into ${UBOOT_DTB}"
        fi

        # 正式打包并签名 u-boot.itb
        ${MKIMAGE} -f ${ITS_UBOOT} -k ${KEY_DIR} -K ${SPL_DTB} -E -p ${OFFS_DATA} -r ${ITB_UBOOT} -v ${ARG_VER_UBOOT}
        mv ${SIG_BIN} ${SIG_UBOOT}

        # 如果需要烧录密钥哈希
        if [ "${ARG_BURN_KEY_HASH}" == "y" ]; then
            if grep -q '^CONFIG_SPL_FIT_HW_CRYPTO=y' .config ; then
                fdtput -tx ${SPL_DTB} ${SIGNATURE_KEY_NODE} burn-key-hash 0x1
            else
                echo "ERROR: --burn-key-hash requires CONFIG_SPL_FIT_HW_CRYPTO=y"
                exit 1
            fi
        fi

        # 检查回滚索引是否设置成功
        if [ "${ARG_SPL_ROLLBACK_PROTECT}" == "y" ]; then
            VERSION=`fdtget -ti ${ITB_UBOOT} /configurations/conf rollback-index`
            if [ "${VERSION}" != "${ARG_ROLLBACK_IDX_UBOOT}" ]; then
                echo "ERROR: Failed to set rollback-index for ${ITB_UBOOT}";
                exit 1
            fi
        fi

        # 检查密钥哈希是否设置成功
        if [ "${ARG_BURN_KEY_HASH}" == "y" ]; then
            if [ "`fdtget -ti ${SPL_DTB} ${SIGNATURE_KEY_NODE} burn-key-hash`" != "1" ]; then
                echo "ERROR: Failed to set burn-key-hash for ${SPL_DTB}";
                exit 1
            fi
        fi

        # 主机端校验签名
        if [ "${ARG_NO_CHECK}" != "y" ]; then
            if [ "${ARG_SPL_NEW}" == "y" ]; then
                 ${CHECK_SIGN} -f ${ITB_UBOOT} -k ${SPL_DTB} -s
            else
                spl_file="../rkbin/"`sed -n "/FlashBoot=/s/FlashBoot=//p" ${ARG_INI_LOADER}  |tr -d '\r'`
                offs=`fdtdump -s ${spl_file} | head -1 | awk -F ":" '{ print $2 }' | sed "s/ found fdt at offset //g" | tr -d " "`
                if [ -z ${offs}  ]; then
                    echo "ERROR: invalid ${spl_file} , unable to find fdt blob"
                fi
                offs=`printf %d ${offs} ` # 十六进制转十进制
                dd if=${spl_file} of=spl/u-boot-spl-old.dtb bs=${offs} skip=1 >/dev/null 2>&1
                ${CHECK_SIGN} -f ${ITB_UBOOT} -k spl/u-boot-spl-old.dtb -s
            fi
        fi

        # 最小化 u-boot-spl.dtb，清空部分属性但不删除
        if grep -q '^CONFIG_SPL_FIT_HW_CRYPTO=y' .config ; then
            fdtput -tx ${SPL_DTB} ${SIGNATURE_KEY_NODE} rsa,r-squared 0x0
            if grep -q '^CONFIG_SPL_ROCKCHIP_CRYPTO_V1=y' .config ; then
                fdtput -tx ${SPL_DTB} ${SIGNATURE_KEY_NODE} rsa,np 0x0
                fdtput -r ${SPL_DTB} ${SIGNATURE_KEY_NODE}/hash@np
            else
                fdtput -tx ${SPL_DTB} ${SIGNATURE_KEY_NODE} rsa,c 0x0
                fdtput -r ${SPL_DTB} ${SIGNATURE_KEY_NODE}/hash@c
            fi
        else
            fdtput -tx ${SPL_DTB} ${SIGNATURE_KEY_NODE} rsa,c 0x0
            fdtput -tx ${SPL_DTB} ${SIGNATURE_KEY_NODE} rsa,np 0x0
            fdtput -tx ${SPL_DTB} ${SIGNATURE_KEY_NODE} rsa,exponent-BN 0x0
            fdtput -r ${SPL_DTB} ${SIGNATURE_KEY_NODE}/hash@c
            fdtput -r ${SPL_DTB} ${SIGNATURE_KEY_NODE}/hash@np
        fi

        # 重新打包 spl
        if [ "${ARG_SPL_NEW}" == "y" ]; then
            cat spl/u-boot-spl-nodtb.bin > spl/u-boot-spl.bin
            if ! grep -q '^CONFIG_SPL_SEPARATE_BSS=y' .config ; then
                cat spl/u-boot-spl-pad.bin >> spl/u-boot-spl.bin
            fi
            cat ${SPL_DTB} >> spl/u-boot-spl.bin

            ./make.sh --spl ${ARG_INI_LOADER}
            echo "## pack loader with new: spl/u-boot-spl.bin"
        else
            ./make.sh loader ${ARG_INI_LOADER}
        fi

        if [ "${ARG_BURN_KEY_HASH}" == "y" ]; then
            echo "## ${SPL_DTB}: burn-key-hash=1"
        fi
    fi

    # 清理旧的 u-boot.itb、u-boot.img、u-boot-dtb.img 文件
    rm -f u-boot.itb u-boot.img u-boot-dtb.img
    # 将 u-boot.its 移动到 fit 目录
    mv ${ITS_UBOOT} ${FIT_DIR}
}
~~~

`u-boot.itb` 文件是 U-Boot 的 FIT（Flattened Image Tree）格式镜像，其主要作用如下：

1. **多镜像封装**
   它可以将 U-Boot 的多个组件（如 u-boot-nodtb.bin、设备树 dtb、TEE、BL31、MCU、loadable 等）封装在一个文件里，便于统一管理和加载。
2. **安全启动与签名**
   支持 RSA 签名和回滚保护等安全机制，保证镜像的完整性和可信性。启动时 SPL 会校验签名，只有合法的 u-boot.itb 才能被加载。
3. **灵活配置**
   通过 its 文件（如 u-boot.its）描述各个镜像的加载地址、属性、签名等，方便定制和扩展。
4. **平台适配**
   支持 ARM TrustZone、TEE、MCU 等多种安全和硬件配置，适用于复杂的嵌入式启动流程。

**总结**：
`u-boot.itb` 是 U-Boot 的安全启动入口镜像，集成了多种启动组件和安全机制，是现代 Rockchip 平台推荐的 U-Boot启动格式。

~~~shell
function fit_gen_uboot_img()
{
    # 获取传入的 ITB 文件名，如果未指定则使用默认的 u-boot.itb
    ITB=$1

    if [ -z ${ITB} ]; then
        ITB=${ITB_UBOOT}
    fi

    # 从 .config 文件中读取最大镜像数量和单个镜像最大容量（KB）
    ITB_MAX_NUM=`sed -n "/SPL_FIT_IMAGE_MULTIPLE/p" .config | awk -F "=" '{ print $2 }'`
    ITB_MAX_KB=`sed  -n "/SPL_FIT_IMAGE_KB/p" .config | awk -F "=" '{ print $2 }'`
    ITB_MAX_BS=$((ITB_MAX_KB*1024))   # 转换为字节
    ITB_BS=`ls -l ${ITB} | awk '{ print $5 }'`  # 获取实际文件大小

    # 检查 ITB 文件大小是否超出限制
    if [ ${ITB_BS} -gt ${ITB_MAX_BS} ]; then
        echo "ERROR: pack ${IMG_UBOOT} failed! ${ITB} actual: ${ITB_BS} bytes, max limit: ${ITB_MAX_BS} bytes"
        exit 1
    fi

    # 删除旧的 uboot.img 文件
    rm -f ${IMG_UBOOT}
    # 按最大数量循环，将 ITB 文件内容追加到 uboot.img，并按最大容量截断
    for ((i = 0; i < ${ITB_MAX_NUM}; i++));
    do
        cat ${ITB} >> ${IMG_UBOOT}
        truncate -s %${ITB_MAX_KB}K ${IMG_UBOOT}
    done
}
~~~

# U-Boot（RK3588 / ARMv8-A）启动流程最终版

> 适用场景：Rockchip RK3588 平台，ARM64（AArch64）路径；以 U-Boot 近年的主线组织方式为基准（Kconfig 体系，`board_f.c / board_r.c` 双阶段初始化）。

------

## 1. 入口如何确定（从 `Makefile` 到 `start.S`）

### 1.1 顶层 `Makefile` 与 `head-y`

顶层 `Makefile` 有如下关键片段：

```makefile
u-boot-init := $(head-y)
u-boot-main := $(libs-y)

u-boot: $(u-boot-init) $(u-boot-main) u-boot.lds FORCE
	+$(call if_changed,u-boot__)
```

- `u-boot-init` 由 `head-y` 决定，表示“镜像最前面、决定入口的目标文件”。
- 对 ARM 平台，`head-y` 在 `arch/arm/Makefile` 中定义。

### 1.2 `CONFIG_SYS_ARCH` / `CONFIG_SYS_CPU` 的来历

- 现代 U-Boot 由 **Kconfig** 生成配置，变量会出现在
   `include/generated/autoconf.mk`（供 make 使用）与 `include/generated/autoconf.h`（供 C/汇编使用）。
- `CONFIG_SYS_ARCH="arm"`、`CONFIG_SYS_CPU="armv8"` 由所选 SoC/板级 defconfig 决定。

> 你文中说“config 文件”可理解为 `.config`（Kconfig 的输出），或其导出的 `autoconf.mk/.h`。要点：这些配置不是手写散落的宏，而是 Kconfig 统一产物。

### 1.3 `head-y` 最终指向

在 `arch/arm/Makefile`（ARM 通用）：

```makefile
head-y := arch/arm/cpu/$(CPU)/start.o
```

结合 `CONFIG_SYS_CPU="armv8"`，则：

```
head-y = arch/arm/cpu/armv8/start.o
```

对应源码：`arch/arm/cpu/armv8/start.S`，这就是 **AArch64** 路径的真正入口。

> ✅ 你的结论“入口为 `arch/arm/cpu/armv8/start.S`”是正确的。

------

## 2. `start.S`（AArch64）详解

> 注意区分：AArch64 使用 **x0~x30** 等 64 位寄存器，且 **gd 全局数据寄存器通常是 x18**（由 `DECLARE_GLOBAL_DATA_PTR` 约定）。你原文有部分 ARM32（`r0/r9`）示例，下面全部按 ARM64 口径说明。

### 2.1 入口与 Boot0 hook

```asm
.globl  _start
_start:
#ifdef CONFIG_ENABLE_ARM_SOC_BOOT0_HOOK
#include <asm/arch/boot0.h>
#else
    b  reset
#endif
```

- 如果 SoC 需要非常早期的自定义处理（如 Rockchip 的极早期时钟/复位钩子），会走 `boot0.h`。
- 否则直接跳到 `reset`。

### 2.2 `save_boot_params`（弱符号）

```asm
reset:
    b  save_boot_params

WEAK(save_boot_params)
    b  save_boot_params_ret
ENDPROC(save_boot_params)
```

- 这是 **弱定义**：若 SoC/板级未提供自定义实现，默认啥也不做直接返回。
- 用途是“从前一阶段（ROM/SPL/BL31）拿启动参数/寄存器快照”，常见约定包括 *x0* 保留值（例如 Linux AArch64 要求 x0 = FDT 指针，但对 U-Boot proper 不强制）。

### 2.3 PIE 重定位修正（可选）

```asm
#if CONFIG_POSITION_INDEPENDENT
/* 遍历 .rela.dyn，对 R_AARCH64_RELATIVE (1027) 做加偏写回 */
#endif
```

- 仅当打开 `CONFIG_POSITION_INDEPENDENT`（多用于 SPL/小型镜像）才执行。
- 作用：运行地址 ≠ 链接地址 时，修正相对重定位项。

### 2.4 异常向量表与 EL 级别

```asm
adr x0, vectors
switch_el x1, 3f, 2f, 1f
3:  msr vbar_el3, x0
    mrs x0, scr_el3
    orr x0, x0, #0xf           // NS|IRQ|FIQ|EA
    msr scr_el3, x0
    msr cptr_el3, xzr          // 允许 FP/SIMD
    ...
    b 0f
2:  msr vbar_el2, x0
    msr cptr_el2, xzr          // 允许 FP/SIMD
    ...
    b 0f
1:  msr vbar_el1, x0
    mov x0, #3 << 20
    msr cpacr_el1, x0          // 允许 FP/SIMD
0:
```

- **RK3588 常见链路**：BootROM →（可选 TPL）→ SPL → TF-A（BL31，驻留 EL3）→ U-Boot proper 通常在 **EL2** 运行（也可能 EL3/EL1，取决于集成）。因此这里会写 `VBAR_EL2` 或 `EL3/EL1`。
- 同时设置 `SCR_EL3` 等，以允许中断与 FP/SIMD。

### 2.5 打开 I-Cache / 允许 SError

```asm
#ifndef CONFIG_SYS_ICACHE_OFF
    mov x1, #CR_I
#else
    mov x1, #0
#endif
switch_el ...
msr sctlr_elX, x0|CR_I
msr daifclr, #4   // 清除 A 位，允许 SError
isb
```

### 2.6 SMP 相关（可选）

```asm
#ifdef CONFIG_ARMV8_SET_SMPEN
mrs x0, S3_1_c15_c2_1  // cpuectlr_el1
orr x0, x0, #0x40      // 置 SMPEN
msr S3_1_c15_c2_1, x0
#endif
```

- 一些核（早期 Cortex-A 系列）需要手动置位 SMPEN 以保证一致性语义。

### 2.7 已知 errata 修复 + 低级硬件初始化

```asm
bl apply_core_errata
bl lowlevel_init
```

- `lowlevel_init` 通常在 `arch/arm/mach-rockchip/` 路径有平台实现，用于最早期时钟/Pin/安全设置等（非驱动模型阶段）。

### 2.8 多核启动（主/从核分流）

- 根据 `CONFIG_ARMV8_SPIN_TABLE` 或 `CONFIG_ARMV8_MULTIENTRY` 路径：
  - 主核进入后续流程；
  - 从核通常 `wfe` 等待释放地址或保留在 ROM/TF-A 控制下。
- **现实里**：RK3588 多由 TF-A 管理次核，U-Boot proper 常只在 CPU0 运行。

### 2.9 跳转到 C 世界过渡点 `_main`

```asm
master_cpu:
    bl  _main
```

- **答案你的疑问：`_main` 如何落到 `crt0.S`？**
   `_main` 是一个全局符号，由 `arch/arm/lib/crt0.S` 定义。链接时，`start.o` 引用 `_main`，连接器根据 `u-boot.lds` 和输入目标文件列表把它解析到 `crt0.S` 的实现。**不需要“搜索文件”，是链接器符号解析的自然结果**。

------

## 3. `crt0.S`：从“过渡点”到 C 运行时

> ARM64 路径与 ARM32 相似但寄存器不同。以下描述按 **AArch64** 口径，关键寄存器为 **x18 = gd**。

### 3.1 设置初始栈（Early stack）

- 分三种构建：`TPL / SPL / U-Boot proper`，优先级大致如下：
  - `CONFIG_TPL_STACK` → `CONFIG_SPL_STACK` → `CONFIG_SYS_INIT_SP_ADDR`
- 对 AArch64，会装入 x0 为 “top 地址”，调用 `board_init_f_alloc_reserve(top)` 计算并返回保留后的新栈顶，再把 **sp** 设为该值，同时把 **x18**（gd）暂定位到该区。

> 你原文的示例使用了 `r0/r9`（ARM32），概念对，但在 AArch64 实际上是 **x0/x18**。

### 3.2 早期全局数据（gd）与 early malloc

- 逻辑顺序（抽象）：
  1. `board_init_f_alloc_reserve(top)`：从 **top 向下**预留 `gd` 与 `malloc_f` 空间，返回新 top。
  2. `board_init_f_init_reserve(base)`：对 `gd` 清零、设置 `gd->malloc_base` 等。
  3. `board_init_f_boot_flags()`：弱定义，默认返回 0；通常不在此处写串口/地址等硬件细节（这些在后续 init 阶段完成）。

> 纠错①：你贴的 `board_init_f_init_reserve()` 示例里有两个小问题：
>
> - `memset(gd_ptr, '\0', sizeof(*gd));` 应该是 `sizeof(*gd_ptr)`；
> - `base += roundup(sizeof(struct(struct global_data), 16);` 明显拼写/括号错误，且 `roundup` 参数应是 `sizeof(struct global_data)`。
>    正确意图如下（伪代码）：

```c
void board_init_f_init_reserve(ulong base) {
    struct global_data *gd_ptr = (struct global_data *)base;
    memset(gd_ptr, 0, sizeof(*gd_ptr));
#if !defined(CONFIG_ARM)
    arch_setup_gd(gd_ptr);  // ARM64 一般在汇编里设置了 x18=gd
#endif
    base += ROUNDUP(sizeof(struct global_data), 16);
#if CONFIG_VAL(SYS_MALLOC_F_LEN)
    gd_ptr->malloc_base = base;
    base += CONFIG_VAL(SYS_MALLOC_F_LEN);
#endif
    /* 还会把 x18 指向 gd_ptr，具体在汇编/宏里处理 */
}
```

> 纠错②：`gd->serial.baudrate` / `gd->serial.addr` 这类成员并非主线 U-Boot 的 `gd_t` 标准字段，常见做法是 **`gd->baudrate`**，串口基址/时钟等由驱动/设备树提供，而不是塞进 `gd->serial.*`。

### 3.3 `board_init_f()`（**在 `common/board_f.c`，不是 `board_r.c`**）

- 你原文把文件路径写成 `common/board_r.c`，这处要修正：
   **`board_init_f()` 在 `common/board_f.c`**，负责 **重定位前** 的一系列早期初始化。
- 核心是依次运行 `init_sequence_f[]`，典型包括：
  - FDT 解析器（`fdtdec_setup`）、early malloc（`initf_malloc`）、日志、bootstage、早期 console record；
  - `arch_cpu_init` / `mach_cpu_init`（最早的 CPU/SoC init，不依赖驱动模型）；
  - 驱动模型预初始化（`initf_dm`）、定时器（`timer_init`）；
  - `env_init`、`init_baud_rate`、`serial_init`、`console_init_f`；
  - DRAM 初始化（`dram_init` / `dram_init_banksize`）；
  - **内存预留与重定位筹备**：`reserve_*` 系列（uboot镜像、malloc、FDT、video、board、stacks、mmu 等），计算 `gd->relocaddr / gd->reloc_off`；
  - `setup_reloc` 等。
- 到了这里，**DRAM 已可用**，即将搬家到 `gd->relocaddr`。

### 3.4 代码重定位与向量表

- `relocate_code(x0 = gd->relocaddr)`：在 `arch/arm/lib/relocate_64.S`，把 `.text/.data` 从当前执行位置 **拷贝到 DRAM 目标地址**，并设置 `gd->reloc_off`。
- `relocate_vectors()`：ARM64 上通常会在重定位后 **重新写一次 VBAR_ELx**（让向量表指向新地址）。不同版本实现位置略有差异（可能由 `relocate_vectors` 或 `c_runtime_cpu_setup` 统一完成）。

### 3.5 C 运行时的最后准备

- `c_runtime_cpu_setup()`：与 CPU 相关的最终设置。
- **清 BSS**：把 `__bss_start ~ __bss_end` 清零（ARM64 也会做）。
- **跳转到 `board_init_r(gd, relocaddr)`**：进入 “重定位后” 阶段。

------

## 4. `board_init_r()`（在 `common/board_r.c`）

> 这里才是“驱动模型全面生效、外设真正初始化、环境存储介质接入、网络/存储/命令行设备就绪”的阶段。

### 4.1 gd 指针处理

- ARM64 下 `gd` 通过 `x18` 维护；`board_init_r(gd, dest)` 里会确保全局可见的 `gd` 指向新位置（大多数架构由宏/编译器约定完成，无需手动赋值）。

### 4.2 `init_sequence_r[]`

按顺序调用，典型步骤（择要）：

- `initr_trace / initr_reloc / initr_reloc_global_data`：标记重定位完成、修正全局数据。
- **缓存/MMU**：`initr_caches`（ARM64 会开启 dcache/MMU，性能与一致性所需）。
- `initr_malloc`：正式的动态内存分配器（非 early 版）。
- `interrupt_init` / `initr_enable_interrupts`：GIC/中断。
- **驱动模型 DM**：`initr_dm`（真正枚举设备，绑定 probed 设备树）。
- `board_early_init_r` / `board_init`：板级差异初始化。
- **串口**：`initr_serial`（此时 console 完全作为设备抽象）。
- **环境变量**：`initr_env`（若不是“内核 DTB 模式”）；环境的介质（eMMC/SD/NAND/SPI-NOR）也在此或之后初始化。
- **电源/时钟/PCI/USB/MMC/网络**：按配置逐项初始化（`power_init_board / initr_pci / initr_mmc / initr_net ...`）。
- **最后进入主循环**：`run_main_loop`。

------

## 5. 主循环 `run_main_loop() / main_loop()`

- `run_main_loop()` 只是个外壳，反复调用 `main_loop()`。
- `main_loop()` 做这些事：
  1. `cli_init()` 初始化命令行；
  2. 执行 `preboot` 环境变量指定的命令；
  3. `bootdelay_process()` 处理倒计时与自动启动，得到 `bootcmd`；
  4. `autoboot_command(s)`：如果用户未打断，执行自动启动（典型是加载内核/DTB/ramdisk 启动 OS）；
  5. 如果被打断，进入 `cli_loop()` 命令行交互；
  6. 理论上不会退出；若退出，`panic("No CLI available")`。

------

## 6. “重定位”机制的关键点（ARM64）

- **两类场景不要混淆**：
  1. **PIE 修正**：若开启 `CONFIG_POSITION_INDEPENDENT`，在镜像开头就扫描 `.rela.dyn`，对 `R_AARCH64_RELATIVE` 逐条加“运行时偏移量”，确保当前位置执行无关。
  2. **搬家到 DRAM**：`relocate_code()` 把 `.text/.data` 从当前介质映射/缓存处拷到 `gd->relocaddr`；之后 `gd->reloc_off` 生效，符号地址统一“+偏移”理解。
- **BSS 清零时机**：通常在重定位后的 C 过渡代码中进行（确保未初始化全局变量为 0）。
- **VBAR 重指向**：重定位后需确保异常向量表基址寄存器指向新地址。

------

## 7. RK3588 / TF-A 相关的现实补充

- 量产/主线组合里，**TF-A（BL31）在 EL3**，它负责 PSCI、电源域、次核管理、安全世界等；**U-Boot proper 多在 EL2**，因此向量表一般写 `VBAR_EL2`。
- 次核通常由 TF-A 掌控；U-Boot 不一定主动唤醒所有 CPU。多核路径代码在很多量产配置中“走不到”。

## 9. 一图流（文字版）

1. **`start.S`**：早期 EL/向量表 → I-Cache/SError → SMPEN → errata → `lowlevel_init` → 主从核分流 → `bl _main`
2. **`crt0.S`**：设置 early SP → 预留 `gd + malloc_f` → `board_init_f()` → 筹备重定位 → `relocate_code`（搬家到 DRAM）→（重写 VBAR）→ 清 BSS → `board_init_r(gd, relocaddr)`
3. **`board_f.c`**：`init_sequence_f[]`（FDT/early malloc/console/DRAM/内存预留/计算 reloc 参数…）
4. **`board_r.c`**：`init_sequence_r[]`（缓存/MMU/中断/DM/串口/环境/存储/网络/外设…）→ `run_main_loop()`
5. **`main_loop()`**：preboot → bootdelay/bootcmd → 自动启动或 CLI 交互
