除了通过ioctl接口操作I2C设备外，还需要实现I2C的读写函数i2c_read_reg()和i2c_write_reg()。调用i2c_read_reg()函数时，可以向指定地址的从设备发送寄存器地址并读取相应的数据；调用i2c_write_reg()函数则可向指定寄存器写入数据，实现多字节写入操作，其代码如如代码清单14.12所示。

~~~c
//读取指定寄存器的值
int i2c_read_reg(int fd, unsigned char addr, unsigned char reg) {
    unsigned char data;
    //构造两个I2C消息：先发送寄存器地址，再读取数据
    struct i2c_msg msgs[2] = {
        {
            .addr = addr,
            .flags = 0,
            .len = 1,
            .buf = &reg
        },
        {
            .addr = addr,
            .flags = I2C_M_RD,
            .len = 1,
            .buf = &data
        }
    };
    //将消息组合成ioctl可用的数据结构
    struct i2c_rdwr_ioctl_data msgset = {
        .msgs = msgs,
        .nmsgs = 2
    };
    //执行I2C读写操作
    if (ioctl(fd, I2C_RDWR, &msgset) < 0) {
        perror("I2C read failed");
        return -1;
    }
    return data;
}
//向指定寄存器写入多个字节数据
int i2c_write_reg(int fd, unsigned char addr, unsigned char reg, const unsigned char *data, int len) {
    unsigned char buf[256];
    //检查数据长度是否超出缓冲区大小（256字节）
    if (len >= sizeof(buf)) return -1;
    //第一个字节为寄存器地址，后面跟随要写入的数据
    buf[0] = reg;
    memcpy(&buf[1], data, len);
    //构造I2C消息：发送寄存器地址+数据
    struct i2c_msg msg = {
        .addr = addr,
        .flags = 0,
        .len = len + 1,   //寄存器地址+数据长度
        .buf = buf
    };
    //将消息打包成ioctl可识别的结构体
    struct i2c_rdwr_ioctl_data msgset = {
        .msgs = &msg,
        .nmsgs = 1
    };
    //执行I2C写操作
    if (ioctl(fd, I2C_RDWR, &msgset) < 0) {
        perror("I2C write failed");
        return -1;
    }
    return 0;
}

~~~c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <fcntl.h>
#include <unistd.h>
#include <poll.h>
#include <signal.h>

#define MAX_DEVICES 5
#define POLL_TIMEOUT -1  // 无限等待

// 设备路径数组
static const char* device_paths[MAX_DEVICES] = {
    "/dev/gpioirq162",
    "/dev/gpioirq164",
    "/dev/gpioirq168",
    "/dev/gpioirq204",
    "/dev/gpioirq205"
};

// 设备状态数组
static int device_fds[MAX_DEVICES] = {-1};
static unsigned int device_counts[MAX_DEVICES] = {0};
static volatile sig_atomic_t running = 1;

// 信号处理函数
static void signal_handler(int signum) {
    running = 0;
}

int main() {
    struct pollfd fds[MAX_DEVICES];
    int i, ret, nready;
    unsigned int delta;
    
    // 安装信号处理
    signal(SIGINT, signal_handler);
    signal(SIGTERM, signal_handler);
    
    // 打开所有设备
    for (i = 0; i < MAX_DEVICES; i++) {
        device_fds[i] = open(device_paths[i], O_RDONLY);
        if (device_fds[i] < 0) {
            printf("无法打开设备 %s: %m\n", device_paths[i]);
            device_fds[i] = -1;
        } else {
            printf("已打开设备: %s (fd=%d)\n", device_paths[i], device_fds[i]);
        }
    }
    
    // 初始化pollfd数组
    for (i = 0; i < MAX_DEVICES; i++) {
        if (device_fds[i] >= 0) {
            fds[i].fd = device_fds[i];
            fds[i].events = POLLIN;
        } else {
            fds[i].fd = -1; // 标记为无效
        }
    }
    
    printf("开始监测%d个设备的中断事件，按Ctrl+C退出\n", MAX_DEVICES);
    
    // 主循环
    while (running) {
        // 等待任意设备有中断事件
        nready = poll(fds, MAX_DEVICES, POLL_TIMEOUT);
        
        if (nready < 0) {
            if (errno == EINTR && running)
                continue;
            perror("poll失败");
            break;
        } else if (nready == 0) {
            // 超时，这里设置为无限等待，理论上不会到达
            continue;
        }
        
        // 检查每个设备是否有事件
        for (i = 0; i < MAX_DEVICES; i++) {
            if (device_fds[i] < 0)
                continue;
                
            if (fds[i].revents & POLLIN) {
                // 读取中断事件
                ret = read(device_fds[i], &delta, sizeof(delta));
                if (ret == sizeof(delta)) {
                    device_counts[i] += delta;
                    printf("[%s] 检测到%d次中断，累计%d次\n", 
                           device_paths[i], delta, device_counts[i]);

	__u16 flags;
#define I2C_M_RD		0x0001	/* read data, from slave to master */
					/* I2C_M_RD is guaranteed to be 0x0001! */
#define I2C_M_TEN		0x0010	/* this is a ten bit chip address */
#define I2C_M_DMA_SAFE		0x0200	/* the buffer of this message is DMA safe */

					/* userspace buffers are copied anyway */
#define I2C_M_RECV_LEN		0x0400	/* length will be first received byte */

#define I2C_M_IGNORE_NAK	0x1000	/* if I2C_FUNC_PROTOCOL_MANGLING */
#define I2C_M_REV_DIR_ADDR	0x2000	/* if I2C_FUNC_PROTOCOL_MANGLING */
#define I2C_M_NOSTART		0x4000	/* if I2C_FUNC_NOSTART */
#define I2C_M_STOP		0x8000	/* if I2C_FUNC_PROTOCOL_MANGLING */
	__u16 len;		/* msg length				*/
	__u8 *buf;		/* pointer to msg data			*/
};

### struct i2c_msg 结构体详细解析

#### 一、结构体概述

`struct i2c_msg` 是 Linux 内核中用于定义 I2C 通信消息的核心结构体，主要用于描述一次 I2C 数据传输的具体参数，包括从设备地址、传输标志、数据长度及数据缓冲区等信息。它是 I2C 驱动开发中与硬件交互的关键数据结构。

#### 二、结构体成员详解

```c
struct i2c_msg {
    __u16 addr;     /* 从设备地址 */
    __u16 flags;    /* 传输标志位，包含多种操作模式 */
    __u16 len;      /* 消息数据长度 */
    __u8 *buf;      /* 指向消息数据的缓冲区指针 */
};
```

##### 1. addr 成员

- **含义**：从设备的地址（slave address），用于指定 I2C 通信的目标设备。
- **数据类型**：`__u16`（无符号 16 位整数）
- **使用场景**：在 I2C 总线中，每个设备通过唯一地址识别，通常为 7 位或 10 位地址。当使用 10 位地址时，需配合 `I2C_M_TEN` 标志位。

##### 2. flags 成员

- **含义**：传输标志位，通过位运算组合不同标志来控制 I2C 通信行为。

- **数据类型**：`__u16`（无符号 16 位整数）

- 常用标志位解析：

  - `I2C_M_RD (0x0001)`：
    表示读操作（从从设备到主设备传输数据），该标志位固定为 0x0001。
  - `I2C_M_TEN (0x0010)`：
    标识使用 10 位从设备地址（默认是 7 位地址模式）。
  - `I2C_M_DMA_SAFE (0x0200)`：
    表示消息缓冲区可安全用于 DMA（直接内存访问），仅在内核空间有效（用户空间缓冲区会被复制，该标志无意义）。
  - `I2C_M_RECV_LEN (0x0400)`：
    接收数据时，将第一个字节作为实际数据长度（适用于变长数据接收）。
  - `I2C_M_NO_RD_ACK (0x0800)`：
    读操作时不发送 ACK（应答信号），需硬件支持 `I2C_FUNC_PROTOCOL_MANGLING` 功能。
  - `I2C_M_IGNORE_NAK (0x1000)`：
    忽略从设备的 NAK（非应答信号），继续传输，适用于特定协议场景。
  - `I2C_M_REV_DIR_ADDR (0x2000)`：
    反转传输方向时重新发送地址，用于特殊协议处理。
  - `I2C_M_NOSTART (0x4000)`：
    不发送 START 信号（适用于连续传输场景，需硬件支持 `I2C_FUNC_NOSTART`）。
  - `I2C_M_STOP (0x8000)`：
    传输完成后发送 STOP 信号，用于控制总线时序。

##### 3. len 成员

- **含义**：消息数据的长度（单位：字节）。
- **数据类型**：`__u16`（无符号 16 位整数）
- **注意事项**：当使用 `I2C_M_RECV_LEN` 标志时，实际数据长度由接收的第一个字节决定，此时 `len` 需设置为预期最大长度。

##### 4. buf 成员

- **含义**：指向数据缓冲区的指针，用于存储待发送或接收的数据。

- **数据类型**：`__u8 *`（无符号 8 位整数指针）

- 使用场景

  ：

  - 写操作时，`buf` 存储主设备发送给从设备的数据；
  - 读操作时，`buf` 用于接收从设备返回的数据。

#### 三、典型应用场景

##### 1. 单字节写操作（例：向设备寄存器写入数据）

```c
struct i2c_msg msg = {
    .addr = 0x50,         /* 从设备地址（7位） */
    .flags = 0,           /* 默认为写操作（无 I2C_M_RD 标志） */
    .len = 2,             /* 数据长度：寄存器地址 + 写入值 */
    .buf = (u8[]){0x01, 0xAA}  /* 写入寄存器 0x01 的值为 0xAA */
};
```

##### 2. 多字节读操作（例：读取设备传感器数据）

```c
struct i2c_msg msgs[2] = {
    /* 第一帧：发送寄存器地址（写操作） */
    {
        .addr = 0x50,
        .flags = 0,
        .len = 1,
        .buf = (u8[]){0x02}  /* 读取寄存器 0x02 的数据 */
    },
    /* 第二帧：读取数据（读操作） */
    {
        .addr = 0x50,
        .flags = I2C_M_RD,
        .len = 4,
        .buf = (u8[4]){0}    /* 存储读取的 4 字节数据 */
    }
};
```

#### 四、标志位组合示例

| 场景                 | 标志位组合                     | 说明                                                |
| -------------------- | ------------------------------ | --------------------------------------------------- |
| 标准 7 位地址写操作  | `flags = 0`                    | 常规写数据，发送 START 和 STOP 信号                 |
| 10 位地址读操作      | `flags = I2C_M_RD | I2C_M_TEN` | 使用 10 位从设备地址，并执行读操作                  |
| 连续传输（无 START） | `flags = I2C_M_NOSTART`        | 在多帧传输中，后续帧不发送 START 信号（需硬件支持） |
| 忽略 NAK 继续传输    | `flags = I2C_M_IGNORE_NAK`     | 从设备无应答时不中断传输，适用于容错性要求高的场景  |

#### 五、与 I2C 核心函数的关联

在 Linux 内核中，`struct i2c_msg` 通常与 `i2c_transfer()` 函数配合使用，该函数可批量处理多个 `i2c_msg` 结构体：

```c
/* 函数原型 */
int i2c_transfer(struct i2c_client *client, struct i2c_msg *msgs, int num);
```



- `client`：I2C 客户端设备句柄（包含适配器和从设备信息）；
- `msgs`：`i2c_msg` 数组，支持多帧连续传输（如先写地址再读数据）；
- `num`：消息数组的长度（即传输的帧数）。

#### 六、总结

`struct i2c_msg` 是 Linux I2C 子系统中定义传输行为的基础结构，通过 `addr` 指定通信目标，`flags` 控制传输模式，`len` 和 `buf` 管理数据交互。理解该结构体的成员及标志位组合，是开发 I2C 设备驱动（如传感器、EEPROM、ADC 等）的核心基础。在实际应用中，需根据硬件特性和协议要求合理配置各成员，确保数据传输的准确性和可靠性。`struct i2c_msg` 是 Linux 内核中用于 I2C（Inter-Integrated Circuit）总线通信的核心数据结构，定义在 `<linux/i2c.h>` 头文件中。它封装了一次 I2C 传输的基本要素，支持灵活配置各种传输选项。以下是对该结构体的详细解析：

    - 在传输结束时发送 STOP 信号。

#### 3. **`len` - 消息长度**

- **类型**：`__u16`（16 位无符号整数）

- 含义

- 含义

### SPI_MSGSIZE 宏定义解析

这个宏 `SPI_MSGSIZE(N)` 是用于计算 SPI 通信中批量传输消息的总大小，并进行合法性检查。下面逐步解析其功能和原理：

### 宏定义拆解

```c
SPI_MSGSIZE(N) \
    ((((N)*(sizeof (struct spi_ioc_transfer))) < (1 << _IOC_SIZEBITS)) \
        ? ((N)*(sizeof (struct spi_ioc_transfer))) : 0)
```

#### 核心组成部分

1. **`N \* sizeof(struct spi_ioc_transfer)`**
   计算 N 个`struct spi_ioc_transfer`结构体的总字节数。其中：
   - `N` 是传输消息的数量
   - `struct spi_ioc_transfer` 是 SPI 批量传输的基本单元（见下文）
2. **`(1 << _IOC_SIZEBITS)`**
   通过左移运算得到 ioctl 系统调用允许的最大数据大小。其中：
   - `_IOC_SIZEBITS` 是内核中定义的常量（通常为 14 或 16）
   - `1 << _IOC_SIZEBITS` 表示最大支持的字节数（如 14 对应 16KB，16 对应 64KB）
3. **条件判断**
   若总大小不超过系统限制，返回实际大小；否则返回 0（表示非法大小）

### 关键结构体与常量说明

#### 1. `struct spi_ioc_transfer` 结构体

```c
struct spi_ioc_transfer {
    __u64          tx_buf;        // 发送缓冲区地址
    __u64          rx_buf;        // 接收缓冲区地址
    __u32          len;           // 传输字节数
    __u32          speed_hz;      // 本次传输的时钟频率
    __u16          delay_usecs;   // 传输延迟（微秒）
    __u8           bits_per_word; // 字长
    __u8           cs_change;     // 片选信号变化标志
    __u8           pad[7];        // 保留字段
};
```

- 该结构体描述一次 SPI 传输的参数，包括缓冲区、传输长度、速度等
- `sizeof(struct spi_ioc_transfer)` 通常为 32 字节（不同架构可能有差异）

#### 2. `_IOC_SIZEBITS` 常量

- 定义在 Linux 内核头文件 `include/uapi/asm-generic/ioctl.h` 中
- 表示 ioctl 命令中数据大小字段的位数，决定了最大传输尺寸
- 常见值：
  - `_IOC_SIZEBITS = 14` 时，最大传输尺寸为 `1 << 14 = 16384` 字节（16KB）
  - `_IOC_SIZEBITS = 16` 时，最大传输尺寸为 `1 << 16 = 65536` 字节（64KB）

### 宏的功能与用途

#### 1. 计算批量传输的总大小

当需要通过`SPI_IOC_MESSAGE(N)`命令进行 N 次 SPI 消息传输时，`SPI_MSGSIZE(N)`用于计算这些消息所需的总内存空间：

```c
struct spi_ioc_transfer transfers[N];
int size = SPI_MSGSIZE(N);  // 计算N个transfer的总大小
```

#### 2. 合法性检查

- 防止传入过大的 N 值导致内存溢出或系统错误
- 若 N 过大（如`N * sizeof(transfer) >= 1 << _IOC_SIZEBITS`），宏返回 0，提示调用者参数非法

#### 3. 与 ioctl 命令配合

在 SPI 编程中，该宏通常用于构造 ioctl 命令的参数长度：

```c
#define SPI_IOC_MESSAGE(N) \
    _IOWR(SPI_IOC_MAGIC, 8 + (N), struct spi_ioc_transfer[0])

// 使用示例
struct spi_ioc_transfer xfers[2];
ioctl(fd, SPI_IOC_MESSAGE(2), xfers);  // 传输2条消息
```

### `spi_ioc_transfer`结构体详解

#### 结构体背景与作用

`spi_ioc_transfer`是 Linux 系统中用于 SPI (Serial Peripheral Interface) 通信的核心结构体，定义在内核头文件`linux/spi/spidev.h`中。它主要用于通过`ioctl`系统调用与 SPI 设备进行数据传输，支持全双工通信模式，允许在一次传输中同时发送和接收数据。

#### 结构体定义

在 Linux 内核中，该结构体的典型定义如下：

```c
struct spi_ioc_transfer {
    __u64         tx_buf;         /* 发送缓冲区地址 */
    __u64         rx_buf;         /* 接收缓冲区地址 */
    __u32         len;            /* 传输的字节数 */
    __u32         speed_hz;       /* 传输速度(Hz) */
    __u16         delay_usecs;    /* 片选激活到传输开始的延迟(微秒) */
    __u8          bits_per_word;  /* 每个字的位数 */
    __u8          cs_change;      /* 传输后是否改变片选状态 */
    __u8          pad;            /* 填充字节，保持结构体对齐 */
    __u8          reserved[6];    /* 保留字段，必须为0 */
};
```

#### 字段详细解析

| 字段名称        | 数据类型  | 描述与用途                                                   |
| --------------- | --------- | ------------------------------------------------------------ |
| `tx_buf`        | `__u64`   | 发送数据缓冲区的物理地址或虚拟地址（取决于内核配置）。 若无需发送数据，可设为 0。 |
| `rx_buf`        | `__u64`   | 接收数据缓冲区的物理地址或虚拟地址。 若无需接收数据，可设为 0。 |
| `len`           | `__u32`   | 本次传输的字节数，必须大于 0。                               |
| `speed_hz`      | `__u32`   | 期望的传输速度（单位：Hz）。 实际速度可能受硬件限制，需通过驱动确认。 |
| `delay_usecs`   | `__u16`   | 片选信号激活后到数据传输开始的延迟时间（单位：微秒）。 某些设备需要此延迟来稳定状态。 |
| `bits_per_word` | `__u8`    | 每个数据字的位数，常见值为 8（字节）或 16（半字），需与设备兼容。 |
| `cs_change`     | `__u8`    | 布尔值（0 或 1），表示本次传输完成后是否取消片选。 0：保持片选激活 1：取消片选 |
| `pad`           | `__u8`    | 填充字节，用于结构体对齐，无需手动设置。                     |
| `reserved[6]`   | `__u8[6]` | 保留字段，必须全部设为 0，否则可能导致不可预测的行为。       |

#### 关键字段使用说明

##### 1. 数据缓冲区字段（tx_buf, rx_buf）

- **物理地址与虚拟地址**：
  - 在大多数用户空间程序中，使用虚拟地址（即内存指针）
  - 若在内核模块中使用，可能需要物理地址
  - 示例：`tx_buf = (unsigned long)tx_buffer;`
- **全双工传输**：
  - 同时设置`tx_buf`和`rx_buf`可实现全双工通信
  - SPI 是同步通信，发送和接收同时进行
  - 发送和接收的字节数必须相等（由`len`指定）

##### 2. 传输速度（speed_hz）

- 可在每次传输时指定不同速度
- 实际速度受以下因素限制：
  - 硬件 SPI 控制器的最大速度
  - 设备支持的最大速度
  - 内核驱动的配置
- 在`spi_configure`函数中设置的全局速度会被单次传输的`speed_hz`覆盖

##### 3. 片选控制（cs_change）

- ```
  cs_change = 0
  ```

  ：

  - 保持片选信号激活
  - 适用于需要连续传输的场景（如多字节数据传输）

- ```
  cs_change = 1
  ```

  ：

  - 传输完成后取消片选
  - 适用于单次传输或需要片选隔离的场景

##### 4. 延迟（delay_usecs）

- 某些 SPI 设备需要片选激活后等待一段时间才能开始传输
- 典型应用场景：
  - EEPROM 等存储设备
  - 某些传感器芯片
- 延迟时间根据设备数据手册设置，通常为 0-100 微秒

#### 使用示例

下面是一个使用`spi_ioc_transfer`结构体进行 SPI 传输的完整示例：

```c
#include <stdio.h>
#include <stdlib.h>
#include <fcntl.h>
#include <unistd.h>
#include <sys/ioctl.h>
#include <linux/spi/spidev.h>

int main() {
    const char *dev = "/dev/spidev0.0";
    int fd = open(dev, O_RDWR);
    if (fd < 0) {
        perror("open");
        return EXIT_FAILURE;
    }
    
    // 配置SPI设备基本参数
    uint8_t mode = 0;         // SPI模式0
    uint8_t bits = 8;         // 8位/字
    uint32_t speed = 1000000; // 1MHz
    
    ioctl(fd, SPI_IOC_WR_MODE, &mode);
    ioctl(fd, SPI_IOC_WR_BITS_PER_WORD, &bits);
    ioctl(fd, SPI_IOC_WR_MAX_SPEED_HZ, &speed);
    
    // 准备传输数据
    uint8_t tx_buf[4] = {0x01, 0x02, 0x03, 0x04};
    uint8_t rx_buf[4] = {0};
    
    // 初始化spi_ioc_transfer结构体
    struct spi_ioc_transfer xfer = {
        .tx_buf = (unsigned long)tx_buf,
        .rx_buf = (unsigned long)rx_buf,
        .len = 4,
        .speed_hz = 1000000,  // 可覆盖全局速度
        .delay_usecs = 0,
        .bits_per_word = 8,
        .cs_change = 1,       // 传输后取消片选
    };
    
    // 执行SPI传输
    int ret = ioctl(fd, SPI_IOC_MESSAGE(1), &xfer);
    if (ret < 0) {
        perror("ioctl");
        close(fd);
        return EXIT_FAILURE;
    }
    
    // 打印接收结果
    printf("发送数据: ");
    for (int i = 0; i < 4; i++) {
        printf("%02X ", tx_buf[i]);
    }
    printf("\n接收数据: ");
    for (int i = 0; i < 4; i++) {
        printf("%02X ", rx_buf[i]);
    }
    printf("\n");
    
    close(fd);
    return 0;
}
```

#### 高级用法：批量传输

`spi_ioc_transfer`结构体支持批量传输，通过`SPI_IOC_MESSAGE(N)`宏实现：

```c
// 定义多个传输结构体
struct spi_ioc_transfer xfers[2] = {
    {
        .tx_buf = (unsigned long)tx_buf1,
        .rx_buf = (unsigned long)rx_buf1,
        .len = 4,
        .cs_change = 0,  // 保持片选
    },
    {
        .tx_buf = (unsigned long)tx_buf2,
        .rx_buf = (unsigned long)rx_buf2,
        .len = 8,
        .cs_change = 1,  // 取消片选
    }
};

// 执行批量传输
ioctl(fd, SPI_IOC_MESSAGE(2), xfers);
```

#### 常见问题与解决方案

1. **缓冲区地址问题**：
   - 用户空间程序中使用`(unsigned long)`强制转换指针即可
   - 内核模块中可能需要使用`phys_to_virt`或`dma_map_single`获取物理地址
2. **传输速度异常**：
   - 检查设备驱动是否支持指定速度
   - 使用`SPI_IOC_RD_MAX_SPEED_HZ`获取实际支持的最大速度
   - 单次传输的`speed_hz`会覆盖全局配置
3. **数据传输错误**：
   - 确认`bits_per_word`与设备兼容
   - 检查`cs_change`设置是否符合设备要求
   - 增加适当的`delay_usecs`解决时序问题
4. **结构体对齐问题**：
   - 确保编译时结构体对齐正确（通常由编译器自动处理）
   - 避免手动修改`pad`和`reserved`字段

#### 与内核版本的兼容性

- `spi_ioc_transfer`结构体在不同 Linux 内核版本中可能有细微差异
- 较新的内核版本（如 4.0+）可能包含额外的保留字段
- 为确保兼容性，建议：
  - 始终将保留字段设为 0
  - 使用内核头文件中定义的结构体
  - 避免直接访问未文档化的字段

通过深入理解`spi_ioc_transfer`结构体的各个字段及其作用，你可以更灵活地控制 SPI 通信过程，解决传输速度、时序控制等方面的问题，实现与各种 SPI 设备的可靠通信。

#include <stdio.h>
#include <stdlib.h>
#include <fcntl.h>
#include <unistd.h>
#include <sys/ioctl.h>
#include <linux/spi/spidev.h>

/* SPI设备句柄 */
static int spi_fd = -1;

/* SPI设备初始化函数 */
int spi_init(const char *device, char mode, unsigned long speed_hz, char bits) {
    // 打开SPI设备
    spi_fd = open(device, O_RDWR);
    if (spi_fd < 0) {
        perror("打开SPI设备失败");
        return -1;
    }
    
    // 配置SPI参数
    if (ioctl(spi_fd, SPI_IOC_WR_MODE, &mode) < 0) {
        perror("设置SPI模式失败");
        close(spi_fd);
        spi_fd = -1;
        return -1;
    }
    
    if (ioctl(spi_fd, SPI_IOC_WR_BITS_PER_WORD, &bits) < 0) {
        perror("设置字长失败");
        close(spi_fd);
        spi_fd = -1;
        return -1;
    }
    
    if (ioctl(spi_fd, SPI_IOC_WR_MAX_SPEED_HZ, &speed_hz) < 0) {
        perror("设置时钟频率失败");
        close(spi_fd);
        spi_fd = -1;
        return -1;
    }
    
    printf("SPI设备初始化成功: 模式=%d, 速度=%luHz, 字长=%d位\n", 
           mode, speed_hz, bits);
    return 0;
}

/* SPI单字节收发函数 */
int spi_transfer(char tx_data, char *rx_data) {
    if (spi_fd < 0) {
        printf("错误：SPI设备未初始化\n");
        return -1;
    }
    
    // 准备传输结构体
    struct spi_ioc_transfer xfer = {
        .tx_buf = (unsigned long)&tx_data,  // 发送字节地址
        .rx_buf = (unsigned long)rx_data,   // 接收字节地址
        .len = 1,                          // 传输1字节
        .cs_change = 1,                    // 传输后取消片选
    };
    
    // 执行SPI传输
    if (ioctl(spi_fd, SPI_IOC_MESSAGE(1), &xfer) < 0) {
        perror("SPI传输失败");
        return -1;
    }
    
    return 0;
}

/* SPI设备关闭函数 */
void spi_close() {
    if (spi_fd >= 0) {
        close(spi_fd);
        spi_fd = -1;
    }
}

int main() {
    const char *dev = "/dev/spidev0.0";  // SPI设备节点
    char tx_data = 0xAA;                 // 要发送的数据
    char rx_data = 0;                    // 接收的数据
    
    // 初始化SPI设备（Mode 0, 10MHz, 8位字长）
    if (spi_init(dev, 0, 10000000, 8) < 0) {
        return EXIT_FAILURE;
    }
    
    // 执行单字节收发
    if (spi_transfer(tx_data, &rx_data) < 0) {
        spi_close();
        return EXIT_FAILURE;
    }
    
    // 验证结果
    printf("发送数据: 0x%02X\n", (unsigned char)tx_data);
    printf("接收数据: 0x%02X\n", (unsigned char)rx_data);
    
    // 关闭设备
    spi_close();
    return 0;
}
