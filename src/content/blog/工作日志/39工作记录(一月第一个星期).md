~~~c
void decode_outputs(float *prob,std::string &plate_no)
{
    int pre_index = 0;
    int time_step = 21;   //时间步长，这里值得是最长的字符长度
    int num_char=78; //字符类别数  车牌78种类别字符
    for(int i = 0; i<time_step;i++)
    {
        float *row_ptr = prob+num_char*i;
        int max_value = row_ptr[0];
        // printf("%lf\n",max_value);
        int max_index = 0;
        for(int j = 1; j<num_char;j++)
        {
            if (row_ptr[j]>max_value)
            {
                max_value= row_ptr[j];
                max_index = j;
            }
        }
        if (max_index!=0 && max_index!=pre_index)
        {
            plate_no+=plate_string[max_index];
        }
        pre_index= max_index;
    }
}
~~~
