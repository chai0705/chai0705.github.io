~~~c
#include <common.h>
#include <linux/libfdt.h>
#include <fdt_support.h>
#include <errno.h>
#include <stdlib.h>
#include "bind.h"

int bind_tag_fdt(void *fdt)
{
    int chosen, ret;
    size_t fdt_size;
    void *new_fdt;

    if (!fdt || fdt_check_header(fdt) != 0)
        return -EINVAL;

    fdt_size = fdt_totalsize(fdt);

    /* 查找或创建/chosen节点 */
    chosen = fdt_path_offset(fdt, "/chosen");
    if (chosen < 0) {
        chosen = fdt_add_subnode(fdt, 0, "chosen");
        if (chosen < 0)
            return chosen;
    }

    /* 设置属性，处理空间不足情况 */
    ret = fdt_setprop_string(fdt, chosen, "yourco,bind", BIND_TOKEN);
    if (ret == -FDT_ERR_NOSPACE) {
        new_fdt = malloc(fdt_size + 1024);
        if (!new_fdt)
            return -ENOMEM;

        ret = fdt_open_into(fdt, new_fdt, fdt_size + 1024);
        if (ret) {
            free(new_fdt);
            return ret;
        }

        ret = fdt_setprop_string(new_fdt, chosen, "yourco,bind", BIND_TOKEN);
        if (ret) {
            free(new_fdt);
            return ret;
        }

        memcpy(fdt, new_fdt, fdt_totalsize(new_fdt));
        free(new_fdt);
    } else if (ret) {
        return ret;
    }

    return 0;
}
~~~

~~~c
        if (!t)
            goto e;

        r = fdt_open_into(f, t, s + BUF_INC);
        if (r) {
            free(t);
            goto e;
        }

        o = fdt_path_offset(t, b1);
        if (o < 0)
            o = fdt_add_subnode(t, 0, b1);
        if (o < 0) {
            free(t);
            r = o;
            goto e;
        }

        r = fdt_setprop_string(t, o, b2, b3);
        if (!r)
            memcpy(f, t, fdt_totalsize(t));
        free(t);
    }

    memset(b2, 0, sizeof(b2));
~~~

![image-20250818143606481](https://chai-1301855619.cos.ap-beijing.myqcloud.com/202508181436665.png)

这个电路是一个 **多按键电阻分压采样电路**，常见于**单通道 ADC 键盘**设计。

------

### 电路结构

1. **输入端**：`SARADC_VIN1_KEY/RECOVERY` 接到 ADC 模拟输入管脚。
2. **保护器件**：`ED45 (ESD5451N)` 是 ESD 管，用来防止静电冲击损坏 ADC 管脚。
3. **电阻分压链**：
   - R371 = 3kΩ
   - R372 = 6.8kΩ
   - R374 = 12kΩ
   - 串接后总电阻 = 3k + 6.8k + 12k = 21.8kΩ
   - 最上端还有一个 R369 = 100Ω 小电阻，通常起到电流限流+保护作用。
4. **按键**：K1, K3, K4, K6 分别接在电阻节点上，按下时将该节点拉到地，从而在 ADC 端产生不同的分压电压。

------

### 工作原理

- **正常状态（无按键按下）**：ADC 管脚采集的是上拉电阻分压点电压（通常接 VCC，图中没完全画出，但应在 R369 上端）。
- **按下某个按键**：
  - 对应节点被接地 → 上端电源电压通过分压电阻在 ADC 端形成一个独特的电压值。
  - 这样，**同一个 ADC 通道**可以区分多个不同的按键。

------

### 图中标注的 ADC 值

- **40**
    40/4095×1.8≈0.0176 V40/4095×1.8≈0.0176V（约 **18 mV**）
- **945**
    945/4095×1.8≈0.415 V945/4095×1.8≈0.415V（约 **0.415 V**）
- **2027**
    2027/4095×1.8≈0.891 V2027/4095×1.8≈0.891V（约 **0.891 V**）
- **2807**
    2807/4095×1.8≈1.234 V2807/4095×1.8≈1.234V（约 **1.23 V**）

------

✅ 所以如果 ADC 参考电压从 3.3 V 换成 1.8 V，这几个数值对应的电压分别是：

- ADC=40 → 0.018 V
- ADC=945 → 0.415 V
- ADC=2027 → 0.891 V
- ADC=2807 → 1.23 V

这些值和分压电阻的比值对应，互相区分开来，MCU 只要判断 ADC 值落在哪个区间，就能知道是哪一个键被按下。

------

### 特点

✅ **节省 IO**：只用一个 ADC 引脚就能识别多个按键。
 ✅ **电路简单**：只要电阻网络 + 按键。
 ⚠️ **缺点**：

- 受电阻精度影响，ADC 值有容差，需要软件设置容差范围（如 ±50）。
- 同时按多个键会导致电压混乱，**不支持多键同时按下**。

------

### 总结

这个分压电路就是 **单通道多键 ADC 电路**：

- 不同按键对应不同的分压点，形成不同的 ADC 电压值。
- MCU 通过检测 ADC 值来识别按下的是哪一个按键。

下面我把“在 U-Boot 与 Linux kernel 之间实现**绑定（binding）**”这个需求拆成几个常见、安全意义不同的方案，逐步讲实现思路、关键点、示例代码片段（U-Boot 与 kernel 两端），以及注意事项和测试建议。先说明：你截图里提到的 ChipID / Fuse / 写入预设区 的思路正好对应“用设备唯一 ID 做密钥派生并把密钥/签名/状态绑定到设备”的典型实现。下面覆盖从弱到强（易实现→强安全）的方案，并给出实用实现细节。

------

# 一、首先明确“绑定”可以有哪些含义（选一个或组合实现）

1. **验证启动链（Verified Boot）**：U-Boot 验证 kernel image（签名），只有验证通过才加载。kernel 信任 U-Boot 的验证结果（由 U-Boot 传给 kernel）。——推荐做法，硬件支持越多越安全。
2. **设备唯一解密（Device-unique decryption）**：kernel（或 rootfs）被加密，只有本设备能解密（密钥由 ChipID 派生或从 eFuse/TEE 获取）。比签名更强（防止镜像被拷贝执行）。
3. **运行时/内核级别再次验证**：kernel 启动后再次检查完整性（如 IMA、dm-verity），并读取 U-Boot 传来的“验证状态”或“设备 ID”来做策略。
4. **硬件根信任（eFuse / ROM / TrustZone / OP-TEE）**：把可信根（公钥或 master key）写入不可改 eFuse 或由 SoC ROM/TEE 管理，实现链式信任（最安全）。

------

# 二、先决条件（你必须确认的东西）

- SoC 是否能读取 **ChipID / unique id**（哪个寄存器或 efuse 区）？
- 是否支持写/读 eFuse（并了解不可逆性）？
- U-Boot 是否能编译进所需的 crypto / FIT / libfdt 支持（CONFIG_FIT、CONFIG_RSA、CONFIG_SHA256 等）？
- Kernel 是否启用了 IMA/dm-verity 或你愿意在 kernel 加模块去校验？

> 没硬件支持就无法实现强安全方案（比如 device-unique decrypt 或 eFuse 根信任）。

------

# 三、方案 A — 最容易实现：U-Boot 验签 Kernel（FIT 签名），并把验证状态与 ChipID 传给 kernel

**思路**：用 FIT image（或其它签名格式）对 kernel（及 dtb、initramfs）签名。U-Boot 在引导前验证签名；验证结果通过 device tree 的 `/chosen` 节点或 kernel cmdline 传递给 kernel。kernel 在启动早期读取该信息（可由 initramfs 或内核模块检验并决定是否继续）。

## U-Boot 端要做的事

1. 编译 U-Boot 时启用 FIT 签名支持（`CONFIG_FIT`, `CONFIG_RSA`, `CONFIG_FIT_SIGNATURE` 等）。
2. 把公钥以只读方式放到 U-Boot（最好放到 eFuse 或编译进只读区，如果 eFuse 可用更安全）。
3. 在启动流程中对 FIT image 做 `verify`，如果失败则拒绝启动或进入 recovery。
4. 把验证结果和设备 ChipID 写入 FDT 的 `/chosen` 节点，或把结果追加到 kernel cmdline（`bootargs`）。

### 示例：把 ChipID 和验证状态写入 `/chosen`

（放在 `board_late_init()` 或合适的 hook 中；伪代码，按你 SoC 的读 ChipID 函数实现 `read_chipid_hex()`）

```c
#include <common.h>
#include <fdt_support.h>
#include <libfdt.h>

int board_late_init(void)
{
    void *fdt = (void *)gd->fdt_blob;
    int chosen_off;
    char chipid[64];
    char bootargs[512];

    /* 读取 SoC ChipID（你需要实现）*/
    read_chipid_hex(chipid, sizeof(chipid)); /* e.g. "A1B2C3..." */

    /* 找 /chosen 节点 */
    chosen_off = fdt_path_offset(fdt, "/chosen");
    if (chosen_off < 0)
        chosen_off = fdt_add_subnode(fdt, 0, "chosen");

    /* 写入属性 */
    fdt_setprop_string(fdt, chosen_off, "u-boot,chip-id", chipid);

    /* 假设你在 verify 后设置 env var verified="1" 或 "0" */
    fdt_setprop_string(fdt, chosen_off, "u-boot,verified", env_get("verified") ? env_get("verified") : "0");

    /* 可选：把 ChipID 加入 kernel cmdline */
    snprintf(bootargs, sizeof(bootargs), "%s device_chipid=%s", env_get("bootargs") ? env_get("bootargs") : "", chipid);
    env_set("bootargs", bootargs);

    return 0;
}
```

> 注意：`read_chipid_hex()` 需根据你的 SoC（寄存器/efuse）实现。FDT 修改要在 U-Boot 最终传递给 kernel 的 blob 上进行。

## Kernel 端读取（early init 或模块）

内核启动后可从 `/proc/device-tree/chosen/u-boot,chip-id` 读取，或在驱动里用 device tree API：

```c
static int __init show_uboot_info(void)
{
    struct device_node *chosen;
    const char *chipid;
    chosen = of_find_node_by_path("/chosen");
    if (chosen && !of_property_read_string(chosen, "u-boot,chip-id", &chipid))
        pr_info("Device ChipID from U-Boot: %s\n", chipid);

    /* 读取验证标志 */
    const char *verified;
    if (chosen && !of_property_read_string(chosen, "u-boot,verified", &verified))
        pr_info("Kernel image verified: %s\n", verified);

    return 0;
}
late_initcall(show_uboot_info);
```

# 四、方案 B — 更强：用 ChipID 派生对称 key 做“设备专属解密”

**思路**：镜像（kernel 或 rootfs）被加密。只有设备能用 ChipID（结合制造时 secret）派生出实际解密 key。U-Boot 在启动时用 ChipID 派生 key 并解密内核镜像，仅本设备能成功。

## 实现要点

1. **密钥来源**：需要一个制造时的 master secret（最好写入 eFuse 或保存在 TEE）。不能把 master secret 放在普通 flash。
2. **派生函数**：使用 HMAC-SHA256 或 KDF：`device_key = HMAC(master_secret, ChipID)`。
3. **镜像加密**：在生产线端，用 `device_key` 加密（每台机单独加密）或用对称密钥包裹每台设备的对称 key。
4. **U-Boot 实现**：U-Boot 内实现 HMAC/KDF + AES decrypt，完成后把解密的 kernel 放内存并启动。
5. **安全存储**：master_secret 应放在 eFuse 或由 SoC 的 secure world 管理（OP-TEE），否则安全性不高。

## 风险

- 若 master_secret 泄露，所有设备都会被攻破。
- eFuse 写入不可逆，生产线必须严格测试。

## U-Boot 伪代码示例（派生 key 并解密）

```c
/* 伪代码，需链接 U-Boot 的 crypto lib 或自行实现 */
void derive_device_key(const uint8_t *chipid, size_t chipid_len,
                       const uint8_t *master_secret, size_t ms_len,
                       uint8_t *out_key) {
    /* out_key 32 bytes */
    hmac_sha256(master_secret, ms_len, chipid, chipid_len, out_key);
}

int decrypt_kernel(u8 *enc_image, size_t enc_len, u8 *out, u8 *key) {
    /* 使用 AES-CTR 或 AES-CBC + padding，根据你选的加密方案 */
    aes_decrypt(enc_image, enc_len, key, out);
    return 0;
}
```

# 五、方案 C（最强）— 使用 SoC ROM / eFuse / TrustZone 做根信任（推荐用于高安全场景）

**思路**：把公钥（或 key hash）烧到 eFuse，SoC ROM 在 bootROM 级别就验证第一级 bootloader（U-Boot），然后 U-Boot 验证下一层，形成链式信任；密钥永不出现在可写存储。若 SoC 支持 OP-TEE，可在 secure world 托管密钥和解密操作，U-Boot 向 secure world 请求解密/签名校验。

## 实现要点

- 在制造时写入公钥哈希或公钥到 eFuse（一次性），或使用 SoC 的 secure boot fuse 写入工具（厂商提供）。
- ROM -> (验证) -> U-Boot -> (验证) -> kernel（chain of trust）。
- U-Boot 只存放公钥指纹或调用 secure engine，不保存私钥。

## 注意

- eFuse 写入不可逆，请在生产前在开发板上彻底测试。
- 需要厂商文档：如何在 eFuse 写入数据、boot ROM 的行为、secure monitor 接口等。

# 六、如何把验证/绑定状态“传给” kernel（两种常用方式）

1. **Device Tree `/chosen`**（推荐）：在 U-Boot 修改 FDT，在 `/chosen` 下写 `u-boot,chip-id`、`u-boot,verified` 等属性。Kernel/early initramfs 可直接读取 `/proc/device-tree/chosen/...`。示例见上。
2. **Kernel cmdline**：在 U-Boot 更新 `bootargs` 环境变量，追加参数，例如 `device_chipid=... uboot_verified=1`，kernel 可使用 `early_param` 或 initrd 脚本读取 `cmdline`。

------

# 七、内核侧进一步保证（可选但建议）

- **IMA/EVM**：内核级完整性（文件/模块完整性）检查。U-Boot 负责引导验证并把 `verified` 状态提供给 kernel；kernel 的 IMA 可以基于该状态决定是否 enforce。
- **dm-verity**：对 rootfs 做块级只读校验，配合 signed metadata 确保 rootfs 没被替换。
- **early userspace（initramfs）校验**：在 initramfs 中再次检查签名或密钥材料，决定是否挂载 rootfs。

------

# 八、典型生产/开发流程（示例步骤）

1. 在开发板上实现并测试读取 ChipID、U-Boot 写 FDT、kernel 读取 FDT。
2. 做 FIT 签名方案（先用单个 test key）：生成密钥对 → 生成签名 FIT → 在 U-Boot 加载前验证成功/失败并记录 env `verified`。
3. 试验把 `verified` 传给 kernel，并在 initramfs 中读取并打印。
4. （可选）实现以 ChipID 派生 key 解密镜像（先在开发阶段用 master_secret 存在于 U-Boot env，确认流程后迁移到 eFuse/TEE）。
5. 如果一切 OK，再考虑把公钥/secret 写入 eFuse（谨慎，先在少量设备上试）。

方法3：

> ### 方法 3：利用核心板 MCU 自带的唯一 ID
>
> **原理**：多数微控制器（如 STM32、ESP32、MSP430 等）在出厂时会固化一个唯一硬件 ID（通常为 64 位或 128 位，全球唯一），存储在芯片内部的只读区域。
> **实现逻辑**：
>
> - 预先读取目标核心板 MCU 的唯一 ID，并在源码中写入该 ID 作为 “合法标识”；
> - 源码运行时，通过 MCU 的专用寄存器（如 STM32 的`UID_BASE`地址）读取当前核心板的 ID；
> - 比对读取到的 ID 与源码中预设的 ID，一致则正常运行，否则终止程序。
>   **优势**：无需额外硬件，依赖 MCU 原生特性，实现简单，成本低；
>   **注意**：需确保源码中预设 ID 的存储方式安全（如加密存储），避免被直接篡改。
>
> ### 方法 4：通过硬件加密模块 / 加密芯片验证
>
> **原理**：利用核心板集成的硬件加密模块（如 AES 引擎、RSA 加速器）或外接独立加密芯片（如 AT88SC、SH200H），通过密钥验证实现绑定。
> **实现逻辑**：
>
> - 在加密模块 / 芯片中预先写入核心板的唯一密钥（或用核心板 ID 生成密钥）；
> - 源码运行时，向加密模块发送验证请求（如用核心板 ID 加密一段随机数）；
> - 加密模块解密并返回验证结果，源码确认结果正确后才继续运行。
>   **优势**：硬件加密抗破解能力强，安全性远高于软件加密；
>   **适用场景**：对安全性要求高的场景（如工业控制、金融设备）。
>
> ### 方法 5：基于外设的唯一特性验证
>
> **原理**：核心板上的外设（如 SPI Flash、I2C 传感器、ADC 校准电阻等）存在个体差异（如厂商 ID、硬件参数偏差），可作为 “唯一标识”。
> **实现逻辑**：
>
> 
>
> - 选取核心板上固定外设（如板载 SPI Flash），读取其唯一信息（如厂商 ID + 设备 ID + 出厂序列号组合）；
> - 源码中预设目标核心板的外设信息，运行时读取当前外设信息并比对；
> - 比对一致则绑定成功，例如：某 SPI Flash 的 “厂商 ID=0x1F，设备 ID=0x48，序列号 = 0x123456” 为目标标识。
>   **优势**：利用现有外设，无需额外硬件，兼容性强；
>   **注意**：需确保外设信息的唯一性（避免同一批次设备信息重复）。
>
> ### 方法 6：物理不可克隆函数（PUF）绑定
>
> **原理**：PUF 利用硬件制造过程中随机产生的物理差异（如晶体管阈值电压、导线电阻）生成唯一 “指纹”，无法被复制或克隆。
> **实现逻辑**：
>
> 
>
> - 核心板集成 PUF 模块（部分高端 MCU 已支持，如 NXP 的 PUF-enabled 芯片），预先提取核心板的 PUF 指纹并存储；
> - 源码运行时，通过 PUF 模块重新生成当前核心板的指纹，与预设指纹比对；
> - 指纹匹配则绑定成功，因物理差异不可复制，安全性极高。
>   **优势**：理论上不可破解，唯一性和安全性远超传统 ID；
>   **局限**：依赖核心板硬件支持，目前仅在中高端设备中普及。
>
> ### 方法 7：基于时钟 / 晶振的特性差异验证
>
> **原理**：核心板的时钟源（如外部晶振、内部 RC 振荡器）存在微小个体差异（如频率偏差），可通过软件测量并作为标识。
> **实现逻辑**：
>
> 
>
> - 预先测量目标核心板的时钟特性（如 1MHz 晶振的实际频率偏差值 ±50Hz）；
> - 源码运行时，用定时器计数测量当前核心板的时钟偏差（如计数 1 秒内的时钟脉冲数）；
> - 比对偏差值是否在预设范围内，符合则绑定成功。
>   **优势**：利用硬件固有特性，难以仿造；
>   **注意**：受温度、电压影响较大，需结合校准机制（如定期修正偏差范围）。
>
> 
>
> 这些方法可单独使用，也可组合验证（如 “MCU 唯一 ID + 加密芯片” 双重验证），进一步提升绑定的安全性。选择时需平衡成本、硬件兼容性和安全性需求。

## 问题2：公钥、私钥各自的作用是什么

公钥和私钥都是由RK提供的工具rk_sign_tool生成的，具体命令为

~~~shell
../rkbin/tools/rk_sign_tool kk --bits 2048 --out .
ln -s ../private_key.pem keys/dev.key
ln -s ../public_key.pem keys/dev.pubkey
~~~

但是后面又通过-x509和私钥生成一个自签名证书：keys/dev.crt，看百度说的是dev.crt和公钥dev.pubkey的作用是相同的，它封装公钥并提供额外的身份信息，在安全验证环节替代或辅助公钥完成固件签名的合法性校验

~~~shell
openssl req -batch -new -x509 -key keys/dev.key -out keys/dev.crt
~~~

下面是公钥和dev.crt的补充

> ### 1. **本质：公钥的 “包装形式”**
>
> 自签名证书 `dev.crt` 本质上是将之前生成的公钥（`dev.pubkey`）进行了标准化封装，包含以下关键信息：
>
> - 公钥本身（与 `dev.pubkey` 内容一致，用于验证签名）；
> - 证书持有者信息（如开发者 / 厂商名称，通过 `openssl req` 命令的参数或默认配置生成）；
> - 签名算法（如 RSA2048 + SHA256）；
> - 有效期；
> - 用私钥（`dev.key`）对证书自身的签名（证明证书未被篡改）。
>
> 因此，它和公钥的核心功能一致 —— 用于验证固件的数字签名，但形式上更符合行业通用的证书标准（X.509），便于硬件或软件系统按标准流程解析和使用。
>
> ### 2. **在安全启动中的具体作用**
>
> 在 RK 平台的 Secure Boot 流程中，`dev.crt` 主要用于以下场景：
>
> - **替代公钥写入硬件**：部分硬件或启动程序（如 U-Boot 的安全启动模块）可能要求以 X.509 证书格式存储公钥，而非原始的 PEM 公钥文件。此时 `dev.crt` 会被烧录到 OTP 或安全存储区域，设备启动时通过解析证书提取公钥，验证固件签名。
> - **身份验证辅助**：证书中包含的持有者信息（如厂商名称）可用于额外的身份校验，确保固件不仅签名合法，还来自指定的授权方。
> - **兼容性适配**：部分开源工具或第三方组件（如 OP-TEE 信任执行环境）对 X.509 证书有原生支持，使用 `dev.crt` 可简化与这些组件的集成。
>
> ### 3. **与公钥的关系**
>
> `dev.crt` 和 `dev.pubkey` 是 “同源同效” 的：
>
> - 两者包含的公钥内容完全一致（均来自同一 RSA 密钥对）；
> - 验证固件签名时，无论是直接使用公钥还是从证书中提取公钥，最终的验证逻辑和结果完全相同；
> - 区别仅在于形式：`dev.pubkey` 是原始公钥文件，`dev.crt` 是包含公钥及附加信息的标准化证书。
>
> 总结来说，`dev.crt` 是公钥的标准化封装形式，主要用于适配硬件 / 软件的证书验证需求，确保在安全启动流程中能合规、高效地完成固件签名的合法性校验。

**然后回到公钥和私钥各自的作用：**

> 在瑞芯微（Rockchip）平台的 Secure Boot（安全启动）流程中，公钥和私钥是实现**数字签名与验证**的核心，二者配合完成对固件的合法性校验，防止未授权的恶意固件被加载运行。具体作用如下：
>
> ### **1. 私钥（privateKey.pem）的作用**
>
> - **生成数字签名**：
>   私钥由开发者 / 厂商保管，用于对固件（如 `loader.bin`、`uboot.img`、`update.img` 等）进行 “签名” 操作。
>   签名过程本质是：
>   1. 对固件内容计算哈希值（如 SHA256），得到一个固定长度的 “指纹”；
>   2. 用私钥对这个哈希值进行加密，生成 “数字签名”。
>      这个签名会附加到固件中，成为固件的一部分。
> - **唯一性与保密性**：
>   私钥是保密的，仅掌握在合法开发者手中。一旦泄露，第三方可能伪造签名，绕过安全验证，因此必须妥善保管（如离线存储）。
>
> ### **2. 公钥（publicKey.pem）的作用**
>
> - **验证签名合法性**：
>   公钥是公开的（但需预先写入硬件），用于在设备启动时验证固件签名的有效性。
>   验证过程与签名对应：
>   1. 设备启动时，读取固件中的 “数字签名” 和固件内容；
>   2. 用公钥解密签名，得到一个哈希值（即签名时用私钥加密的原始哈希）；
>   3. 对当前固件内容重新计算哈希值，与解密得到的哈希值比对。
>      若一致，说明固件未被篡改且来自持有对应私钥的合法来源，允许启动；否则验证失败，终止启动。
> - **硬件部署要求**：
>   公钥必须预先烧录到设备的安全存储区域（如瑞芯微的 OEM_S_OTP、熔丝或专用安全寄存器），且通常不可修改（或仅允许一次写入），确保攻击者无法替换公钥。
>
> ### **总结：公私钥的配合逻辑**
>
> - **签名阶段（开发端）**：用私钥对固件签名，确保 “只有合法开发者能生成可验证的固件”。
> - **验证阶段（设备端）**：用预存的公钥验证签名，确保 “只有经过合法签名的固件能被启动”。
>
> 这种 “非对称加密” 机制（私钥签名、公钥验证）是 Secure Boot 的核心，通过数学算法保证了固件的**完整性**（未被篡改）和**真实性**（来自合法来源）。

上面涉及到了哈希算法和哈希值，这里简单学习一下哈希

> ### 什么是哈希值？
>
> 哈希值（Hash Value）是通过**哈希算法**（如 SHA-256、MD5 等）对任意长度的数据（如文件、字符串）计算得到的**固定长度的二进制值**（通常以十六进制字符串形式展示），具有以下核心特性：
>
> 1. **固定长度**：无论输入数据大小（1KB 的文件或 1GB 的固件），同一算法生成的哈希值长度固定（如 SHA-256 始终生成 64 个字符的十六进制值）。
> 2. **唯一性**：不同的输入数据几乎不可能生成相同的哈希值（概率极低，称为 “哈希碰撞”）。
> 3. **不可逆性**：无法从哈希值反推出原始数据。
> 4. **敏感性**：原始数据哪怕只修改 1 个字节，哈希值会发生巨大变化。
>
> 哈希值的核心作用是**验证数据的完整性**—— 通过比对数据的哈希值，可快速判断数据是否被篡改（如固件是否被恶意修改）。
>
> ### 如何计算固件的哈希值？
>
> 计算固件（如`.bin`、`.img`文件）的哈希值可通过命令行工具（如`openssl`、`sha256sum`）或编程语言实现，以下是最常用的**命令行方法**：
>
> #### 1. 使用 `openssl` 工具（支持多种算法）
>
> `openssl` 是跨平台工具，支持 SHA-1、SHA-256、MD5 等主流算法，命令格式：
>
> ```bash
> # 计算SHA-256哈希值（推荐，安全性高）
> openssl dgst -sha256 固件文件名
> 
> # 示例：计算uboot.img的SHA-256哈希值
> openssl dgst -sha256 uboot.img
> # 输出类似：SHA256(uboot.img)= a1b2c3d4e5f6...（64个字符）
> 
> # 其他算法示例：
> openssl dgst -sha1 uboot.img   # SHA-1哈希值（40个字符）
> openssl dgst -md5 uboot.img    # MD5哈希值（32个字符，安全性较低，不推荐用于安全场景）
> ```
>
> #### 2. 使用系统自带工具（Linux/macOS）
>
> **Linux/macOS** 通常预装了专门的哈希工具（如`sha256sum`），更简洁：
>
> ```bash
> # 计算SHA-256哈希值
> sha256sum 固件文件名
> 
> # 示例
> sha256sum update.img
> # 输出类似：a1b2c3d4e5f6...  update.img（哈希值 + 文件名）
> ```
>
> **Windows** 可使用 PowerShell：
>
> ```powershell
> # 计算SHA-256哈希值
> Get-FileHash -Algorithm SHA256 固件文件名
> 
> # 示例
> Get-FileHash -Algorithm SHA256 uboot.img
> ```
>
> #### 3. 验证哈希值是否匹配
>
> 如果需要确认固件是否被篡改，可比对计算出的哈希值与官方提供的哈希值是否一致：
>
> ```bash
> # 假设官方提供的哈希值为 official_hash.txt
> # 方法1：手动比对
> sha256sum uboot.img
> cat official_hash.txt
> 
> # 方法2：自动比对（若不一致会提示错误）
> echo "官方哈希值  uboot.img" | sha256sum --check
> ```
>
> ### 总结
>
> - 哈希值是数据的 “数字指纹”，用于验证完整性。
> - 计算固件哈希值最便捷的方式是用`openssl dgst -sha256 文件名`（跨平台）或`sha256sum 文件名`（Linux/macOS）。
> - 安全场景（如 Secure Boot）推荐使用 SHA-256 及以上强度的算法，避免使用 MD5、SHA-1（存在碰撞风险）。

然后就有了一个问题，公钥和私钥具体是怎样计算出来的呢？我这里是通过rk提供的可执行程序来得到的

> 公钥和私钥（非对称对称密钥对）是通过**密码学算法**（如 RSA、ECC 等）生成的，两者是 “成对出现” 的数学关系 —— 由私钥可以推导出公钥，但无法从公钥反推出私钥。生成过程完全依赖算法的数学特性，无需人工干预。
>
> ### 以 RSA 算法为例（最常用），密钥对的生成原理
>
> RSA 密钥对的生成基于 “大数分解难题”（两个大质数相乘容易，但若只知道乘积，分解出原始质数极难），核心步骤如下：
>
> 1. 随机生成两个**大质数**（如 1024 位或 2048 位，位数越长安全性越高），记为`p`和`q`；
> 2. 计算乘积`n = p × q`（`n`称为 “模”，是公钥和私钥的共同部分）；
> 3. 计算欧拉函数`φ(n) = (p-1) × (q-1)`；
> 4. 随机选择一个整数`e`（`1 < e < φ(n)`），且`e`与`φ(n)`互质（最大公约数为 1），`e`作为**公钥的指数**；
> 5. 计算`e`的逆元`d`（满足`(e × d) mod φ(n) = 1`），`d`作为**私钥的指数**；
> 6. 最终：
>    - 公钥为 `(e, n)`
>    - 私钥为 `(d, n)`
>
> ### 实际操作：如何生成公钥和私钥？
>
> 在实际开发中，无需手动计算（数学过程极其复杂），而是通过工具自动生成。以瑞芯微平台常用的工具为例：
>
> #### 1. 用 `rk_sign_tool` 生成（适配瑞芯微 Secure Boot）
>
> 如之前提到的命令，直接生成 RSA 密钥对：
>
> ```bash
> # 生成2048位RSA密钥对，输出到当前目录
> ../rkbin/tools/rk_sign_tool kk --bits 2048 --out .
> ```
>
> 执行后会自动生成：
>
> - `privateKey.pem`（私钥，PEM 格式，包含`d`和`n`）
> - `publicKey.pem`（公钥，PEM 格式，包含`e`和`n`）
>
> #### 2. 用 `openssl` 生成（通用方法，跨平台）
>
> 若没有专用工具，`openssl` 是更通用的选择，支持多种算法：
>
> ```bash
> # 生成2048位RSA私钥（PEM格式，带密码保护可选）
> openssl genrsa -out private.key 2048
> 
> # 从私钥中提取公钥
> openssl rsa -in private.key -pubout -out public.key
> ```
>
> 执行后得到：
>
> - `private.key`（私钥）
> - `public.key`（公钥）
>
> ### 密钥对的核心特性
>
> 1. **成对性**：公钥和私钥必须成对使用，用公钥加密的数据只能用对应私钥解密，反之亦然。
> 2. **保密性**：私钥必须严格保密（仅持有者知晓），公钥可公开（用于加密或验证）。
> 3. **唯一性**：每次生成的密钥对都是唯一的，即使参数相同（如 2048 位 RSA），随机质数的选择也会导致密钥对不同。
>
> ### 总结
>
> 公钥和私钥通过密码学算法（如 RSA）自动生成，核心是数学上的不可逆关系。实际操作中，通过工具（`rk_sign_tool`、`openssl`等）一键生成即可，无需关注底层数学细节。生成后，私钥用于签名 / 解密，公钥用于验证 / 加密，共同构成非对称加密的安全基础。

## 问题4 公钥、私钥是如何集成到镜像里面去的

这里的问题现在看的话是有些问题的，这就又涉及到公钥和私钥各自的作用，公钥是可以公开的，我就是用的这个公钥进行的加密，然后再由私钥进行解密，

> 私钥对 U-Boot 或内核镜像的签名过程，本质是通过**加密镜像的哈希值**生成 “数字签名”，以此证明镜像的合法性和完整性。具体步骤结合工具（如`rk_sign_tool`）和密码学原理如下：
>
> ### 一、私钥签名的核心原理
>
> 签名过程依赖 “哈希算法” 和 “非对称加密” 的结合，核心逻辑是：
>
> 1. 先通过哈希算法提取镜像的 “数字指纹”（哈希值）；
> 2. 再用私钥对这个哈希值进行加密，生成 “数字签名”；
> 3. 签名会与镜像绑定（附加到镜像中或单独存储），作为镜像的 “身份证明”。
>
> ### 二、私钥对镜像签名的详细步骤（以 U-Boot 为例）
>
> 假设已通过`rk_sign_tool`生成私钥`privateKey.pem`（或`keys/dev.key`），并加载到工具中，签名流程如下：
>
> #### 1. **计算 U-Boot 镜像的哈希值**
>
> 工具首先对`uboot.img`进行哈希计算（如 SHA256 算法），得到一个固定长度的哈希值（例如：`a1b2c3d4...`）。
>
> 哈希值的作用：相当于镜像的 “唯一指纹”，镜像内容哪怕有 1 字节变化，哈希值会完全不同。
>
> #### 2. **用私钥对哈希值进行加密（生成签名）**
>
> 工具调用私钥（`privateKey.pem`）对第一步得到的哈希值进行加密，生成 “数字签名”。
>
> 这里的 “加密” 并非加密整个镜像（镜像通常很大，效率低），而是加密短小的哈希值（固定长度，效率极高）。
>
> 由于私钥是唯一的，只有持有该私钥的人才能生成这个签名，确保了签名的 “唯一性” 和 “不可伪造性”。
>
> #### 3. **将签名与镜像绑定**
>
> 生成的签名会以特定格式附加到`uboot.img`中（或生成单独的签名文件，如`uboot.img.sig`），随镜像一起发布。
>
> 例如，用rk_sign_tool对 U-Boot 签名的命令：
>
> ```bash
> rk_sign_tool si --img uboot.img  # 工具内部自动完成哈希计算+私钥加密+签名绑定
> ```
>
> #### 4. **签名的作用**
>
> 当设备启动时，会通过公钥验证这个签名：
>
> - 用公钥解密签名，得到 “原始哈希值”（即签名时用私钥加密的哈希）；
> - 重新计算当前加载的`uboot.img`的哈希值；
> - 比对两个哈希值：一致则说明镜像未被篡改且来自合法来源（私钥持有者），允许启动；否则拒绝启动。
>
> ### 三、为何必须用私钥签名？
>
> - 私钥的 “保密性”：只有合法开发者持有私钥，确保其他人无法伪造签名。
> - 非对称加密的 “单向性”：用私钥加密的内容（签名）只能用对应的公钥解密，而公钥是公开且预先烧录到设备中的，确保设备能验证签名合法性。
>
> ### 总结
>
> 私钥对 U-Boot / 内核镜像的签名过程，是通过 “哈希值提取→私钥加密哈希值→生成签名” 三个步骤完成的。核心目的是生成一个 “唯一且不可伪造” 的数字凭证，让设备能够通过公钥验证镜像的合法性，这是 Secure Boot（安全启动）的核心机制。

~~~c
/**
 * 加载操作系统内核到内存
 * @param images 包含镜像信息的结构体指针
 * @param load_end 输出参数，内核加载结束的内存地址
 * @param boot_progress 启动进度标识（未使用）
 * @return 0表示成功，非0表示错误码
 */
static int bootm_load_os(bootm_headers_t *images, unsigned long *load_end,
                         int boot_progress)
{
    // 从镜像信息中获取操作系统相关信息
    image_info_t os = images->os;
    // 内核加载的目标地址
    ulong load = os.load;
    // 镜像数据的起始地址
    ulong blob_start = os.start;
    // 镜像数据的结束地址
    ulong blob_end = os.end;
    // 内核镜像在内存中的起始地址
    ulong image_start = os.image_start;
    // 内核镜像的长度
    ulong image_len = os.image_len;
    // 标记是否没有内存重叠
    bool no_overlap;
    // 加载缓冲区和镜像缓冲区指针
    void *load_buf, *image_buf;
    // 错误码
    int err;

    // 将加载地址映射到系统内存地址
    load_buf = map_sysmem(load, 0);
    // 将镜像起始地址映射到系统内存地址（长度为image_len）
    image_buf = map_sysmem(os.image_start, image_len);

    // 解压并加载内核镜像
    // 参数包括：压缩类型、加载地址、镜像起始地址、镜像类型、缓冲区、长度等
    err = bootm_decomp_image(os.comp, load, os.image_start, os.type,
                             load_buf, image_buf, image_len,
                             CONFIG_SYS_BOOTM_LEN, load_end);
    // 如果解压或加载失败，记录错误并返回
    if (err) {
        bootstage_error(BOOTSTAGE_ID_DECOMP_IMAGE);
        return err;
    }

    // 刷新缓存，确保数据正确写入内存
    flush_cache(load, ALIGN(*load_end - load, ARCH_DMA_MINALIGN));

    // 打印内核加载信息：加载地址和结束地址
    printf("   kernel loaded at 0x%08lx, end = 0x%08lx\n", load, *load_end);
    // 标记内核加载完成的启动阶段
    bootstage_mark(BOOTSTAGE_ID_KERNEL_LOADED);

    // 判断是否没有内存重叠：未压缩且加载地址等于镜像起始地址
    no_overlap = (os.comp == IH_COMP_NONE && load == image_start);

    // 检查是否存在内存重叠（加载的内核覆盖了原始镜像数据）
    if (!no_overlap && (load < blob_end) && (*load_end > blob_start)) {
        // 打印重叠的地址信息
        printf("images.os.start = 0x%lX, images.os.end = 0x%lx\n",
               blob_start, blob_end);
        printf("images.os.load = 0x%lx, load_end = 0x%lx\n", load,
               *load_end);

        // 检查镜像类型
        if (images->legacy_hdr_valid) {
            // 传统多组件镜像，打印警告并返回重叠错误
            if (image_get_type(&images->legacy_hdr_os_copy)
                    == IH_TYPE_MULTI)
                puts("WARNING: legacy format multi component image overwritten\n");
            return BOOTM_ERR_OVERLAP;
        } else {
            // 新格式镜像被覆盖，提示必须重启，并返回重置错误
            puts("ERROR: new format image overwritten - must RESET the board to recover\n");
            bootstage_error(BOOTSTAGE_ID_OVERWRITTEN);
            return BOOTM_ERR_RESET;
        }
    }

    // 更新镜像长度为解压后的内核大小，供后续使用
    images->os.image_len = *load_end - load;

    // 加载成功
    return 0;
}
~~~

EMMC（嵌入式多媒体卡）的**RPMB（Replay Protected Memory Block，重放保护内存块）** 是 EMMC 中一个特殊的安全存储区域，专为存储敏感数据设计，核心功能是防止 “重放攻击”（Replay Attack）—— 即攻击者重复使用之前的有效指令或数据来伪造操作。以下从定义、作用、工作原理、技术细节等方面详细讲解：

### 一、RPMB 的核心定义与作用

RPMB 是 EMMC 规范中定义的一个**受保护的物理存储分区**，位于 EMMC 芯片内部，独立于用户数据区（User Data Area）、引导分区（Boot Partition）等其他区域。其核心作用是：

- 安全存储敏感数据：如设备密钥、数字证书、DRM（数字版权管理）信息、安全计数器、设备身份标识（ID）等。
- 防止重放攻击：通过硬件级机制确保写入的数据无法被重复使用（例如，防止攻击者复用旧的 “写入指令 + 数据” 来篡改内容）。
- 确保数据完整性与真实性：通过加密和消息认证码（MAC）验证，保证数据在传输和存储过程中未被篡改。

### 二、RPMB 的硬件与协议基础

#### 1. 硬件结构

- **物理隔离**：RPMB 是 EMMC 芯片内部一块独立的存储区域，与其他分区在硬件上隔离，仅通过特定的安全命令访问。
- **大小**：由厂商定义，通常为 64KB~1MB（例如常见的 512KB），远小于用户数据区（GB 级），仅用于存储少量敏感数据。
- **硬件加密**：多数 EMMC 芯片内置加密引擎，支持 AES 等算法，用于 RPMB 的密钥存储和数据加密。

#### 2. 通信协议（基于 EMMC 标准）

RPMB 的访问通过 EMMC 的专用命令集实现，核心命令包括：

- **CMD23**：用于配置 RPMB 传输的块大小和数量（预处理命令）。
- **CMD18（读）**：读取 RPMB 中的数据（需携带认证信息）。
- **CMD25（写）**：向 RPMB 写入数据（需携带 MAC 认证码）。
- **CMD42**：用于 RPMB 的密钥设置（仅首次使用时配置，通常由厂商或设备厂商预烧录）。

所有访问 RPMB 的命令必须包含**认证信息**（如 MAC），否则 EMMC 会拒绝执行。

### 三、防重放攻击的核心原理

RPMB 通过 “**计数器 + MAC 认证**” 机制防止重放攻击，具体流程如下：

#### 1. 预配置：RPMB 密钥

- 首次使用 RPMB 前，需通过**RPMB 密钥**（通常为 128 位或 256 位）进行初始化。该密钥由设备厂商预烧录到 EMMC 的不可修改区域（如熔丝或 OTP），仅 EMMC 内部硬件可访问，外部无法读取。
- 密钥用于生成和验证 MAC（消息认证码），确保访问者是 “授权设备”。

#### 2. 写入操作（防重放核心）

当向 RPMB 写入数据时，流程如下：

1. **主机（如手机 CPU）生成命令包**：
   - 包含要写入的数据（如密钥、计数器值）。
   - 包含一个**递增的计数器（Counter）**（每次写入必须 + 1，由主机维护）。
   - 用 RPMB 密钥计算整个数据包的**MAC**（如 HMAC-SHA256），附加到命令中。
2. **EMMC 验证并执行**：
   - 验证 MAC：EMMC 用内部存储的 RPMB 密钥重新计算 MAC，与命令中的 MAC 比对，不一致则拒绝。
   - 验证计数器：EMMC 内部也维护一个计数器，仅当主机发送的计数器**大于**EMMC 内部记录的计数器时，才允许写入，并更新内部计数器。
   - 写入数据：验证通过后，将数据存入 RPMB 区域。

**防重放逻辑**：若攻击者重放之前的写入命令（包含旧的计数器和 MAC），由于其计数器小于 EMMC 内部的当前计数器，会被直接拒绝，无法生效。

#### 3. 读取操作

读取 RPMB 数据时，流程类似：

1. 主机发送读取请求，包含计数器和 MAC（用 RPMB 密钥生成）。
2. EMMC 验证 MAC 和计数器（需大于等于上次读取的计数器），验证通过后返回数据，并附加 MAC（供主机验证数据未被篡改）。

### 四、RPMB 的访问限制与特性

1. **严格的访问控制**：
   - 任何读写操作必须通过 MAC 认证，无密钥或认证失败则无法访问。
   - 首次配置 RPMB 密钥后，通常无法修改（硬件级锁定），防止密钥泄露。
2. **写入次数限制**：
   - 虽然没有明确的 “寿命限制”，但 RPMB 的计数器是 32 位（最大值约 40 亿），理论上最多支持 40 亿次写入（远超设备生命周期）。
3. **数据持久性**：
   - RPMB 属于非易失性存储（与用户数据区一样），断电后数据不丢失，适合存储长期有效的敏感信息。

### 五、应用场景

RPMB 的安全性使其在以下场景中广泛应用：



1. **移动设备安全**：存储设备解锁密钥（如 Android 的 SELinux 密钥、安全启动证书）。
2. **DRM 版权保护**：存储媒体内容的解密密钥，防止盗版。
3. **支付安全**：存储移动支付的安全芯片信息、交易计数器。
4. **设备认证**：存储设备唯一身份标识（ID），用于物联网设备的身份验证。
5. **安全启动**：存储引导镜像的哈希值，验证启动过程中镜像的完整性。

### 六、与 EMMC 其他分区的区别

| 分区类型          | 作用                   | 安全性                 | 访问方式                 |
| ----------------- | ---------------------- | ---------------------- | ------------------------ |
| 用户数据区（UDA） | 存储用户数据、应用等   | 低（无特殊保护）       | 普通读写命令（CMD18/25） |
| 引导分区（Boot）  | 存储启动代码（如 BL1） | 中（只读或厂商锁定）   | 专用引导命令（CMD17/24） |
| RPMB              | 存储敏感数据           | 高（防重放、加密认证） | 带 MAC 的安全命令        |

### 总结

RPMB 是 EMMC 中核心的安全组件，通过 “密钥认证 + 计数器防重放” 机制，为敏感数据提供硬件级的安全存储。其设计目标是在嵌入式设备中抵御重放攻击、数据篡改和未授权访问，是移动设备、物联网终端等场景中实现安全功能的关键技术。
