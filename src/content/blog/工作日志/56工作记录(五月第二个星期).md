~~~c
#include <stdio.h>
#include <stdlib.h>
#include <fcntl.h>
#include <unistd.h>
#include <sys/ioctl.h>
#include <sys/mman.h>
#include <linux/fb.h>
#include <string.h>
#include <stdint.h>   // 修复 uint16_t、uint32_t 报错

// 假设 IMX6_Inc.h 是你自己写的头文件，如果没有可以删除这一行
//#include "IMX6_Inc.h"

#pragma pack(push, 1)
typedef struct {
    uint16_t bfType;
    uint32_t bfSize;
    uint16_t bfReserved1;
    uint16_t bfReserved2;
    uint32_t bfOffBits;
} BITMAPFILEHEADER;

typedef struct {
    uint32_t biSize;
    int32_t  biWidth;
    int32_t  biHeight;
    uint16_t biPlanes;
    uint16_t biBitCount;
    uint32_t biCompression;
    uint32_t biSizeImage;
    int32_t  biXPelsPerMeter;
    int32_t  biYPelsPerMeter;
    uint32_t biClrUsed;
    uint32_t biClrImportant;
} BITMAPINFOHEADER;
#pragma pack(pop)

#define BMP_PATH        ("1.bmp")
#define FB_DEV          ("/dev/fb0")

#define PIC_WIDTH       (480)
#define PIC_HEIGHT      (272)
#define DISPLAY_WIDTH   (480)
#define DISPLAY_HEIGHT  (272)
#define DISPLAY_BPP     (3) // 3 for RGB888 or 4 for ARGB8888

void display_bmp()
{
    int fb_fd = -1;
    int bmp_fd = -1;
    char *fb_ptr = NULL;
    char *bmp_data = NULL;
    struct fb_var_screeninfo var_info;
    struct fb_fix_screeninfo fix_info;

    // 打开BMP文件
    bmp_fd = open(BMP_PATH, O_RDONLY);
    if (bmp_fd == -1) {
        perror("open bmp");
        return;
    }

    BITMAPFILEHEADER fh;
    if (read(bmp_fd, &fh, sizeof(fh)) != sizeof(fh)) {
        printf("BMP header read error\r\n");
        close(bmp_fd);
        return;
    }
    printf("fh.bfType:0x%04X\r\n", fh.bfType);

    BITMAPINFOHEADER ih;
    if (read(bmp_fd, &ih, sizeof(ih)) != sizeof(ih)) {
        printf("BMP info read error\r\n");
        close(bmp_fd);
        return;
    }
    printf("ih.biBitCount:%d, ih.biCompression:%d\r\n", ih.biBitCount, ih.biCompression);

    // 检查是否是 24 位 BMP
    if (ih.biBitCount != 24 || ih.biCompression != 0) {
        printf("Only support 24-bit uncompressed BMP.\r\n");
        close(bmp_fd);
        return;
    }

    // 打开帧缓冲设备
    fb_fd = open(FB_DEV, O_RDWR);
    if (fb_fd < 0) {
        perror("FB open failed");
        goto END;
    }

    if (ioctl(fb_fd, FBIOGET_VSCREENINFO, &var_info)) {
        perror("Failed to get variable info");
        goto END;
    }
    printf("Display info: %dx%d, %dbpp\n",
           var_info.xres, var_info.yres, var_info.bits_per_pixel);

    // 获取固定屏幕信息
    if (ioctl(fb_fd, FBIOGET_FSCREENINFO, &fix_info)) {
        perror("Failed to get fixed info");
        goto END;
    }
    printf("Display info, smem_len:%d\n",
           fix_info.smem_len);
    printf("offset, red:%d, green:%d, blue:%d\n",
           var_info.red.offset, var_info.green.offset, var_info.blue.offset);
    printf("fix_info.line_length:%d\n",
           fix_info.line_length);

    // 映射帧缓冲内存
    fb_ptr = mmap(0, fix_info.smem_len, PROT_READ | PROT_WRITE,
                  MAP_SHARED, fb_fd, 0);
    if (fb_ptr == MAP_FAILED) {
        perror("FB mmap failed");
        goto END;
    }

    int width = DISPLAY_WIDTH;
    int height = DISPLAY_HEIGHT;
    int is_top_down = (ih.biHeight > 0) ? 1 : 0;

    printf("pic info, width:%d, height:%d, is_top_down:%d, bfOffBits:%d\n",
           width, height, is_top_down, fh.bfOffBits);

    // 计算BMP行字节数（含对齐填充）
    int bmp_stride = (width * 3 + 3) & ~3;
    lseek(bmp_fd, fh.bfOffBits, SEEK_SET);

    bmp_data = malloc(bmp_stride * height);
    if (!bmp_data || read(bmp_fd, bmp_data, bmp_stride * height) != bmp_stride * height) {
        perror("BMP data read error");
        goto END;
    }

    int fb_stride = fix_info.line_length;

    printf("BMP WIDTH (%d);HEIGHT (%d);BMPSTRIDE (%d);FBSTRIDE (%d);\n",
           width, height, bmp_stride, fb_stride);

    for (int y = 0; y < height; y++) {
        int src_y = is_top_down ? y : (height - 1 - y);
        char *bmp_row = bmp_data + src_y * bmp_stride;
        char *fb_row = fb_ptr + y * fb_stride;

        for (int x = 0; x < width; x++) {
            // BGR转RGB
            if (DISPLAY_BPP == 4) {
                fb_row[x*4] = 0xFF;                 // A
                fb_row[x*4+1] = bmp_row[x*3 + 2];   // R
                fb_row[x*4+2] = bmp_row[x*3 + 1];   // G
                fb_row[x*4+3] = bmp_row[x*3];       // B
            } else {
                fb_row[x*3] = bmp_row[x*3 + 2];     // R
                fb_row[x*3+1] = bmp_row[x*3 + 1];   // G
                fb_row[x*3+2] = bmp_row[x*3];       // B
            }
        }
    }

END:
    if (fb_ptr) {
        munmap(fb_ptr, fix_info.smem_len);
    }

    if (bmp_data) {
        free(bmp_data);
    }

    if (fb_fd >= 0) {
        close(fb_fd);
    }

    if (bmp_fd >= 0) {
        close(bmp_fd);
    }

    printf("display over\r\n");
}

int main()
{
    display_bmp();
    return 0;
}
~~~

## 🔍 整体流程简述：

1. **打开 I²C 设备节点 `/dev/i2c-1`**
2. **定义从设备地址、寄存器地址和要写入的数据**
3. **调用 `i2c_write_reg()` 写入数据到指定寄存器**
4. **调用 `i2c_read_reg()` 从该寄存器读取数据并打印**
5. **关闭文件描述符**

---

## 🧱 重点结构体解析

下面是你代码中用到的关键结构体及其作用说明：

---

### 1. `struct i2c_msg`

这是 Linux 内核定义用于表示一次 I²C 数据传输（发送或接收）的消息结构体。

```c
struct i2c_msg {
    __u16 addr;     // 从设备地址（7位）
    __u16 flags;    // 标志位，如 I2C_M_RD 表示读操作
    __u16 len;      // 要传输的数据长度（字节数）
    __u8 *buf;      // 数据缓冲区指针
};
```

#### 使用场景举例：

在 `i2c_read_reg()` 中用了两个 `i2c_msg`：
- 第一个：向设备发送寄存器地址（写）
- 第二个：从设备读取数据（读）

在 `i2c_write_reg()` 中只用了一个 `i2c_msg`：
- 发送寄存器地址 + 数据（写）

---

### 2. `struct i2c_rdwr_ioctl_data`

这是一个封装多个 `i2c_msg` 的结构体，用于一次性提交多个 I²C 消息（例如先写地址再读数据）。

```c
struct i2c_rdwr_ioctl_data {
    struct i2c_msg *msgs;   // 用于表示一次 I²C 数据传输（发送或接收）的消息结构体
    __u32 nmsgs;            // 消息的数量
};
```

#### 使用方式：

将多个 `i2c_msg` 封装成一个 `i2c_rdwr_ioctl_data` 结构体后，通过 `ioctl(fd, I2C_RDWR, &msgset)` 提交给内核处理。

---

### 3. `ioctl()` 函数

```c
int ioctl(int fd, unsigned long request, ...);
```

- `fd`：已经打开的 I²C 设备文件描述符（如 `/dev/i2c-1`）
- `request`：请求码，这里用的是 `I2C_RDWR`
- 第三个参数是一个指向 `i2c_rdwr_ioctl_data` 的指针

#### 常见请求码：

| 请求码      | 含义                             |
| ----------- | -------------------------------- |
| `I2C_RDWR`  | 执行复合读写操作                 |
| `I2C_SMBUS` | SMBus 类型的操作（如读写单字节） |
| `I2C_SLAVE` | 设置从设备地址                   |

---

## 📌 函数详解

### ✅ `i2c_read_reg()`

```c
int i2c_read_reg(int fd, unsigned char addr, unsigned char reg)
```

#### 功能：
从指定地址 `addr` 的 I²C 从设备中读取寄存器地址为 `reg` 的一个字节数据。

#### 实现逻辑：
1. 构造两个 `i2c_msg`：
   - 第一个是写寄存器地址（`reg`）
   - 第二个是从设备读取一个字节（设置 `I2C_M_RD` 标志）
2. 使用 `ioctl(fd, I2C_RDWR, &msgset)` 完成一次“写地址+读数据”的复合操作
3. 返回读取到的数据（`data`）

---

### ✅ `i2c_write_reg()`

```c
int i2c_write_reg(int fd, unsigned char addr, unsigned char reg, const unsigned char *data, int len)
```

#### 功能：
向指定地址 `addr` 的 I²C 从设备写入一个寄存器地址 `reg` 和后续 `len` 字节的数据。

#### 实现逻辑：
1. 构造一个缓冲区 `buf[]`，第一个字节是寄存器地址 `reg`，后面跟着数据
2. 构造一个 `i2c_msg`，指向这个缓冲区，长度为 `len + 1`
3. 使用 `ioctl(fd, I2C_RDWR, &msgset)` 完成写操作
4. 成功返回 0，失败返回 -1

---

## 🧪 示例流程图（以读操作为例）

```
主设备（CPU）
   │
   ├─ 写 Reg Addr（0x80） → SDA 上发送地址 + 写标志
   │
   └─ 读 Data Byte        ← SDA 上接收从设备返回的一个字节
```

## 💡 补充知识点

### 📌 I²C 地址格式：

- 7位地址模式（常用），所以 `addr` 是 7 位（如 `0x38`）
- 在实际通信中，第8位用来表示读/写方向（`0` 表示写，`1` 表示读）

### 📌 寄存器地址是否需要单独发送？

- 是的，I²C 协议本身没有“寄存器”概念，必须手动先发送寄存器地址，再进行读写。
- 这就是为什么读写函数都要传入 `reg` 参数。

`I2C_RDWR` 的作用是执行一次复合式的 I²C 传输操作（例如先发送寄存器地址再读取数据），并且在整个过程中仅产生一个 STOP 信号，从而保证通信的连续性和高效性。
