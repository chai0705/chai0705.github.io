~~~c
// 静态函数：视频编码（VENC）的正常流程处理
// 返回值：td_s32 类型，TD_SUCCESS 表示成功，其他值表示失败
static td_s32 sample_venc_normal(td_void)
{
    td_s32 ret; // 用于存储函数调用的返回值，判断操作是否成功
    sample_sns_type sns_type = 0;  // 传感器类型变量，用于指定使用的摄像头型号
    sample_vi_cfg vi_cfg = {0};  // 视频输入（VI）模块的配置结构体，初始化清零
    sample_vpss_cfg vpss_cfg = {0};  // 视频预处理（VPSS）模块的配置结构体，初始化清零
    ot_vpss_grp vpss_grp = 0;  // VPSS组号，此处使用第0组
    sample_venc_vpss_chn venc_vpss_chn = {0};  // 编码通道（VENC）与VPSS通道的映射关系结构体，初始化清零
    sample_venc_param enc_param = {0};  // 编码参数结构体，存储编码相关配置（如通道数、分辨率等），初始化清零
    ot_pic_size pic_size[CHN_NUM_MAX] = {0};  // 图像尺寸数组，存储各通道的分辨率，大小为最大支持通道数，初始化清零

    /* *****************************************
      步骤0：准备相关参数（初始化传感器、分辨率等）
    ***************************************** */
    sns_type = SENSOR0_TYPE;  // 设置使用第0个传感器（通常为默认摄像头）
    enc_param.venc_chn_num = CHN_NUM_MAX;  // 设置编码通道数量为最大支持通道数
    sample_venc_set_pic_size_0(sns_type, pic_size);  // 根据传感器类型，初始化第0路图像的分辨率
    pic_size[1] = SMALL_STREAM_SIZE;  // 设置第1路图像为小流尺寸（通常用于低带宽场景）
    sample_venc_get_enc_size(pic_size, &enc_param);  // 根据图像尺寸计算并设置编码所需的参数（如码率、缓冲区大小等）
    sample_set_venc_vpss_chn(&venc_vpss_chn, enc_param.venc_chn_num);  // 配置编码通道与VPSS通道的映射关系

    /* *****************************************
      步骤1：初始化系统并分配通用视频缓冲区（VB）
             同时启动VI（视频输入）和VPSS（视频预处理）模块
    ***************************************** */
    // 调用封装函数启动系统、VI、VPSS，若失败则返回错误码
    if ((ret = sample_venc_online_wrap_start_sys_vi_vpss(sns_type, &vi_cfg, &vpss_cfg, &enc_param)) != TD_SUCCESS) {
        return ret;
    }

    // 启动编码流程：根据编码尺寸、VPSS组和通道映射，初始化并启动VENC模块
    if ((ret = sample_venc_normal_start_encode(&enc_param.enc_size[0], vpss_grp, &venc_vpss_chn)) != TD_SUCCESS) {
        goto EXIT_SYS_VI_VPSS;  // 若编码启动失败，跳转到清理步骤
    }

    /* *****************************************
     退出流程：处理用户退出操作，释放资源
    ***************************************** */

EXIT_SYS_VI_VPSS:  // 清理标签：若启动编码失败，执行此部分释放资源
    sample_venc_online_wrap_stop_sys_vi_vpss(&vi_cfg, &vpss_cfg);  // 停止VI、VPSS模块并释放系统资源

    return ret;  // 返回最终结果（成功或失败的错误码）
}
~~~

这里的重点在于编码，对应的函数为sample_venc_normal_start_encode，具体内容如下所示：

~~~c
/**
 * 静态函数：启动视频编码的核心流程（支持H.265和H.264两路编码）
 * 功能：配置编码参数（GOP、分辨率、码率等）、启动编码通道、绑定VPSS与VENC、启动码流获取
 * @param enc_size：编码分辨率数组（存储各通道的目标编码尺寸）
 * @param vpss_grp：VPSS组号（指定使用哪个VPSS组的预处理数据）
 * @param venc_vpss_chn：编码通道与VPSS通道的映射关系结构体（记录哪路VPSS对应哪路VENC）
 * @return 成功返回TD_SUCCESS，失败返回错误码
 */
static td_s32 sample_venc_normal_start_encode(ot_size *enc_size, ot_vpss_grp vpss_grp,
    sample_venc_vpss_chn *venc_vpss_chn)
{
    td_s32 ret;                          // 存储函数调用返回值，用于判断操作是否成功
    ot_venc_gop_mode gop_mode;           // GOP（图像组）模式（如I帧间隔模式：固定间隔、自适应等）
    ot_venc_gop_attr gop_attr;           // GOP属性结构体（存储I帧间隔、P帧数量等参数）
    // 编码通道参数数组（每个元素对应一路编码通道的配置，如码率、帧率、编码格式）
    sample_comm_venc_chn_param chn_param[CHN_NUM_MAX] = {0};

    /* 第一步：获取GOP模式（决定I帧生成策略） */
    // 调用get_gop_mode函数获取预设的GOP模式（如固定I帧间隔），失败则返回错误
    if (get_gop_mode(&gop_mode) != TD_SUCCESS) {
        return TD_FAILURE;
    }

    /* 第二步：根据GOP模式获取对应的GOP属性（如I帧间隔为30帧、P帧为2帧等） */
    ret = sample_comm_venc_get_gop_attr(gop_mode, &gop_attr);
    if (ret != TD_SUCCESS) {
        sample_print("Venc Get GopAttr for %#x!\n", ret);  // 打印GOP属性获取失败的错误码
        return ret;
    }

    /* 第三步：设置视频编码核心参数（分辨率、GOP、码率、编码格式等） */
    // enc_size：目标编码分辨率；chn_param：输出的通道参数；CHN_NUM_MAX：通道数量；gop_attr：GOP属性；TD_FALSE：不启用特殊模式
    ret = sample_venc_set_video_param(enc_size, chn_param, CHN_NUM_MAX, gop_attr, TD_FALSE);
    if (ret != TD_SUCCESS) {
        return ret;
    }

    /* 第四步：启用编码通道的mini缓冲区模式（优化内存占用，适合低资源场景） */
    ret = sample_comm_venc_mini_buf_en(chn_param, CHN_NUM_MAX);
    if (ret != TD_SUCCESS) {
        return ret;
    }

    /* 第五步：启动第0路编码（H.265格式，通常为高分辨率主码流） */
    // 启动指定VENC通道（venc_vpss_chn->venc_chn[0]），并传入该通道的参数（chn_param[0]）
    ret = sample_comm_venc_start(venc_vpss_chn->venc_chn[0], &(chn_param[0]));
    if (ret != TD_SUCCESS) {
        sample_print("Venc Start failed for %#x!\n", ret);  // 打印H.265编码启动失败的错误码
        return ret;
    }

    /* 第六步：绑定第0路VPSS与第0路VENC（让VPSS预处理后的数据流向VENC编码） */
    if (sample_comm_vpss_bind_venc(vpss_grp, venc_vpss_chn->vpss_chn[0], venc_vpss_chn->venc_chn[0]) != TD_SUCCESS) {
        sample_print("sample_comm_vpss_bind_venc failed!\n");  // 打印VPSS-VENC绑定失败信息
        goto EXIT_VENC_H265_STOP;  // 绑定失败，跳转到停止H.265编码的清理步骤
    }

    /* 第七步：启动第1路编码（H.264格式，通常为低分辨率子码流） */
    ret = sample_comm_venc_start(venc_vpss_chn->venc_chn[1], &(chn_param[1]));
    if (ret != TD_SUCCESS) {
        sample_print("Venc Start failed for %#x!\n", ret);  // 打印H.264编码启动失败的错误码
        goto EXIT_VENC_H265_UnBind;  // 启动失败，跳转到解绑H.265并停止H.265的清理步骤
    }

    /* 第八步：绑定第1路VPSS与第1路VENC */
    if (sample_comm_vpss_bind_venc(vpss_grp, venc_vpss_chn->vpss_chn[1], venc_vpss_chn->venc_chn[1]) != TD_SUCCESS) {
        sample_print("sample_comm_vpss_bind_venc failed!\n");
        goto EXIT_VENC_H264_STOP;  // 绑定失败，跳转到停止H.264编码的清理步骤
    }

    /* 第九步：启动码流获取（开始从VENC通道读取编码后的H.264/H.265码流，用于存储或推流） */
    ret = sample_comm_venc_start_get_stream(venc_vpss_chn->venc_chn, CHN_NUM_MAX);
    if (ret != TD_SUCCESS) {
        sample_print("Start Venc failed!\n");
        goto EXIT_VENC_H264_UnBind;  // 码流获取启动失败，跳转到解绑H.264的清理步骤
    }

    // 所有编码流程启动成功，返回成功状态
    return TD_SUCCESS;


    /* 错误清理步骤：按“反向顺序”释放资源，避免内存/通道泄漏 */
    // 1. H.264绑定失败：解绑第1路VPSS-VENC
EXIT_VENC_H264_UnBind:
    sample_comm_vpss_un_bind_venc(vpss_grp, venc_vpss_chn->vpss_chn[1], venc_vpss_chn->venc_chn[1]);
    // 2. H.264启动失败：停止第1路VENC编码
EXIT_VENC_H264_STOP:
    sample_comm_venc_stop(venc_vpss_chn->venc_chn[1]);
    // 3. H.265绑定失败：解绑第0路VPSS-VENC
EXIT_VENC_H265_UnBind:
    sample_comm_vpss_un_bind_venc(vpss_grp, venc_vpss_chn->vpss_chn[0], venc_vpss_chn->venc_chn[0]);
    // 4. H.265启动失败：停止第0路VENC编码
EXIT_VENC_H265_STOP:
    sample_comm_venc_stop(venc_vpss_chn->venc_chn[0]);

    // 所有清理步骤完成，返回失败状态
    return TD_FAILURE;
}
~~~

- **VI（Video Input）**：视频输入（摄像头采集）；
- **VPSS（Video Process Sub-System）**：视频预处理（如缩放、裁剪、降噪）；
- **NPU（Neural Processing Unit）**：神经网络处理器（AI 推理加速）；
- **VENC（Video Encoder）**：视频编码（如 H.264/H.265 压缩）。

~~~c

/* start get venc stream process thread */
td_s32 sample_comm_venc_start_get_stream(ot_venc_chn *venc_chn, td_s32 cnt)
{
    td_s32 i, ret;
    td_s32 wait_times = WAIT_GET_STREAM_THREAD_START_TIMES;

    g_para.thread_start = TD_TRUE;
    g_para.cnt = cnt;
    if (cnt >= OT_VENC_MAX_CHN_NUM) {
        sample_print("input count invalid\n");
        return TD_FAILURE;
    }
    for (i = 0; (i < cnt) && (i < OT_VENC_MAX_CHN_NUM); i++) {
        g_para.venc_chn[i] = venc_chn[i];
    }

    ret = pthread_create(&g_venc_pid, 0, sample_comm_venc_get_venc_stream_proc, (td_void *)&g_para);
    if (ret < 0) {
        sample_print("create threa failed !!! ret %d\n", ret);
        return TD_FAILURE;
    }

    while (g_para.thread_start && wait_times-- > 0) {
        usleep(WAIT_GET_STREAM_THREAD_START_SLEEP);
        if (!g_para.thread_start) {
            sample_print("start thread failed !!!\n");
            pthread_join(g_venc_pid, 0);
            return TD_FAILURE;
        }
    }

    return ret;
}
~~~

~~~c
{
    td_s32 i, ret;
    fd_set read_fds;
    struct timeval timeout_val = {0};
    ot_payload_type payload_type[OT_VENC_MAX_CHN_NUM] = {0};
    ot_venc_stream_buf_info stream_buf_info[OT_VENC_MAX_CHN_NUM];
    sample_venc_getstream_para *para = (sample_venc_getstream_para *)p;
    sample_comm_venc_stream_proc_info *stream_proc_info = malloc(sizeof(sample_comm_venc_stream_proc_info));

    if (stream_proc_info == TD_NULL || memset_s(stream_proc_info, sizeof(sample_comm_venc_stream_proc_info), 0,
        sizeof(sample_comm_venc_stream_proc_info)) != EOK) {
        goto FREE;
    }

    prctl(PR_SET_NAME, "get_venc_stream", 0, 0, 0);

    stream_proc_info->chn_total = para->cnt;
    ret = sample_comm_set_name_save_stream(stream_proc_info, stream_buf_info, payload_type, para, OT_VENC_MAX_CHN_NUM);
    if (ret == SAMPLE_RETURN_NULL) {
        goto CLOSE_FILE;
    } else if (ret == SAMPLE_RETURN_FAILURE) {
        goto CLOSE_FILE;
    }

    while (para->thread_start == TD_TRUE) {
        FD_ZERO(&read_fds);
        for (i = 0; (i < stream_proc_info->chn_total) && (i < OT_VENC_MAX_CHN_NUM); i++) {
~~~

从你提供的 Hi3516 平台 EfficientNet 模型仿真日志来看，**功能仿真和指令仿真是嵌入式 AI 开发中两种不同层级的仿真验证手段**，核心区别在于 “是否关联目标硬件的具体指令执行”—— 功能仿真验证 “模型计算逻辑对不对”，指令仿真验证 “模型在目标硬件上跑的指令对不对、效率如何”，以下是详细拆解和对比：

### 一、先明确：两种仿真的核心定义与日志对应分析

#### 1. 功能仿真（Functional Simulation，日志中 `./func_main` 执行结果）

**核心定位**：仅验证 “模型的计算功能是否正确”，不依赖具体目标硬件（如 Hi3516 的 CPU/NN 加速器），本质是 “软件层面的逻辑验证”。

简单说：它只关心 “输入一张狗的图片，模型是否能算出正确的分类结果（如 top5 索引）”，不关心 “这些计算在 Hi3516 上是用哪条硬件指令实现的、占用多少硬件资源”。

##### 从你的功能仿真日志看关键特征：

- **仿真对象是 “模型层计算逻辑”**：日志详细打印了每一层的计算过程（如 `layer forward id(1) type(convolution) name(/_conv_stem/Conv)`），包括卷积、池化、激活函数（sigmoid）等层的执行顺序和类型，重点验证 “层与层之间的数据流转是否正确”。
- **无硬件指令 / 资源相关信息**：日志中没有出现 “硬件版本（如 InstSimVersion）”“指令跳转（JumpNum）”“硬件单元（如 CCU）” 等关键词，说明不涉及具体硬件的指令执行细节。
- **目标是 “功能正确性”**：最终输出 `model execute success` 和 top5 分类结果（index [162] 等），核心验证 “模型没有计算逻辑错误（如层漏算、数据维度错配）”，至于这些计算在 Hi3516 上能不能跑、跑多快，功能仿真不关心。
- **速度相对较快**：日志中功能仿真耗时未明确标注，但从 “仅计算层逻辑” 的特性来看，速度比指令仿真快 1-2 个数量级（适合快速验证模型是否跑通）。

#### 2. 指令仿真（Instruction Simulation，日志中 `./inst_main` 执行结果）

**核心定位**：基于目标硬件（如 Hi3516）的**具体指令集**，模拟 “硬件执行模型计算时的每一条指令”，本质是 “硬件层面的兼容性与性能验证”。

简单说：它不仅关心 “模型结果对不对”，更关心 “模型的计算逻辑能否被目标硬件的指令正确实现”（如卷积计算是否对应 Hi3516 的 NN 加速器指令）、“指令执行过程中是否有异常（如死循环）”“硬件资源是否够用”。

##### 从你的指令仿真日志看关键特征：

- **仿真对象是 “硬件指令执行过程”**：日志开头明确标注硬件仿真版本（`InstSimVersion: InstV200R001C001_V1.0.0.0_e7eda17`），这是 Hi3516 平台指令仿真器的版本，说明仿真过程完全贴合该硬件的指令集。
- 包含硬件级别的细节警告：
  - `SetCcuLayerInfo layerinfo size: 0`：CCU（Hi3516 的卷积计算单元）的层信息配置为空，可能是硬件指令与模型层的适配细节问题（不影响功能，但需关注硬件资源配置）；
  - `Model JumpNum: 80000, large JumpNum may indicate dead loop`：模拟硬件执行时的 “指令跳转次数” 过多，警告可能存在潜在的指令死循环风险（需排查硬件指令调度逻辑）。
- **输出 “硬件执行耗时”**：日志明确打印 `RunSimulator take 17.2383 seconds`，这是模拟 Hi3516 硬件执行所有指令的总耗时（功能仿真不会统计该数据，因为不涉及硬件指令）。
- **目标是 “硬件兼容性 + 性能预判”**：最终虽然也输出了正确的 top5 结果（与功能仿真一致，说明硬件指令执行结果正确），但核心价值是验证 “模型在 Hi3516 上的指令级适配性”—— 比如是否有指令不支持、资源占用是否超标、执行效率是否达标。

### 二、功能仿真 vs 指令仿真：7 个核心维度对比

结合你的 Hi3516 模型仿真场景，从 “开发实用角度” 做对比，更易理解两者的定位差异：

| 对比维度         | 功能仿真（Functional Simulation）                            | 指令仿真（Instruction Simulation）                           |
| ---------------- | ------------------------------------------------------------ | ------------------------------------------------------------ |
| **核心目标**     | 验证 “模型计算逻辑是否正确”（如层间数据流转、分类结果对不对） | 验证 “模型在目标硬件上的指令执行是否正确、高效”（如硬件兼容性、指令耗时） |
| **仿真对象**     | 模型的 “计算图逻辑”（如卷积、激活、池化的数学计算）          | 目标硬件的 “指令集执行”（如 Hi3516 的 CCU 指令、CPU 跳转指令） |
| **与硬件关联度** | 完全无关（可在 PC 端运行，无需目标硬件信息）                 | 强关联（必须基于目标硬件的指令集、硬件架构，如 Hi3516 的 InstSim） |
| **日志关键信息** | 层类型（convolution/sigmoid）、层执行进度、分类结果          | 硬件仿真版本、指令跳转次数（JumpNum）、硬件单元警告（CCU）、执行耗时 |
| **仿真速度**     | 快（仅算逻辑，无需模拟硬件指令），通常秒级完成               | 慢（需逐条模拟硬件指令），你的案例中耗时 17 秒（比功能仿真慢数倍） |
| **适用开发阶段** | 模型开发早期（如 PC 端调试模型、修改网络结构后快速验证）     | 模型移植后期（适配目标硬件后，验证硬件兼容性、预判实际运行性能） |
| **典型用途**     | 排查 “模型为什么算错”（如漏层、激活函数选错）                | 排查 “模型在硬件上为什么跑不通 / 跑的慢”（如指令不支持、死循环风险） |

### 三、结合你的场景：两种仿真的实际价值

从你的日志结果来看，两种仿真最终都输出了 `model execute success` 和相同的 top5 分类结果（index [162] 等），这说明：

1. **功能层面**：你的 EfficientNet 模型计算逻辑无错（功能仿真通过）；
2. **硬件适配层面**：模型的计算逻辑能被 Hi3516 的硬件指令正确实现（指令仿真也输出正确结果）；
3. **潜在优化点**：指令仿真的 `JumpNum` 警告（跳转次数过多），可能需要优化模型的硬件指令调度（如调整层融合策略），避免实际硬件运行时出现效率问题。

### 总结：开发中如何配合使用两种仿真？

在 Hi3516 这类嵌入式 AI 平台的模型开发流程中，两者通常是 “先功能、后指令” 的递进关系：

1. **第一步：功能仿真（PC 端 / 开发机）**：先在 PC 上用功能仿真验证模型跑通（如你的 `func_main`），快速排查 “模型结构错、数据预处理错、分类逻辑错” 等问题，避免带着功能错误去适配硬件；
2. **第二步：指令仿真（目标硬件平台）**：模型功能正确后，再用指令仿真（如你的 `inst_main`）验证 “硬件能否正确执行模型指令”，预判性能（耗时）、排查硬件兼容性问题（如指令死循环、资源不足）；
3. **第三步：实际硬件测试**：指令仿真通过后，再烧录到 Hi3516 实际硬件上运行，此时出现问题的概率会大幅降低。
