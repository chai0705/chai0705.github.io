~~~c
import evdev
import uinput

# ================= 配置区域 =================
# 1. 这里填入你在第三步查到的触摸屏名字（必须完全匹配，或者你可以修改代码通过 path 获取）
DEVICE_NAME = "eGalax Inc. USB TouchController" 

# 2. 你的屏幕分辨率或触摸范围 (需要先读取一次原始值确认，或者先试用 4096)
# eGalax 通常是 0-4095 或 0-32767。你可以先运行 evtest /dev/input/eventX 查看这一行：
# "Event code 0 (ABS_X) Value 123 Min 0 Max 4095"
MAX_X_VAL = 4095 
MAX_Y_VAL = 4095 

# 3. Y轴的分界线 (根据你说的上 1/4 翻转)
# 假设 0 是顶部，MAX_Y 是底部。上 1/4 大概是 0 到 1024
Y_SPLIT_POINT = MAX_Y_VAL * 0.25 
# ===========================================

def map_axis(abs_code, value):
    # --- 处理 X 轴 (ABS_X = 0) ---
    if abs_code == 0: 
        # 你的需求：X轴左右翻转
        # 算法：最大值 - 当前值
        return MAX_X_VAL - value

    # --- 处理 Y 轴 (ABS_Y = 1) ---
    elif abs_code == 1:
        # 你的需求：下 3/4 (大于分界线) 正常，上 1/4 (小于分界线) 翻转
        if value > Y_SPLIT_POINT:
            # 下部正常，直接返回
            return value
        else:
            # 上部翻转。这里需要你需要根据实际手感调整逻辑
            # 如果上部是“镜像”的，可能需要用 分界线 - 值
            # 或者你需要把这部分区间反转过来：
            return int(Y_SPLIT_POINT - value)
            
    return value

def main():
    # 1. 寻找设备
    devices = [evdev.InputDevice(path) for path in evdev.list_devices()]
    touch_device = None
    for dev in devices:
        if DEVICE_NAME in dev.name:
            touch_device = dev
            break
    
    if not touch_device:
        print(f"未找到设备: {DEVICE_NAME}")
        return

    print(f"正在抓取设备: {touch_device.name} ({touch_device.path})")
    
    # 2. 独占设备 (Grab)，这样系统就看不到原始的乱数据了
    touch_device.grab()

    # 3. 创建一个虚拟设备 (Virtual Touchscreen)
    # 我们复制原始设备的能力 (capabilities)，但会发送修正后的数据
    virtual_device = uinput.UInput.from_device(touch_device, name="eGalax Fixed Touch")

    print("开始修正数据... 按 Ctrl+C 停止")

    # 4. 循环读取并修改数据
    try:
        for event in touch_device.read_loop():
            # 只处理绝对坐标事件 (ABS)
            if event.type == evdev.ecodes.EV_ABS:
                # event.code 0 是 X轴, 1 是 Y轴
                if event.code == 0 or event.code == 1:
                    new_val = map_axis(event.code, event.value)
                    virtual_device.write(evdev.ecodes.EV_ABS, event.code, new_val)
                else:
                    # 其他 ABS 事件直接透传
                    virtual_device.write(event.type, event.code, event.value)
            else:
                # 按键、同步等其他事件直接透传
                virtual_device.write(event.type, event.code, event.value)
    except KeyboardInterrupt:
        pass
    finally:
        touch_device.ungrab()

if __name__ == "__main__":
    main()
~~~

~~~c

static int mt_process_slot(struct mt_device *td, struct input_dev *input,
			    struct mt_application *app,
			    struct mt_usages *slot)
{
	struct input_mt *mt = input->mt;
	__s32 quirks = app->quirks;
	bool valid = true;
	bool confidence_state = true;
	bool inrange_state = false;
	int active;
	int slotnum;
	int tool = MT_TOOL_FINGER;

	if (!slot)
		return -EINVAL;

	if ((quirks & MT_QUIRK_CONTACT_CNT_ACCURATE) &&
	    app->num_received >= app->num_expected)
		return -EAGAIN;

	if (!(quirks & MT_QUIRK_ALWAYS_VALID)) {
		if (quirks & MT_QUIRK_VALID_IS_INRANGE)
			valid = *slot->inrange_state;
		if (quirks & MT_QUIRK_NOT_SEEN_MEANS_UP)
			valid = *slot->tip_state;
		if (quirks & MT_QUIRK_VALID_IS_CONFIDENCE)
			valid = *slot->confidence_state;

		if (!valid)
			return 0;
	}

	slotnum = mt_compute_slot(td, app, slot, input);
	if (slotnum < 0 || slotnum >= td->maxcontacts)
		return 0;

	if ((quirks & MT_QUIRK_IGNORE_DUPLICATES) && mt) {
		struct input_mt_slot *i_slot = &mt->slots[slotnum];

		if (input_mt_is_active(i_slot) &&
		    input_mt_is_used(mt, i_slot))
			return -EAGAIN;
	}

	if (quirks & MT_QUIRK_CONFIDENCE)
		confidence_state = *slot->confidence_state;

	if (quirks & MT_QUIRK_HOVERING)
		inrange_state = *slot->inrange_state;

	active = *slot->tip_state || inrange_state;

	if (app->application == HID_GD_SYSTEM_MULTIAXIS)
		tool = MT_TOOL_DIAL;
	else if (unlikely(!confidence_state)) {
		tool = MT_TOOL_PALM;
		if (!active && mt &&
		    input_mt_is_active(&mt->slots[slotnum])) {
			/*
			 * The non-confidence was reported for
			 * previously valid contact that is also no
			 * longer valid. We can't simply report
			 * lift-off as userspace will not be aware
			 * of non-confidence, so we need to split
			 * it into 2 events: active MT_TOOL_PALM
			 * and a separate liftoff.
			 */
			active = true;
			set_bit(slotnum, app->pending_palm_slots);
		}
	}

	input_mt_slot(input, slotnum);
	input_mt_report_slot_state(input, tool, active);
	if (active) {
		/* ================= 原有逻辑保持不变 (变量声明) ================= */
		/* this finger is in proximity of the sensor */
		int wide = (*slot->w > *slot->h);
		int major = max(*slot->w, *slot->h);
		int minor = min(*slot->w, *slot->h);
		int orientation = wide;
		int max_azimuth;
		int azimuth;
		__s32 final_x,final_y;
		if (slot->a != DEFAULT_ZERO) {
			/*
			 * Azimuth is counter-clockwise and ranges from [0, MAX)
			 * (a full revolution). Convert it to clockwise ranging
			 * [-MAX/2, MAX/2].
			 */
			azimuth = *slot->a;
			max_azimuth = input_abs_get_max(input, ABS_MT_ORIENTATION);
			if (azimuth > max_azimuth * 2)
				azimuth -= max_azimuth * 4;
			orientation = -azimuth;
		}

		if (quirks & MT_QUIRK_TOUCH_SIZE_SCALING) {
			/*
			 * divided by two to match visual scale of touch
			 * for devices with this quirk
			 */
			major = major >> 1;
			minor = minor >> 1;
		}

		/* ================= START: eGalax 修复补丁 (插入在这里) ================= */
		final_x = *slot->x;
        final_y = *slot->y;

		/* 2. 只有当 Vendor=0xeef, Product=0xc002 且 名字包含 "CE6130" 时才生效 */
        /* 注意：td->hdev->name 存储了 dmesg 中显示的设备名称 */
        if (td->hdev->vendor == 0xeef && td->hdev->product == 0xc002 &&
            strstr(td->hdev->name, "CE6130")) {
            
            /* 定义最大值 */
            int MAX_VAL = 16383;

            /* --- X 轴修复 (包含校准参数) --- */
            /* 对应 xinput Matrix: -1.05 0 0.99 */
            
            /* Step 1: 翻转 */
            final_x = MAX_VAL - final_x;
            
            /* Step 2: 缩放 1.05 并偏移 -0.06 (16383*0.06 = 983) */
            final_x = (int)((long)final_x * 105 / 100) - 983;

            /* Step 3: 防止越界 */
            if (final_x < 0) final_x = 0;
            if (final_x > MAX_VAL) final_x = MAX_VAL;

            /* 注意：我看你之前代码里有一句 final_y = 8192; */
            /* 如果那是为了测试用的，可以保留；如果不是，请记得删除或还原为正确的 Y 轴校准逻辑 */
            final_y = 8192;
        }
		/* ================= END: eGalax 修复补丁 ================= */

		/* 3. 上报坐标 (使用修改后的 final_x 和 final_y) */
		input_event(input, EV_ABS, ABS_MT_POSITION_X, final_x);
		input_event(input, EV_ABS, ABS_MT_POSITION_Y, final_y);
		
		/* 4. 上报其他属性 (使用上面声明的变量) */
		input_event(input, EV_ABS, ABS_MT_TOOL_X, *slot->cx);
		input_event(input, EV_ABS, ABS_MT_TOOL_Y, *slot->cy);
		input_event(input, EV_ABS, ABS_MT_DISTANCE, !*slot->tip_state);
		input_event(input, EV_ABS, ABS_MT_ORIENTATION, orientation);
		input_event(input, EV_ABS, ABS_MT_PRESSURE, *slot->p);
		input_event(input, EV_ABS, ABS_MT_TOUCH_MAJOR, major);
		input_event(input, EV_ABS, ABS_MT_TOUCH_MINOR, minor);

		set_bit(MT_IO_FLAGS_ACTIVE_SLOTS, &td->mt_io_flags);
	}

	return 0;
}
~~~
