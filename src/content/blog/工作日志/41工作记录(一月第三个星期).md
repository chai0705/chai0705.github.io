~~~c
#include <stdio.h>
#include <sys/stat.h>
#include <sys/types.h>
#include <fcntl.h>
#include <unistd.h>
#include <linux/input.h>

int main()
{
    int fd;
    struct input_event in_ev;
    
    fd = open("/dev/input/event3", O_RDONLY);
    
    while(1)
    {
        read(fd, &in_ev, sizeof(struct input_event));
        if(in_ev.type == EV_KEY)
        {
            switch(in_ev.code)
            {
                case KEY_VOLUMEUP:
                    printf("音量加按键");
                    break;
                case KEY_VOLUMEDOWN:
                    printf("音量减按键");
                    break;
                case KEY_BACK:
                    printf("返回按键");
                    break;
                case KEY_MENU:
                    printf("HOME按键");
                    break;
            }
            switch(in_ev.value)
            {
                case 0:
                    printf("弹起\n");
                    break;
                case 1:
                    printf("按下\n");
                    break;                       
            }
        }
    }
    return 0;
}
~~~

为什么可以这么设置呢？

首先来看奇校验，为什么这些标志位的最开始都是PAR呢，这是因为parity英文是校验，而enb是使能的缩写，

在您提供的代码片段中，`|=` 和 `&= ~` 运算符用于设置或清除特定位，以配置串口的校验方式。以下是对这些运算符的详细解释：

### `|=` 运算符

`|=` 是按位或赋值运算符，用于将变量与一个值进行按位或运算，并将结果赋值给该变量。在 C 语言中，按位或运算符 `|` 会对两个操作数的每一位进行比较，只要其中一个操作数的对应位为 1，结果的该位就为 1。例如：

```c
options.c_cflag |= PARENB;
```

这行代码的作用是将 `PARENB` 标志位置为 1，而不改变 `options.c_cflag` 中的其他位。具体操作如下：

1. **`options.c_cflag`** 的当前值（假设为 8 位二进制表示）：`0010 1100`
2. **`PARENB`** 的值：`0000 1000`
3. **按位或运算结果**：`0010 1100 | 0000 1000 = 0010 1100`

由于 `PARENB` 位在 `options.c_cflag` 中已经为 1，因此按位或运算后，`options.c_cflag` 的值保持不变。

### `&= ~` 运算符

`&= ~` 是按位与赋值运算符和按位取反运算符的组合，用于清除特定位。按位与运算符 `&` 会对两个操作数的每一位进行比较，只有当两个操作数的对应位都为 1 时，结果的该位才为 1。按位取反运算符 `~` 会将操作数的每一位取反，即 0 变为 1，1 变为 0。例如：

```c
options.c_cflag &= ~PARODD;
```

这行代码的作用是将 `PARODD` 标志位清零，而不改变 `options.c_cflag` 中的其他位。具体操作如下：

1. **`options.c_cflag`** 的当前值（假设为 8 位二进制表示）：`0010 1100`
2. **`PARODD`** 的值：`0000 0100`
3. **`~PARODD`** 的值：`1111 1011`
4. **按位与运算结果**：`0010 1100 & 1111 1011 = 0010 1000`

通过上述操作，`PARODD` 位被清零，而其他位保持不变。

### 总结

- **`|=` 运算符**：用于设置特定位，将变量与一个值进行按位或运算，结果赋值给该变量。
- **`&= ~` 运算符**：用于清除特定位，先对值进行按位取反，然后与变量进行按位与运算，结果赋值给该变量。

通过使用这些运算符，可以方便地对特定位进行设置或清除，而不影响其他位的值。

1. **`INPCK`**：表示输入奇偶校验使能（**In**put **P**arity **Ch**ec**k**）。设置该标志位后，输入数据的奇偶校验功能被启用。
