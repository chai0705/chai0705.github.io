然后讲解了内核中是如何配置IIO触发器的

~~~C
  │     -> Device Drivers                                                                                │   
  │       -> Industrial I/O support (IIO [=y])                                                           │   
  │         -> Triggers - standalone 
      			<*> SYSFS trigger
~~~

对应的驱动源码为drivers/iio/trigger/iio-trig-sysfs.c，对应的驱动入口函数为：
~~~c
static int __init iio_sysfs_trig_init(void)
{
    // 初始化设备结构体
    device_initialize(&iio_sysfs_trig_dev);

    // 设置设备名称为 "iio_sysfs_trigger"
    dev_set_name(&iio_sysfs_trig_dev, "iio_sysfs_trigger");

    // 将设备添加到内核中
    return device_add(&iio_sysfs_trig_dev);
}
~~~

第4行，调用device_initialize函数初始化设备对象，确保设备结构体处于可用状态。

第7行，调用dev_set_name函数为设备设置一个名称，用于标识设备，这里设置的名称为iio_sysfs_trigger，也就是在sysfs子系统中看到的目录。

第10行，调用device_add函数将设备注册到内核的设备模型中，使其成为系统的一部分。

这三个函数都有一个共同的参数iio_sysfs_trig_dev，他是一个 struct device 结构体类型的变量，具体内容如下所示：
~~~c
static struct device iio_sysfs_trig_dev = {
    .bus = &iio_bus_type,    // 指定设备所属的总线类型为 IIO 总线
    .groups = iio_sysfs_trig_groups,    // 定义设备的属性组，用于在 sysfs 中暴露设备属性
    .release = &iio_trigger_sysfs_release,    // 指定设备释放时的回调函数，用于清理资源
};	
~~~

该结构体中的groups成员用来存放一些iio设备相关的属性信息，接下来查看iio_sysfs_trig_groups内容：
~~~c
static const struct attribute_group *iio_sysfs_trig_groups[] = {
	&iio_sysfs_trig_group,
	NULL
};
static const struct attribute_group iio_sysfs_trig_group = {
	.attrs = iio_sysfs_trig_attrs,
~~~

~~~c
static int iio_sysfs_trigger_probe(int id)
{
    // 定义一个指向 iio_sysfs_trig 结构的指针 t，用于后续操作
    struct iio_sysfs_trig *t;
    int ret; // 用于存储返回值
    bool foundit = false; // 标志变量，用于判断是否找到重复的触发器 ID

    // 加锁，确保对共享资源 iio_sysfs_trig_list 的访问是线程安全的
    mutex_lock(&iio_sysfs_trig_list_mut);

    // 遍历 iio_sysfs_trig_list 链表，检查是否存在相同的 ID
    list_for_each_entry(t, &iio_sysfs_trig_list, l) {
        if (id == t->id) { // 如果找到相同的 ID
            foundit = true; // 设置标志为 true
            break; // 跳出循环
        }
    }

    // 如果找到了重复的 ID，返回错误码 -EINVAL（无效参数）
    if (foundit) {
        ret = -EINVAL;
        goto out1; // 跳转到解锁并返回的代码段
    }

    // 分配内存以创建一个新的 iio_sysfs_trig 结构
    t = kmalloc(sizeof(*t), GFP_KERNEL);
    if (t == NULL) { // 如果内存分配失败
        ret = -ENOMEM; // 返回错误码 -ENOMEM（内存不足）
        goto out1; // 跳转到解锁并返回的代码段
    }

    // 初始化新触发器的 ID
    t->id = id;

    // 分配一个新的 IIO 触发器，并命名为 "sysfstrig%d"，其中 %d 是传入的 ID
    t->trig = iio_trigger_alloc("sysfstrig%d", id);
    if (!t->trig) { // 如果触发器分配失败
        ret = -ENOMEM; // 返回错误码 -ENOMEM（内存不足）
        goto free_t; // 跳转到释放已分配内存的代码段
    }

    // 设置触发器的属性组
    t->trig->dev.groups = iio_sysfs_trigger_attr_groups;

    // 设置触发器的操作函数集
    t->trig->ops = &iio_sysfs_trigger_ops;

    // 设置触发器的父设备
    t->trig->dev.parent = &iio_sysfs_trig_dev;

    // 将触发器的私有数据设置为当前的 iio_sysfs_trig 结构
    iio_trigger_set_drvdata(t->trig, t);

    // 初始化中断工作队列，用于处理触发器的工作
    init_irq_work(&t->work, iio_sysfs_trigger_work);

    // 注册触发器到 IIO 子系统
    ret = iio_trigger_register(t->trig);
    if (ret) // 如果注册失败
        goto out2; // 跳转到释放触发器资源的代码段

    // 将新触发器添加到全局链表 iio_sysfs_trig_list 中
    list_add(&t->l, &iio_sysfs_trig_list);

    // 增加模块的引用计数，防止模块被卸载
    __module_get(THIS_MODULE);

    // 解锁互斥锁，允许其他线程访问共享资源
    mutex_unlock(&iio_sysfs_trig_list_mut);

    return 0; // 返回成功

out2:
    // 如果触发器注册失败，释放触发器资源
    iio_trigger_free(t->trig);

free_t:
    // 如果触发器分配失败，释放 iio_sysfs_trig 结构的内存
    kfree(t);

out1:
    // 解锁互斥锁，确保在任何情况下都释放锁
    mutex_unlock(&iio_sysfs_trig_list_mut);

    return ret; // 返回错误码
}
~~~

该函数只有一个参数为id，表示要创建触发器的ID,接下来对该函数进行详细的分析。

第12-17行：遍历列表，检查是否存在相同 id 的触发器。

第26-40行：动态分配内存以存储新的触发器结构体，然后对触发器对象进行初始化，通过传入的ID值创建文件名，例如传入ID为0时，文件名为sysfstrig0

第43-52行：分别对触发器的属性组、操作函数、父设备以及私有数据进行设置。
