~~~c
// SPDX-License-Identifier: GPL-2.0-only
/*
 * Simple Linux Kernel Module to Read and Print Serial Number from Device Tree Root Node
 * Assumes U-Boot has placed the serial# env var into the DT as /serial-number or /serial#
 * This version reads directly from the root node ('/') on module init and prints it.
 */

#include <linux/init.h>
#include <linux/module.h>
#include <linux/kernel.h>
#include <linux/of.h> // For device tree support
#include <linux/string.h>

#define MODULE_NAME "simple_dt_serial_printer"
#define VENDOR_SN_MAX 513 // Define the maximum length for the serial string

// Module init function
static int __init simple_dt_serial_init(void)
{
    struct device_node *root_node; // Pointer to the root node
    const char *serial_prop;
    char serial_buffer[VENDOR_SN_MAX] = {0}; // Local buffer to hold the serial
    int ret;

    pr_info("%s: Initializing module\n", MODULE_NAME);

    // Find the root node (named "/")
    root_node = of_find_node_by_path("/");
    if (!root_node) {
        pr_err("%s: Failed to find root device tree node '/'.\n", MODULE_NAME);
        return -ENODEV;
    }

    pr_info("%s: Found root device tree node.\n", MODULE_NAME);

    // Try to read the "serial-number" property first (common convention)
    ret = of_property_read_string(root_node, "serial-number", &serial_prop);
    if (ret) {
        pr_info("%s: 'serial-number' property not found, trying 'serial#'...\n", MODULE_NAME);
        // If "serial-number" not found, try the original "serial#" property name
        ret = of_property_read_string(root_node, "serial#", &serial_prop);
        if (ret) {
            pr_warn("%s: Neither 'serial-number' nor 'serial#' property found in DT root node.\n", MODULE_NAME);
            strscpy(serial_buffer, "NOT_FOUND_IN_DT", sizeof(serial_buffer));
        } else {
             pr_info("%s: Found serial using 'serial#' property.\n", MODULE_NAME);
             strscpy(serial_buffer, serial_prop, sizeof(serial_buffer));
        }
    } else {
        // Successfully read from "serial-number"
        pr_info("%s: Found serial using 'serial-number' property.\n", MODULE_NAME);
        strscpy(serial_buffer, serial_prop, sizeof(serial_buffer));
    }

    // Release the reference to the device node
    of_node_put(root_node);

    // --- Print the final result ---
    pr_info("%s: **********************************************\n", MODULE_NAME);
    pr_info("%s: * Serial Number Read from Device Tree:       *\n", MODULE_NAME);
    pr_info("%s: * '%s'                         *\n", MODULE_NAME, serial_buffer); // Adjust padding if needed
    pr_info("%s: **********************************************\n", MODULE_NAME);

    return 0; // Success
}

// Module exit function
static void __exit simple_dt_serial_exit(void)
{
    pr_info("%s: Module unloaded.\n", MODULE_NAME);
}

module_init(simple_dt_serial_init);
module_exit(simple_dt_serial_exit);

MODULE_LICENSE("GPL");
MODULE_AUTHOR("Your Name / Adapted");
MODULE_DESCRIPTION("A simple module to read and print serial number from DT root node");
MODULE_VERSION("1.0");
~~~

~~~c
// 从 efuse/otp 读取芯片序列号（基于 CPUID 计算 CRC）
static int get_chip_serial(char *serialno_str, int len)
{
#define VENDOR_SN_MAX 513
#define CPUID_LEN     0x10
#define CPUID_OFF     0x07

#if defined(CONFIG_ROCKCHIP_EFUSE) || defined(CONFIG_ROCKCHIP_OTP)
	u8 low[CPUID_LEN / 2], high[CPUID_LEN / 2];
	u8 cpuid[CPUID_LEN] = {0};
	struct udevice *dev;
	int ret, i;
	u64 serialno;

	// 获取 efuse/otp 设备
	if (IS_ENABLED(CONFIG_ROCKCHIP_EFUSE))
		ret = uclass_get_device_by_driver(UCLASS_MISC, DM_GET_DRIVER(rockchip_efuse), &dev);
	else
		ret = uclass_get_device_by_driver(UCLASS_MISC, DM_GET_DRIVER(rockchip_otp), &dev);

	if (ret) {
		LOG("could not find efuse/otp device\n");
		return ret;
	}

	// 读取 CPUID
	ret = misc_read(dev, CPUID_OFF, cpuid, sizeof(cpuid));
	if (ret) {
		LOG("read cpuid from efuse/otp failed, ret=%d\n", ret);
		return ret;
	}

	// 交错重组高低字节
	for (i = 0; i < 8; i++) {
		low[i]  = cpuid[1 + (i << 1)];
		high[i] = cpuid[i << 1];
	}

	// 使用 CRC32 构造 64 位序列号
	serialno = crc32_no_comp(0, low, 8);
	serialno |= (u64)crc32_no_comp(serialno, high, 8) << 32;
	snprintf(serialno_str, len, "%llx", serialno);
	return 0;
#else
	return -1;
#endif
}

// AES-CBC 加解密封装
static int do_aes(int enc, const uint8_t *key, uint8_t *src, uint8_t *dst, int len)
{
	uint8_t key_exp[AES_EXPAND_KEY_LENGTH];
	uint32_t aes_blocks = DIV_ROUND_UP(len, AES_KEY_LENGTH);

	aes_expand_key((uint8_t *)key, key_exp);
	if (enc)
		aes_cbc_encrypt_blocks(key_exp, src, dst, aes_blocks);
	else
		aes_cbc_decrypt_blocks(key_exp, src, dst, aes_blocks);
	return 0;
}

// 验证 RPMB 中的许可是否匹配当前序列号
static int do_verify_permit(char *serial)
{
	uint8_t src[256] = {0};
	uint8_t dst[256] = {0};
	char cmd[128];

	// 从 RPMB 读取加密数据
	sprintf(cmd, "mmc rpmb read 0x%08lx 0 %d 0x%08lx", (ulong)src, 1, (ulong)rpmb_test_key);
	if (run_command(cmd, 0) != CMD_RET_SUCCESS) {
		LOG("RPMB read failed\n");
		return CMD_RET_FAILURE;
	}

	// 解密并与序列号比对
	do_aes(0, rpmb_test_key, src, dst, sizeof(src));
	if (memcmp(dst, serial, strlen(serial)) != 0) {
		LOG("permit verify failed\n");
		return CMD_RET_FAILURE;
	}
	LOG("permit verify success\n");
	return CMD_RET_SUCCESS;
}

// verify 命令入口：自动验证，失败时可选自动烧录（调试模式）
static int do_verify(cmd_tbl_t *cmdtp, int flag, int argc, char *const argv[])
{
	char serial[128] = {0};

	if (argc < 2) {
		if (get_chip_serial(serial, sizeof(serial)))
			return CMD_RET_FAILURE;
	} else {
		strncpy(serial, argv[1], sizeof(serial) - 1);

#if DEBUG_AUTO_BURNPERMIT && DEBUG_BURNMODE

		return CMD_RET_SUCCESS;

	do_rpmb_burnkey(cmdtp, flag, argc, argv);
	do_rpmb_burnpermit(cmdtp, flag, argc, argv);
	return do_verify_permit(serial);
#else
	return do_verify_permit(serial);
#endif
}

// RPMB 自检：写入固定数据并读回校验
static int do_selftest(cmd_tbl_t *cmdtp, int flag, int argc, char *const argv[])
{
	char cmd[128];
	uint8_t data[256];

	// 初始化测试数据
	for (int i = 0; i < 256; i++)
		data[i] = i;

	// 写入 RPMB
	LOG("RPMB self test: write\n");
	sprintf(cmd, "mmc rpmb write 0x%08lx 0 %d 0x%08lx", (ulong)data, 1, (ulong)rpmb_test_key);
	if (run_command(cmd, 0) != CMD_RET_SUCCESS) {
		LOG("RPMB self test failed\n");
		return CMD_RET_FAILURE;
	}

	// 读回数据
	memset(data, 0, sizeof(data));
	LOG("RPMB self test: read\n");
	sprintf(cmd, "mmc rpmb read 0x%08lx 0 %d 0x%08lx", (ulong)data, 1, (ulong)rpmb_test_key);
	if (run_command(cmd, 0) != CMD_RET_SUCCESS) {
		LOG("RPMB self test failed\n");
		return CMD_RET_FAILURE;
	}

	// 校验一致性
	LOG("RPMB self test: check\n");
	for (int i = 0; i < 256; i++) {
		if (data[i] != i) {
			LOG("RPMB self test failed\n");
			return CMD_RET_FAILURE;
		}
	}
	return CMD_RET_SUCCESS;
}
~~~

~~~c
#include <linux/module.h>
#include <linux/init.h>
#include <linux/kernel.h>
#include <linux/crypto.h>
#include <crypto/skcipher.h>
#include <linux/scatterlist.h>
#include <linux/slab.h>
#include <linux/string.h>

#define DRIVER_NAME "aes_print"
#define AES_BLOCK_SIZE 16
#define BUF_SIZE   32

// 固定的测试 key（取前 16 字节，用于 AES-128）
static const u8 rpmb_key_128[16] = {
    0x5F, 0x2A, 0xCB, 0x15, 0x5C, 0x18, 0xBE, 0xE8,
    0xD3, 0x5D, 0x5A, 0x14, 0x31, 0x7F, 0xDA, 0xED
};

// 固定 serial
static const char serial_str[] = "95ee8a4e093468fe";

static int __init aes_print_init(void)
{
    struct crypto_skcipher *tfm;
    struct skcipher_request *req;
    struct scatterlist sg_src, sg_dst;
    u8 *src_buf, *dst_buf;
    u8 iv[AES_BLOCK_SIZE] = {0};
    int ret, i;

    pr_info(DRIVER_NAME ": init\n");

    // 分配 cipher
    tfm = crypto_alloc_skcipher("cbc(aes)", 0, 0);
    if (IS_ERR(tfm)) {
        pr_err(DRIVER_NAME ": crypto_alloc_skcipher failed\n");
        return PTR_ERR(tfm);
    }

    ret = crypto_skcipher_setkey(tfm, rpmb_key_128, sizeof(rpmb_key_128));
    if (ret) {
        pr_err(DRIVER_NAME ": setkey failed %d\n", ret);
        crypto_free_skcipher(tfm);
        return ret;
    }

    req = skcipher_request_alloc(tfm, GFP_KERNEL);
    if (!req) {
        pr_err(DRIVER_NAME ": request alloc failed\n");
        crypto_free_skcipher(tfm);
        return -ENOMEM;
    }

    src_buf = kzalloc(BUF_SIZE, GFP_KERNEL);
    dst_buf = kzalloc(BUF_SIZE, GFP_KERNEL);
    if (!src_buf || !dst_buf) {
        pr_err(DRIVER_NAME ": buffer alloc failed\n");
        ret = -ENOMEM;
        goto out_free;
    }

    // 拷贝 serial 到 src_buf 前面，其余补 0
    memcpy(src_buf, serial_str, strlen(serial_str));

    sg_init_one(&sg_src, src_buf, BUF_SIZE);
    sg_init_one(&sg_dst, dst_buf, BUF_SIZE);

    skcipher_request_set_crypt(req, &sg_src, &sg_dst, BUF_SIZE, iv);

    ret = crypto_skcipher_encrypt(req);
    if (ret) {
        pr_err(DRIVER_NAME ": encrypt failed %d\n", ret);
        goto out_free;
    }

    // 打印结果
~~~

~~~c
extern unsigned int system_serial_low;
extern unsigned int system_serial_high;

#define DRIVER_NAME "cpuid_aes"
#define AES_BLOCK_SIZE 16
#define BUF_SIZE   32   // 足够存放 CPU ID 字符串

// 固定的 AES-128 密钥（取前 16 字节）
static const u8 rpmb_key_128[16] = {
    0x5F, 0x2A, 0xCB, 0x15, 0x5C, 0x18, 0xBE, 0xE8,
    0xD3, 0x5D, 0x5A, 0x14, 0x31, 0x7F, 0xDA, 0xED
};

static int __init cpuid_aes_init(void)
{
    struct crypto_skcipher *tfm;
    struct skcipher_request *req;
    struct scatterlist sg_src, sg_dst;
    u8 *src_buf, *dst_buf;
    u8 iv[AES_BLOCK_SIZE] = {0}; // CBC 模式需要 IV，固定为 0
    int ret, i;
    char serial_str[BUF_SIZE];

    pr_info(DRIVER_NAME ": init\n");

    // 拼接 CPU ID 成字符串（高 32 位 + 低 32 位）
    snprintf(serial_str, sizeof(serial_str), "%08x%08x",
             system_serial_high, system_serial_low);

    pr_info(DRIVER_NAME ": CPU Serial = %s\n", serial_str);

    // 分配 AES 算法
    tfm = crypto_alloc_skcipher("cbc(aes)", 0, 0);
    if (IS_ERR(tfm)) {
        pr_err(DRIVER_NAME ": crypto_alloc_skcipher failed\n");
        return PTR_ERR(tfm);
    }

    // 设置密钥
    ret = crypto_skcipher_setkey(tfm, rpmb_key_128, sizeof(rpmb_key_128));
    if (ret) {
        pr_err(DRIVER_NAME ": setkey failed %d\n", ret);
        crypto_free_skcipher(tfm);
        return ret;
    }

    // 分配请求
    req = skcipher_request_alloc(tfm, GFP_KERNEL);
    if (!req) {
        pr_err(DRIVER_NAME ": request alloc failed\n");
        crypto_free_skcipher(tfm);
        return -ENOMEM;
    }

    // 分配源/目标 buffer
    src_buf = kzalloc(BUF_SIZE, GFP_KERNEL);
    dst_buf = kzalloc(BUF_SIZE, GFP_KERNEL);
    if (!src_buf || !dst_buf) {
        pr_err(DRIVER_NAME ": buffer alloc failed\n");
        ret = -ENOMEM;
        goto out_free;
    }

    // 拷贝 CPU ID 到源 buffer，剩余补 0
    memcpy(src_buf, serial_str, strlen(serial_str));

    // 初始化 scatterlist
    sg_init_one(&sg_src, src_buf, BUF_SIZE);
    sg_init_one(&sg_dst, dst_buf, BUF_SIZE);

    skcipher_request_set_crypt(req, &sg_src, &sg_dst, BUF_SIZE, iv);

    // 执行加密
    ret = crypto_skcipher_encrypt(req);
    if (ret) {
        pr_err(DRIVER_NAME ": encrypt failed %d\n", ret);
        goto out_free;
    }

    // 打印加密结果
    printk(KERN_INFO "AES-128-CBC result (hex):\n");
    for (i = 0; i < BUF_SIZE; i++) {
        printk(KERN_CONT "%02x%s",
               dst_buf[i],
               (i % 16 == 15) ? "\n" : " ");
    }

out_free:
    kfree(src_buf);
    kfree(dst_buf);
    skcipher_request_free(req);
    crypto_free_skcipher(tfm);
    return ret;
}
~~~

~~~c
/**
 * 从CPUID生成序列号
 */
static int generate_serial_from_cpuid(const u8 *cpuid, char *serial_str, int len)
{
    u8 cpuid_low[CPUID_HALF_LEN], cpuid_high[CPUID_HALF_LEN];
    u64 serialno = 0;
    int i;

    if (!cpuid || !serial_str || len <= 0)
        return -EINVAL;

    // 重组CPUID字节
    for (i = 0; i < CPUID_HALF_LEN; i++) {
        cpuid_low[i] = cpuid[1 + (i << 1)];  // 奇数索引
        cpuid_high[i] = cpuid[i << 1];       // 偶数索引
    }

    // 计算CRC32生成64位序列号
    serialno = crc32_no_comp(0, cpuid_low, CPUID_HALF_LEN);
    serialno |= (u64)crc32_no_comp(serialno, cpuid_high, CPUID_HALF_LEN) << 32;

    // 转换为十六进制字符串
    if (snprintf(serial_str, len, "%llx", serialno) >= len)
        return -ENOSPC;

    return 0;
}

/**
 * 获取官方格式序列号（优先厂商存储，其次EFUSE/OTP生成）

{
    u8 cpuid[CPUID_LEN] = {0};

    if (!serial_str || len <= 0)
        return -EINVAL;

    // 尝试读取厂商存储的序列号

        return 0;

    // 读取CPUID并生成序列号
    if ((ret = read_cpuid_from_efuse_otp(cpuid, sizeof(cpuid))) != 0)

    return generate_serial_from_cpuid(cpuid, serial_str, len);
}

        }
    }
~~~

~~~c
    if (offset < 0) {
        LOG("Warning: /chosen node not found, creating...\n");
        offset = fdt_add_subnode(fdt, 0, "chosen");
        if (offset < 0) {
            LOG("Failed to create /chosen node: %s\n", fdt_strerror(offset));
            return -1;
        }
    }

    // 尝试写入属性
    ret = fdt_setprop(fdt, offset, "rpmb_permit", permit_data, len);
    
    // 如果空间不足，尝试扩展设备树
    if (ret == -FDT_ERR_NOSPACE) {
        size = fdt_totalsize(fdt);
        temp_buffer = malloc(size + BUF_INC);
        if (!temp_buffer) {
            LOG("Failed to allocate temp FDT buffer (size=%zu)\n", size + BUF_INC);
            return -ENOMEM;
        }

        // 扩展设备树
        ret = fdt_open_into(fdt, temp_buffer, size + BUF_INC);
        if (ret < 0) {
            LOG("fdt_open_into failed: %s\n", fdt_strerror(ret));
            free(temp_buffer);
            return ret;
        }

        // 重新写入属性
        offset = fdt_path_offset(temp_buffer, chosen_path);
        if (offset < 0) {
            LOG("Failed to find /chosen node after expansion\n");
            free(temp_buffer);
            return -1;
        }

        ret = fdt_setprop(temp_buffer, offset, "rpmb_permit", permit_data, len);
        if (ret >= 0) {
            // ✅ 修复点：添加强制类型转换
            memcpy((void *)gd->fdt_blob, temp_buffer, fdt_totalsize(temp_buffer));
            gd->fdt_size = fdt_totalsize(gd->fdt_blob);
        }
        free(temp_buffer);
    } else if (ret < 0) {
        LOG("Failed to set rpmb_permit: %s\n", fdt_strerror(ret));
        return ret;
    }

    LOG("RPMB permit written to FDT at /chosen/rpmb_permit\n");
    return 0;
}

{

#define CPUID_LEN     0x10
#define CPUID_OFF     0x07

#if defined(CONFIG_ROCKCHIP_EFUSE) || defined(CONFIG_ROCKCHIP_OTP)
	u8 low[CPUID_LEN / 2], high[CPUID_LEN / 2];
	u8 cpuid[CPUID_LEN] = {0};

	if (IS_ENABLED(CONFIG_ROCKCHIP_EFUSE))
		ret = uclass_get_device_by_driver(UCLASS_MISC, DM_GET_DRIVER(rockchip_efuse), &dev);
	else

	snprintf(serialno_str, len, "%llx", serialno);
	return 0;
#else
	return -1;
#endif
}

// AES-CBC 加解密封装
static int do_aes(int enc, const uint8_t *key, uint8_t *src, uint8_t *dst, int len)
{

	uint32_t aes_blocks = DIV_ROUND_UP(len, AES_KEY_LENGTH);

	else
		aes_cbc_decrypt_blocks(key_exp, src, dst, aes_blocks);
	return 0;
}
~~~

~~~c
    if (IS_ERR(tfm)) {
        pr_err(DRIVER_NAME ": crypto_alloc_skcipher failed\n");
        return PTR_ERR(tfm);
    }

    ret = crypto_skcipher_setkey(tfm, rpmb_key_128, sizeof(rpmb_key_128));
    if (ret) {
        pr_err(DRIVER_NAME ": setkey failed %d\n", ret);
        crypto_free_skcipher(tfm);
        return ret;
    }

    req = skcipher_request_alloc(tfm, GFP_KERNEL);
    if (!req) {
        pr_err(DRIVER_NAME ": request alloc failed\n");
        crypto_free_skcipher(tfm);
        return -ENOMEM;
    }

    enc_buf = kzalloc(BUF_SIZE, GFP_KERNEL);
    if (!enc_buf) {
        pr_err(DRIVER_NAME ": buffer alloc failed\n");
        skcipher_request_free(req);
        crypto_free_skcipher(tfm);
        return -ENOMEM;
    }

    /* 拷贝 CPU ID 到源 buffer，剩余补0 */
    memcpy(enc_buf, serial_str, strlen(serial_str));
    sg_init_one(&sg_src, enc_buf, BUF_SIZE);
    sg_init_one(&sg_dst, enc_buf, BUF_SIZE);
    skcipher_request_set_crypt(req, &sg_src, &sg_dst, BUF_SIZE, iv);

    ret = crypto_skcipher_encrypt(req);
    if (ret) {
        pr_err(DRIVER_NAME ": encrypt failed %d\n", ret);
        goto out_free;
    }

    pr_info(DRIVER_NAME ": AES-128-CBC result (hex):\n");
    for (i = 0; i < BUF_SIZE; i++) {
        pr_cont("%02x%s", enc_buf[i], (i % 16 == 15) ? "\n" : " ");
    }

    /* ------------------ 对比 RPMB permit 和 AES 加密结果 ------------------ */
    if (memcmp(rpmb_permit, enc_buf, 32) == 0) {
        pr_info(DRIVER_NAME ": RPMB permit matches AES CPU ID\n");
    } else {
        pr_info(DRIVER_NAME ": RPMB permit does NOT match AES CPU ID\n");
    }

out_free:
    kfree(enc_buf);
    skcipher_request_free(req);
    crypto_free_skcipher(tfm);

    return ret;
}

static void __exit rpmb_check_exit(void)
{
    pr_info(DRIVER_NAME ": exit\n");
}

module_init(rpmb_check_init);
module_exit(rpmb_check_exit);

MODULE_LICENSE("GPL");
MODULE_AUTHOR("ChatGPT-assisted");
MODULE_DESCRIPTION("Read /chosen rpmb_permit, encrypt CPU ID, and compare");
~~~

~~~c
#define AES_BLOCK_SIZE 16

static const u8 rpmb_key_128[16] = {
    0x5F,0x2A,0xCB,0x15,0x5C,0x18,0xBE,0xE8,

static int check_rpmb_permit(void)
{
    struct device_node *chosen_node;
    const void *rpmb_permit;
    int len, i;
    char serial_str[BUF_SIZE];
    u8 *enc_buf;
    struct crypto_skcipher *tfm;
    struct skcipher_request *req;
    struct scatterlist sg_src, sg_dst;
    u8 iv[AES_BLOCK_SIZE] = {0};
    int ret = 0;

    printk(KERN_INFO "RPMB check init\n");

    chosen_node = of_find_node_by_path("/chosen");
    if (!chosen_node) {
        printk(KERN_ERR "Failed to find /chosen node\n");
        /* 没找到 /chosen 节点也必须宕机 */
        for (;;)
            cpu_relax();
    }

    rpmb_permit = of_get_property(chosen_node, "rpmb_permit", &len);
    if (!rpmb_permit || len != 32) {
        printk(KERN_ERR "Invalid rpmb_permit\n");
        of_node_put(chosen_node);
        for (;;)
            cpu_relax();
    }

    printk(KERN_INFO "RPMB permit:\n");
    for (i = 0; i < len; i++)
        printk(KERN_CONT "%02x%s", ((u8 *)rpmb_permit)[i],
               (i % 16 == 15) ? "\n" : " ");
    of_node_put(chosen_node);

    snprintf(serial_str, sizeof(serial_str), "%08x%08x",
             system_serial_high, system_serial_low);
    printk(KERN_INFO "CPU Serial = %s\n", serial_str);

    tfm = crypto_alloc_skcipher("cbc(aes)", 0, 0);
    if (IS_ERR(tfm)) {
        printk(KERN_ERR "crypto_alloc_skcipher failed\n");
        for (;;)
            cpu_relax();
    }

    ret = crypto_skcipher_setkey(tfm, rpmb_key_128, sizeof(rpmb_key_128));
    if (ret) {
        printk(KERN_ERR "setkey failed %d\n", ret);
        crypto_free_skcipher(tfm);
        for (;;)
            cpu_relax();
    }

    req = skcipher_request_alloc(tfm, GFP_KERNEL);
    if (!req) {
        printk(KERN_ERR "request alloc failed\n");
        crypto_free_skcipher(tfm);
        for (;;)
            cpu_relax();
    }

    enc_buf = kzalloc(BUF_SIZE, GFP_KERNEL);
    if (!enc_buf) {
        printk(KERN_ERR "buffer alloc failed\n");
        skcipher_request_free(req);
        crypto_free_skcipher(tfm);
        for (;;)
            cpu_relax();
    }

    memcpy(enc_buf, serial_str, strlen(serial_str));
    sg_init_one(&sg_src, enc_buf, BUF_SIZE);
    sg_init_one(&sg_dst, enc_buf, BUF_SIZE);
    skcipher_request_set_crypt(req, &sg_src, &sg_dst, BUF_SIZE, iv);

    ret = crypto_skcipher_encrypt(req);
    if (ret) {
        printk(KERN_ERR "encrypt failed %d\n", ret);
        goto out_free;
    }

    printk(KERN_INFO "AES-128-CBC result:\n");
    for (i = 0; i < BUF_SIZE; i++)
        printk(KERN_CONT "%02x%s", enc_buf[i], (i % 16 == 15) ? "\n" : " ");

    if (!memcmp(rpmb_permit, enc_buf, 32)) {
        printk(KERN_INFO "RPMB permit matches AES CPU ID\n");
    } else {
        printk(KERN_INFO "RPMB permit does NOT match AES CPU ID\n");
        for (;;)
            cpu_relax();
    }

out_free:
    kfree(enc_buf);
    skcipher_request_free(req);
    crypto_free_skcipher(tfm);
    return ret;
}

device_initcall(check_rpmb_permit);
~~~
