388|# target: what to clean
389|# "make" - "make clean" for selected kernel and u-boot
390|# "debs" - delete output/debs for board&branch
391|# "ubootdebs" - delete output/debs for uboot&board&branch
392|# "alldebs" - delete output/debs

394|# "oldcache" - remove old output/cache

396|# "sources" - delete output/sources

399|cleaning()
400|{
401|	case $1 in
402|		debs) # delete ${DEB_STORAGE} for current branch and family
403|		if [[ -d "${DEB_STORAGE}" ]]; then
404|			display_alert "Cleaning ${DEB_STORAGE} for" "$BOARD $BRANCH" "info"
405|			# easier than dealing with variable expansion and escaping dashes in file names
406|			find "${DEB_STORAGE}" -name "${CHOSEN_UBOOT}_*.deb" -delete
407|			find "${DEB_STORAGE}" \( -name "${CHOSEN_KERNEL}_*.deb" -o \
408|				-name "orangepi-*.deb" -o \
409|				-name "plymouth-theme-orangepi_*.deb" -o \
410|				-name "${CHOSEN_KERNEL/image/dtb}_*.deb" -o \
411|				-name "${CHOSEN_KERNEL/image/headers}_*.deb" -o \
412|				-name "${CHOSEN_KERNEL/image/source}_*.deb" -o \
413|				-name "${CHOSEN_KERNEL/image/firmware-image}_*.deb" \) -delete
414|			[[ -n $RELEASE ]] && rm -f "${DEB_STORAGE}/${RELEASE}/${CHOSEN_ROOTFS}"_*.deb
415|			[[ -n $RELEASE ]] && rm -f "${DEB_STORAGE}/${RELEASE}/orangepi-desktop-${RELEASE}"_*.deb
416|		fi
417|		;;
418|
419|		ubootdebs) # delete ${DEB_STORAGE} for uboot, current branch and family
420|		if [[ -d "${DEB_STORAGE}" ]]; then
421|			display_alert "Cleaning ${DEB_STORAGE} for u-boot" "$BOARD $BRANCH" "info"
