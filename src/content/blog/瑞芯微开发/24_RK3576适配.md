ddrbin_tool工具简介

`ddrbin_tool` 是一款用于处理 DDR 配置文件（ddr.bin）的工具。它提供了两种功能，分别用于修改 DDR 配置以及从 DDR 配置文件中提取配置参数。

**1.修改 `ddr.bin` 文件（功能1）**

该功能通过读取 `ddrbin_param.txt` 文件中的配置参数，生成新的 `ddr.bin` 文件。可以修改 DDR 频率、UART 信息等内容。

**步骤**：

修改 `ddrbin_param.txt` 文件，填入你想设置的 DDR 频率、UART 信息等参数。如果要保持默认值，可以将这些项目留空。

使用以下命令运行工具：
```bash
./ddrbin_tool <platform> ddrbin_param.txt <output_ddr_file>
```

**示例**：

```bash
./ddrbin_tool px30 ddrbin_param.txt px30_ddr_333MHz_v1.13.bin
```
此示例将在 px30 平台上，根据 `ddrbin_param.txt` 中的参数生成 `px30_ddr_333MHz_v1.13.bin` 文件。

**2.提取 `ddr.bin` 配置（功能2）**

该功能从现有的 `ddr.bin` 文件中读取配置参数，并将这些参数导出到 `gen_param.txt` 文件中。

**步骤**：

使用以下命令运行工具：
```bash
./ddrbin_tool <platform> -g <output_param_file> <input_ddr_file>
```

**示例**：

```bash
./ddrbin_tool px30 -g gen_param.txt px30_ddr_333MHz_v1.15.bin
```
此示例会将 `px30_ddr_333MHz_v1.15.bin` 文件中的配置参数提取并导出到 `gen_param.txt` 文件。

 **注意事项**

- 功能1和功能2是独立的，生成的 `gen_param.txt` 文件并不需要用于功能1。
- 详细说明和参数含义请参考工具随附的 `ddrbin_tool_user_guide.txt` 文档。

**工具使用示例**

```bash
# 修改 DDR 配置文件
./ddrbin_tool px30 ddrbin_param.txt px30_ddr_333MHz_v1.13.bin

# 提取 DDR 配置参数
./ddrbin_tool px30 -g gen_param.txt px30_ddr_333MHz_v1.15.bin
```

整理好的脚本如下所示：
~~~shell
#!/bin/bash

# 定义芯片型号及对应的参数文件
declare -A chips=(
    ["rk3568"]="rk3568_ddr_1560MHz_v1.23.bin"
    ["rk3588"]="rk3588_ddr_lp4_2112MHz_lp5_2400MHz_v1.18.bin"
    ["rk3562"]="rk3562_ddr_1332MHz_v1.06.bin"
    ["rk3576"]="rk3576_ddr_lp4_2112MHz_lp5_2736MHz_v1.05.bin"
)

# 颜色代码
GREEN='\033[0;32m'
NC='\033[0m' # No Color

# 显示选择菜单
echo -e "${GREEN}请选择要处理的芯片型号：${NC}"
echo -e "${GREEN}0: rk3568${NC}"
echo -e "${GREEN}1: rk3588${NC}"
echo -e "${GREEN}2: rk3562${NC}"
echo -e "${GREEN}3: rk3576${NC}"
echo -e "${GREEN}4: 全部处理${NC}"
read -p "请输入你的选择（0-4）: " choice

# 定义选择对应的芯片名称数组
selected_chips=()
case $choice in
    0) selected_chips=("rk3568");;
    1) selected_chips=("rk3588");;
    2) selected_chips=("rk3562");;
    3) selected_chips=("rk3576");;
    4) selected_chips=("rk3568" "rk3588" "rk3562" "rk3576");;
    *) echo "无效选择！请重新运行脚本并输入有效选项。"; exit 1;;
esac

# 遍历选择的芯片型号，执行相应的操作
for chip in "${selected_chips[@]}"; do
    param_file="${chip}_gen_param.txt"
    ddr_file="${chips[$chip]}"

    echo "处理 $chip ..."

    # 读取 DDR 配置到参数文件
    ../../tools/ddrbin_tool $chip -g $param_file $ddr_file

    # 修改波特率
    sed -i 's/uart baudrate=1500000/uart baudrate=115200/' $param_file

    # 写回修改后的参数
    ../../tools/ddrbin_tool $chip $param_file $ddr_file

    # 删除参数文件
    rm -rf $param_file

    echo "$chip 处理完成！"
done

echo "处理完毕！"


../../tools/ddrbin_tool rk3576 -g gen_param.txt rk3576_ddr_lp4_2112MHz_lp5_2736MHz_v1.05.bin
~~~

~~~shell

#!/bin/bash

# 预先处理 Makefile 的选项
case "$@" in
	make-*) makefile_options $@ ;;  # 调用 makefile_options 函数
esac

# 保存初始环境变量
unset INITIAL_SESSION
INITIAL_ENV=$(mktemp -u)  # 创建临时文件用于保存初始环境变量
env > "$INITIAL_ENV"  # 将当前环境变量保存到初始环境文件

notice -n "notice "
warning -n "warning "
error -n "error "

echo

# 检查 SDK 的要求
check_sdk

# 检查会话有效性
if [ -z "$INITIAL_SESSION" ] && [ ! -d "$RK_LOG_DIR" ]; then
	warning "会话($RK_SESSION)无效!"

	# 重新设置会话和日志目录
	export RK_SESSION="$(date +%F_%H-%M-%S)"
	export RK_LOG_DIR="$RK_SESSION_DIR/$RK_SESSION"
	INITIAL_SESSION=1

# 定义环境日志文件
export RK_INITIAL_ENV="$RK_LOG_DIR/initial.env"
export RK_CUSTOM_ENV="$RK_LOG_DIR/custom.env"
export RK_FINAL_ENV="$RK_LOG_DIR/final.env"

# 创建固件目录
mkdir -p "$RK_FIRMWARE_DIR"

# 切换到 SDK 目录
cd "$RK_SDK_DIR"
[ -f README.md ] || ln -rsf "$RK_COMMON_DIR/README.md" .
[ -d common ] || ln -rsf "$RK_COMMON_DIR" .

# TODO: 从 manifest.xml 中移除
rm -f envsetup.sh

# 解析传入的选项，若无则使用默认目标
OPTIONS="${@:-$RK_DEFAULT_TARGET}"

# 特殊处理芯片和 defconfig 选项
for opt in $OPTIONS; do
	if [ -d "$RK_CHIPS_DIR/${opt%%:*}" ]; then
		OPTIONS=$(echo "$OPTIONS" | xargs -n 1 | \
			sed "s/^$opt$/chip:$opt/" | xargs)
	elif echo "$opt" | grep -q "^[0-9a-z_]*_defconfig$"; then
		OPTIONS=$(echo "$OPTIONS" | xargs -n 1 | \
			sed "s/^$opt$/defconfig:$opt/" | xargs)
	fi
done

# 解析支持的命令
if [ ! -r "$RK_PARSED_CMDS" ] || \
	[ "$(find "$RK_SCRIPTS_DIR" -cnewer "$RK_PARSED_CMDS")" ]; then
	message "解析支持的命令...\n"
	rm -rf "$RK_PARSED_CMDS"
	run_build_hooks parse-cmds
fi
source "$RK_PARSED_CMDS"  # 导入解析后的命令

# 检查选项有效性
CMDS="$RK_INIT_CMDS $RK_PRE_BUILD_CMDS $RK_BUILD_CMDS $RK_POST_BUILD_CMDS"
for opt in $OPTIONS; do
	case "$opt" in
		help | h | -h | --help | usage | \?) usage ;;
		clean:*)
			for m in $(echo ${opt#clean:} | tr ':' ' '); do
				grep -wq clean_hook "$RK_SCRIPTS_DIR/mk-$m.sh" 2>/dev/null || usage
			done
			;&
		shell | cleanall)
			if [ "$opt" = "$OPTIONS" ]; then
				break
			fi
			error "错误: $opt 不能与其他选项组合使用!"
			;;
		post-rootfs)
			if [ "$opt" = "$1" -a -d "$2" ]; then
				OPTIONS=$opt  # 隐藏参数以便后续检查
				break
			fi
			error "错误: $opt 应作为第一个选项并跟随 rootfs 目录!"
			;;
		*)
			if option_check "$CMDS" $opt; then
				continue
			fi
			error "错误: 未处理的选项: $opt"
			;;
	esac
	usage
done

# 准备日志目录
if [ ! -d "$RK_LOG_DIR" ]; then
	rm -rf "$RK_LOG_BASE_DIR" "$RK_LOG_DIR" "$RK_SESSION_DIR/latest"
	mkdir -p "$RK_LOG_DIR"
	ln -rsf "$RK_SESSION_DIR" "$RK_LOG_BASE_DIR"
	ln -rsf "$RK_LOG_DIR" "$RK_SESSION_DIR/latest"
	message "日志保存在 $RK_LOG_DIR"
fi

# 删除旧日志，仅保留最近 10 个
cd "$RK_LOG_BASE_DIR"
rm -rf $(ls -t | sed '1,10d')
cd "$RK_SDK_DIR"

# 保存初始环境变量
if [ "$INITIAL_SESSION" ]; then
	rm -f "$RK_INITIAL_ENV"
	cp "$INITIAL_ENV" "$RK_INITIAL_ENV"
	ln -rsf "$RK_INITIAL_ENV" "$RK_OUTDIR/"
fi
rm -f "$INITIAL_ENV"

# 初始化阶段（准备 SDK 配置等）
run_build_hooks init $OPTIONS
rm -f "$RK_OUTDIR/.tmpconfig*"

# 不需要继续执行的情况
CMDS="$RK_PRE_BUILD_CMDS $RK_BUILD_CMDS $RK_POST_BUILD_CMDS cleanall clean post-rootfs"
option_check "$CMDS" $OPTIONS || return 0

# 强制导出配置环境
set -a

# 加载配置环境
source "$RK_CONFIG"
cp "$RK_CONFIG" "$RK_LOG_DIR"
export RK_KERNEL_VERSION="$(kernel_version)"

if [ -z "$INITIAL_SESSION" ]; then
	# 继承会话环境
	sed -n 's/^\(RK_.*=\)\(.*\)/\1"\2"/p' "$RK_FINAL_ENV" > "$INITIAL_ENV"
	source "$INITIAL_ENV"
	rm -f "$INITIAL_ENV"
else
	# 检测和保存自定义环境
	rm -f "$RK_CUSTOM_ENV"
	for cfg in $(grep "^RK_" "$RK_INITIAL_ENV" || true); do
		env | grep -q "^${cfg//\"/}$" || echo "$cfg" >> "$RK_CUSTOM_ENV"
	done

	# 允许自定义环境变量覆盖
	if [ -e "$RK_CUSTOM_ENV" ]; then
		ln -rsf "$RK_CUSTOM_ENV" "$RK_OUTDIR/"

		warning "警告: 发现自定义环境变量:"
		cat "$RK_CUSTOM_ENV"

		warning "假定是预期的设置，请清除非预期变量。"
		read -t 10 -p "按回车键继续。"
		source "$RK_CUSTOM_ENV"

		if grep -q "^RK_KERNEL_VERSION=" "$RK_CUSTOM_ENV"; then
			warning "忽略自定义的 RK_KERNEL_VERSION!"
		fi
		if grep -q "^RK_ROOTFS_SYSTEM=" "$RK_CUSTOM_ENV"; then
			warning "忽略自定义的 RK_ROOTFS_SYSTEM!"
			load_config RK_ROOTFS_SYSTEM
		fi
	fi
fi

# 解析分区表
source "$RK_PARTITION_HELPER"
rk_partition_init

set +a

# 设置实际的内核版本
export RK_KERNEL_VERSION_RAW=$(kernel_version_raw)
export RK_KERNEL_VERSION="$(kernel_version)"

# 处理特殊命令
case "$OPTIONS" in
	cleanall)
		run_build_hooks clean
		rm -rf "$RK_OUTDIR" "$RK_SDK_DIR/rockdev"
		finish_build cleanall
		exit 0 ;;
	clean:*)
		MODULES="$(echo ${OPTIONS#clean:} | tr ':' ' ')"
		for m in $MODULES; do
			"$RK_SCRIPTS_DIR/mk-$m.sh" clean
		done
		finish_build clean - $MODULES
		exit 0 ;;
	post-rootfs)
		shift
		TARGET_DIR="$1"

		source "$RK_POST_HELPER"
		POST_DIR="$RK_OUTDIR/$POST_OS"
		mkdir -p "$POST_DIR"

		touch "$POST_DIR/.stamp_post_start"
		run_post_hooks "$TARGET_DIR"
		touch "$POST_DIR/.stamp_post_finish"

		ln -rsf "$TARGET_DIR" "$POST_DIR/target"
		finish_build post-rootfs

		notice "post-rootfs 阶段更改的文件:"
		cd "$TARGET_DIR"
		find . \( -type f -o -type l \) -cnewer "$POST_DIR/.stamp_post_start" | tee "$POST_DIR/.files_post.txt"
		exit 0 ;;
esac

# 保存最终环境变量
rm -f "$RK_FINAL_ENV"
env > "$RK_FINAL_ENV"
ln -rsf "$RK_FINAL_ENV" "$RK_OUTDIR/"

# 日志记录配置
message "\n=========================================="
message "          最终配置"
message "=========================================="
env | grep -E "^RK_.*=.+" | grep -vE "PARTITION_[0-9]" | \
	grep -vE "=\"\"$|_DEFAULT=y|^RK_DEFAULT_TARGET|CMDS=" | \
	grep -vE "^RK_CONFIG|_BASE_CFG=|_LINK=|DIR=|_ENV=|_NAME=|_DTB=" | \
	grep -vE "_HELPER=|_SUPPORTS=|_ARM64=|_ARM=|_HOST=" | \
	grep -vE "^RK_ROOTFS_SYSTEM_|^RK_YOCTO_DISPLAY_PLATFORM_" | sort
echo

# 构建前阶段（子模块配置等）
run_build_hooks pre-build $OPTIONS

# 构建阶段（编译等）
run_build_hooks build $OPTIONS

# 构建后阶段（固件打包等）
run_build_hooks post-build $OPTIONS
~~~
